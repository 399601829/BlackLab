191     private static class TestException extends Exception {
192         // (intentionally left blank)
193     }
194 
195     // Implementation is synchronized so expected file/directory count is deterministic when throwing exceptions in multithreaded tests
196     // In we didn't synchronize then multiple file() or directory() calls may run simultaneously.
197     // (also ArrayList is not thread-safe)
198     private static class LoggingFileHandler implements FileProcessor.FileHandler {
199         private boolean triggerException;
200 
201         public List<File> dirsReceived = new ArrayList<>();
202         public List<String> filesReceived = new ArrayList<>();
203 
204         public LoggingFileHandler(boolean triggerException) {
205             this.triggerException = triggerException;
206         }
207 
208         @Override
209         public synchronized void directory(File dir) throws Exception {
210             this.dirsReceived.add(dir);
211             if (triggerException)
212                 throw new TestException();
213         }
214 
215         @Override
216         public synchronized void file(String path, InputStream is, File file) throws Exception {
217             this.filesReceived.add(FilenameUtils.getName(path));
218             if (triggerException)
219                 throw new TestException();
220         }
221     }
222 
223     private static class LoggingErrorHandler implements FileProcessor.ErrorHandler {
224         public Throwable caughtException = null;
225 
226         @Override
227         public boolean errorOccurred(Throwable e, String path, File f) {
228             this.caughtException = e;
229             return false;
230         }
231     }
232 
233     @Test
234     public void test() throws FileNotFoundException {
235         LoggingFileHandler fileHandler = new LoggingFileHandler(shouldTriggerException);
236         LoggingErrorHandler errorHandler = new LoggingErrorHandler();
237 
238         try (FileProcessor proc = new FileProcessor(useThreads ? 2 : 1, recurseSubdirs, processArchives)) {
239             proc.setFileHandler(fileHandler);
240             proc.setErrorHandler(errorHandler);
241             proc.processFile(this.inputFile, "");
242         }
243 
244         if (!shouldTriggerException) {
245             // Deterministic results
246             assertEquals(expectedDirectories, fileHandler.dirsReceived.size());
247             assertEquals(expectedFiles, fileHandler.filesReceived.size());
248         } else if (!useThreads) {
249             assertEquals(1, fileHandler.dirsReceived.size() + fileHandler.filesReceived.size());
250         } // else both throwing and using threads, results are nondeterministic
251 
252         assertEquals(shouldTriggerException, errorHandler.caughtException instanceof TestException);
253     }
