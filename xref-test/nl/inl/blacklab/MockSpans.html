33  /**
34   * Stub Spans class for testing. Takes arrays and iterates through 'hits'
35   * from these arrays.
36   */
37  public class MockSpans extends BLSpans {
38  
39  	private final class MockPostingsEnum extends PostingsEnum {
40  		private int currentDoc = -1;
41  		private int currentHit = -1;
42  		private boolean alreadyAtFirstMatch = false;
43  
44  		public MockPostingsEnum() {
45  			// NOP
46  		}
47  
48  		@Override
49  		public int nextDoc() throws IOException {
50  			if (currentDoc != NO_MORE_DOCS) {
51  				alreadyAtFirstMatch = false;
52  				while (currentHit < doc.length && (currentHit == -1 || doc[currentHit] == currentDoc) ) {
53  					currentHit++;
54  				}
55  				if (currentHit >= doc.length) {
56  					currentDoc = NO_MORE_DOCS;
57  					return NO_MORE_DOCS;
58  				}
59  				alreadyAtFirstMatch = true;
60  				currentDoc = doc[currentHit];
61  			}
62  			return currentDoc;
63  		}
64  
65  		@Override
66  		public int docID() {
67  			return currentDoc;
68  		}
69  
70  		@Override
71  		public long cost() {
72  			return 0;
73  		}
74  
75  		@Override
76  		public int advance(int target) throws IOException {
77  			if (currentDoc != NO_MORE_DOCS) {
78  				alreadyAtFirstMatch = false;
79  				do {
80  					currentDoc = nextDoc();
81  				} while(currentDoc != NO_MORE_DOCS && currentDoc < target);
82  			}
83  			return currentDoc;
84  		}
85  
86  		@Override
87  		public int startOffset() throws IOException {
88  			if (currentHit < 0 || alreadyAtFirstMatch)
89  				return -1;
90  			if (currentDoc == NO_MORE_DOCS || currentHit >= doc.length || doc[currentHit] != currentDoc)
91  				return NO_MORE_POSITIONS;
92  			return start[currentHit];
93  		}
94  
95  		@Override
96  		public int nextPosition() throws IOException {
97  			if (currentDoc == NO_MORE_DOCS)
98  				return NO_MORE_POSITIONS;
99  			if (alreadyAtFirstMatch) {
100 				alreadyAtFirstMatch = false;
101 				return startOffset();
102 			}
103 			if (currentHit < 0)
104 				throw new RuntimeException("nextDoc() not called yet!");
105 			if (currentHit < doc.length && doc[currentHit] == currentDoc) {
106 				currentHit++;
107 				return startOffset(); // may return NO_MORE_POSITIONS if we're at the next doc
108 			}
109 			return NO_MORE_POSITIONS;
110 		}
111 
112 		@Override
113 		public BytesRef getPayload() throws IOException {
114 			if (payloads == null)
115 				return null;
116 			if (currentHit < 0 || alreadyAtFirstMatch)
117 				return null;
118 			if (currentDoc == NO_MORE_DOCS || currentHit >= doc.length || doc[currentHit] != currentDoc)
119 				return null;
120 			return new BytesRef(payloads[currentHit]);
121 		}
122 
123 		@Override
124 		public int freq() throws IOException {
125 			// Find start of next document
126 			int i;
127 			for (i = currentHit + 1; i < doc.length && doc[i] == currentDoc; i++) {
128 				// NOP
129 			}
130 			return i - currentHit;
131 		}
132 
133 		@Override
134 		public int endOffset() throws IOException {
135 			if (currentHit < 0 || alreadyAtFirstMatch)
136 				return -1;
137 			if (currentDoc == NO_MORE_DOCS || currentHit >= doc.length || doc[currentHit] != currentDoc)
138 				return NO_MORE_POSITIONS;
139 			return end[currentHit];
140 		}
141 	}

143 	int[] doc;
144 
145 	int[] start;
146 
147 	int[] end;

149 	private MockPostingsEnum postings;
150 
151 	private MyTermSpans spans;
152 
153 	private boolean noMoreDocs = false;
154 
155 	private boolean noMoreHitsInDoc = true;
156 
157 	byte[][] payloads = null;
158 
159 	private int endPos = -1;
160 
161 	public MockSpans(int[] doc, int[] start, int[] end) {
162 		this.doc = doc;
163 		this.start = start;
164 		this.end = end;
165 		postings = new MockPostingsEnum();
166 		spans = new MyTermSpans(postings, new Term("test", "dummy"), 1);
167 	}
168 
169 	@Override
170 	public int docID() {
171 		return spans.docID();
172 	}
173 
174 	@Override
175 	public int endPosition() {
176 		return endPos; //spans.endPosition();
177 	}
178 
179 	private void setPayloadsInt(int[] aEnd) {
180 		this.payloads = new byte[aEnd.length][];
181 		for (int i = 0; i < aEnd.length; i++) {
182 			this.payloads[i] = ByteBuffer.allocate(4).putInt(aEnd[i]).array();
183 		}
184 	}
185 
186 	@Override
187 	public int nextDoc() throws IOException {
188 		if (noMoreDocs)
189 			throw new RuntimeException("Called nextDoc() on exhausted spans!");
190 		endPos = -1;
191 		int docId = spans.nextDoc();
192 		if (docId == NO_MORE_DOCS)
193 			noMoreDocs = true;
194 		else
195 			noMoreHitsInDoc = false;
196 		return docId;
197 	}
198 
199 	@Override
200 	public int nextStartPosition() throws IOException {
201 		if (noMoreHitsInDoc)
202 			throw new RuntimeException("Called nextStartPosition() on hit-exhausted spans!");
203 		int startPos = spans.nextStartPosition();
204 		endPos = startPos == NO_MORE_POSITIONS ? NO_MORE_POSITIONS : postings.endOffset();
205 		if (startPos == NO_MORE_POSITIONS) {
206 			noMoreHitsInDoc = true;
207 		}
208 		return startPos;
209 	}
210 
211 	@Override
212 	public int advance(int target) throws IOException {
213 		if (noMoreDocs)
214 			throw new RuntimeException("Called advance() on exhausted spans!");
215 		endPos = -1;
216 		if (target <= spans.docID())
217 			throw new IllegalArgumentException("target <= doc (" + target + " <= " + spans.docID() + ")");
218 		int docId = spans.advance(target);
219 		if (docId == NO_MORE_DOCS)
220 			noMoreDocs = true;
221 		else
222 			noMoreHitsInDoc = false;
223 		return docId;
224 	}

251 	}

253 	public static MockSpans withEndInPayload(int[] aDoc, int[] aStart, int[] aEnd) {
254 		MockSpans spans = MockSpans.singleWordSpans(aDoc, aStart);
255 		spans.setPayloadsInt(aEnd);
256 		return spans;
257 	}

259 	private static MockSpans singleWordSpans(int[] aDoc, int[] aStart) {
260 		int[] aEnd = new int[aStart.length];
261 		for (int i = 0; i < aStart.length; i++) {
262 			aEnd[i] = aStart[i] + 1;
263 		}
264 		return fromLists(aDoc, aStart, aEnd);
265 	}
