32  public class TestHitPropertySerialize {
33      
34      private MetadataField decadeField = new MockMetadataField("decade");
35  
36      private MockBlackLabIndex mockIndex = new MockBlackLabIndex();
37  
38      private Hits hits = Hits.emptyList(mockIndex.createDefaultQueryInfo());
39  
40      private Annotation lemmaAnnotation;
41  
42      @Before
43      public void setUp() {
44          lemmaAnnotation = mockIndex.mainAnnotatedField().annotation("lemma");
45          mockIndex.setForwardIndex(lemmaAnnotation, new MockForwardIndex(new MockTerms("aap", "noot", "mies")));
46      }
47  
48      @Test
49      public void testHitPropertySerialize() {
50          HitProperty prop;
51  
52          prop = new HitPropertyDocumentDecade(mockIndex, decadeField);
53          Assert.assertEquals("decade:decade", prop.serialize());
54  
55          prop = new HitPropertyDocumentId().reverse();
56          String exp = "-docid";
57          Assert.assertEquals(exp, prop.serialize());
58          Assert.assertEquals(exp, HitProperty.deserialize(hits, exp).serialize());
59  
60          prop = new HitPropertyHitText(mockIndex, lemmaAnnotation, MatchSensitivity.SENSITIVE);
61          Assert.assertEquals("hit:lemma:s", prop.serialize());
62  
63          List<ContextPart> contextParts = Arrays.asList(
64                  ContextPart.get(ContextStart.LEFT_OF_HIT, 1, 1), // second word to left of hit
65                  ContextPart.get(ContextStart.HIT_TEXT_FROM_START, 0, 1), // first two hit words
66                  ContextPart.get(ContextStart.HIT_TEXT_FROM_END, 0, 0) // last hit word
67          );
68          prop = new HitPropertyContextWords(mockIndex, lemmaAnnotation, MatchSensitivity.SENSITIVE, contextParts);
69          Assert.assertEquals("context:lemma:s:L2-2;H1-2;E1-1", prop.serialize());
70      }
71  
72      @Test
73      public void testDocPropertySerialize() {
74          DocProperty prop;
75  
76          prop = new DocPropertyDecade(mockIndex, "decade").reverse();
77          String exp = "-decade:decade";
78          Assert.assertEquals(exp, prop.serialize());
79          Assert.assertEquals(exp, DocProperty.deserialize(mockIndex, exp).serialize());
80      }
81  
82      @Test
83      public void testHitPropValueSerialize() {
84          PropertyValue val, val1;
85  
86          val1 = new PropertyValueContextWord(hits.index(), lemmaAnnotation, MatchSensitivity.SENSITIVE, 2);
87          String exp = "cwo:lemma:s:mies";
88          Assert.assertEquals(exp, val1.serialize());
89          Assert.assertEquals(exp, PropertyValue.deserialize(hits, exp).serialize());
90  
91          val1 = new PropertyValueDecade(1980);
92          exp = "dec:1980";
93          Assert.assertEquals(exp, val1.serialize());
94          Assert.assertEquals(exp, PropertyValue.deserialize(hits, exp).serialize());
95  
96          val = new PropertyValueMultiple(new PropertyValue[] { val1, new PropertyValueString("blabla") });
97          exp = "dec:1980,str:blabla";
98          Assert.assertEquals(exp, val.serialize());
99          Assert.assertEquals(exp, PropertyValue.deserialize(hits, exp).serialize());
100 
101         val = new PropertyValueMultiple(new PropertyValue[] { val1, new PropertyValueString("$bl:ab,la") });
102         exp = "dec:1980,str:$DLbl$CLab$CMla";
103         Assert.assertEquals(exp, val.serialize());
104         Assert.assertEquals(exp, PropertyValue.deserialize(hits, exp).serialize());
105     }
106 }
