35      @BeforeClass
36      public static void setUp() {
37          testIndex = new TestIndex();
38          index = testIndex.index();
39          wordAnnotation = index.mainAnnotatedField().annotation("word");
40          terms = index.annotationForwardIndex(wordAnnotation).terms();
41      }
42  
43      private static int term(String word) {
44          return terms.indexOf(word);
45      }
46  
47      @AfterClass
48      public static void tearDown() {
49          if (testIndex != null)
50              testIndex.close();
51      }

53      @Test
54      public void testHitPropHitText() {
55          Hits hits = testIndex.find(" 'the' ");
56          HitProperty p = new HitPropertyHitText(index, MatchSensitivity.SENSITIVE);
57          HitGroups g = hits.group(p, Results.NO_LIMIT);
58          HitGroup group = g.get(new PropertyValueContextWords(index, wordAnnotation, MatchSensitivity.SENSITIVE, new int[] { term("the") }, false));
59          Assert.assertEquals(3, group.size());
60          group = g.get(new PropertyValueContextWords(index, wordAnnotation, MatchSensitivity.SENSITIVE, new int[] { term("The") }, false));
61          Assert.assertEquals(1, group.size());
62      }
63  
64      @Test
65      public void testHitPropContextWords() {
66          Hits hits = testIndex.find(" 'the' ");
67          HitProperty p = new HitPropertyContextWords(index, wordAnnotation, MatchSensitivity.SENSITIVE, "L1-1;H1-2");
68          HitGroups g = hits.group(p, Results.NO_LIMIT);
69          Assert.assertEquals(4, g.size());
70          HitGroup group;
71          group = g.get(
72                  new PropertyValueContextWords(index, wordAnnotation, MatchSensitivity.SENSITIVE, new int[] { NO_TERM, term("The"), NO_TERM }, false));
73          Assert.assertEquals(1, group.size());
74          group = g.get(
75                  new PropertyValueContextWords(index, wordAnnotation, MatchSensitivity.SENSITIVE, new int[] { term("over"), term("the"), NO_TERM }, false));
76          Assert.assertEquals(1, group.size());
77          group = g.get(
78                  new PropertyValueContextWords(index, wordAnnotation, MatchSensitivity.SENSITIVE, new int[] { term("May"), term("the"), NO_TERM }, false));
79          Assert.assertEquals(1, group.size());
80          group = g.get(
81                  new PropertyValueContextWords(index, wordAnnotation, MatchSensitivity.SENSITIVE, new int[] { term("is"), term("the"), NO_TERM }, false));
82          Assert.assertEquals(1, group.size());
83      }

85      @Test
86      public void testHitPropContextWordsReverse() {
87          Hits hits = testIndex.find(" 'the' 'lazy' ");
88          HitProperty p = new HitPropertyContextWords(index, wordAnnotation, MatchSensitivity.SENSITIVE, "L1;H2-1;R1");
89          HitGroups g = hits.group(p, Results.NO_LIMIT);
90          Assert.assertEquals(1, g.size());
91          HitGroup group;
92          group = g.get(new PropertyValueContextWords(index, wordAnnotation,
93                  MatchSensitivity.SENSITIVE, new int[] { term("over"), term("lazy"), term("the"), term("dog") }, false));
94          Assert.assertEquals(1, group.size());
95      }

97      @Test
98      public void testTermSerialization() {
99          String[] words = { "aap", "~", "~~", "" };
100         String[] expected = { "aap", "~~", "~~~", "" };
101         MockTerms mockTerms = new MockTerms(words);
102         for (int i = 0; i < mockTerms.numberOfTerms(); i++) {
103             Assert.assertEquals(expected[i], mockTerms.serializeTerm(i));
104             Assert.assertEquals(expected[i], mockTerms.serializeTerm(mockTerms.deserializeToken(expected[i])));
105         }
106         Assert.assertEquals("~", mockTerms.serializeTerm(Terms.NO_TERM));
107         Assert.assertEquals("~", mockTerms.serializeTerm(mockTerms.deserializeToken("~")));
108     }
