332     @Test
333     public void testConstraintImplication1() {
334         expected = Arrays.asList(
335                 "[noot mier aap mier] mier", // left side matches, right side holds
336                 "noot [mier aap mier] mier", // left side doesn't match
337                 "noot [noot aap aap] aap", // left side doesn't match
338                 "noot [noot aap aap aap] aap", // left side matches, right side holds
339                 "noot [aap aap aap] aap", // left side doesn't match
340                 "aap [aap aap aap]" // left side doesn't match
341         );
342         // If left side of implication is true, right side must also be true
343         Assert.assertEquals(expected, testIndex.findConc("(c:'noot')? a:[] 'aap' b:[] :: c -> (a.word = b.word)"));
344     }

357     @Test
358     public void testSort() {
359         expected = Arrays.asList(
360                 "aap [aap aap aap]",
361                 "noot [aap aap aap] aap",
362                 "noot [mier aap mier] mier",
363                 "noot [noot aap aap] aap"
364                 );
365         // If left side of implication is always false, right side is ignored
366         HitProperty hit = new HitPropertyHitText(testIndex.index(), MatchSensitivity.INSENSITIVE);
367         HitProperty left = new HitPropertyLeftContext(testIndex.index(), MatchSensitivity.INSENSITIVE);
368         HitProperty sortBy = new HitPropertyMultiple(hit, left);
369         Assert.assertEquals(expected, testIndex.findConc("(c:'NOTININDEX')? a:[] 'aap' b:[] :: c -> a.word = b.word", sortBy));
370     }

387     @Test
388     public void testFilter() {
389         expected = Arrays.asList(
390                 "noot [noot aap aap] aap"
391                 );
392         // If left side of implication is always false, right side is ignored
393         BlackLabIndex index = testIndex.index();
394         HitProperty prop = new HitPropertyHitText(index, MatchSensitivity.INSENSITIVE);
395         Annotation annotation = index.mainAnnotatedField().mainAnnotation();
396         Terms terms = index.annotationForwardIndex(annotation).terms();
397         int[] words = { terms.indexOf("noot"), terms.indexOf("aap"), terms.indexOf("aap") };
398         PropertyValue value = new PropertyValueContextWords(index, annotation, MatchSensitivity.INSENSITIVE, words, false);
399         Assert.assertEquals(expected, testIndex.findConc("(c:'NOTININDEX')? a:[] 'aap' b:[] :: c -> a.word = b.word", prop, value));
400     }
