18  import org.junit.AfterClass;
19  import org.junit.Assert;
20  import org.junit.BeforeClass;
21  import org.junit.Ignore;
22  import org.junit.Test;
23  
24  import nl.inl.blacklab.TestIndex;
25  import nl.inl.blacklab.exceptions.BlackLabRuntimeException;
26  import nl.inl.blacklab.exceptions.InvalidQuery;
27  import nl.inl.blacklab.queryParser.corpusql.CorpusQueryLanguageParser;
28  import nl.inl.blacklab.search.BlackLabIndex;
29  import nl.inl.blacklab.search.QueryExplanation;
30  import nl.inl.blacklab.search.lucene.BLSpanQuery;
31  import nl.inl.blacklab.search.lucene.optimize.ClauseCombinerNfa;
32  import nl.inl.blacklab.search.results.QueryInfo;
33  import nl.inl.blacklab.search.textpattern.TextPattern;
34  
35  public class TestQueryRewriteNfa {
36  
37      static TestIndex testIndex;
38  
39      private static BlackLabIndex index;
40  
41      @BeforeClass
42      public static void setUp() {
43          testIndex = new TestIndex();
44          index = testIndex.index();
45          ClauseCombinerNfa.setNfaThreshold(ClauseCombinerNfa.MAX_NFA_MATCHING);
46      }
47  
48      @AfterClass
49      public static void tearDown() {
50          ClauseCombinerNfa.setNfaThreshold(ClauseCombinerNfa.DEFAULT_NFA_THRESHOLD);
51          if (index != null)
52              index.close();
53          if (testIndex != null)
54              testIndex.close();
55      }
56  
57      static TextPattern getPatternFromCql(String cqlQuery) {
58          cqlQuery = cqlQuery.replaceAll("'", "\""); // makes queries more readable in tests
59          try {
60              return CorpusQueryLanguageParser.parse(cqlQuery);
61          } catch (InvalidQuery e) {
62              throw BlackLabRuntimeException.wrap(e);
63          }
64      }
65  
66      void assertNoRewrite(String cql, String result) {
67          assertRewrite(cql, result, result);
68      }
69  
70      void assertRewrite(String cql, String before, String after) {
71          try {
72              QueryExplanation explanation = index.explain(QueryInfo.create(index), getPatternFromCql(cql), null);
73              if (before != null) {
74                  BLSpanQuery original = explanation.originalQuery();
75                  Assert.assertEquals(before, original.toString());
76              }
77              BLSpanQuery rewritten = explanation.rewrittenQuery();
78              Assert.assertEquals(after, rewritten.toString());
79          } catch (InvalidQuery e) {
80              throw BlackLabRuntimeException.wrap(e);
81          }
82      }
83  
84      void assertRewriteResult(String cql, String after) {
85          assertRewrite(cql, null, after);
86      }
87  
88      @Test
89      public void testRewrite() {
90          assertRewriteResult("\"the\" [word=\"quick\" & lemma=\"quick\"] [lemma=\"brown\"]",
91                  "FISEQ(FISEQ(AND(TERM(contents%word@i:quick), TERM(contents%lemma@i:quick)), NFA:#1:TOKEN(brown,DANGLING), 1), NFA:#1:TOKEN(the,DANGLING), -1)");
92      }
93  
94      @Ignore // hard to test properly with tiny indices
95      @Test
96      public void testRewriteSuffix() {
97          assertRewriteResult("\"noot\" \".*p\"",
98                  "FISEQ(TERM(contents%word@i:noot), NFA:#1:REGEX(^.*p$,DANGLING), 1)");
99      }
100 
101     @Test
102     public void testRewritePrefix() {
103         assertRewriteResult("\"a.*\" \"b.*\" \"c.*\"",
104                 "FISEQ(TERM(contents%word@i:aap), NFA:#1:REGEX(b.*,#2:REGEX(c.*,DANGLING)), 1)");
105     }
106 
107 }
