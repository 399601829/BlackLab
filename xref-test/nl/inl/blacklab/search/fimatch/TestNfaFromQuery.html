29  	static class MockForwardIndexAccessor extends ForwardIndexAccessor {
30  
31  		int[] termIds;
32  
33  		private final Map<String, Integer> terms = new HashMap<>();
34  
35  		public MockForwardIndexAccessor(String... document) {
36  			this.termIds = new int[document.length];
37  			for (int i = 0; i < document.length; i++) {
38  				Integer termId = terms.get(document[i]);
39  				if (termId == null) {
40  					termId = terms.size();
41  					terms.put(document[i], termId);
42  				}
43  				termIds[i] = termId;
44  			}
45  		}
46  
47  		@Override
48  		public int getPropertyNumber(String propertyName) {
49  			if (propertyName.equals("word"))
50  				return 0;
51  			throw new IllegalArgumentException("Unknown property " + propertyName);
52  		}
53  
54  		@Override
55  		public void getTermNumbers(MutableIntSet results, int propertyNumber, String propertyValue, boolean caseSensitive, boolean diacSensitive) {
56  			if (propertyNumber != 0)
57  				throw new IllegalArgumentException("Unknown property " + propertyNumber);
58  			if (caseSensitive) {
59  				results.add(terms.get(propertyValue));
60  				return;
61  			}
62  			for (Entry<String, Integer> e: terms.entrySet()) {
63  				if (e.getKey().equalsIgnoreCase(propertyValue)) {
64  					results.add(e.getValue());
65  				}
66  			}
67  		}
68  
69  		@Override
70  		public int numberOfProperties() {
71  			return 1;
72  		}

98  					return Arrays.copyOfRange(termIds, start, end);
99  				}

104 				}
105 
106 			};

159 	private static void test(BLSpanQuery q, ForwardIndexAccessor fiAccessor, int startPos, int direction, int tests, List<Integer> matches) {
160 		// The NFA
161 		Nfa frag = q.getNfa(fiAccessor, direction);
162 		frag.finish();
163 		frag.lookupPropertyNumbers(fiAccessor, new IdentityHashMap<NfaState, Boolean>());
164 		//System.err.println(frag);
165 		NfaState start = frag.getStartingState(); //finish();
166 
167 		ForwardIndexDocument fiDoc = fiAccessor.getForwardIndexAccessorLeafReader(null).getForwardIndexDoc(0);
168 		for (int i = 0; i < tests; i++) {
169 			Assert.assertEquals("Test " + i, matches.contains(i), start.matches(fiDoc, startPos + direction * i, direction));
170 		}
171 	}
