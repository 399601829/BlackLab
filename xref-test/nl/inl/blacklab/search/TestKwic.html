1   package nl.inl.blacklab.search;

3   import java.util.Arrays;
4   import java.util.List;
5   
6   import org.junit.Assert;
7   import org.junit.Test;
8   
9   public class TestKwic {
10  
11  	List<String> props = Arrays.asList("punct", "lemma", "pos", "word");
12  	List<String> tokens = Arrays.asList(
13  			" ", "de", "lw", "De",
14  			" ", "snel", "bn", "snelle",
15  			" ", "bruin", "bn", "bruine",
16  			" ", "vos", "zn", "vos");
17  	List<String> left = Arrays.asList(" ", "de", "lw", "De", " ", "snel", "bn", "snelle");
18  	List<String> match = Arrays.asList(" ", "bruin", "bn", "bruine");
19  	List<String> right = Arrays.asList(" ", "vos", "zn", "vos");
20  	String expLeft = "<w lemma=\"de\" pos=\"lw\">De</w> <w lemma=\"snel\" pos=\"bn\">snelle</w> ";
21  	String expMatch = "<w lemma=\"bruin\" pos=\"bn\">bruine</w>";
22  	String expRight = " <w lemma=\"vos\" pos=\"zn\">vos</w>";
23  
24  	@Test
25  	public void testKwicToConcordance() {
26  		Kwic kwic = new Kwic(props, tokens, 2, 3);
27  		Concordance conc = kwic.toConcordance();
28  
29  		Assert.assertEquals(expLeft, conc.left());
30  		Assert.assertEquals(expMatch, conc.match());
31  		Assert.assertEquals(expRight, conc.right());
32  	}
33  
34  	@SuppressWarnings("deprecation")
35  	@Test
36  	public void testKwicOldConstructor() {
37  		Kwic kwic = new Kwic(props, left, match, right);
38  
39  		Assert.assertEquals(expLeft + expMatch + expRight, kwic.getFullXml());
40  		Assert.assertEquals(left, kwic.getLeft());
41  		Assert.assertEquals(match, kwic.getMatch());
42  		Assert.assertEquals(right, kwic.getRight());
43  		Assert.assertEquals(Arrays.asList("De", "snelle"), kwic.getLeft("word"));
44  	}
45  
46  	@Test
47  	public void testKwicNewConstructor() {
48  		Kwic kwic = new Kwic(props, tokens, 2, 3);
49  
50  		Assert.assertEquals(expLeft + expMatch + expRight, kwic.getFullXml());
51  		Assert.assertEquals(left, kwic.getLeft());
52  		Assert.assertEquals(match, kwic.getMatch());
53  		Assert.assertEquals(right, kwic.getRight());
54  		Assert.assertEquals(Arrays.asList("De", "snelle"), kwic.getLeft("word"));
55  	}
56  }
