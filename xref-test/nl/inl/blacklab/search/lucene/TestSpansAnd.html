16  package nl.inl.blacklab.search.lucene;
17  
18  import java.io.IOException;
19  
20  import org.apache.lucene.search.spans.Spans;
21  import org.junit.Assert;
22  import org.junit.Test;
23  
24  public class TestSpansAnd {
25  	@Test
26  	public void testAndSpans() throws IOException {
27  		int[] aDoc = new int[] { 1, 1, 2, 2, 2, 3 };
28  		int[] aStart = new int[] { 10, 20, 10, 10, 30, 10 };
29  		int[] aEnd = new int[] { 15, 25, 15, 20, 35, 15 };
30  		Spans a = new SpansStub(aDoc, aStart, aEnd);
31  
32  		int[] bDoc = new int[] { 1, 2, 2, 3 };
33  		int[] bStart = new int[] { 10, 10, 20, 20 };
34  		int[] bEnd = new int[] { 15, 20, 25, 25 };
35  		Spans b = new SpansStub(bDoc, bStart, bEnd);
36  
37  		SpansAnd andSpans = new SpansAnd(a, b);
38  
39  		// First hit
40  		Assert.assertTrue(andSpans.next());
41  		Assert.assertEquals(1, andSpans.doc());
42  		Assert.assertEquals(10, andSpans.start());
43  		Assert.assertEquals(15, andSpans.end());
44  
45  		// Second hit
46  		Assert.assertTrue(andSpans.next());
47  		Assert.assertEquals(2, andSpans.doc());
48  		Assert.assertEquals(10, andSpans.start());
49  		Assert.assertEquals(20, andSpans.end());
50  
51  		// Done
52  		Assert.assertFalse(andSpans.next());
53  	}
54  }
