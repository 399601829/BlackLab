78      /**
79       * Handle Document element. Starts a new Lucene document and adds the attributes
80       * of this element (if any) as metadata fields.
81       */
82      public class DocumentElementHandler extends ElementHandler {
83  
84          /** Open tag: start indexing this document */
85          @Override
86          public void startElement(String uri, String localName, String qName,
87                  Attributes attributes) {
88              startCaptureContent(contentsField.name());
89  
90              currentLuceneDoc = new Document();
91              // Store attribute values from the tag as metadata fields
92              for (int i = 0; i < attributes.getLength(); i++) {
93                  addMetadataField(attributes.getLocalName(i), attributes.getValue(i));
94              }
95              addMetadataField("fromInputFile", documentName);
96              addMetadataFieldsFromParameters();
97              docWriter.listener().documentStarted(documentName);
98          }
99  
100         /** Open tag: end indexing the document */
101         @Override
102         public void endElement(String uri, String localName, String qName) {
103             // Make sure all the properties have an equal number of values.
104             // See what annotation has the highest position
105             // (in practice, only starttags and endtags should be able to have
106             // a position one higher than the rest)
107             int lastValuePos = 0;
108             for (AnnotationWriter prop: contentsField.annotationWriters()) {
109                 if (prop.lastValuePosition() > lastValuePos)
110                     lastValuePos = prop.lastValuePosition();
111             }
112 
113             // Make sure we always have one more token than the number of
114             // words, so there's room for any tags after the last word, and we
115             // know we should always skip the last token when matching.
116             if (propMain.lastValuePosition() == lastValuePos)
117                 lastValuePos++;
118 
119             // Add empty values to all lagging properties
120             for (AnnotationWriter prop: contentsField.annotationWriters()) {
121                 while (prop.lastValuePosition() < lastValuePos) {
122                     prop.addValue("");
123                     if (prop.hasPayload())
124                         prop.addPayload(null);
125                     if (prop == propMain) {
126                         contentsField.addStartChar(getCharacterPosition());
127                         contentsField.addEndChar(getCharacterPosition());
128                     }
129                 }
130             }
131 
132             // Finish storing the document in the document store (parts of it
133             // may already have been written because we write in chunks to save memory),
134             // retrieve the content id, and store that in Lucene.
135             // (Note that we do this after adding the dummy token, so the character
136             // positions for the dummy token still make (some) sense)
137             int contentId = storeCapturedContent();
138             currentLuceneDoc.add(new IntField(AnnotatedFieldNameUtil
139                     .contentIdField(contentsField.name()), contentId,
140                     Store.YES));
141 
142             // Store the different properties of the annotated contents field that
143             // were gathered in
144             // lists while parsing.
145             contentsField.addToLuceneDoc(currentLuceneDoc);
146 
147             // Add field with all its annotations to the forward index
148             addToForwardIndex(contentsField);
149 
150             // If there's an external metadata fetcher, call it now so it can
151             // add the metadata for this document and (optionally) store the
152             // metadata
153             // document in the content store (and the corresponding id in the
154             // Lucene doc)
155             MetadataFetcher m = getMetadataFetcher();
156             if (m != null) {
157                 m.addMetadata();
158             }
159 
160             addMetadataToDocument();
161 
162             try {
163                 // Add Lucene doc to indexer
164                 docWriter.add(currentLuceneDoc);
165             } catch (Exception e) {
166                 throw BlackLabRuntimeException.wrap(e);
167             }
168 
169             // Report progress
170             reportCharsProcessed();
171             reportTokensProcessed();
172 
173             docWriter.listener().documentDone(documentName);
174 
175             // Reset contents field for next document
176             contentsField.clear(true);
177             currentLuceneDoc = null;
178 
179             // Stop if required
180             if (!docWriter.continueIndexing())
181                 throw new MaxDocsReached();
182         }
183     }

253     /** Handle tags. */
254     public class InlineTagHandler extends ElementHandler {
255 
256         /** Open tag: store the start tag location and the attribute values */
257         @Override
258         public void startElement(String uri, String localName, String qName,
259                 Attributes attributes) {
260             int lastStartTagPos = propTags.lastValuePosition();
261             int currentPos = propMain.lastValuePosition() + 1;
262             int posIncrement = currentPos - lastStartTagPos;
263             propTags.addValue(localName, posIncrement);
264             propTags.addPayload(null);
265             int startTagIndex = propTags.lastValueIndex();
266             openTagIndexes.add(startTagIndex);
267             for (int i = 0; i < attributes.getLength(); i++) {
268                 // Index element attribute values
269                 String name = attributes.getLocalName(i);
270                 String value = attributes.getValue(i);
271                 propTags.addValue("@" + name.toLowerCase() + "__" + value.toLowerCase(), 0);
272                 propTags.addPayload(null);
273             }
274         }
275 
276         /** Close tag: store the end tag location */
277         @Override
278         public void endElement(String uri, String localName, String qName) {
279             int currentPos = propMain.lastValuePosition() + 1;
280 
281             // Add payload to start tag annotation indicating end position
282             Integer openTagIndex = openTagIndexes.remove(openTagIndexes.size() - 1);
283             byte[] payload = ByteBuffer.allocate(4).putInt(currentPos).array();
284             propTags.setPayloadAtIndex(openTagIndex, new BytesRef(payload));
285         }

427     /**
428      * StringBuffer re-used for building start/end tags and processing instructions.
429      */
430     StringBuilder elementBuilder = new StringBuilder();
431 
432     public void startElement(String uri, String localName, String qName,
433             Attributes attributes) {
434         // Call any hooks associated with this element
435         hookableHandler.startElement(uri, localName, qName, attributes);
436 
437         elementBuilder.setLength(0); // clear
438         elementBuilder.append("<").append(qName);
439         for (int i = 0; i < attributes.getLength(); i++) {
440             String value = StringEscapeUtils.escapeXml10(attributes.getValue(i));
441             elementBuilder.append(" ").append(attributes.getQName(i))
442                     .append("=\"").append(value).append("\"");
443         }
444         // Append any namespace mapping not yet outputted
445         if (!outputPrefixMapping.isEmpty()) {
446             for (Map.Entry<String, String> e: outputPrefixMapping.entrySet()) {
447                 if (e.getKey().length() == 0)
448                     elementBuilder.append(" xmlns=\"").append(e.getValue())
449                             .append("\"");
450                 else
451                     elementBuilder.append(" xmlns:").append(e.getKey())
452                             .append("=\"").append(e.getValue()).append("\"");
453             }
454             outputPrefixMapping.clear(); // outputted all prefix mappings for
455                                          // now
456         }
457         elementBuilder.append(">");
458         processContent(elementBuilder.toString());
459     }

479     public void characters(char[] buffer, int start, int length) {
480         // Capture character content in string builder
481         if (characterContent.length() < MAX_CHARACTER_CONTENT_CAPTURE_LENGTH)
482             characterContent.append(buffer, start, length);
483 
484         String s = StringEscapeUtils.escapeXml10(new String(buffer, start, length));
485         processContent(s);
486 
487         // Call any hooks associated with this element
488         hookableHandler.characters(buffer, start, length);
489     }

512     /**
513      * Get the external metadata fetcher for this indexer, if any.
514      *
515      * The metadata fetcher can be configured through the "metadataFetcherClass"
516      * parameter.
517      *
518      * @return the metadata fetcher if any, or null if there is none.
519      */
520     MetadataFetcher getMetadataFetcher() {
521         if (metadataFetcher == null) {
522             @SuppressWarnings("deprecation")
523             String metadataFetcherClassName = getParameter("metadataFetcherClass");
524             if (metadataFetcherClassName != null) {
525                 try {
526                     Class<? extends MetadataFetcher> metadataFetcherClass = Class.forName(metadataFetcherClassName)
527                             .asSubclass(MetadataFetcher.class);
528                     Constructor<? extends MetadataFetcher> ctor = metadataFetcherClass.getConstructor(DocIndexer.class);
529                     metadataFetcher = ctor.newInstance(this);
530                 } catch (ReflectiveOperationException e) {
531                     throw new BlackLabRuntimeException(e);
532                 }
533             }
534         }
535         return metadataFetcher;
536     }

591     @Override
592     public void index() throws IOException, MalformedInputFile {
593         SAXParserFactory factory = SAXParserFactory.newInstance();
594         factory.setNamespaceAware(true);
595         SAXParser parser;
596         try {
597             parser = factory.newSAXParser();
598         } catch (SAXException | ParserConfigurationException e1) {
599             // Unrecoverable error, throw runtime exception
600             throw BlackLabRuntimeException.wrap(e1);
601         }
602         try {
603             InputSource is = new InputSource(reader);
604             XMLReader xmlReader = parser.getXMLReader();
605             xmlReader.setEntityResolver(new DummyEntityResolver());
606             xmlReader.setContentHandler(saxParseHandler);
607             xmlReader.parse(is);
608         } catch (SAXException e) {
609             throw new MalformedInputFile(e);
610         } catch (MaxDocsReached e) {
611             // OK; just stop indexing prematurely
612         }
613 
614         if (nDocumentsSkipped > 0)
615             System.err.println("Skipped " + nDocumentsSkipped + " large documents");
616     }
