16  package nl.inl.blacklab.index;
17  
18  import java.io.File;
19  
20  import nl.inl.util.TimeUtil;
21  
22  /**
23   * Used to report progress while indexing, so we can give feedback to the user.
24   */
25  public class IndexListenerReportConsole extends IndexListener {
26  	final int REPORT_INTERVAL_SEC = 10;
27  
28  	long prevCharsDoneReported = 0;
29  
30  	long prevTokensDoneReported = 0;
31  
32  	double prevReportTime = 0;
33  
34  	double curSpeed = -1;
35  
36  	double curTokensSpeed = -1;
37  
38  	@Override
39  	public synchronized void charsDone(long charsDone) {
40  		super.charsDone(charsDone);
41  
42  		reportProgress();
43  	}
44  
45  	private void reportProgress() {
46  		reportProgress(false);
47  	}
48  
49  	private void reportProgress(boolean force) {
50  		double elapsed = getElapsed();
51  		if (elapsed == 0)
52  			elapsed = 0.1;
53  		double secondsSinceLastReport = elapsed - prevReportTime;
54  		if (force || secondsSinceLastReport >= REPORT_INTERVAL_SEC) {
55  			long totalCharsDone = getCharsProcessed();
56  			long charsDoneSinceLastReport = totalCharsDone - prevCharsDoneReported;
57  
58  			double lastMbDone = charsDoneSinceLastReport / 1000000.0;
59  			double lastSpeed = lastMbDone / secondsSinceLastReport;
60  
61  			double mbDone = totalCharsDone / 1000000.0;
62  			double overallSpeed = mbDone / elapsed;
63  
64  			if (curSpeed < 0)
65  				curSpeed = overallSpeed;
66  			curSpeed = curSpeed * 0.7 + lastSpeed * 0.3;
67  
68  			long totalTokensDone = getTokensProcessed();
69  			long tokensDoneSinceLastReport = totalTokensDone - prevTokensDoneReported;
70  
71  			double lastKDone = tokensDoneSinceLastReport / 1000.0;
72  			double lastTokensSpeed = lastKDone / secondsSinceLastReport;
73  
74  			double kTokensDone = totalTokensDone / 1000.0;
75  			double overallTokenSpeed = kTokensDone / elapsed;
76  
77  			if (curTokensSpeed < 0)
78  				curTokensSpeed = overallTokenSpeed;
79  			curTokensSpeed = curTokensSpeed * 0.7 + lastTokensSpeed * 0.3;
80  
81  			System.out
82  					.printf("%d docs done (%d MB, %dk tokens). Average speed %.1fk tokens/s (%.1f MB/s), currently %.1fk tokens/s (%.1f MB/s)\n",
83  							getDocsDone(), (int) mbDone, (int) kTokensDone, overallTokenSpeed,
84  							overallSpeed, curTokensSpeed, curSpeed);
85  
86  			prevCharsDoneReported = totalCharsDone;
87  			prevTokensDoneReported = totalTokensDone;
88  			prevReportTime = elapsed;
89  		}
90  	}
91  
92  	private double getElapsed() {
93  		return (System.currentTimeMillis() - getIndexStartTime()) / 1000.0;
94  	}
95  
96  	@Override
97  	public void indexEnd() {
98  		super.indexEnd();
99  		reportProgress(true);
100 		System.out.println("Done. Elapsed time: " + TimeUtil.describeInterval(System.currentTimeMillis() - getIndexStartTime()));
101 	}
102 
103 	/**
104 	 * An index error occurred. Report it.
105 	 *
106 	 * @param error type of error, i.e. "not found"
107 	 * @param unitType type of indexing unit, i.e. "file", "zip", "tgz"
108 	 * @param unit the indexing unit in which the error occurred
109 	 * @param subunit optional subunit (i.e. which file inside zip, or null for regular files)
110 	 * @return true if indexing should continue
111 	 */
112 	@Override
113 	public boolean errorOccurred(String error, String unitType, File unit, File subunit) {
114 		System.out.println("An error occurred during indexing!");
115 		System.out.println("error: " + error + ", unitType: " + unitType +
116 				", unit: " + unit + ", subunit: " + subunit);
117 		return super.errorOccurred(error, unitType, unit, subunit);
118 	}
119 
120 }
