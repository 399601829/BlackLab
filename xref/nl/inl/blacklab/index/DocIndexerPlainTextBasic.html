44  /**
45   * Simple example indexer for plain text files. Reads a line, chops it into
46   * words to index and keeps track of word positions.
47   */
48  public class DocIndexerPlainTextBasic extends DocIndexerAbstract {

50  	/**
51  	 * If any metadata fields were supplied in the indexer parameters,
52  	 * add them now.
53  	 */
54  	void addMetadataFieldsFromParameters() {
55  		for (Entry<String, String> e: parameters.entrySet()) {
56  			if (e.getKey().startsWith("meta-")) {
57  				String fieldName = e.getKey().substring(5);
58  				String fieldValue = e.getValue();
59  				currentLuceneDoc.add(new Field(fieldName, fieldValue, indexer.metadataFieldTypeUntokenized));
60  			}
61  		}
62  	}

64  	SensitivitySetting getSensitivitySetting(String propName) {
65  		// See if it's specified in a parameter
66  		String strSensitivity = getParameter(propName + "_sensitivity");
67  		if (strSensitivity != null) {
68  			if (strSensitivity.equals("i"))
69  				return SensitivitySetting.ONLY_INSENSITIVE;
70  			if (strSensitivity.equals("s"))
71  				return SensitivitySetting.ONLY_SENSITIVE;
72  			if (strSensitivity.equals("si") || strSensitivity.equals("is"))
73  				return SensitivitySetting.SENSITIVE_AND_INSENSITIVE;
74  			if (strSensitivity.equals("all"))
75  				return SensitivitySetting.CASE_AND_DIACRITICS_SEPARATE;
76  		}
77  
78  		// Not in parameter (or unrecognized value), use default based on
79  		// propName
80  		if (propName.equals(ComplexFieldUtil.getDefaultMainPropName()) || propName.equals(ComplexFieldUtil.LEMMA_PROP_NAME)) {
81  			// Word: default to sensitive/insensitive
82  			return SensitivitySetting.SENSITIVE_AND_INSENSITIVE;
83  		}
84  		if (propName.equals(ComplexFieldUtil.PUNCTUATION_PROP_NAME)) {
85  			// Punctuation: default to only insensitive
86  			return SensitivitySetting.ONLY_INSENSITIVE;
87  		}
88  		if (propName.equals(ComplexFieldUtil.START_TAG_PROP_NAME) || propName.equals(ComplexFieldUtil.END_TAG_PROP_NAME)) {
89  			// XML tag properties: default to only sensitive
90  			return SensitivitySetting.ONLY_SENSITIVE;
91  		}
92  
93  		// Unrecognized; default to only insensitive
94  		return SensitivitySetting.ONLY_INSENSITIVE;
95  	}

105 	public DocIndexerPlainTextBasic(Indexer indexer, String fileName, Reader reader) {
106 		super(indexer, fileName, reader);
107 
108 		// Define the properties that make up our complex field
109 		String mainPropName = ComplexFieldUtil.getDefaultMainPropName();
110 		contentsField = new ComplexField(Searcher.DEFAULT_CONTENTS_FIELD_NAME, mainPropName, getSensitivitySetting(mainPropName), false);
111 		propMain = contentsField.getMainProperty();
112 		propPunct = addProperty(ComplexFieldUtil.PUNCTUATION_PROP_NAME);
113 		IndexStructure indexStructure = indexer.getSearcher().getIndexStructure();
114 		indexStructure.registerComplexField(contentsField.getName(), propMain.getName());
115 
116 		// If the indexmetadata file specified a list of properties that shouldn't get a forward
117 		// index,
118 		// make the new complex field aware of this.
119 		Set<String> noForwardIndexProps = indexStructure.getComplexFieldDesc(Searcher.DEFAULT_CONTENTS_FIELD_NAME).getNoForwardIndexProps();
120 		contentsField.setNoForwardIndexProps(noForwardIndexProps);
121 	}

163 	/**
164 	 * Get the external metadata fetcher for this indexer, if any.
165 	 *
166 	 * The metadata fetcher can be configured through the "metadataFetcherClass"
167 	 * parameter.
168 	 *
169 	 * @return the metadata fetcher if any, or null if there is none.
170 	 */
171 	MetadataFetcher getMetadataFetcher() {
172 		if (metadataFetcher == null) {
173 			String metadataFetcherClassName = getParameter("metadataFetcherClass");
174 			if (metadataFetcherClassName != null) {
175 				try {
176 					Class<? extends MetadataFetcher> metadataFetcherClass = Class.forName(metadataFetcherClassName)
177 							.asSubclass(MetadataFetcher.class);
178 					Constructor<? extends MetadataFetcher> ctor = metadataFetcherClass.getConstructor(DocIndexer.class);
179 					metadataFetcher = ctor.newInstance(this);
180 				} catch (Exception e) {
181 					throw new RuntimeException(e);
182 				}
183 			}
184 		}
185 		return metadataFetcher;
186 	}

215 	public void addMetadataField(String name, String value) {
216 
217 		IndexStructure struct = indexer.getSearcher().getIndexStructure();
218 		struct.registerMetadataField(name);
219 
220 		MetadataFieldDesc desc = struct.getMetadataFieldDesc(name);
221 		FieldType type = desc.getType();
222 		desc.addValue(value);
223 
224 		FieldType shouldBeType = getMetadataFieldTypeFromIndexerProperties(name);
225 		if (type == FieldType.TEXT && shouldBeType != FieldType.TEXT) {
226 			// indexer.properties overriding default type
227 			type = shouldBeType;
228 		}
229 		if (type != FieldType.NUMERIC) {
230 			currentLuceneDoc.add(new Field(name, value, luceneTypeFromIndexStructType(type)));
231 		}
232 		if (type == FieldType.NUMERIC || numericFields.contains(name)) {
233 			String numFieldName = name;
234 			if (type != FieldType.NUMERIC) {
235 				numFieldName += "Numeric";
236 			}
237 			// Index these fields as numeric too, for faster range queries
238 			// (we do both because fields sometimes aren't exclusively numeric)
239 			int n = 0;
240 			try {
241 				n = Integer.parseInt(value);
242 			} catch (NumberFormatException e) {
243 				// This just happens sometimes, e.g. given multiple years, or
244 				// descriptive text like "around 1900". OK to ignore.
245 			}
246 			IntField nf = new IntField(numFieldName, n, Store.YES);
247 			currentLuceneDoc.add(nf);
248 		}
249 	}

251 	@Override
252 	public void index() throws IOException, InputFormatException {
253 		BufferedReader r = new BufferedReader(reader);
254 		boolean firstWord = true;
255 
256 		// Start a new Lucene document
257 		currentLuceneDoc = new Document();
258 		currentDocumentName = fileName;
259 		if (currentDocumentName == null)
260 			currentDocumentName = "?";
261 		currentLuceneDoc.add(new Field("fromInputFile", currentDocumentName, indexer.metadataFieldTypeUntokenized));
262 		addMetadataFieldsFromParameters();
263 		indexer.getListener().documentStarted(currentDocumentName);
264 
265 		while (true) {
266 			// For each line, split on whitespace and index each word.
267 			String line = r.readLine();
268 			if (line == null)
269 				break;
270 			String[] words = line.trim().split("\\s+");
271 			for (int i = 0; i < words.length; i++) {
272 				// Handle space and punctuation between words. Instead of always using a hardcoded
273 				// space,
274 				// you would want to use smarter tokenization and actually capture the space and
275 				// punctuation between words (right now the punctuation becomes part of the word,
276 				// because
277 				// we naively split on whitespace).
278 				String punctuation = "";
279 				if (!firstWord) {
280 					punctuation = " ";
281 					processContent(punctuation);
282 				}
283 				firstWord = false;
284 				propPunct.addValue(punctuation);
285 
286 				// Handle the word itself, including character positions.
287 				contentsField.addStartChar(getCharacterPosition());
288 				processContent(words[i]); // add word to content store
289 				contentsField.addEndChar(getCharacterPosition());
290 				propMain.addValue(words[i]); // add word to index
291 
292 				// Report progress regularly but not too often
293 				wordsDone++;
294 				if (wordsDone >= 5000) {
295 					reportCharsProcessed();
296 					reportTokensProcessed(wordsDone);
297 					wordsDone = 0;
298 				}
299 			}
300 
301 			// Make sure all the properties have an equal number of values.
302 			// See what property has the highest position
303 			// (in practice, only starttags and endtags should be able to have
304 			// a position one higher than the rest)
305 			int lastValuePos = 0;
306 			for (ComplexFieldProperty prop: contentsField.getProperties()) {
307 				if (prop.lastValuePosition() > lastValuePos)
308 					lastValuePos = prop.lastValuePosition();
309 			}
310 
311 			// Make sure we always have one more token than the number of
312 			// words, so there's room for any tags after the last word, and we
313 			// know we should always skip the last token when matching.
314 			if (propMain.lastValuePosition() == lastValuePos)
315 				lastValuePos++;
316 
317 			// Add empty values to all lagging properties
318 			for (ComplexFieldProperty prop: contentsField.getProperties()) {
319 				while (prop.lastValuePosition() < lastValuePos) {
320 					prop.addValue("");
321 					if (prop.hasPayload())
322 						prop.addPayload(null);
323 					if (prop == propMain) {
324 						contentsField.addStartChar(getCharacterPosition());
325 						contentsField.addEndChar(getCharacterPosition());
326 					}
327 				}
328 			}
329 
330 			// Finish storing the document in the document store (parts of it
331 			// may already have been written because we write in chunks to save memory),
332 			// retrieve the content id, and store that in Lucene.
333 			// (Note that we do this after adding the dummy token, so the character
334 			// positions for the dummy token still make (some) sense)
335 			int contentId = storeCapturedContent();
336 			currentLuceneDoc.add(new IntField(ComplexFieldUtil.contentIdField(contentsField.getName()), contentId, Store.YES));
337 
338 			// Store the different properties of the complex contents field that
339 			// were gathered in lists while parsing.
340 			contentsField.addToLuceneDoc(currentLuceneDoc);
341 
342 			// Add all properties to forward index
343 			for (ComplexFieldProperty prop: contentsField.getProperties()) {
344 				if (!prop.hasForwardIndex())
345 					continue;
346 
347 				// Add property (case-sensitive tokens) to forward index and add
348 				// id to Lucene doc
349 				String propName = prop.getName();
350 				String fieldName = ComplexFieldUtil.propertyField(contentsField.getName(), propName);
351 				int fiid = indexer.addToForwardIndex(fieldName, prop);
352 				currentLuceneDoc.add(new IntField(ComplexFieldUtil.forwardIndexIdField(fieldName), fiid, Store.YES));
353 			}
354 
355 			// If there's an external metadata fetcher, call it now so it can
356 			// add the metadata for this document and (optionally) store the
357 			// metadata document in the content store (and the corresponding id in the
358 			// Lucene doc)
359 			MetadataFetcher m = getMetadataFetcher();
360 			if (m != null) {
361 				m.addMetadata();
362 			}
363 
364 			// See what metadatafields are missing or empty and add unknown value
365 			// if desired.
366 			IndexStructure struct = indexer.getSearcher().getIndexStructure();
367 			for (String fieldName: struct.getMetadataFields()) {
368 				MetadataFieldDesc fd = struct.getMetadataFieldDesc(fieldName);
369 				boolean missing = false, empty = false;
370 				String currentValue = currentLuceneDoc.get(fieldName);
371 				if (currentValue == null)
372 					missing = true;
373 				else if (currentValue.length() == 0)
374 					empty = true;
375 				UnknownCondition cond = fd.getUnknownCondition();
376 				boolean useUnknownValue = false;
377 				switch (cond) {
378 				case EMPTY:
379 					useUnknownValue = empty;
380 					break;
381 				case MISSING:
382 					useUnknownValue = missing;
383 					break;
384 				case MISSING_OR_EMPTY:
385 					useUnknownValue = missing | empty;
386 					break;
387 				case NEVER:
388 					useUnknownValue = false;
389 					break;
390 				}
391 				if (useUnknownValue)
392 					addMetadataField(fieldName, fd.getUnknownValue());
393 			}
394 
395 			try {
396 				// Add Lucene doc to indexer
397 				indexer.add(currentLuceneDoc);
398 			} catch (Exception e) {
399 				throw ExUtil.wrapRuntimeException(e);
400 			}
401 
402 			// Report progress
403 			reportCharsProcessed();
404 			reportTokensProcessed(wordsDone);
405 			wordsDone = 0;
406 			indexer.getListener().documentDone(currentDocumentName);
407 
408 			// Reset contents field for next document
409 			contentsField.clear();
410 			currentLuceneDoc = null;
411 
412 			// Stop if required
413 			if (!indexer.continueIndexing())
414 				throw new MaxDocsReachedException();
415 
416 		}
417 
418 		if (nDocumentsSkipped > 0)
419 			System.err.println("Skipped " + nDocumentsSkipped + " large documents");
420 	}
