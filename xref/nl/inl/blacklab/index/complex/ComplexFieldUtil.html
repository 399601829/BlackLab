110 	public enum BookkeepFieldType {
111 		CONTENT_ID,
112 		FORWARD_INDEX_ID,
113 		LENGTH_TOKENS
114 	}
115 
116 	public static boolean isBookkeepingSubfield(String bookkeepName) {
117 		return BOOKKEEPING_SUBFIELDS.contains(bookkeepName);
118 	}
119 
120 	public static BookkeepFieldType whichBookkeepingSubfield(String bookkeepName) {
121 		switch(BOOKKEEPING_SUBFIELDS.indexOf(bookkeepName)) {
122 		case 0:
123 			return BookkeepFieldType.CONTENT_ID;
124 		case 1:
125 			return BookkeepFieldType.FORWARD_INDEX_ID;
126 		case 2:
127 			return BookkeepFieldType.LENGTH_TOKENS;
128 		default:
129 			throw new IllegalArgumentException("Unknown bookkeeping field: " + bookkeepName);
130 		}
131 	}

158 	/** Set what field name separators to use.
159 	 * @param avoidSpecialChars if true, use only standard identifier characters for the separators. If false, use special chars %, @, #.
160 	 */
161 	public static void setFieldNameSeparators(boolean avoidSpecialChars) {
162 		if (avoidSpecialChars) {
163 			// Avoid using special characters in fieldnames, in case
164 			// this clashes with other Lucene-based software (such as e.g. Solr)
165 			PROP_SEP = "_PR_";
166 			ALT_SEP = "_AL_";
167 			BOOKKEEPING_SEP = "_BK_";
168 		} else {
169 			// Lucene doesn't have any restrictions on characters in field names;
170 			// use the short, symbolic ones.
171 			PROP_SEP = "%";
172 			ALT_SEP = "@";
173 			BOOKKEEPING_SEP = "#";
174 		}
175 
176 		ALT_SEP_LEN = ALT_SEP.length();
177 		PROP_SEP_LEN = PROP_SEP.length();
178 		BOOKKEEPING_SEP_LEN = BOOKKEEPING_SEP.length();
179 	}

181 	/**
182 	 * Construct a complex field bookkeeping subfield name.
183 	 *
184 	 * @param fieldName
185 	 *            the base field name
186 	 * @param propName
187 	 *            the property name, or null if this is bookkeeping for the whole complex field
188 	 * @param bookkeepName
189 	 *            name of the bookkeeping value
190 	 * @return the combined complex field name
191 	 */
192 	public static String bookkeepingField(String fieldName, String propName, String bookkeepName) {
193 		String fieldPropName;
194 		boolean propGiven = propName != null && propName.length() > 0;
195 		if (fieldName == null || fieldName.length() == 0) {
196 			if (propGiven) {
197 				fieldPropName = propName;
198 			} else
199 				throw new IllegalArgumentException("Must specify a base name, a property name or both: " + fieldName + ", " + propName + ", " + bookkeepName);
200 		} else {
201 			fieldPropName = fieldName + (propGiven ? PROP_SEP + propName : "");
202 		}
203 
204 		if (bookkeepName == null || bookkeepName.length() == 0)
205 			return fieldPropName;
206 		return fieldPropName + BOOKKEEPING_SEP + bookkeepName;
207 	}

222 	/**
223 	 * Construct a complex field property name.
224 	 *
225 	 * @param fieldName
226 	 *            the base field name
227 	 * @param propName
228 	 *            the property name
229 	 * @param altName
230 	 *            the alternative name
231 	 * @return the combined complex field name
232 	 */
233 	public static String propertyField(String fieldName, String propName, String altName) {
234 		String fieldPropName;
235 		boolean propGiven = propName != null && propName.length() > 0;
236 		if (!propGiven) {
237 			throw new IllegalArgumentException("Must specify a property name");
238 		}
239 		if (fieldName == null || fieldName.length() == 0) {
240 			fieldPropName = propName;
241 		} else {
242 			fieldPropName = fieldName + PROP_SEP + propName;
243 		}
244 
245 		if (altName == null || altName.length() == 0) {
246 			return fieldPropName;
247 		}
248 		return fieldPropName + ALT_SEP + altName;
249 	}

277 	/**
278 	 * Gets the different components of a complex field property (alternative) name.
279 	 *
280 	 * @param luceneFieldName
281 	 *   the Lucene index field name, with possibly a property and/or alternative added
282 	 * @return an array of size 1-4, containing the field name, and optionally the property name,
283 	 *   alternative name and bookkeeping field name.
284 	 *
285 	 *  Property name may be null if this is a main bookkeeping field.
286 	 *  Alternative name may be null if this is a bookkeeping field or if it indicates
287 	 *  the main property (not an alternative).
288 	 */
289 	public static String[] getNameComponents(String luceneFieldName) {
290 
291 		/*
292 		Field names can be one of the following:
293 
294 		(1) Property:              base%prop (e.g. word, lemma, pos)
295 		(2) Main bookkeeping:      base#cid
296 		(3) Property bookkeeping:  base%prop#fiid
297 		(4) Property alternative:  base%prop@alt
298 
299 		(Old style, now unsupported:)
300 
301 		(A) Main property:          base       (implicitly, "word")
302 		(1) Additional property:    base__prop (e.g. lemma, pos)
303 		(B) Main bookkeeping:       base__fiid
304 		(3) Property bookkeeping:   base__prop__fiid
305 		(4) Property alternative:   base__prop_ALT_alt
306 		(C) Main prop alternative:  base_ALT_alt
307 		*/
308 
309 		String baseName, propName, altName, bookkeepingName;
310 
311 		int propSepPos = luceneFieldName.indexOf(PROP_SEP);
312 		int altSepPos = luceneFieldName.indexOf(ALT_SEP);
313 		int bookkeepingSepPos;
314 			bookkeepingSepPos = luceneFieldName.indexOf(BOOKKEEPING_SEP);
315 
316 		// Strip off property and possible alternative
317 		if (propSepPos >= 0) {
318 			// Property given (1/3/4)
319 			baseName = luceneFieldName.substring(0, propSepPos);
320 			int afterPropSepPos = propSepPos + PROP_SEP_LEN;
321 
322 			if (altSepPos >= 0) {
323 				// Property and alternative given (4)
324 				propName = luceneFieldName.substring(afterPropSepPos, altSepPos);
325 				altName = luceneFieldName.substring(altSepPos + ALT_SEP_LEN);
326 				return new String[] {baseName, propName, altName};
327 			}
328 
329 			// Maybe it's a bookkeeping field?
330 			if (bookkeepingSepPos >= 0 && bookkeepingSepPos > propSepPos) {
331 				// Property plus bookkeeping subfield given. (3)
332 				propName = luceneFieldName.substring(afterPropSepPos, bookkeepingSepPos);
333 				bookkeepingName = luceneFieldName.substring(bookkeepingSepPos + BOOKKEEPING_SEP_LEN);
334 				return new String[] {baseName, propName, null, bookkeepingName};
335 			}
336 
337 			// Plain property, no alternative or bookkeeping (1)
338 			propName = luceneFieldName.substring(afterPropSepPos);
339 			return new String[] {baseName, propName};
340 		}
341 
342 		// No property given. Alternative?
343 		if (altSepPos >= 0) {
344 			throw new IllegalArgumentException("Basename+altname is not a valid field name! (" + luceneFieldName + ")");
345 		}
346 
347 		if (bookkeepingSepPos >= 0) {
348 			// Main bookkeeping field (2)
349 			baseName = luceneFieldName.substring(0, bookkeepingSepPos);
350 			altName = luceneFieldName.substring(bookkeepingSepPos + BOOKKEEPING_SEP_LEN);
351 			return new String[] {baseName, null, null, altName};
352 		}
353 
354 		// Just the base name given (A).
355 		// This means it's a metadata field.
356 		return new String[] {luceneFieldName};
357 	}

359 	/**
360 	 * Gets the base complex field name from a Lucene index field name. So "contents" and
361 	 * "contents%pos" would both yield "contents".
362 	 *
363 	 * @param luceneFieldName
364 	 *            the Lucene index field name, with possibly a property added
365 	 * @return the base complex field name
366 	 */
367 	public static String getBaseName(String luceneFieldName) {
368 		// Strip off property and possible alternative
369 		int pos = luceneFieldName.indexOf(PROP_SEP);
370 		if (pos >= 0) {
371 			return luceneFieldName.substring(0, pos);
372 		}
373 		pos = luceneFieldName.indexOf(BOOKKEEPING_SEP);
374 		if (pos >= 0) {
375 			return luceneFieldName.substring(0, pos);
376 		}
377 		pos = luceneFieldName.indexOf(ALT_SEP);
378 		if (pos >= 0) {
379 			throw new IllegalArgumentException("Illegal field name: " + luceneFieldName);
380 		}
381 		return luceneFieldName;
382 	}

384 	/**
385 	 * Checks if the given fieldName actually points to an alternative property (for example, a
386 	 * case-sensitive version of a property).
387 	 *
388 	 * Example: the fieldName "contents%lemma@s" indicates the "s" alternative of the "lemma"
389 	 * property of the "contents" complex field.
390 	 *
391 	 * @param fieldPropAltName
392 	 *            the full fieldname, possibly including a property and/or alternative
393 	 * @param altName
394 	 *            the alternative to test for
395 	 * @return true if the fieldName indicates the specified alternative
396 	 */
397 	public static boolean isAlternative(String fieldPropAltName, String altName) {
398 		if (altName.length() == 0) {
399 			// No alternative, therefore no alternative separator
400 			return fieldPropAltName.indexOf(ALT_SEP) < 0;
401 		}
402 
403 		// We're looking for an alternative
404 		String altSuffix = ALT_SEP + altName;
405 		return fieldPropAltName.endsWith(altSuffix);
406 	}

436 	/**
437 	 * Does this Lucene field name refer to a case-sensitive alternative?
438 	 *
439 	 * Case-sensitive alternatives are "s" (case- and diacritics-sensitive)
440 	 * and "di" (diacritics-insensitive but case-sensitive).
441 	 *
442 	 * @param fieldPropAltName Lucene field name including property and alt name
443 	 * @return true if the field name refers to a case-sensitive alternative
444 	 */
445 	public static boolean isCaseSensitive(String fieldPropAltName) {
446 		// both-sensitive or diacritics-insensitive
447 		return fieldPropAltName.endsWith(ALT_SEP + "s") || fieldPropAltName.endsWith(ALT_SEP + "di");
448 	}
449 
450 	/**
451 	 * Does this Lucene field name refer to a diacritics-sensitive alternative?
452 	 *
453 	 * Diacritics-sensitive alternatives are "s" (case- and diacritics-sensitive)
454 	 * and "ci" (case-insensitive but diacritics-sensitive).
455 	 *
456 	 * @param fieldPropAltName Lucene field name including property and alt name
457 	 * @return true if the field name refers to a diacritics-sensitive alternative
458 	 */
459 	public static boolean isDiacriticsSensitive(String fieldPropAltName) {
460 		// both-sensitive or case-insensitive
461 		return fieldPropAltName.endsWith(ALT_SEP + "s") || fieldPropAltName.endsWith(ALT_SEP + "ci");
462 	}
