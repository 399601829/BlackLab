21  /**
22   * Responsible for loading file conversion and tagging plugins.
23   * <p>
24   * It will attempt to load all conversion and tagging plugins (according to the
25   * {@link ServiceLoader} system) and initialize them with their respective
26   * settings from the main blacklab config.
27   * <p>
28   * A jar that wishes to register a plugin must contain a file named
29   * "nl.inl.blacklab.indexers.preprocess.(Tag|Covert)Plugin" inside
30   * "META-INF/services/", containing the qualified classNames of the
31   * implementations they contain.
32   */
33  public class PluginManager {
34      private static class PluginData<T extends Plugin> {
35          public boolean initialized;
36          public PluginException initializationException;
37  
38          public final T plugin;
39          private final Map<String, String> configMap;
40  
41          public PluginData(T plugin, Map<String, String> configMap) {
42              this.configMap = configMap;
43              this.plugin = plugin;
44          }
45      }
46  
47      private static final Logger logger = LogManager.getLogger(PluginManager.class);
48  
49      /** a pluginId may only contain a-z, A-Z, 0-9, and _ */
50      private static final Pattern PLUGINID_PATTERN = Pattern.compile("[\\w]+");
51  
52      /**
53       * Delay initialization of plugins until they are first used. Useful for
54       * development
55       */
56      private static final String PROP_DELAY_INITIALIZATION = "delayInitialization";
57  
58      /** Is the plugin system itself initialized */
59      private static boolean isInitialized = false;
60  
61      private static Map<String, PluginData<ConvertPlugin>> convertPlugins = new HashMap<>();
62      private static Map<String, PluginData<TagPlugin>> tagPlugins = new HashMap<>();
63  
64      // Nothing to do; initialization happens when the blacklab config is loaded.
65      // The blacklab Config is automatically loaded when the first BlackLabIndex is
66      // opened, or earlier by a user library.
67      // So plugin formats should always be visible by the time they're needed.
68      // (except when trying to query available formats before opening an index or
69      // loading a config...this is an edge case)
70      private PluginManager() {
71      }
72  
73      /**
74       * Attempts to load and initialize all plugin classes on the classpath, passing
75       * the values in the config to the matching plugin.
76       *
77       * @param pluginConfig the plugin configurations collection object. The format
78       *            for this object is
79       *
80       *            <pre>
81       * {
82       *   "pluginId": {
83       *     // arbitrary plugin config here
84       *   },
85       *
86       *   "anotherPluginId": { ... },
87       *   ...
88       * }
89       *            </pre>
90       */
91      public static void initPlugins(BLConfigPlugins pluginConfig) {
92          if (isInitialized)
93              throw new IllegalStateException("PluginManager already initialized");
94          isInitialized = true;
95  
96          logger.debug("Initializing plugin system...");
97          
98          boolean delayInitialization = pluginConfig.isDelayInitialization();
99  
100         // First load all plugins, so we have the full list of plugins available.
101         convertPlugins = loadPlugins(ConvertPlugin.class, pluginConfig);
102         tagPlugins = loadPlugins(TagPlugin.class, pluginConfig);
103 
104         // Some plugins take a LONG time to init, if we block, we block the loading of the config
105         // Which in turn blocks the whole of blacklab(-server), so don't do that
106         if (!delayInitialization) {
107             CompletableFuture.runAsync(() -> {
108                 // only now they're all located, initialize them
109                 logger.trace("Config setting " + PROP_DELAY_INITIALIZATION + " is false, initializing plugins...");
110                 initializePlugins(convertPlugins);
111                 initializePlugins(tagPlugins);
112                 logger.trace("Finished Initializing plugin system");
113             });
114         } else {
115             logger.trace("Config setting " + PROP_DELAY_INITIALIZATION
116                     + " is true, plugins will be initialized on first use.");
117             logger.trace("Finished Initializing plugin system");
118         }
119     }
120 
121     private static <T extends Plugin> Map<String, PluginData<T>> loadPlugins(Class<T> pluginClass,
122             BLConfigPlugins pluginConfig) {
123         Map<String, PluginData<T>> plugins = new HashMap<>();
124 
125         Iterator<T> it = ServiceLoader.load(pluginClass).iterator();
126         Map<String, Map<String, String>> pluginParamConfig = pluginConfig.getPlugins();
127         while (it.hasNext()) {
128             String id = null;
129 
130             try {
131                 T plugin = it.next();
132                 id = plugin.getId();
133                 if (!PLUGINID_PATTERN.matcher(id).matches()) {
134                     logger.warn("Plugin id " + id + " (class " + plugin.getClass().getCanonicalName() +
135                             ") is not a valid id; ignoring plugin.");
136                     continue;
137                 }
138 
139                 // Config available, or plugin needs no config?
140                 if (!plugin.needsConfig() || pluginParamConfig.containsKey(id)) {
141                     // Yes, add the plugin data to our map.
142                     PluginData<T> data = new PluginData<>(plugin, pluginParamConfig.get(id));
143                     plugins.put(id, data);
144                 }
145             } catch (ServiceConfigurationError e) {
146                 logger.error("Plugin failed to load: " + e.getMessage(), e);
147             } catch (Exception e) {
148                 logger.error("Plugin " + (id == null ? "(unknown)" : id) + " failed to load: " + e.getMessage(), e);
149             }
150         }
151 
152         return plugins;
153     }

155     public static Optional<ConvertPlugin> getConverter(String convertPluginId) throws PluginException {
156         if (!isInitialized)
157             throw new UnsupportedOperationException(
158                     "Plugin system is not initialized, place a top-level key \"plugins\" with " +
159                             "per-plugin configuration in your blacklab config to use plugins.");
160 
161         Optional<PluginData<ConvertPlugin>> data = Optional.ofNullable(convertPlugins.get(convertPluginId));
162         if (data.isPresent())
163             initializePlugin(data.get());
164 
165         return data.map(d -> d.plugin);
166     }
167 
168     public static Optional<TagPlugin> getTagger(String tagPluginId) throws PluginException {
169         if (!isInitialized)
170             throw new UnsupportedOperationException(
171                     "Plugin system is not initialized, place a top-level key \"plugins\" with " +
172                             "per-plugin configuration in your blacklab config to use plugins.");
173 
174         Optional<PluginData<TagPlugin>> data = Optional.ofNullable(tagPlugins.get(tagPluginId));
175         if (data.isPresent())
176             initializePlugin(data.get());
177 
178         return data.map(d -> d.plugin);
179     }

181     /**
182      * Initialize the plugin, if it exists and is currently uninitialized.
183      * Previously encountered errors are rethrown. If am error is encountered, it is
184      * stored in the plugin data and rethrown.
185      *
186      * @param data plugin data
187      * @throws PluginException when the plugin fails to initialize, care should be
188      *             taken by the caller to remove it from the list of plugins when
189      *             this occurs.
190      */
191     private static void initializePlugin(PluginData<?> data) throws PluginException {
192         synchronized (data.plugin) {
193             if (data.initializationException != null)
194                 throw data.initializationException;
195             if (data.initialized)
196                 return;
197 
198             try {
199                 logger.debug("Initializing plugin " + data.plugin.getDisplayName());
200                 data.plugin.init(data.configMap);
201                 logger.debug("Initialized plugin " + data.plugin.getDisplayName());
202             } catch (PluginException e) {
203                 data.initializationException = e;
204                 throw e;
205             } catch (Exception e) {
206                 data.initializationException = new PluginException("Error during initialization.", e);
207                 throw data.initializationException;
208             } finally {
209                 data.initialized = true;
210             }
211         }
212     }
213 
214     /**
215      * Used to initialize all plugins in one go when
216      * {@link #PROP_DELAY_INITIALIZATION} is false.
217      *
218      * @param plugins
219      */
220     private static <T extends Plugin> void initializePlugins(Map<String, PluginData<T>> plugins) {
221         plugins.values().forEach(pd -> {
222             try {
223                 initializePlugin(pd);
224             } catch (PluginException e) {
225                 // exception already cached in plugindata, no need to throw.
226                 logger.error("Plugin " + pd.plugin.getId() + " failed to initialize: " + e.getMessage(), e);
227             }
228         });
229     }
