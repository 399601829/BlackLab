76          @Override
77          public void file(String path, InputStream is, File file) throws IOException, MalformedInputFile, PluginException {
78              // Attempt to detect the encoding of our inputStream, falling back to DEFAULT_INPUT_ENCODING if the stream
79              // doesn't contain a a BOM This doesn't do any character parsing/decoding itself, it just detects and skips
80              // the BOM (if present) and exposes the correct character set for this stream (if present)
81              // This way we can later use the charset to decode the input
82              // There is one gotcha however, and that is that if the inputstream contains non-textual data, we pass the
83              // default encoding to our DocIndexer
84              // This usually isn't an issue, since docIndexers work exclusively with either binary data or text.
85              // In the case of binary data docIndexers, they should always ignore the encoding anyway
86              // and for text docIndexers, passing a binary file is an error in itself already.
87              try (
88                      UnicodeStream inputStream = new UnicodeStream(is, DEFAULT_INPUT_ENCODING);
89                      DocIndexer docIndexer = DocumentFormats.get(IndexerImpl.this.formatIdentifier, IndexerImpl.this, path,
90                              inputStream, inputStream.getEncoding());) {
91                  impl(docIndexer, path);
92              }
93          }
94  
95          public void file(String path, Reader reader) throws MalformedInputFile, PluginException, IOException {
96              try (DocIndexer docIndexer = DocumentFormats.get(IndexerImpl.this.formatIdentifier, IndexerImpl.this, path,
97                      reader)) {
98                  impl(docIndexer, path);
99              }
100         }
101 
102         private void impl(DocIndexer indexer, String documentName) throws MalformedInputFile, PluginException, IOException {
103 
104             if (!indexer.continueIndexing())
105                 return;
106 
107             // FIXME progress reporting is broken in multithreaded indexing, as the listener is shared between threads
108             // So a docIndexer that didn't index anything can slip through if another thread did index some data in the
109             // meantime
110             listener().fileStarted(documentName);
111             int docsDoneBefore = indexWriter.writer().numDocs();
112             long tokensDoneBefore = listener().getTokensProcessed();
113 
114             indexer.index();
115             listener().fileDone(documentName);
116             int docsDoneAfter = indexWriter.writer().numDocs();
117             if (docsDoneAfter == docsDoneBefore) {
118                 logger.warn("No docs found in " + documentName + "; wrong format?");
119             }
120             long tokensDoneAfter = listener().getTokensProcessed();
121             if (tokensDoneAfter == tokensDoneBefore) {
122                 logger.warn("No words indexed in " + documentName + "; wrong format?");
123             }
124         }

239      * @throws IOException

247     /**
248      * Open an indexer for the provided writer.
249      *
250      * @param writer the writer
251      * @param formatIdentifier (optional) - the formatIdentifier to use when indexing data through this indexer.
252      *      If omitted, uses the default formatIdentifier stored in the indexMetadata. If that is missing too, throws DocumentFormatNotFound.
253      */
254     IndexerImpl(BlackLabIndexWriter writer, String formatIdentifier) throws DocumentFormatNotFound {
255         if (writer == null) {
256             throw new BlackLabRuntimeException("writer == null");
257         }
258 
259         this.indexWriter = writer;
260 
261         if (!DocumentFormats.isSupported(formatIdentifier)) {
262             formatIdentifier = writer.metadataWriter().documentFormat();
263             if (!DocumentFormats.isSupported(formatIdentifier)) {
264                 String message = formatIdentifier == null ? "No formatIdentifier"
265                         : "Unknown formatIdentifier '" + formatIdentifier + "'";
266                 throw new DocumentFormatNotFound(
267                         message + ", and could not determine the default documentFormat for index " + writer.name());
268             }
269         }
270         this.formatIdentifier = formatIdentifier;
271 
272         // TODO
273         metadataFieldTypeTokenized = new FieldType();
274         metadataFieldTypeTokenized.setStored(true);
275         //metadataFieldTypeTokenized.setIndexed(true);
276         metadataFieldTypeTokenized.setIndexOptions(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS);
277         metadataFieldTypeTokenized.setTokenized(true);
278         metadataFieldTypeTokenized.setOmitNorms(true); // <-- depending on setting?
279         metadataFieldTypeTokenized.setStoreTermVectors(true);
280         metadataFieldTypeTokenized.setStoreTermVectorPositions(true);
281         metadataFieldTypeTokenized.setStoreTermVectorOffsets(true);
282         metadataFieldTypeTokenized.freeze();
283 
284         metadataFieldTypeUntokenized = new FieldType(metadataFieldTypeTokenized);
285         metadataFieldTypeUntokenized.setIndexOptions(IndexOptions.DOCS_AND_FREQS);
286         //metadataFieldTypeUntokenized.setTokenized(false);  // <-- this should be done with KeywordAnalyzer, otherwise untokenized fields aren't lowercased
287         metadataFieldTypeUntokenized.setStoreTermVectors(false);
288         metadataFieldTypeUntokenized.setStoreTermVectorPositions(false);
289         metadataFieldTypeUntokenized.setStoreTermVectorOffsets(false);
290         metadataFieldTypeUntokenized.freeze();
291     }

293     protected void init(File directory, boolean create, String formatIdentifier, File indexTemplateFile)
294             throws DocumentFormatNotFound, ErrorOpeningIndex {
295 
296         if (create) {
297             if (indexTemplateFile != null) {
298                 indexWriter = BlackLab.openForWriting(directory, true, indexTemplateFile);
299 
300                 // Read back the formatIdentifier that was provided through the indexTemplateFile now that the index
301                 // has written it (might be null)
302                 final String defaultFormatIdentifier = indexWriter.metadataWriter().documentFormat();
303 
304                 if (DocumentFormats.isSupported(formatIdentifier)) {
305                     this.formatIdentifier = formatIdentifier;
306                     if (defaultFormatIdentifier == null || defaultFormatIdentifier.isEmpty()) {
307                         // indexTemplateFile didn't provide a default formatIdentifier,
308                         // overwrite it with our provided formatIdentifier
309                         indexWriter.metadataWriter().setDocumentFormat(formatIdentifier);
310                         indexWriter.metadataWriter().save();
311                     }
312                 } else if (DocumentFormats.isSupported(defaultFormatIdentifier)) {
313                     this.formatIdentifier = defaultFormatIdentifier;
314                 } else {
315                     // TODO we should delete the newly created index here as it failed, how do we clean up files properly?
316                     indexWriter.close();
317                     String formatError = DocumentFormats.formatError(formatIdentifier);
318                     if (formatError == null)
319                         formatError = "format not found";
320                     throw new DocumentFormatNotFound("Cannot create new index in " + directory + " with format " + formatIdentifier + ": " +
321                             formatError);
322 //                    throw new DocumentFormatNotFound("Input format config '" + formatIdentifier
323 //                            + "' not found (or format config contains an error) when creating new index in "
324 //                            + directory);
325                 }
326             } else if (DocumentFormats.isSupported(formatIdentifier)) {
327                 this.formatIdentifier = formatIdentifier;
328 
329                 // No indexTemplateFile, but maybe the formatIdentifier is backed by a ConfigInputFormat (instead of some other DocIndexer implementation)
330                 // this ConfigInputFormat could then still be used as a minimal template to setup the index
331                 // (if there's no ConfigInputFormat, that's okay too, a default index template will be used instead)
332                 ConfigInputFormat format = null;
333                 for (Format desc : DocumentFormats.getFormats()) {
334                     if (desc.getId().equals(formatIdentifier) && desc.getConfig() != null) {
335                         format = desc.getConfig();
336                         break;
337                     }
338                 }
339 
340                 // template might still be null, in that case a default will be created
341                 indexWriter = BlackLab.create(directory, format);
342 
343                 String defaultFormatIdentifier = indexWriter.metadata().documentFormat();
344                 if (defaultFormatIdentifier == null || defaultFormatIdentifier.isEmpty()) {
345                     // ConfigInputFormat didn't provide a default formatIdentifier,
346                     // overwrite it with our provided formatIdentifier
347                     indexWriter.metadataWriter().setDocumentFormat(formatIdentifier);
348                     indexWriter.metadataWriter().save();
349                 }
350             } else {
351                 String formatError = DocumentFormats.formatError(formatIdentifier);
352                 if (formatError == null)
353                     formatError = "format not found";
354                 throw new DocumentFormatNotFound("Cannot create new index in " + directory + " with format " + formatIdentifier + ": " +
355                         formatError);
356 //                throw new DocumentFormatNotFound("Input format config '" + formatIdentifier
357 //                        + "' not found (or format config contains an error) when creating new index in " + directory);
358             }
359         } else { // opening an existing index
360 
361             this.indexWriter = BlackLab.openForWriting(directory, false);
362             String defaultFormatIdentifier = this.indexWriter.metadata().documentFormat();
363 
364             if (DocumentFormats.isSupported(formatIdentifier))
365                 this.formatIdentifier = formatIdentifier;
366             else if (DocumentFormats.isSupported(defaultFormatIdentifier))
367                 this.formatIdentifier = defaultFormatIdentifier;
368             else {
369                 indexWriter.close();
370                 String message = formatIdentifier == null ? "No formatIdentifier"
371                         : "Unknown formatIdentifier '" + formatIdentifier + "'";
372                 throw new DocumentFormatNotFound(
373                         message + ", and could not determine the default documentFormat for index " + directory);
374             }
375         }
376 
377         metadataFieldTypeTokenized = new FieldType();
378         metadataFieldTypeTokenized.setStored(true);
379         //metadataFieldTypeTokenized.setIndexed(true);
380         metadataFieldTypeTokenized.setIndexOptions(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS);
381         metadataFieldTypeTokenized.setTokenized(true);
382         metadataFieldTypeTokenized.setOmitNorms(true); // <-- depending on setting?
383         metadataFieldTypeTokenized.setStoreTermVectors(true);
384         metadataFieldTypeTokenized.setStoreTermVectorPositions(true);
385         metadataFieldTypeTokenized.setStoreTermVectorOffsets(true);
386         metadataFieldTypeTokenized.freeze();
387 
388         metadataFieldTypeUntokenized = new FieldType(metadataFieldTypeTokenized);
389         metadataFieldTypeUntokenized.setIndexOptions(IndexOptions.DOCS_AND_FREQS);
390         //metadataFieldTypeUntokenized.setTokenized(false);  // <-- this should be done with KeywordAnalyzer, otherwise untokenized fields aren't lowercased
391         metadataFieldTypeUntokenized.setStoreTermVectors(false);
392         metadataFieldTypeUntokenized.setStoreTermVectorPositions(false);
393         metadataFieldTypeUntokenized.setStoreTermVectorOffsets(false);
394         metadataFieldTypeUntokenized.freeze();
395     }

463     // TODO this should call close() on running FileProcessors
464     @Override
465     public synchronized void close() {
466 
467         // Signal to the listener that we're done indexing and closing the index (which might take a
468         // while)
469         listener().indexEnd();
470         listener().closeStart();
471 
472         if (!hasRollback) {
473             indexWriter.metadataWriter().addToTokenCount(listener().getTokensProcessed());
474             indexWriter.metadataWriter().save();
475         }
476         indexWriter.close();
477 
478         // Signal that we're completely done now
479         listener().closeEnd();
480         listener().indexerClosed();
481 
482         closed = true;
483     }

531     @Override
532     public void addToForwardIndex(AnnotatedFieldWriter fieldWriter, Document currentLuceneDoc) {
533         Map<Annotation, List<String>> annotations = new HashMap<>();
534         Map<Annotation, List<Integer>> posIncr = new HashMap<>();
535         for (AnnotationWriter annotationWriter: fieldWriter.annotationWriters()) {
536             if (annotationWriter.hasForwardIndex()) {
537                 Annotation annotation = annotationWriter.annotation();
538                 annotations.put(annotation, annotationWriter.values());
539                 posIncr.put(annotation, annotationWriter.positionIncrements());
540             }
541         }
542         indexWriter().forwardIndex(fieldWriter.field()).addDocument(annotations, posIncr, currentLuceneDoc);
543     }

559         } catch (Exception e) {
560             listener().errorOccurred(e, documentName, null);

562             e.printStackTrace();
563             logger.error("(continuing indexing)");
564         }

597     /**
598      * Should we continue indexing or stop?
599      *
600      * We stop if we've reached the maximum that was set (if any), or if a fatal
601      * error has occurred (indicated by terminateIndexing).
602      *
603      * @return true if we should continue, false if not
604      */
605     @Override
606     public synchronized boolean continueIndexing() {
607         if (!indexWriter.isOpen())
608             return false;
609         if (terminateIndexing)
610             return false;
611         if (maxNumberOfDocsToIndex >= 0) {
612             return docsToDoLeft() > 0;
613         }
614         return true;
615     }
616 
617     /**
618      * How many more documents should we process?
619      *
620      * @return the number of documents
621      */
622     @Override
623     public synchronized int docsToDoLeft() {
624         if (maxNumberOfDocsToIndex < 0)
625             return maxNumberOfDocsToIndex;
626         int docsDone = indexWriter.writer().numDocs();
627         return Math.max(0, maxNumberOfDocsToIndex - docsDone);
628     }

709     @Override
710     public File linkedFile(String inputFile) {
711         File f = new File(inputFile);
712         if (f.exists())
713             return f; // either absolute or relative to current dir
714         if (f.isAbsolute())
715             return null; // we tried absolute, but didn't find it
716 
717         // Look in the configured directories for the relative path
718         f = FileUtil.findFile(linkedFileDirs, inputFile, null);
719         if (f == null && this.linkedFileResolver != null)
720             f = this.linkedFileResolver.apply(inputFile);
721 
722         return f;
723     }
