191     /**
192      * Index a linked document.
193      *
194      * @param inputFile where the linked document can be found (file or http(s) reference)
195      * @param pathInsideArchive if input file is an archive: the path to the file we need inside the archive
196      * @param documentPath XPath to the specific linked document we need
197      * @param inputFormatIdentifier input format of the linked document
198      * @param storeWithName if set, store the linked document and store the id to it in a field with this name with "Cid" (content id) appended to it
199      * @throws IOException on error
200      */
201     protected void indexLinkedDocument(String inputFile, String pathInsideArchive, String documentPath, String inputFormatIdentifier, String storeWithName) throws IOException {
202         // Fetch the input file (either by downloading it to a temporary location, or opening it from disk)
203         File f = resolveFileReference(inputFile);
204 
205         // Get the data
206         byte [] data;
207         String completePath = inputFile;
208         if (inputFile.endsWith(".zip") || inputFile.endsWith(".tar") || inputFile.endsWith(".tar.gz") || inputFile.endsWith(".tgz")) {
209             // It's an archive. Unpack the right file from it.
210             completePath += "/" + pathInsideArchive;
211             data = Indexer.fetchFileFromArchive(f, pathInsideArchive);
212         } else {
213             // Regular file.
214             try(InputStream is = new FileInputStream(f)) {
215                 data = IOUtils.toByteArray(is);
216             } catch (IOException e) {
217                 throw new RuntimeException(e);
218             }
219         }
220         if (data == null) {
221             throw new RuntimeException("Error reading linked document");
222         }
223 
224         // Index the data
225         try (DocIndexer docIndexer = DocumentFormats.get(inputFormatIdentifier, indexer, completePath, data, Indexer.DEFAULT_INPUT_ENCODING)) {
226             if (docIndexer instanceof DocIndexerBase) {
227                 @SuppressWarnings("resource")
228                 DocIndexerBase ldi = (DocIndexerBase)docIndexer;
229                 ldi.indexingIntoExistingLuceneDoc = true;
230                 ldi.currentLuceneDoc = currentLuceneDoc;
231                 if (storeWithName != null) {
232                     // If specified, store in this content store and under this name instead of the default
233                     ldi.contentStoreName = storeWithName;
234                 }
235                 ldi.indexSpecificDocument(documentPath);
236             } else {
237                 throw new RuntimeException("Linked document indexer must be subclass of DocIndexerBase, but is " + docIndexer.getClass().getName());
238             }
239         } catch (Exception e1) {
240             throw new RuntimeException(e1);
241         }
242 
243     }

254     /**
255      * Given a URL or file reference, either download to a temp file or find file and return it.
256      *
257      * @param inputFile URL or (relative) file reference
258      * @return the file
259      * @throws IOException
260      */
261     protected File resolveFileReference(String inputFile) throws IOException {
262         if (inputFile.startsWith("http://") || inputFile.startsWith("https://")) {
263             return DownloadCache.downloadFile(inputFile);
264         }
265         if (inputFile.startsWith("file://"))
266             inputFile = inputFile.substring(7);
267         if (indexer == null) // TEST
268             return new File(inputFile);
269         File f = indexer.getLinkedFile(inputFile);
270         if (f == null)
271             throw new FileNotFoundException("References file not found: " + f);
272         if (!f.canRead())
273             throw new IOException("Cannot read referenced file " + f);
274         return f;
275     }

309     protected void startDocument() {
310 
311         traceln("START DOCUMENT");
312         if (!indexingIntoExistingLuceneDoc) {
313             currentLuceneDoc = new Document();
314             if (indexer != null)
315                 currentLuceneDoc.add(new Field("fromInputFile", documentName, indexer.getMetadataFieldType(false)));
316 
317             // DEPRECATED for these types of indexer, but still supported for now
318             addMetadataFieldsFromParameters();
319         }
320         if (indexer != null)
321             indexer.getListener().documentStarted(documentName);
322     }
323 
324     protected void endDocument() {
325         traceln("END DOCUMENT");
326 
327         for (ComplexField complexField: getComplexFields().values()) {
328             ComplexFieldProperty propMain = complexField.getMainProperty();
329 
330             // Make sure all the properties have an equal number of values.
331             // See what property has the highest position
332             // (in practice, only starttags and endtags should be able to have
333             // a position one higher than the rest)
334             int lastValuePos = 0;
335             for (ComplexFieldProperty prop: complexField.getProperties()) {
336                 if (prop.lastValuePosition() > lastValuePos)
337                     lastValuePos = prop.lastValuePosition();
338             }
339 
340             // Make sure we always have one more token than the number of
341             // words, so there's room for any tags after the last word, and we
342             // know we should always skip the last token when matching.
343             if (propMain.lastValuePosition() == lastValuePos)
344                 lastValuePos++;
345 
346             // Add empty values to all lagging properties
347             for (ComplexFieldProperty prop: complexField.getProperties()) {
348                 while (prop.lastValuePosition() < lastValuePos) {
349                     prop.addValue("");
350                     if (prop.hasPayload())
351                         prop.addPayload(null);
352                     if (prop == propMain) {
353                         complexField.addStartChar(getCharacterPosition());
354                         complexField.addEndChar(getCharacterPosition());
355                     }
356                 }
357             }
358             // Store the different properties of the complex field that
359             // were gathered in lists while parsing.
360 //            System.out.println("Adding to lucene doc: " + complexField.getName());
361 //            System.out.println("Values: " + complexField.getMainProperty().getValues());
362             complexField.addToLuceneDoc(currentLuceneDoc);
363 
364             // Add all properties to forward index
365             for (ComplexFieldProperty prop: complexField.getProperties()) {
366                 if (!prop.hasForwardIndex())
367                     continue;
368 
369                 // Add property (case-sensitive tokens) to forward index and add
370                 // id to Lucene doc
371                 String propName = prop.getName();
372                 String fieldName = ComplexFieldUtil.propertyField(
373                         complexField.getName(), propName);
374                 if (indexer != null) {
375                     int fiid = indexer.addToForwardIndex(fieldName, prop);
376                     currentLuceneDoc.add(new IntField(ComplexFieldUtil
377                             .forwardIndexIdField(fieldName), fiid, Store.YES));
378                 }
379             }
380 
381         }
382 
383         if (getStoreDocuments()) {
384             storeDocument();
385         }
386 
387         if (indexer != null) {
388             // If there's an external metadata fetcher, call it now so it can
389             // add the metadata for this document and (optionally) store the
390             // metadata
391             // document in the content store (and the corresponding id in the
392             // Lucene doc)
393             MetadataFetcher m = getMetadataFetcher();
394             if (m != null) {
395                 m.addMetadata();
396             }
397 
398             // See what metadatafields are missing or empty and add unknown value
399             // if desired.
400             IndexStructure struct = indexer.getSearcher().getIndexStructure();
401             for (String fieldName: struct.getMetadataFields()) {
402                 MetadataFieldDesc fd = struct.getMetadataFieldDesc(fieldName);
403                 boolean missing = false, empty = false;
404                 String currentValue = currentLuceneDoc.get(fieldName);
405                 if (currentValue == null)
406                     missing = true;
407                 else if (currentValue.length() == 0)
408                     empty = true;
409                 UnknownCondition cond = fd.getUnknownCondition();
410                 boolean useUnknownValue = false;
411                 switch (cond) {
412                 case EMPTY:
413                     useUnknownValue = empty;
414                     break;
415                 case MISSING:
416                     useUnknownValue = missing;
417                     break;
418                 case MISSING_OR_EMPTY:
419                     useUnknownValue = missing || empty;
420                     break;
421                 case NEVER:
422                     useUnknownValue = false;
423                     break;
424                 }
425                 if (useUnknownValue)
426                     addMetadataField(optTranslateFieldName(fieldName), fd.getUnknownValue());
427             }
428         }
429 
430         try {
431             // Add Lucene doc to indexer, if not existing already
432             if (indexer != null && !indexingIntoExistingLuceneDoc)
433                 indexer.add(currentLuceneDoc);
434         } catch (Exception e) {
435             throw ExUtil.wrapRuntimeException(e);
436         }
437 
438         for (ComplexField complexField: getComplexFields().values()) {
439             // Reset complex field for next document
440             // don't reuse buffers, they're still referenced by the lucene doc.
441             complexField.clear(!indexingIntoExistingLuceneDoc);
442         }
443 
444         // Report progress
445         if (indexer != null) {
446         	reportCharsProcessed();
447             reportTokensProcessed();
448         }
449         if (indexer != null)
450             indexer.getListener().documentDone(documentName);
451 
452         currentLuceneDoc = null;
453 
454         // Stop if required
455         if (indexer != null) {
456             if (!indexer.continueIndexing())
457                 throw new MaxDocsReachedException();
458         }
459 
460         uniqueStrings.clear();
461     }

463     /**
464      * Store the entire document at once.
465      *
466      * Subclasses that simply capture the entire document can use this in their storeDocument implementation.
467      * @param document document to store
468      */
469     protected void storeWholeDocument(String document) {
470         // Finish storing the document in the document store,
471         // retrieve the content id, and store that in Lucene.
472         // (Note that we do this after adding the dummy token, so the character
473         // positions for the dummy token still make (some) sense)
474         String contentIdFieldName;
475         String contentStoreName = getContentStoreName();
476         if (contentStoreName == null) {
477             ComplexField main = getMainComplexField();
478             if (main == null) {
479                 contentStoreName = "metadata";
480                 contentIdFieldName = "metadataCid";
481             } else {
482                 contentStoreName = main.getName();
483                 contentIdFieldName = ComplexFieldUtil.contentIdField(main.getName());
484             }
485         } else {
486             contentIdFieldName = contentStoreName + "Cid";
487         }
488         int contentId = -1;
489         if (indexer != null) {
490             ContentStore contentStore = indexer.getContentStore(contentStoreName);
491             contentId = contentStore.store(document);
492         }
493         currentLuceneDoc.add(new IntField(contentIdFieldName, contentId, Store.YES));
494     }

516     protected void inlineTag(String tagName, boolean isOpenTag, Map<String, String> attributes) {
517         if (isOpenTag) {
518             trace("<" + tagName + ">");
519 
520             int lastStartTagPos = propTags().lastValuePosition();
521             int currentPos = getCurrentTokenPosition();
522             int posIncrement = currentPos - lastStartTagPos;
523             propTags().addValue(tagName, posIncrement);
524             propTags().addPayload(null);
525             int startTagIndex = propTags().getLastValueIndex();
526             openInlineTags.add(new OpenTagInfo(tagName, currentPos, startTagIndex));
527 
528             for (Entry<String, String> e: attributes.entrySet()) {
529                 // Index element attribute values
530                 String name = e.getKey();
531                 String value = e.getValue();
532                 propTags().addValue("@" + name.toLowerCase() + "__" + value.toLowerCase(), 0);
533                 propTags().addPayload(null);
534             }
535 
536         } else {
537             traceln("</" + tagName + ">");
538 
539             int currentPos = getCurrentTokenPosition();
540 
541             // Add payload to start tag property indicating end position
542             if (openInlineTags.size() == 0)
543                 throw new MalformedInputFileException("Close tag " + tagName + " found, but that tag is not open");
544             OpenTagInfo openTag = openInlineTags.remove(openInlineTags.size() - 1);
545             if (!openTag.name.equals(tagName))
546                 throw new MalformedInputFileException("Close tag " + tagName + " found, but " + openTag.name + " expected");
547             byte[] payload = ByteBuffer.allocate(4).putInt(currentPos).array();
548             propTags().setPayloadAtIndex(openTag.index, new BytesRef(payload));
549         }
550     }

571     protected void endWord() {
572         String punct;
573         if (punctuation.length() == 0)
574             punct = addDefaultPunctuation && !preventNextDefaultPunctuation ? " " : "";
575         else
576             punct = punctuation.toString();
577 
578         preventNextDefaultPunctuation = false;
579         // Normalize once more in case we hit more than one adjacent punctuation
580         propPunct().addValue(StringUtil.normalizeWhitespace(punct));
581         addEndChar(getCharacterPosition());
582         wordsDone++;
583         if (wordsDone > 0 && wordsDone % 5000 == 0) {
584         	reportCharsProcessed();
585             reportTokensProcessed();
586         }
587         if (punctuation.length() > 10000)
588             punctuation = new StringBuilder(); // let's not hold on to this much memory
589         else
590             punctuation.setLength(0);
591     }

594      * Index an annotation.
595      *
596      * Can be used to add annotation(s) at the current position (indexAtPositions == null),
597      * or to add annotations at specific positions (indexAtPositions contains positions). The latter
598      * is used for standoff annotations.
599      *
600      * Also called for subannotations (with the value already prepared)
601      *
602      * @param name annotation name
603      * @param value annotation value
604      * @param increment if indexAtPosition == null: the token increment to use
605      * @param indexAtPositions if null: index at the current position; otherwise: index at these positions
606      */
607     protected void annotation(String name, String value, int increment, List<Integer> indexAtPositions) {
608         ComplexFieldProperty property = getProperty(name);
609         if (indexAtPositions == null) {
610             if (name.equals("word"))
611                 trace(value + " ");
612 //          if (name.equals("xmlid"))
613 //              trace(value + " ");
614             property.addValue(value, increment);
615             //System.out.println("Field " + currentComplexField.getName() + ", property " + property.getName() + ": added " + value + ", lastValuePos = " + property.lastValuePosition());
616         } else {
617 //            if (name.equals("rating"))
618 //                traceln("{" + value + "}: " + indexAtPositions);
619             //System.out.println("Field " + currentComplexField.getName() + ", property " + property.getName() + ", value " + value + " (STANDOFF)");
620             for (Integer position: indexAtPositions) {
621                 property.addValueAtPosition(value, position);
622                 //System.out.println("  added at position " + position);
623             }
624         }

652     @Override
653     public final void reportCharsProcessed() {
654     	final int charsDone = getCharacterPosition();
655     	final int charsDoneSinceLastReport;
656     	if (charsDoneAtLastReport > charsDone)
657     		charsDoneSinceLastReport = charsDone;
658     	else
659     		charsDoneSinceLastReport = charsDone - charsDoneAtLastReport;
660 
661     	indexer.getListener().charsDone(charsDoneSinceLastReport);
662     	charsDoneAtLastReport = charsDone;
663     }

665     /**
666      * Report the change in wordsDone since the last report
667      */
668     @Override
669     public final void reportTokensProcessed() {
670     	final int wordsDoneSinceLastReport;
671     	if (wordsDoneAtLastReport > wordsDone) // wordsDone reset by child class? report everything then
672     		wordsDoneSinceLastReport = wordsDone;
673     	else
674     		wordsDoneSinceLastReport = wordsDone - wordsDoneAtLastReport;
675 
676     	indexer.getListener().tokensDone(wordsDoneSinceLastReport);

680     /**
681      * Get the external metadata fetcher for this indexer, if any.
682      *
683      * The metadata fetcher can be configured through the "metadataFetcherClass"
684      * parameter.
685      *
686      * @return the metadata fetcher if any, or null if there is none.
687      */
688     protected MetadataFetcher getMetadataFetcher() {
689         if (metadataFetcher == null) {
690             @SuppressWarnings("deprecation")
691             String metadataFetcherClassName = getParameter("metadataFetcherClass");
692             if (metadataFetcherClassName != null) {
693                 try {
694                     Class<? extends MetadataFetcher> metadataFetcherClass = Class.forName(metadataFetcherClassName)
695                             .asSubclass(MetadataFetcher.class);
696                     Constructor<? extends MetadataFetcher> ctor = metadataFetcherClass.getConstructor(DocIndexer.class);
697                     metadataFetcher = ctor.newInstance(this);
698                 } catch (Exception e) {
699                     throw new RuntimeException(e);
700                 }
701             }
702         }
703         return metadataFetcher;
704     }
