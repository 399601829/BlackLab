39      /**
40       *
41       * @param r
42       * @param isJson
43       * @param cfg
44       * @param finder responsible for getting (optionally locating/loading) other configs that this config depends on. (for config keys "baseFormat" and "inputFormat")
45       * @throws IOException
46       * @throws InputFormatConfigException if the file is not a valid config
47       */
48      public static void read(Reader r, boolean isJson, ConfigInputFormat cfg, Function<String, Optional<ConfigInputFormat>> finder) throws IOException {
49          ObjectMapper mapper = isJson ? Json.getJsonObjectMapper() : Json.getYamlObjectMapper();
50  
51          JsonNode root;
52          try {
53              root = mapper.readTree(r);
54          } catch (JsonParseException e) {
55              throw new InputFormatConfigException("Could not parse config file: " + e.getMessage());
56          }
57          read(root, cfg, finder);
58      }

60      /**
61       *
62       * @param file
63       * @param cfg
64       * @param finder responsible for getting (optionally locating/loading) other configs that this config depends on. ("baseFormat" and "inputFormat")
65       * @throws IOException
66       * @throws InputFormatConfigException if the file is not a valid config
67       */
68      public static void read(File file, ConfigInputFormat cfg, Function<String, Optional<ConfigInputFormat>> finder) throws IOException {
69          read(FileUtil.openForReading(file), file.getName().endsWith(".json"), cfg, finder);
70          cfg.setReadFromFile(file);
71      }

73      protected static void read(JsonNode root, ConfigInputFormat cfg, Function<String, Optional<ConfigInputFormat>> finder) {
74          obj(root, "root node");
75          Iterator<Entry<String, JsonNode>> it = root.fields();
76          while (it.hasNext()) {
77              Entry<String, JsonNode> e = it.next();
78              switch (e.getKey()) {
79              case "displayName": cfg.setDisplayName(str(e)); break;
80              case "description": cfg.setDescription(str(e)); break;
81              case "helpUrl": cfg.setHelpUrl(str(e)); break;
82              case "baseFormat": {
83                  String formatIdentifier = str(e);
84                  if (finder == null)
85                      throw new InputFormatConfigException("Format depends on base format " + formatIdentifier + " but no BaseFormatFinder provided.");
86  
87                  ConfigInputFormat baseFormat = finder
88                          .apply(formatIdentifier)
89                          .orElseThrow(() ->
90                              new InputFormatConfigException("Base format " + formatIdentifier + " not found for format " + cfg.getName()));
91  
92                  cfg.setBaseFormat(baseFormat);
93                  break;
94              }
95              case "type": cfg.setType(str(e)); break;
96              case "fileType": cfg.setFileType(FileType.fromStringValue(str(e))); break;
97              case "fileTypeOptions": readFileTypeOptions(e, cfg); break;
98  //            case "tabularOptions": cfg.setTabularOptions(readTabularOptions(e)); break;
99              case "corpusConfig": readCorpusConfig(e, cfg.getCorpusConfig()); break;
100             case "namespaces": readStringMap(e, cfg.namespaces); break;
101             case "documentPath": cfg.setDocumentPath(str(e)); break;
102             case "store": cfg.setStore(bool(e)); break;
103             case "indexFieldAs": readStringMap(e, cfg.indexFieldAs); break;
104             case "annotatedFields": readAnnotatedFields(e, cfg); break;
105             case "metadataDefaultAnalyzer": cfg.setMetadataDefaultAnalyzer(str(e)); break;
106             case "metadata": readMetadata(e, cfg); break;
107             case "linkedDocuments": readLinkedDocuments(e, cfg); break;
108             case "convertPlugin": cfg.setConvertPluginId(str(e)); break;
109             case "tagPlugin": cfg.setTagPluginId(str(e)); break;
110             case "isVisible": cfg.setVisible(bool(e)); break;
111             default:
112                 throw new InputFormatConfigException("Unknown top-level key " + e.getKey());
113             }
114         }
115     }

117     private static void readCorpusConfig(Entry<String, JsonNode> ccEntry, ConfigCorpus corpusConfig) {
118         ObjectNode node = obj(ccEntry.getValue(), "");
119         Iterator<Entry<String, JsonNode>> it = node.fields();
120         while (it.hasNext()) {
121             Entry<String, JsonNode> e = it.next();
122             switch(e.getKey()) {
123             case "displayName": corpusConfig.setDisplayName(str(e)); break;
124             case "description": corpusConfig.setDescription(str(e)); break;
125             case "contentViewable": corpusConfig.setContentViewable(bool(e)); break;
126             case "textDirection": corpusConfig.setTextDirection(TextDirection.fromCode(str(e))); break;
127             case "specialFields": readStringMap(e, corpusConfig.specialFields); break;
128             case "metadataFieldGroups": readMetadataFieldGroups(e, corpusConfig); break;
129             default:
130                 throw new InputFormatConfigException("Unknown key " + e.getKey() + " in corpusConfig");
131             }
132         }

160     private static void readMetadataFieldGroups(Entry<String, JsonNode> mfgEntry, ConfigCorpus cfg) {
161         Iterator<JsonNode> itGroups = array(mfgEntry).elements();
162         while (itGroups.hasNext()) {
163             JsonNode group = itGroups.next();
164             Iterator<Entry<String, JsonNode>> itGroup = obj(group, "metadata field group").fields();
165             ConfigMetadataFieldGroup g = new ConfigMetadataFieldGroup();
166             List<String> fields = new ArrayList<>();
167             while (itGroup.hasNext()) {
168                 Entry<String, JsonNode> e = itGroup.next();
169                 switch (e.getKey()) {
170                 case "name": g.setName(str(e)); break;
171                 case "fields":
172                     readStringList(e, fields);
173                     g.addFields(fields);
174                     break;
175                 case "addRemainingFields":
176                     g.setAddRemainingFields(bool(e));
177                     break;
178                 default:
179                     throw new InputFormatConfigException("Unknown key " + e.getKey() + " in metadata field group " + g.getName());
180                 }
181             }
182             cfg.addMetadataFieldGroup(g);
183         }
184     }

186     private static void readAnnotatedFields(Entry<String, JsonNode> afsEntry, ConfigInputFormat cfg) {
187         Iterator<Entry<String, JsonNode>> itFields = obj(afsEntry).fields();
188         while (itFields.hasNext()) {
189             Entry<String, JsonNode> field = itFields.next();
190             String fieldName = field.getKey();
191             Iterator<Entry<String, JsonNode>> itField = obj(field).fields();
192             ConfigAnnotatedField af = cfg.getOrCreateAnnotatedField(fieldName);
193             while (itField.hasNext()) {
194                 Entry<String, JsonNode> e = itField.next();
195                 switch (e.getKey()) {
196                 case "displayName": af.setDisplayName(str(e)); break;
197                 case "description": af.setDescription(str(e)); break;
198                 case "containerPath": af.setContainerPath(str(e)); break;
199                 case "wordPath": af.setWordPath(str(e)); break;
200                 case "tokenPositionIdPath": af.setTokenPositionIdPath(str(e)); break;
201                 case "punctPath": af.setPunctPath(str(e)); break;
202                 case "annotations": readAnnotations(e, af); break;
203                 case "standoffAnnotations": readStandoffAnnotations(e, af); break;
204                 case "inlineTags": readInlineTags(e, af); break;
205                 default:
206                     throw new InputFormatConfigException("Unknown key " + e.getKey() + " in annotated field " + fieldName);
207                 }
208             }
209         }
210     }

226     protected static ConfigAnnotation readAnnotation(boolean isSubAnnotation, JsonNode a) {
227         Iterator<Entry<String, JsonNode>> itAnnotation = obj(a, "annotation").fields();
228         ConfigAnnotation annot = new ConfigAnnotation();
229         while (itAnnotation.hasNext()) {
230             Entry<String, JsonNode> e = itAnnotation.next();
231             switch (e.getKey()) {
232             case "name": annot.setName(str(e)); break;
233             case "value": annot.setValuePath(fixedStringToXpath(str(e))); break;
234             case "valuePath": annot.setValuePath(str(e)); break;
235             case "captureValuePaths":
236                 ArrayNode paths = (ArrayNode) e.getValue();
237                 paths.iterator().forEachRemaining((t) -> {
238                     annot.addCaptureValuePath(t.asText());
239                 });
240                 break;
241             case "forEachPath":
242                 if (!isSubAnnotation)
243                     throw new InputFormatConfigException("Only subannotations may have forEachPath/namePath");
244                 annot.setForEachPath(str(e)); break;
245             case "namePath":
246                 if (!isSubAnnotation)
247                     throw new InputFormatConfigException("Only subannotations may have forEachPath/namePath");
248                 annot.setName(str(e)); break;
249             case "process": annot.setProcess(readProcess(e)); break;
250             case "displayName": annot.setDisplayName(str(e)); break;
251             case "description": annot.setDescription(str(e)); break;
252             case "basePath": annot.setBasePath(str(e)); break;
253             case "sensitivity":
254                 if (isSubAnnotation)
255                     throw new InputFormatConfigException("Subannotations may not have their own sensitivity settings");
256                 annot.setSensitivity(SensitivitySetting.fromStringValue(str(e))); break;
257             case "uiType": annot.setUiType(str(e)); break;
258             case "subAnnotations":
259                 if (isSubAnnotation)
260                     throw new InputFormatConfigException("Subannotations may not have their own subannotations");
261                 readSubAnnotations(e, annot); break;
262             case "forwardIndex": annot.setForwardIndex(bool(e)); break;
263             case "multipleValues": annot.setMultipleValues(bool(e)); break;
264             default:
265                 throw new InputFormatConfigException("Unknown key " + e.getKey() + " in annotation " + StringUtil.nullToEmpty(annot.getName()));
266             }
267         }
268         return annot;
269     }

271     /**
272      * Convert a fixed string value to an XPath expression yielding that value.
273      * @param s fixed string the XPath should evaluate to
274      * @return XPath expression
275      */
276     public static String fixedStringToXpath(String s) {
277         return "\"" + s.replaceAll("\\\\", "\\\\").replaceAll("\"", "\\\"") + "\"";
278     }

319     private static void readMetadata(Entry<String, JsonNode> mdEntry, ConfigInputFormat cfg) {
320         JsonNode node = mdEntry.getValue();
321         if (node instanceof ObjectNode) {
322             // Single metadata block
323             readMetadataBlock(node, cfg);
324         } else if (node instanceof ArrayNode) {
325             // List of metadata blocks
326             Iterator<JsonNode> itAnnotations = node.elements();
327             while (itAnnotations.hasNext()) {
328                 JsonNode as = itAnnotations.next();
329                 readMetadataBlock(as, cfg);
330             }
331         } else {
332             throw new InputFormatConfigException("Wrong node type for metadata (must be object or array)");
333         }
334     }

336     private static void readMetadataBlock(JsonNode as, ConfigInputFormat cfg) {
337         ConfigMetadataBlock b = cfg.createMetadataBlock();
338         Iterator<Entry<String, JsonNode>> it = obj(as, "metadata block").fields();
339         while (it.hasNext()) {
340             Entry<String, JsonNode> e = it.next();
341             switch(e.getKey()) {
342             case "containerPath": b.setContainerPath(str(e)); break;
343             case "defaultAnalyzer": b.setDefaultAnalyzer(str(e)); break;
344             case "fields": readMetadataFields(e, b); break;
345             default:
346                 throw new InputFormatConfigException("Unknown key " + e.getKey() + " in metadata block");
347             }
348         }
349     }

351     private static void readMetadataFields(Entry<String, JsonNode> mfsEntry, ConfigMetadataBlock b) {
352         Iterator<JsonNode> itFields = array(mfsEntry).elements();
353         while (itFields.hasNext()) {
354             JsonNode fld = itFields.next();
355             Iterator<Entry<String, JsonNode>> itField = obj(fld, "metadata field").fields();
356             ConfigMetadataField f = new ConfigMetadataField();
357             while (itField.hasNext()) {
358                 Entry<String, JsonNode> e = itField.next();
359                 switch (e.getKey()) {
360                 case "name": case "namePath": f.setName(str(e)); break;
361                 case "value": f.setValuePath(fixedStringToXpath(str(e))); break;
362                 case "valuePath": f.setValuePath(str(e)); break;
363                 case "forEachPath": f.setForEachPath(str(e)); break;
364                 case "process": f.setProcess(readProcess(e)); break;
365                 case "displayName": f.setDisplayName(str(e)); break;
366                 case "description": f.setDescription(str(e)); break;
367                 case "type": f.setType(FieldType.fromStringValue(str(e))); break;
368                 case "uiType": f.setUiType(str(e)); break;
369                 case "unknownCondition": f.setUnknownCondition(UnknownCondition.fromStringValue(str(e))); break;
370                 case "unknownValue": f.setUiType(str(e)); break;
371                 case "analyzer": f.setAnalyzer(str(e)); break;
372                 case "displayOrder":
373                     List<String> fields = new ArrayList<>();
374                     readStringList(e, fields);
375                     f.addDisplayOrder(fields);
376                     break;
377                 case "displayValues":
378                     Map<String,String> values = new HashMap<>();
379                     readStringMap(e, values);
380                     f.addDisplayValues(values);
381                     break;
382                 default:
383                     throw new InputFormatConfigException("Unknown key " + e.getKey() + " in metadata field " + f.getName());
384                 }
385             }
386             b.addMetadataField(f);
387         }
388     }

390     private static void readLinkedDocuments(Entry<String, JsonNode> ldsEntry, ConfigInputFormat cfg) {
391         Iterator<Entry<String, JsonNode>> itLinkedDocs = obj(ldsEntry).fields();
392         while (itLinkedDocs.hasNext()) {
393             Entry<String, JsonNode> linkedDoc = itLinkedDocs.next();
394             ConfigLinkedDocument ld = cfg.getOrCreateLinkedDocument(linkedDoc.getKey());
395             Iterator<Entry<String, JsonNode>> itLinkedDoc = obj(linkedDoc).fields();
396             while (itLinkedDoc.hasNext()) {
397                 Entry<String, JsonNode> e = itLinkedDoc.next();
398                 switch (e.getKey()) {
399                 case "store": ld.setStore(bool(e)); break;
400                 case "linkValues": readLinkValues(e, ld); break;
401                 case "ifLinkPathMissing": ld.setIfLinkPathMissing(MissingLinkPathAction.fromStringValue(str(e))); break;
402                 case "inputFile": ld.setInputFile(str(e)); break;
403                 case "pathInsideArchive": ld.setPathInsideArchive(str(e)); break;
404                 case "documentPath": ld.setDocumentPath(str(e)); break;
405                 case "inputFormat": readInputFormat(ld, e); break;
406                 default:
407                     throw new InputFormatConfigException("Unknown key " + e.getKey() + " in linked document " + ld.getName());
408                 }
409             }
410         }
411     }

413     protected static void readInputFormat(ConfigLinkedDocument ld, Entry<String, JsonNode> e) {
414         // Resolve the inputFormat right now, instead of potentially failing later when the format is actually needed at some point during indexing
415         String formatIdentifier = str(e);
416         Format format = DocumentFormats.getFormat(formatIdentifier);
417         if (format == null)
418             throw new InputFormatConfigException("Unknown input format " + str(e) + " in linked document " + ld.getName());
419 
420         ld.setInputFormatIdentifier(formatIdentifier);
421     }

423     private static void readLinkValues(Entry<String, JsonNode> lvsEntry, ConfigLinkedDocument ld) {
424         Iterator<JsonNode> itLinkValues = array(lvsEntry).elements();
425         while (itLinkValues.hasNext()) {
426             JsonNode linkValue = itLinkValues.next();
427             ConfigLinkValue lv = new ConfigLinkValue();
428             Iterator<Entry<String, JsonNode>> itLinkValue = obj(linkValue, "link value").fields();
429             while (itLinkValue.hasNext()) {
430                 Entry<String, JsonNode> e = itLinkValue.next();
431                 switch (e.getKey()) {
432                 case "value": lv.setValuePath(fixedStringToXpath(str(e))); break;
433                 case "valuePath": lv.setValuePath(str(e)); break;
434                 case "valueField": lv.setValueField(str(e)); break;
435                 case "process": lv.setProcess(readProcess(e)); break;
436                 default:
437                     throw new InputFormatConfigException("Unknown key " + e.getKey() + " in linked document " + ld.getName());
438                 }
439             }
440             ld.addLinkValue(lv);
441         }

444     private static List<ConfigProcessStep> readProcess(Entry<String, JsonNode> prEntry) {
445         Iterator<JsonNode> itSteps = array(prEntry).elements();
446         List<ConfigProcessStep> p = new ArrayList<>();
447         while (itSteps.hasNext()) {
448             JsonNode step = itSteps.next();
449             ConfigProcessStep s = new ConfigProcessStep();
450             Iterator<Entry<String, JsonNode>> itStep = obj(step, "processing step").fields();
451             while (itStep.hasNext()) {
452                 Entry<String, JsonNode> e = itStep.next();
453                 switch(e.getKey()) {
454                 case "action": s.setMethod(str(e)); break;
455                 default: s.addParam(e.getKey(), str(e)); break;
456                 }
457             }
458             p.add(s);
459         }
460         return p;
461     }
