31  /**
32   * A DocIndexerFactory implementation that automatically adds configurations in the formats/ dir in the jar
33   * and scans the default config directories for subdirectories called "config", then loads any configurations within those directories.
34   * Instances of this class can be used to add additional configs within customized directories.
35   */
36  public class DocIndexerFactoryConfig implements DocIndexerFactory {
37  	static final Logger logger = LogManager.getLogger(DocIndexerConfig.class);
38  
39  	protected boolean isInitialized = false;
40  
41  	protected Map<String, File> unloaded = new HashMap<>(); // Entries here are short-lived, the list should be clean as long as there is no scanDirectories call running
42  	protected Map<String, ConfigInputFormat> supported = new HashMap<>();
43  
44  
45  	/**
46  	 * Return a config from the supported list, or load it if it's in the unloaded list.
47  	 * <br><br>
48  	 * Why this function? Formats can depend on each other,
49  	 * and this is a mechanism to allow formats to get/lazy-load other formats registered with this factory.
50  	 * Why lazy-loading? Formats only refer to other formats by their name, not their file location, so we need to find them all before we actually load them.
51  	 * Those found config files are kept in the {@link DocIndexerFactoryConfig#unloaded} map until they are loaded
52  	 */
53  	protected Function<String, Optional<ConfigInputFormat>> finder = formatIdentifier -> {
54  		if(!isSupported(formatIdentifier)) // Give our wrapping DocIndexerFactory a chance to load a new format (in case it's a derived class)
55  			return Optional.empty();
56  
57  		if (unloaded.containsKey(formatIdentifier)) {
58  			File f = unloaded.get(formatIdentifier);
59  			unloaded.remove(formatIdentifier); // remove before load to avoid infinite recursion on circular dependencies
60  			return load(formatIdentifier, f);
61  		}
62  
63  		// If unknown return null, can't help it.
64  		return Optional.ofNullable(supported.get(formatIdentifier));
65  	};

72  	@Override
73  	public void init() throws InputFormatConfigException {
74  		if (isInitialized)
75  			return;
76  		isInitialized = true;
77  
78  		// Note that these names should not collide with the abbreviations used by DocIndexerFactoryClass, or this will override those classes.
79  		String[] formats = { "chat", "cmdi", "csv", "eaf", "folia", "sketch-wpl", "tcf", "tei-p4", "tei", "tsv-frog", "tsv", "txt" };
80  		for (String formatIdentifier: formats) {
81  			try (InputStream is = DocumentFormats.class	.getClassLoader().getResourceAsStream("formats/" + formatIdentifier + ".blf.yaml")) {
82  				if (is == null)
83  					continue; // not found
84  
85  				try (Reader reader = new BufferedReader(new InputStreamReader(is))) {
86  					ConfigInputFormat format = new ConfigInputFormat(formatIdentifier);
87  
88  					format.setReadFromFile(new File("$BLACKLAB_JAR/formats/" + formatIdentifier + ".blf.yaml"));
89  					InputFormatReader.read(reader, false, format, finder);
90  					addFormat(format);
91  				}
92  			} catch (IOException e) {
93  				throw new RuntimeException(e);
94  			}
95  		}
96  
97  		List<File> configDirs = ConfigReader.getDefaultConfigDirs();
98  		List<File> formatsDirs = new ArrayList<>();
99  		for (File dir: configDirs) {
100 			formatsDirs.add(new File(dir, "formats"));
101 		}
102 		addFormatsInDirectories(formatsDirs);
103 
104 		// Load any configs that may have been added using addFormatsInDirectories, from before we were initialized
105 		loadUnloaded();
106 	}
107 
108 	@Override
109 	public boolean isSupported(String formatIdentifier) {
110 		// Do not load yet, this is called during discovery of new formats (to detect duplicates)
111 		// We shouldn't load any new format until we've discovered them all,
112 		// so that if a format depends on another yet to be loaded format, we know where to find it
113 		return (supported.containsKey(formatIdentifier) || unloaded.containsKey(formatIdentifier));
114 	}
115 
116 	@Override
117 	public List<Format> getFormats() {
118 		List<Format> ret = new ArrayList<>();
119 		for (Entry<String, ConfigInputFormat> e: this.supported.entrySet()) {
120 			ConfigInputFormat config = e.getValue();
121             Format desc = new Format(config.getName(), config.getDisplayName(), config.getDescription(), config.getHelpUrl());
122 			desc.setConfig(config);
123 			desc.setVisible(config.isVisible());
124 			ret.add(desc);
125 		}
126 
127 		return ret;
128 	}

130 	@Override
131 	public Format getFormat(String formatIdentifier) {
132 		if (!isSupported(formatIdentifier))
133 			return null;
134 
135 		ConfigInputFormat config = this.supported.get(formatIdentifier);
136 		if (config == null) {
137 		    loadUnloaded();
138 		    config = this.supported.get(formatIdentifier);
139 		}
140 		Format desc = new Format(config.getName(), config.getDisplayName(), config.getDescription(), config.getHelpUrl());
141 		desc.setConfig(config);
142 		desc.setVisible(config.isVisible());
143 		return desc;
144 	}

146 	protected void addFormat(ConfigInputFormat format) throws InputFormatConfigException {
147 		if (isSupported(format.getName()))
148 			throw new IllegalArgumentException("A config format with this name already exists.");
149 
150 		format.validate();
151 		supported.put(format.getName(), format);
152 	}
153 
154 	/**
155 	 * Locate all config files (files ending with .blf.yaml, .blf.yml, .blf.json) within the list of directories and load them.
156 	 * Formats will use their filename (excluding the .blf.* extension) as a unique identifier. Duplicate formats are ignored.
157 	 * Note that the actual loading of the configs is deferred until this factory is initialized by registering it with {@link DocumentFormats}, if not already done.
158 	 *
159 	 * This is a one-time scan, so configs placed in these directories after this scan will not be picked up.
160 	 * <br><br>
161 	 * If any of the config files depend on other config files, the locations of those dependencies must also be provided in this list of directories.
162 	 * If config A refers to config B, then the directory where config B is located must also be present in the dirs list.
163 	 * @param dirs
164 	 * @throws InputFormatConfigException when one of the formats could not be loaded
165 	 */
166 	public void addFormatsInDirectories(List<File> dirs) throws InputFormatConfigException {
167 		// Finds all new configs and add them to the "unloaded" list
168 		FileTask configLocator = new FileTask() {
169 			@Override
170             public void process(File f) {
171                 if (f.getName().matches("^[\\-\\w]+\\.blf\\.(ya?ml|json)$")) {
172                 	String formatIdentifier = ConfigInputFormat.stripExtensions(f.getName());
173                     if (isSupported(formatIdentifier)) {
174                     	logger.info("Skipping config format " + f + "; a format with this name already exists.");
175                     	return;
176                     }
177 
178                     if (!Files.isReadable(f.toPath()) || !Files.isRegularFile(f.toPath())) {
179                     	logger.trace("Skipping unreadable config file " + f);
180                     	return;
181                     }
182 
183                 	unloaded.put(formatIdentifier, f);
184                 }
185             }
186 		};
187 
188 		// Run the configLocation on the directory - not recursive
189 		for(File dir : dirs) {
190 			if (Files.isReadable(dir.toPath()) && Files.isDirectory(dir.toPath())) {
191 				FileUtil.processTree(dir, "*", false, configLocator);
192 			}
193 		}
194 
195 		// Don't load until we're initialized,
196 		// or the configs won't be able to depend on one of the default configs (which are not loaded until after initialization)
197 		if (isInitialized)
198 			loadUnloaded();
199 	}
200 
201 	protected Optional<ConfigInputFormat> load(String formatIdentifier, File f) {
202 		try {
203 			ConfigInputFormat format = new ConfigInputFormat(formatIdentifier);
204 			InputFormatReader.read(f, format, finder);
205 			format.setReadFromFile(f);
206 
207 			addFormat(format);
208 			return Optional.of(format);
209 		} catch (IOException e) {
210 			return Optional.empty();
211 		}
212 	}
213 
214 	protected void loadUnloaded() {
215 		// use !isEmpty so we can remove entries during iteration (using the finder above) without messing up iterators
216 		while (!unloaded.isEmpty()) {
217 			Entry<String, File> e = unloaded.entrySet().iterator().next();
218 			unloaded.remove(e.getKey());
219 
220 			try {
221 				load(e.getKey(), e.getValue());
222 			} catch (InputFormatConfigException ex) {
223 				logger.warn("Cannot load user format " + e.getValue() + ": " + ex.getMessage());
224 				// an invalid format somehow got saved, or something else went wrong, just ignore this file then
225 			}
226 		}
227 	}

229     @Override
230     public DocIndexerConfig get(String formatIdentifier, Indexer indexer, String documentName, Reader reader) {
231        if (!isSupported(formatIdentifier))
232         	throw new UnsupportedOperationException("Unknown format '"+formatIdentifier+"', call isSupported(formatIdentifier) before attempting to get()");
233 
234     	DocIndexerConfig d = DocIndexerConfig.fromConfig(supported.get(formatIdentifier));
235         d.setIndexer(indexer);
236         d.setDocumentName(documentName);
237         d.setDocument(reader);
238         return d;
239     }
