38      CompleteQuery clause(BlackLabIndex index, String annotation, String relation, String term,
39              String defaultAnnotation) {
40          if (relation == null)
41              relation = "=";
42          if (annotation == null)
43              annotation = defaultAnnotation;
44  
45          if (relation.equals("=")) {
46              return contains(index, annotation, term);
47          } else if (relation.equals("any")) {
48              throw new UnsupportedOperationException("any not yet supported");
49          } else if (relation.equals("all")) {
50              throw new UnsupportedOperationException("all not yet supported");
51          } else if (relation.equals("exact")) {
52              throw new UnsupportedOperationException("exact not supported");
53          } else if (relation.equals("<") || relation.equals(">") || relation.equals("<=") || relation.equals(">=")
54                  || relation.equals("<>")) {
55              throw new UnsupportedOperationException("Only contains (=) relation is supported!");
56          } else {
57              throw new UnsupportedOperationException("Unknown relation operator: " + relation);
58          }
59      }

61      CompleteQuery combineClauses(CompleteQuery a, String op, CompleteQuery b) {
62          if (op.equalsIgnoreCase("and")) {
63              return a.and(b);
64          } else if (op.equalsIgnoreCase("or")) {
65              return a.or(b);
66          } else if (op.equalsIgnoreCase("not")) {
67              return a.not(b);
68          } else if (op.equalsIgnoreCase("prox")) {
69              throw new UnsupportedOperationException("prox is not yet supported!");
70          }
71          throw new UnsupportedOperationException("Unrecognized operator " + op);
72      }

74      /**
75       * Return a clause meaning "field contains value".
76       *
77       * Depending on the field name, this will contain a contents query or a filter
78       * query.
79       *
80       * @param index our index
81       * @param field the field name. If it starts with "contents.", it is a contents
82       *            query. "contents" by itself means "contents.word". "word", "lemma"
83       *            and "pos" by themselves are interpreted as being prefixed with
84       *            "contents."
85       * @param value the value, optionally with wildcards, to search for
86       * @return the query
87       */
88      CompleteQuery contains(BlackLabIndex index, String field, String value) {
89  
90          boolean isContentsSearch = false;
91          String prop = "word";
92          boolean isProperty;
93          if (index != null && !index.getClass().getSimpleName().startsWith("Mock")) // FIXME: ARGH...
94              isProperty = index.mainAnnotatedField().annotations().exists(field);
95          else
96              isProperty = field.equals("word") || field.equals("lemma") || field.equals("pos"); // common case
97          if (isProperty) {
98              isContentsSearch = true;
99              prop = field;
100         } else if (field.equals("contents")) {
101             isContentsSearch = true;
102         } else if (field.startsWith("contents.")) {
103             isContentsSearch = true;
104             prop = field.substring(9);
105         }
106 
107         String[] parts = value.trim().split("\\s+");
108         TextPattern tp = null;
109         Query q = null;
110         if (parts.length == 1) {
111             // Single term, possibly with wildcards
112             if (isContentsSearch)
113                 tp = new TextPatternWildcard(value.trim());
114             else
115                 q = new WildcardQuery(new Term(field, value));
116         } else {
117             // Phrase query
118             if (isContentsSearch) {
119                 List<TextPattern> clauses = new ArrayList<>();
120                 for (int i = 0; i < parts.length; i++) {
121                     clauses.add(new TextPatternWildcard(parts[i]));
122                 }
123                 tp = new TextPatternSequence(clauses.toArray(new TextPattern[0]));
124             } else {
125                 PhraseQuery.Builder pb = new PhraseQuery.Builder();
126                 for (int i = 0; i < parts.length; i++) {
127                     pb.add(new Term(field, parts[i]));
128                 }
129                 q = pb.build();
130             }
131         }
132 
133         if (isContentsSearch)
134             return new CompleteQuery(new TextPatternAnnotation(prop, tp), null);
135         return new CompleteQuery(null, q);
136     }

146     public CompleteQuery parse(String query) throws InvalidQuery {
147         try {
148             GeneratedContextualQueryLanguageParser parser = new GeneratedContextualQueryLanguageParser(new StringReader(query));
149             parser.wrapper = this;
150             return parser.query();

164     String chopEnds(String input) {
165         if (input.length() >= 2)
166             return input.substring(1, input.length() - 1);
167         throw new BlackLabRuntimeException();
168     }
