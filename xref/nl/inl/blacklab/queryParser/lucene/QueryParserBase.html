149   /** Parses a query string, returning a {@link nl.inl.blacklab.search.TextPattern}.
150    *  @param query  the query string to be parsed.
151    *  @throws ParseException if the parsing fails
152    */
153   public TextPattern parse(String query) throws ParseException {
154     ReInit(new FastCharStream(new StringReader(query)));
155     try {
156       // TopLevelQuery is a Query followed by the end-of-input (EOF)
157       TextPattern res = TopLevelQuery(field);
158       return res!=null ? res : newBooleanQuery(false);
159     }
160     catch (ParseException tme) {
161       // rethrow to include the original query:
162       ParseException e = new ParseException("Cannot parse '" +query+ "': " + tme.getMessage());
163       e.initCause(tme);
164       throw e;
165     }
166     catch (TokenMgrError tme) {
167       ParseException e = new ParseException("Cannot parse '" +query+ "': " + tme.getMessage());
168       e.initCause(tme);
169       throw e;
170     }
171     catch (BooleanQuery.TooManyClauses tmc) {
172       ParseException e = new ParseException("Cannot parse '" +query+ "': too many boolean clauses");
173       e.initCause(tmc);
174       throw e;
175     }
176   }

438    *
439    */
440   public DateTools.Resolution getDateResolution(String fieldName) {
441     if (fieldName == null) {
442       throw new IllegalArgumentException("Field cannot be null.");
443     }
444 
445     if (fieldToDateResolution == null) {
446       // no field specific date resolutions set; return default date resolution instead
447       return this.dateResolution;
448     }
449 
450     DateTools.Resolution resolution = fieldToDateResolution.get(fieldName);
451     if (resolution == null) {
452       // no date resolutions set for the given field; return default date resolution instead
453       resolution = this.dateResolution;
454     }
455 
456     return resolution;

477   // BL: BooleanClause -> TPBooleanClause
478   protected void addClause(List<TPBooleanClause> clauses, int conj, int mods, TextPattern q) {
479     boolean required, prohibited;
480 
481     // If this term is introduced by AND, make the preceding term required,
482     // unless it's already prohibited
483     if (clauses.size() > 0 && conj == CONJ_AND) {
484       TPBooleanClause c = clauses.get(clauses.size()-1);
485       if (!c.isProhibited())
486         c.setOccur(BooleanClause.Occur.MUST);
487     }
488 
489     if (clauses.size() > 0 && operator == AND_OPERATOR && conj == CONJ_OR) {
490       // If this term is introduced by OR, make the preceding term optional,
491       // unless it's prohibited (that means we leave -a OR b but +a OR b-->a OR b)
492       // notice if the input is a OR b, first term is parsed as required; without
493       // this modification a OR b would parsed as +a OR b
494     	TPBooleanClause c = clauses.get(clauses.size()-1);
495       if (!c.isProhibited())
496         c.setOccur(BooleanClause.Occur.SHOULD);
497     }
498 
499     // We might have been passed a null query; the term might have been
500     // filtered away by the analyzer.
501     if (q == null)
502       return;
503 
504     if (operator == OR_OPERATOR) {
505       // We set REQUIRED if we're introduced by AND or +; PROHIBITED if
506       // introduced by NOT or -; make sure not to set both.
507       prohibited = (mods == MOD_NOT);
508       required = (mods == MOD_REQ);
509       if (conj == CONJ_AND && !prohibited) {
510         required = true;
511       }
512     } else {
513       // We set PROHIBITED if we're introduced by NOT or -; We set REQUIRED
514       // if not PROHIBITED and not introduced by OR
515       prohibited = (mods == MOD_NOT);
516       required   = (!prohibited && conj != CONJ_OR);
517     }
518     if (required && !prohibited)
519       clauses.add(newBooleanClause(q, BooleanClause.Occur.MUST));
520     else if (!required && !prohibited)
521       clauses.add(newBooleanClause(q, BooleanClause.Occur.SHOULD));
522     else if (!required && prohibited)
523       clauses.add(newBooleanClause(q, BooleanClause.Occur.MUST_NOT));
524     else
525       throw new RuntimeException("Clause cannot be both required and prohibited");
526   }

535   /**
536    * @exception org.apache.lucene.queryparser.classic.ParseException throw in overridden method to disallow
537    */
538   protected TextPattern newFieldQuery(Analyzer analyzer, String field, String queryText, boolean quoted)  throws ParseException {
539     // Use the analyzer to get all the tokens, and then build a TermQuery,
540     // PhraseQuery, or nothing based on the term count
541 
542     TokenStream source;
543     try {
544       source = analyzer.tokenStream(field, new StringReader(queryText));
545       source.reset();
546     } catch (IOException e) {
547       ParseException p = new ParseException("Unable to initialize TokenStream to analyze query text");
548       p.initCause(e);
549       throw p;
550     }
551     CachingTokenFilter buffer = new CachingTokenFilter(source);
552     TermToBytesRefAttribute termAtt = null;
553     PositionIncrementAttribute posIncrAtt = null;
554     int numTokens = 0;
555 
556     buffer.reset();
557 
558     if (buffer.hasAttribute(TermToBytesRefAttribute.class)) {
559       termAtt = buffer.getAttribute(TermToBytesRefAttribute.class);
560     }
561     if (buffer.hasAttribute(PositionIncrementAttribute.class)) {
562       posIncrAtt = buffer.getAttribute(PositionIncrementAttribute.class);
563     }
564 
565     int positionCount = 0;
566     boolean severalTokensAtSamePosition = false;
567 
568     boolean hasMoreTokens = false;
569     if (termAtt != null) {
570       try {
571         hasMoreTokens = buffer.incrementToken();
572         while (hasMoreTokens) {
573           numTokens++;
574           int positionIncrement = (posIncrAtt != null) ? posIncrAtt.getPositionIncrement() : 1;
575           if (positionIncrement != 0) {
576             positionCount += positionIncrement;
577           } else {
578             severalTokensAtSamePosition = true;
579           }
580           hasMoreTokens = buffer.incrementToken();
581         }
582       } catch (IOException e) {
583         // ignore
584       }
585     }
586     try {
587       // rewind the buffer stream
588       buffer.reset();
589 
590       // close original stream - all tokens buffered
591       source.close();
592     }
593     catch (IOException e) {
594       ParseException p = new ParseException("Cannot close TokenStream analyzing query text");
595       p.initCause(e);
596       throw p;
597     }
598 
599     BytesRef bytes = termAtt == null ? null : termAtt.getBytesRef();
600 
601     if (numTokens == 0)
602       return null;
603     else if (numTokens == 1) {
604       try {
605         boolean hasNext = buffer.incrementToken();
606         assert hasNext == true;
607         termAtt.fillBytesRef();
608       } catch (IOException e) {
609         // safe to ignore, because we know the number of tokens
610       }
611       return newTermQuery(new Term(field, BytesRef.deepCopyOf(bytes)));
612     } else {
613       if (severalTokensAtSamePosition || (!quoted && !autoGeneratePhraseQueries)) {
614         if (positionCount == 1 || (!quoted && !autoGeneratePhraseQueries)) {
615           // no phrase query:
616           TextPatternBoolean q = newBooleanQuery(positionCount == 1); // BL: BooleanQuery -> TextPatternBoolean
617 
618           BooleanClause.Occur occur = positionCount > 1 && operator == AND_OPERATOR ?
619             BooleanClause.Occur.MUST : BooleanClause.Occur.SHOULD;
620 
621           for (int i = 0; i < numTokens; i++) {
622             try {
623               boolean hasNext = buffer.incrementToken();
624               assert hasNext == true;
625               termAtt.fillBytesRef();
626             } catch (IOException e) {
627               // safe to ignore, because we know the number of tokens
628             }
629             TextPattern currentQuery = newTermQuery(
630                 new Term(field, BytesRef.deepCopyOf(bytes)));
631             q.add(currentQuery, occur);
632           }
633           return q;
634         }
635         else {
636           // phrase query:
637           TPMultiPhrase mpq = newMultiPhraseQuery(); // BL: MultiPhraseQuery -> TPMultiPhrase
638           mpq.setSlop(phraseSlop);
639           List<Term> multiTerms = new ArrayList<Term>();
640           int position = -1;
641           for (int i = 0; i < numTokens; i++) {
642             int positionIncrement = 1;
643             try {
644               boolean hasNext = buffer.incrementToken();
645               assert hasNext == true;
646               termAtt.fillBytesRef();
647               if (posIncrAtt != null) {
648                 positionIncrement = posIncrAtt.getPositionIncrement();
649               }
650             } catch (IOException e) {
651               // safe to ignore, because we know the number of tokens
652             }
653 
654             if (positionIncrement > 0 && multiTerms.size() > 0) {
655               if (enablePositionIncrements) {
656                 mpq.add(multiTerms.toArray(new Term[0]),position);
657               } else {
658                 mpq.add(multiTerms.toArray(new Term[0]));
659               }
660               multiTerms.clear();
661             }
662             position += positionIncrement;
663             multiTerms.add(new Term(field, BytesRef.deepCopyOf(bytes)));
664           }
665           if (enablePositionIncrements) {
666             mpq.add(multiTerms.toArray(new Term[0]),position);
667           } else {
668             mpq.add(multiTerms.toArray(new Term[0]));
669           }
670           return mpq;
671         }
672       }
673       else {
674         TPPhrase pq = newPhraseQuery(); // BL: PhraseQuery -> TPPhrase
675         pq.setSlop(phraseSlop);
676         int position = -1;
677 
678         for (int i = 0; i < numTokens; i++) {
679           int positionIncrement = 1;
680 
681           try {
682             boolean hasNext = buffer.incrementToken();
683             assert hasNext == true;
684             termAtt.fillBytesRef();
685             if (posIncrAtt != null) {
686               positionIncrement = posIncrAtt.getPositionIncrement();
687             }
688           } catch (IOException e) {
689             // safe to ignore, because we know the number of tokens
690           }
691 
692           if (enablePositionIncrements) {
693             position += positionIncrement;
694             pq.add(new Term(field, BytesRef.deepCopyOf(bytes)),position);
695           } else {

853   protected BytesRef analyzeMultitermTerm(String field, String part, Analyzer analyzerIn) {
854     TokenStream source;
855 
856     if (analyzerIn == null) analyzerIn = analyzer;
857 
858     try {
859       source = analyzerIn.tokenStream(field, new StringReader(part));
860       source.reset();
861     } catch (IOException e) {
862       throw new RuntimeException("Unable to initialize TokenStream to analyze multiTerm term: " + part, e);
863     }
864 
865     TermToBytesRefAttribute termAtt = source.getAttribute(TermToBytesRefAttribute.class);
866     BytesRef bytes = termAtt.getBytesRef();
867 
868     try {
869       if (!source.incrementToken())
870         throw new IllegalArgumentException("analyzer returned no terms for multiTerm term: " + part);
871       termAtt.fillBytesRef();
872       if (source.incrementToken())
873         throw new IllegalArgumentException("analyzer returned too many terms for multiTerm term: " + part);
874     } catch (IOException e) {
875       throw new RuntimeException("error analyzing range part: " + part, e);
876     }
877 
878     try {
879       source.end();
880       source.close();
881     } catch (IOException e) {
882       throw new RuntimeException("Unable to end & close TokenStream after analyzing multiTerm term: " + part, e);
883     }
884 
885     return BytesRef.deepCopyOf(bytes);
886   }

1175   /**
1176    * Returns a String where the escape char has been
1177    * removed, or kept only once if there was a double escape.
1178    *
1179    * Supports escaped unicode characters, e. g. translates
1180    * <code>\\u0041</code> to <code>A</code>.
1181    *
1182    */
1183   String discardEscapeChar(String input) throws ParseException {
1184     // Create char array to hold unescaped char sequence
1185     char[] output = new char[input.length()];
1186 
1187     // The length of the output can be less than the input
1188     // due to discarded escape chars. This variable holds
1189     // the actual length of the output
1190     int length = 0;
1191 
1192     // We remember whether the last processed character was
1193     // an escape character
1194     boolean lastCharWasEscapeChar = false;
1195 
1196     // The multiplier the current unicode digit must be multiplied with.
1197     // E. g. the first digit must be multiplied with 16^3, the second with 16^2...
1198     int codePointMultiplier = 0;
1199 
1200     // Used to calculate the codepoint of the escaped unicode character
1201     int codePoint = 0;
1202 
1203     for (int i = 0; i < input.length(); i++) {
1204       char curChar = input.charAt(i);
1205       if (codePointMultiplier > 0) {
1206         codePoint += hexToInt(curChar) * codePointMultiplier;
1207         codePointMultiplier >>>= 4;
1208         if (codePointMultiplier == 0) {
1209           output[length++] = (char)codePoint;
1210           codePoint = 0;
1211         }
1212       } else if (lastCharWasEscapeChar) {
1213         if (curChar == 'u') {
1214           // found an escaped unicode character
1215           codePointMultiplier = 16 * 16 * 16;
1216         } else {
1217           // this character was escaped
1218           output[length] = curChar;
1219           length++;
1220         }
1221         lastCharWasEscapeChar = false;
1222       } else {
1223         if (curChar == '\\') {
1224           lastCharWasEscapeChar = true;
1225         } else {
1226           output[length] = curChar;
1227           length++;
1228         }
1229       }
1230     }
1231 
1232     if (codePointMultiplier > 0) {
1233       throw new ParseException("Truncated unicode escape sequence.");
1234     }
1235 
1236     if (lastCharWasEscapeChar) {
1237       throw new ParseException("Term can not end with escape character.");
1238     }
1239 
1240     return new String(output, 0, length);
1241   }

1243   /** Returns the numeric value of the hexadecimal character */
1244   static final int hexToInt(char c) throws ParseException {
1245     if ('0' <= c && c <= '9') {
1246       return c - '0';
1247     } else if ('a' <= c && c <= 'f'){
1248       return c - 'a' + 10;
1249     } else if ('A' <= c && c <= 'F') {
1250       return c - 'A' + 10;
1251     } else {
1252       throw new ParseException("Non-hex character in Unicode escape sequence: " + c);
1253     }
1254   }

1256   /**
1257    * Returns a String where those characters that QueryParser
1258    * expects to be escaped are escaped by a preceding <code>\</code>.
1259    */
1260   public static String escape(String s) {
1261     StringBuilder sb = new StringBuilder();
1262     for (int i = 0; i < s.length(); i++) {
1263       char c = s.charAt(i);
1264       // These characters are part of the query syntax and must be escaped
1265       if (c == '\\' || c == '+' || c == '-' || c == '!' || c == '(' || c == ')' || c == ':'
1266         || c == '^' || c == '[' || c == ']' || c == '\"' || c == '{' || c == '}' || c == '~'
1267         || c == '*' || c == '?' || c == '|' || c == '&' || c == '/') {
1268         sb.append('\\');
1269       }
1270       sb.append(c);
1271     }
1272     return sb.toString();
1273   }
