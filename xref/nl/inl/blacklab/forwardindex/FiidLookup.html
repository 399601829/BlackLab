3   import java.io.IOException;
4   import java.util.Map;
5   import java.util.Map.Entry;
6   import java.util.TreeMap;
7   
8   import org.apache.lucene.index.IndexReader;
9   import org.apache.lucene.index.LeafReader;
10  import org.apache.lucene.index.LeafReaderContext;
11  import org.apache.lucene.index.NumericDocValues;
12  import org.apache.lucene.uninverting.UninvertingReader;
13  
14  import nl.inl.blacklab.index.complex.ComplexFieldUtil;
15  
16  /**
17   * Class for looking up forward index id, using DocValues or stored fields.
18   */
19  class FiidLookup {
20  
21  	/** Index reader, for getting documents (for translating from Lucene doc id to fiid) */
22  	private IndexReader reader;
23  
24  	/** fiid field name in the Lucene index (for translating from Lucene doc id to fiid) */
25  	private String fiidFieldName;
26  
27  	/** The DocValues per segment (keyed by docBase) */
28  	private Map<Integer, NumericDocValues> cachedFiids;
29  
30  	public FiidLookup(IndexReader reader, String lucenePropFieldName) {
31  		this.fiidFieldName = ComplexFieldUtil.forwardIndexIdField(lucenePropFieldName);
32  		this.reader = reader;
33  		Map<String, UninvertingReader.Type> fields = new TreeMap<>();
34  		fields.put(fiidFieldName, UninvertingReader.Type.INTEGER);
35  		cachedFiids = new TreeMap<>();
36  		try {
37  			for (LeafReaderContext rc: reader.leaves()) {
38  				LeafReader r = rc.reader();
39  				@SuppressWarnings("resource")
40  				UninvertingReader uninv = new UninvertingReader(r, fields);
41  				cachedFiids.put(rc.docBase, uninv.getNumericDocValues(fiidFieldName));
42  			}
43  
44  			int numToCheck = Math.min(ForwardIndexImplV3.NUMBER_OF_CACHE_ENTRIES_TO_CHECK, reader.maxDoc());
45  			if (!hasFiids(numToCheck))
46  				cachedFiids = null;
47  		} catch (IOException e) {
48  			throw new RuntimeException(e);
49  		}
50  	}
51  
52  	public long get(int docId) {
53  		if (cachedFiids != null) {
54  			// Find the fiid in the correct segment
55  			Entry<Integer, NumericDocValues> prev = null;
56  			for (Entry<Integer, NumericDocValues> e: cachedFiids.entrySet()) {
57  				Integer docBase = e.getKey();
58  				if (docBase > docId) {
59  					// Previous segment (the highest docBase lower than docId) is the right one
60  					Integer prevDocBase = prev.getKey();
61  					NumericDocValues prevDocValues = prev.getValue();
62  					return prevDocValues.get(docId - prevDocBase);
63  				}
64  				prev = e;
65  			}
66  			// Last segment is the right one
67  			Integer prevDocBase = prev.getKey();
68  			NumericDocValues prevDocValues = prev.getValue();
69  			return prevDocValues.get(docId - prevDocBase);
70  		}
71  
72  		// Not cached; find fiid by reading stored value from Document now
73  		try {
74  			return Integer.parseInt(reader.document(docId).get(fiidFieldName));
75  		} catch (Exception e) {
76  			throw new RuntimeException(e);
77  		}
78  
79  	}
80  
81  	public boolean hasFiids(int numToCheck) {
82  		// Check if the cache was retrieved OK
83  		boolean allZeroes = true;
84  		for (int i = 0; i < numToCheck; i++) {
85  			// (NOTE: we don't check if document wasn't deleted, but that shouldn't matter here)
86  			if (get(i) != 0) {
87  				allZeroes = false;
88  				break;
89  			}
90  		}
91  		return !allZeroes;
92  	}
93  }
