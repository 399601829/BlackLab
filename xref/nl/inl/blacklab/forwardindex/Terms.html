23      protected static final Charset DEFAULT_CHARSET = Charset.forName("utf-8");

25      /**
26       * We set this to a lower value on Windows because we can't properly truncate
27       * the file due to the file still being mapped (there is no clean way to unmap a
28       * mapped file in Java, and Windows doesn't allow truncating a mapped file). The
29       * lower value on Windows prevents too much wasted space.
30       */
31      protected static final int DEFAULT_MAX_MAP_SIZE = File.separatorChar == '\\' ? 100_000_000 : Integer.MAX_VALUE - 100;
32  
33      /** Number of bytes per int */
34      protected static final int BYTES_PER_INT = Integer.SIZE / Byte.SIZE;

36      /** How many terms total are there? (always valid) */
37      int numberOfTerms = 0;

50      Collator collatorInsensitive;

52      /** Use new blocks-based terms file, that can grow larger than 2 GB? */
53      boolean useBlockBasedTermsFile = true;
54  
55      /**
56       * The maximum block size to use while writing the terms file. Ususally around
57       * the limit of 2GB, but for testing, we can set this to a lower value. Note
58       * that this should be significantly larger than maxBlockSize, because we also
59       * need to store offsets.
60       */
61      int maxMapSize = DEFAULT_MAX_MAP_SIZE;

67      /**
68       * Get the existing index number of a term, or add it to the term list and
69       * assign it a new index number.

71       * In index mode, this is fast. In search mode, this is slower, because we have
72       * to do a binary search through the memory-mapped terms file. However, this is

101     protected IntBuffer readFromFileChannel(FileChannel fc, long fileLength) throws IOException {
102         long fileMapStart = 0;
103         long fileMapLength = Math.min(maxMapSize, fileLength);
104         MappedByteBuffer buf = fc.map(MapMode.READ_ONLY, fileMapStart, fileMapLength);
105         int n = buf.getInt();
106         IntBuffer ib = buf.asIntBuffer();
107         numberOfTerms = n;
108         int[] termStringOffsets = new int[n + 1];
109         terms = new String[n];
110 
111         if (useBlockBasedTermsFile) {
112             // New format, multiple blocks of term strings if necessary,
113             // so term strings may total over 2 GB.
114 
115             // Read the term string offsets and string data block
116             int currentTerm = 0;
117             while (currentTerm < n) {
118 
119                 int numTermsThisBlock = ib.get();
120                 ib.get(termStringOffsets, currentTerm, numTermsThisBlock); // term
121                                                                            // string
122                                                                            // offsets
123 
124                 // Read term strings data
125                 int dataBlockSize = termStringOffsets[currentTerm + numTermsThisBlock] = ib.get();
126                 buf.position(buf.position() + BYTES_PER_INT * (numTermsThisBlock + 2));
127                 byte[] termStringsThisBlock = new byte[dataBlockSize];
128                 buf.get(termStringsThisBlock);
129 
130                 // Now instantiate String objects from the offsets and byte data
131                 int firstTermInBlock = currentTerm;
132                 for (; currentTerm < firstTermInBlock + numTermsThisBlock; currentTerm++) {
133                     int offset = termStringOffsets[currentTerm];
134                     int length = termStringOffsets[currentTerm + 1] - offset;
135                     String str = new String(termStringsThisBlock, offset, length, DEFAULT_CHARSET);
136 
137                     // We need to find term for id while searching
138                     terms[currentTerm] = str;
139                 }
140 
141                 // Re-map a new part of the file before we read the next block.
142                 // (and before we read the sort buffers)
143                 long bytesRead = buf.position();
144                 fileMapStart += bytesRead;
145                 fileMapLength = Math.min(maxMapSize, fileLength - fileMapStart);
146                 if (fileMapLength > 0) {
147                     buf = fc.map(MapMode.READ_ONLY, fileMapStart, fileMapLength);
148                     ib = buf.asIntBuffer();
149                 }
150             }
151 
152         } else {
153             // Old format, single term strings block.
154             // Causes problems when term strings total over 2 GB.
155 
156             ib.get(termStringOffsets); // term string offsets
157             int termStringsByteSize = ib.get(); // data block size
158 
159             // termStringByteSize fits in an int, and terms
160             // fits in a single byte array. Use the old code.
161             buf.position(buf.position() + BYTES_PER_INT + BYTES_PER_INT * termStringOffsets.length);
162             byte[] termStrings = new byte[termStringsByteSize];
163             buf.get(termStrings);
164             ib = buf.asIntBuffer();
165 
166             // Now instantiate String objects from the offsets and byte data
167             terms = new String[n];
168             for (int id = 0; id < n; id++) {
169                 int offset = termStringOffsets[id];
170                 int length = termStringOffsets[id + 1] - offset;
171                 String str = new String(termStrings, offset, length, DEFAULT_CHARSET);
172 
173                 // We need to find term for id while searching
174                 terms[id] = str;
175             }
176         }
177         return ib;
178     }

200     /**
201      * Get the sort position for a term based on its term id

204      * @param sensitivity whether we want the sensitive or insensitive sort position
205      * @return the sort position
206      */
207     public abstract int idToSortPosition(int id, MatchSensitivity sensitivity);

209     /**
210      * Convert an array of term ids to sort positions
211      * 
212      * @param termId the term ids
213      * @param sortOrder the sort positions
214      * @param sensitivity whether we want the sensitive or insensitive sort positions
215      */
216     public void toSortOrder(int[] termId, int[] sortOrder, MatchSensitivity sensitivity) {
217         for (int i = 0; i < termId.length; i++) {
218             sortOrder[i] = idToSortPosition(termId[i], sensitivity);
219         }
220     }

222     /**
223      * Compare two terms (from their term ids) based on their sort positions
224      * 
225      * @param termId1 id of the first term
226      * @param termId2 id of the second term
227      * @param sensitivity whether we want to compare sensitively or insensitively
228      * @return the comparison result (negative if term1 < term2, zero if equal,
229      *         positive if term1 > term2)
230      */
231     public int compareSortPosition(int termId1, int termId2, MatchSensitivity sensitivity) {
232         return idToSortPosition(termId1, sensitivity) - idToSortPosition(termId2, sensitivity);
233     }

247     public int deserializeToken(String term) {
248         int termId;
249         if (term.equals("~"))
250             termId = Terms.NO_TERM; // no token, effectively a "null" value
251         else {
252             if (term.startsWith("~~")) {
253                 // tilde in first position has to be escaped
254                 // because of how null value is encoded
255                 term = term.substring(1);
256             }
257             termId = indexOf(term);
258         }
259         return termId;
260     }
261 
262     public String serializeTerm(int valueTokenId) {
263         String token;
264         if (valueTokenId < 0)
265             token = "~"; // no token, effectively a "null" value
266         else {
267             token = get(valueTokenId);
268             if (token.length() > 0 && token.charAt(0) == '~') {
269                 // tilde in first position has to be escaped
270                 // because of how null value is encoded
271                 token = "~" + token;
272             }
273         }
274         return token;
275     }
