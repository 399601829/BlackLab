41  /**
42   * Keeps a forward index of documents, to quickly answer the question "what word
43   * occurs in doc X at position Y"?
44   * 
45   * This implementation is thread-safe.
46   */
47  class AnnotationForwardIndexReader extends AnnotationForwardIndex {
48  
49      protected static final Logger logger = LogManager.getLogger(AnnotationForwardIndexReader.class);
50  
51      /** Mapping into the tokens file */
52      private List<ByteBuffer> tokensFileChunks = null;
53  
54      /** Offsets of the mappings into the token file */
55      private List<Long> tokensFileChunkOffsetBytes = null;
56      
57      /** Collators to use for terms file */
58      private Collators collators;
59  
60      /** Offset of each document */
61      long[] offset;
62      
63      /** Length of each document */
64      int[] length;
65      
66      /** Deleted status of each document */
67      byte[] deleted;
68  
69      /** Deleted TOC entries. Always sorted by size. */
70      List<Integer> deletedTocEntries = null;
71  
72      /** Build term indexes right away or lazily? */
73      private boolean buildTermIndexesOnInit;
74  
75      AnnotationForwardIndexReader(Annotation annotation, File dir, Collators collators, boolean largeTermsFileSupport, boolean buildTermIndexesOnInit) {
76          super(annotation, dir, collators, largeTermsFileSupport);
77          
78          if (!dir.exists()) {
79              throw new IllegalArgumentException("ForwardIndex doesn't exist: " + dir);
80          }
81  
82          if (!tocFile.exists())
83              throw new IllegalArgumentException("No TOC found, and not in index mode!");
84          this.collators = collators; // for reading terms file in initialize()
85          this.buildTermIndexesOnInit = buildTermIndexesOnInit;
86      }
87  
88      /**
89       * Memory-map the tokens file for reading.
90       */
91      @Override
92      public synchronized void initialize() {
93          if (initialized)
94              return;
95          
96          //logger.debug("  START read TOC " + tocFile);
97          readToc();
98          //logger.debug("  END   read TOC " + tocFile);
99          
100         //logger.debug("  START read Terms " + tocFile);
101         terms = Terms.openForReading(collators, termsFile, useBlockBasedTermsFile, buildTermIndexesOnInit);
102         //logger.debug("  END   read Terms " + tocFile);
103         //logger.debug("  START Terms.initialize() " + tocFile);
104         terms.initialize();
105         //logger.debug("  END   Terms.initialize() " + tocFile);
106         
107         //logger.debug("  START map tokens file " + tocFile);
108         try (RandomAccessFile tokensFp = new RandomAccessFile(tokensFile, "r");
109                 FileChannel tokensFileChannel = tokensFp.getChannel()) {
110             // Map the tokens file in chunks of 2GB each. When retrieving documents, we always
111             // read it from just one chunk, not multiple, but because each chunk begins at a
112             // document start, documents of up to 2G tokens can be processed. We could get around
113             // this limitation by reading from multiple chunks, but this would make the code
114             // more complex.
115             tokensFileChunks = new ArrayList<>();
116             tokensFileChunkOffsetBytes = new ArrayList<>();
117             long mappedBytes = 0;
118             long tokenFileEndBytes = tokenFileEndPosition * SIZEOF_INT;
119             while (mappedBytes < tokenFileEndBytes) {
120                 // Find the last TOC entry start point that's also in the previous mapping
121                 // (or right the first byte after the previous mapping).
122 
123                 // Look for the largest entryOffset that's no larger than mappedBytes.
124                 int mapNextChunkFrom = -1;
125                 for (int i = 0; i < offset.length; i++) {
126                     if (offset[i] <= mappedBytes && (mapNextChunkFrom < 0 || offset[i] > offset[mapNextChunkFrom]))
127                         mapNextChunkFrom = i;
128                 }
129 
130                 // Uses binary search.
131                 int min = 0, max = offset.length;
132                 while (max - min > 1) {
133                     int middle = (min + max) / 2;
134                     long middleVal = offset[middle] * SIZEOF_INT;
135                     if (middleVal <= mappedBytes) {
136                         min = middle;
137                     } else {
138                         max = middle;
139                     }
140                 }
141                 long startOfNextMappingBytes = offset[min] * SIZEOF_INT;
142 
143                 // Map this chunk
144                 long sizeBytes = tokenFileEndBytes - startOfNextMappingBytes;
145                 if (sizeBytes > preferredChunkSizeBytes)
146                     sizeBytes = preferredChunkSizeBytes;
147 
148                 ByteBuffer mapping = tokensFileChannel.map(FileChannel.MapMode.READ_ONLY, startOfNextMappingBytes,
149                         sizeBytes);
150                 tokensFileChunks.add(mapping);
151                 tokensFileChunkOffsetBytes.add(startOfNextMappingBytes);
152                 mappedBytes = startOfNextMappingBytes + sizeBytes;
153             }
154         } catch (FileNotFoundException e1) {
155             throw BlackLabRuntimeException.wrap(e1);
156         } catch (IOException e1) {
157             throw BlackLabRuntimeException.wrap(e1);
158         }
159         //logger.debug("  END map tokens file " + tocFile);
160         
161         //logger.debug("END initialize AFI " + tocFile.getParent());
162         initialized = true;
163     }

167         // NOP

170     /**
171      * Read the table of contents from the file
172      */
173     protected void readToc() {
174         try (RandomAccessFile raf = new RandomAccessFile(tocFile, "r");
175                 FileChannel fc = raf.getChannel()) {
176             long fileSize = tocFile.length();
177             MappedByteBuffer buf = fc.map(MapMode.READ_ONLY, 0, fileSize);
178             int n = buf.getInt();
179             offset = new long[n];
180             length = new int[n];
181             deleted = new byte[n];
182             LongBuffer lb = buf.asLongBuffer();
183             lb.get(offset);
184             buf.position(buf.position() + SIZEOF_LONG * n);
185             IntBuffer ib = buf.asIntBuffer();
186             ib.get(length);
187             buf.position(buf.position() + SIZEOF_INT * n);
188             buf.get(deleted);
189             deletedTocEntries = new ArrayList<>();
190             for (int i = 0; i < n; i++) {
191                 if (deleted[i] != 0) {
192                     deletedTocEntries.add(i);
193                 }
194                 long end = offset[i] + length[i];
195                 if (end > tokenFileEndPosition)
196                     tokenFileEndPosition = end;
197             }
198             sortDeletedTocEntries();
199         } catch (IOException e) {
200             throw BlackLabRuntimeException.wrap(e);
201         }
202     }
203 
204     protected void sortDeletedTocEntries() {
205         deletedTocEntries.sort( (o1, o2) -> length[o1] - length[o2] );
206     }

214     @Override
215     public List<int[]> retrievePartsInt(int fiid, int[] start, int[] end) {
216         if (!initialized)
217             initialize();
218         
219         if (deleted[fiid] != 0)
220             return null;
221 
222         int n = start.length;
223         if (n != end.length)
224             throw new IllegalArgumentException("start and end must be of equal length");
225         List<int[]> result = new ArrayList<>(n);
226 
227         for (int i = 0; i < n; i++) {
228             if (start[i] == -1)
229                 start[i] = 0;
230             if (end[i] == -1)
231                 end[i] = length[fiid];
232             if (start[i] < 0 || end[i] < 0) {
233                 throw new IllegalArgumentException("Illegal values, start = " + start[i] + ", end = "
234                         + end[i]);
235             }
236             if (end[i] > length[fiid]) // Can happen while making KWICs because we don't know the
237                                    // doc length until here
238                 end[i] = length[fiid];
239             if (start[i] > length[fiid] || end[i] > length[fiid]) {
240                 throw new IllegalArgumentException("Value(s) out of range, start = " + start[i]
241                         + ", end = " + end[i] + ", content length = " + length[fiid]);
242             }
243             if (end[i] <= start[i]) {
244                 throw new IllegalArgumentException(
245                         "Tried to read empty or negative length snippet (from " + start[i]
246                                 + " to " + end[i] + ")");
247             }
248 
249             // Get an IntBuffer to read the desired content
250             IntBuffer ib = null;
251             
252             // The tokens file has has been mapped to memory.
253             // Get an int buffer into the file.
254 
255             // Figure out which chunk to access.
256             ByteBuffer whichChunk = null;
257             long chunkOffsetBytes = -1;
258             long entryOffsetBytes = offset[fiid] * SIZEOF_INT;
259             for (int j = 0; j < tokensFileChunkOffsetBytes.size(); j++) {
260                 long offsetBytes = tokensFileChunkOffsetBytes.get(j);
261                 ByteBuffer buffer = tokensFileChunks.get(j);
262                 if (offsetBytes <= entryOffsetBytes + start[i] * SIZEOF_INT
263                         && offsetBytes + buffer.capacity() >= entryOffsetBytes + end[i]
264                                 * SIZEOF_INT) {
265                     // This one!
266                     whichChunk = buffer;
267                     chunkOffsetBytes = offsetBytes;
268                     break;
269                 }
270             }
271 
272             if (whichChunk == null) {
273                 throw new BlackLabRuntimeException("Tokens file chunk containing document not found. fiid = " + fiid);
274             }
275             whichChunk.position((int) (offset[fiid] * SIZEOF_INT - chunkOffsetBytes));
276             ib = whichChunk.asIntBuffer();
277 
278             int snippetLength = end[i] - start[i];
279             int[] snippet = new int[snippetLength];
280             
281             // The file is mem-mapped (search mode).
282             // Position us at the correct place in the file.
283             ib.position(start[i]);
284             ib.get(snippet);
285             result.add(snippet);
286         }
287 
288         return result;
289     }

297      * @return the number of documents in the forward index
298      */
299     @Override
300     public int numDocs() {
301         if (!initialized)
302             initialize();
303         return offset.length;
304     }

306     /**
307      * @return the amount of space in free blocks in the forward index.
308      */
309     @Override
310     public long freeSpace() {
311         if (!initialized)
312             initialize();
313         long freeSpace = 0;
314         for (Integer e : deletedTocEntries) {
315             freeSpace += length[e];
316         }
317         return freeSpace;
318     }
319 
320     /**
321      * @return the number of free blocks in the forward index.
322      */
323     @Override
324     public int freeBlocks() {
325         if (!initialized)
326             initialize();
327         return deletedTocEntries.size();
328     }

344     /** @return the set of all forward index ids */
345     @Override
346     public Set<Integer> idSet() {
347         if (!initialized)
348             initialize();
349         return new AbstractSet<Integer>() {
350             @Override
351             public boolean contains(Object o) {
352                 return deleted[(Integer)o] == 0;
353             }
354 
355             @Override
356             public boolean isEmpty() {
357                 return offset.length == deletedTocEntries.size();
358             }
359 
360             @Override
361             public Iterator<Integer> iterator() {
362                 return new Iterator<Integer>() {
363                     int current = -1;
364                     int next = -1;
365 
366                     @Override
367                     public boolean hasNext() {
368                         if (next < 0)
369                             findNext();
370                         return next < offset.length;
371                     }
372 
373                     private void findNext() {
374                         next = current + 1;
375                         while (next < offset.length && deleted[next] != 0) {
376                             next++;
377                         }
378                     }
379 
380                     @Override
381                     public Integer next() {
382                         if (next < 0)
383                             findNext();
384                         if (next >= offset.length)
385                             throw new NoSuchElementException();
386                         current = next;
387                         next = -1;
388                         return current;
389                     }
390 
391                     @Override
392                     public void remove() {
393                         throw new UnsupportedOperationException();
394                     }
395                 };
396             }
397 
398             @Override
399             public int size() {
400                 return offset.length - deletedTocEntries.size();
401             }
402         };
403     }
