39  /**
40   * Keeps a first-come-first-serve list of unique terms. Each term gets a unique
41   * index number. These numbers are stored in the forward index to conserve space
42   * and allow quick lookups of terms occurring in specific positions.
43   *
44   * This version of the class stores the terms in a more efficient way so it
45   * saves and loads faster, and includes the case-insensitive sorting order.
46   * 
47   * This implementation is not thread-safe.
48   */
49  @NotThreadSafe
50  class TermsWriter extends Terms {
51  
52      protected static final Logger logger = LogManager.getLogger(TermsWriter.class);
53  
54      /** Maximum size for blocks of term strings. */
55      private static final int DEFAULT_MAX_BLOCK_SIZE = DEFAULT_MAX_MAP_SIZE;
56  
57      /**
58       * Number of sort buffers we store in the terms file (case-sensitive/insensitive
59       * and inverted buffers for both as well)
60       */
61      private static final int NUM_SORT_BUFFERS = 4;
62  
63      /**
64       * Mapping from term to its unique index number. We use a SortedMap because we
65       * wish to store the sorted index numbers later (to speed up sorting).
66       */
67      Map<CollationKey, Integer> termIndex;
68  
69      /**
70       * The maximum block size to use while writing the terms file. Ususally around
71       * the limit of 2GB, but for testing, we can set this to a lower value.
72       */
73      private int maxBlockSize = DEFAULT_MAX_BLOCK_SIZE;
74  
75      TermsWriter(Collators collators, File termsFile, boolean useBlockBasedTermsFile) {
76          this.collator = collators.get(MatchSensitivity.SENSITIVE);
77          this.collatorInsensitive = collators.get(MatchSensitivity.INSENSITIVE);
78  
79          // Index mode: create a SortedMap based on the specified Collator.
80          // (used later to get the terms in sort order)
81          this.termIndex = new TreeMap<>();
82  
83          setBlockBasedFile(useBlockBasedTermsFile);
84          if (termsFile != null && termsFile.exists())
85              read(termsFile);
86          
87          // We need to find id for term quickly while indexing
88          // Build the case-sensitive term index.
89          for (int i = 0; i < numberOfTerms; i++) {
90              termIndex.put(collator.getCollationKey(terms[i]), i);
91          }
92          terms = null; // useless in index mode because we can't add to it, and we don't need it anyway
93      }
94  
95      @Override
96      public int indexOf(String term) {
97          synchronized (this) {
98              // Yes, use the available term index.
99              CollationKey key = this.collator.getCollationKey(term);
100             Integer index = termIndex.get(key);
101             if (index != null)
102                 return index;
103             index = termIndex.size();
104             termIndex.put(key, index);
105             return index;
106         }
107     }

124     private synchronized void read(File termsFile) {
125         termIndex.clear();
126         try {
127             try (RandomAccessFile raf = new RandomAccessFile(termsFile, "r")) {
128                 try (FileChannel fc = raf.getChannel()) {
129                     long fileLength = termsFile.length();
130                     readFromFileChannel(fc, fileLength);
131                 }
132             }
133         } catch (IOException e) {
134             throw BlackLabRuntimeException.wrap(e);
135         }
136     }
137 
138     @Override
139     public synchronized void write(File termsFile) {
140         try {
141             // Open the terms file
142             try (RandomAccessFile raf = new RandomAccessFile(termsFile, "rw")) {
143                 try (FileChannel fc = raf.getChannel()) {
144                     int n = termIndex.size();
145 
146                     // Fill the terms[] array
147                     terms = new String[n];
148                     long termStringsByteSize = 0;
149                     for (Map.Entry<CollationKey, Integer> entry : termIndex.entrySet()) {
150                         String term = entry.getKey().getSourceString();
151                         terms[entry.getValue()] = term;
152                         termStringsByteSize += term.getBytes(DEFAULT_CHARSET).length;
153                     }
154 
155                     // Calculate the file length and map the file
156                     MappedByteBuffer buf;
157                     IntBuffer ib;
158                     if (!useBlockBasedTermsFile) {
159                         long fileLength = 2 * BYTES_PER_INT + (n + 1) * BYTES_PER_INT + termStringsByteSize
160                                 + NUM_SORT_BUFFERS * BYTES_PER_INT * n;
161                         fc.truncate(fileLength); // truncate if necessary
162                         buf = fc.map(MapMode.READ_WRITE, 0, fileLength);
163                         buf.putInt(n); // Start with the number of terms
164                         ib = buf.asIntBuffer();
165 
166                         // Terms file is small enough to fit in a single byte array.
167                         // Use the old code.
168 
169                         // Calculate byte offsets for all the terms and fill data array
170                         int currentOffset = 0;
171                         int[] termStringOffsets = new int[n + 1];
172                         byte[] termStrings = new byte[(int) termStringsByteSize];
173                         for (int i = 0; i < n; i++) {
174                             termStringOffsets[i] = currentOffset;
175                             byte[] termBytes = terms[i].getBytes(DEFAULT_CHARSET);
176                             System.arraycopy(termBytes, 0, termStrings, currentOffset, termBytes.length);
177                             currentOffset += termBytes.length;
178                         }
179                         termStringOffsets[n] = currentOffset;
180 
181                         // Write offset and data arrays to file
182                         ib.put(termStringOffsets);
183                         ib.put((int) termStringsByteSize); // size of the data block to follow
184                         buf.position(buf.position() + BYTES_PER_INT + BYTES_PER_INT * termStringOffsets.length); // advance past offsets array
185                         buf.put(termStrings);
186                         ib = buf.asIntBuffer();
187                     } else {
188                         long fileMapStart = 0, fileMapLength = maxMapSize;
189                         buf = fc.map(MapMode.READ_WRITE, fileMapStart, fileMapLength);
190                         buf.putInt(n); // Start with the number of terms      //@4
191                         ib = buf.asIntBuffer();
192                         long fileLength = BYTES_PER_INT;
193 
194                         // Terms file is too large to fit in a single byte array.
195                         // Use the new code.
196                         int currentTerm = 0;
197                         long bytesLeftToWrite = termStringsByteSize;
198                         int[] termStringOffsets = new int[n];
199                         while (currentTerm < n) {
200                             int firstTermInBlock = currentTerm;
201                             int blockSize = (int) Math.min(bytesLeftToWrite, maxBlockSize);
202 
203                             // Calculate byte offsets for all the terms and fill data array
204                             int currentOffset = 0;
205                             byte[] termStrings = new byte[blockSize];
206                             while (currentTerm < n) {
207                                 termStringOffsets[currentTerm] = currentOffset;
208                                 byte[] termBytes = terms[currentTerm].getBytes(DEFAULT_CHARSET);
209                                 if (currentOffset + termBytes.length > blockSize) {
210                                     // Block is full. Write it and continue with next block.
211                                     break;
212                                 }
213                                 System.arraycopy(termBytes, 0, termStrings, currentOffset, termBytes.length);
214                                 currentOffset += termBytes.length;
215                                 currentTerm++;
216                                 bytesLeftToWrite -= termBytes.length;
217                             }
218 
219                             // Write offset and data arrays to file
220                             int numTermsThisBlock = currentTerm - firstTermInBlock;
221 
222                             long blockSizeBytes = 2 * BYTES_PER_INT + numTermsThisBlock * BYTES_PER_INT + currentOffset;
223 
224                             ib.put(numTermsThisBlock); //@4
225                             ib.put(termStringOffsets, firstTermInBlock, numTermsThisBlock); //@4 * numTermsThisBlock
226                             ib.put(currentOffset); // include the offset after the last term at position termStringOffsets[n]
227                                                    // (doubles as the size of the data block to follow) //@4
228                             int newPosition = buf.position() + BYTES_PER_INT * (2 + numTermsThisBlock);
229                             buf.position(newPosition); // advance past offsets array
230                             buf.put(termStrings, 0, currentOffset); //@blockSize (max. maxBlockSize)
231                             ib = buf.asIntBuffer();
232                             fileLength += blockSizeBytes;
233 
234                             // Re-map a new part of the file before we write the next block.
235                             // (and eventually, the sort buffers, see below)
236                             fileMapStart += buf.position();
237                             buf = fc.map(MapMode.READ_WRITE, fileMapStart, fileMapLength);
238                             ib = buf.asIntBuffer();
239                         }
240 
241                         // Determine total file length (by adding the sort buffer byte length to the
242                         // running total) and truncate the file if necessary
243                         // (we can do this now, even though we still have to write the sort buffers,
244                         // because we know how large the file will eventually be)
245                         fileLength += NUM_SORT_BUFFERS * BYTES_PER_INT * n;
246                         if (File.separatorChar != '\\') // causes problems on Windows
247                             fc.truncate(fileLength);
248                     }
249 
250                     // Write the case-sensitive sort order
251                     // Because termIndex is a SortedMap, values are returned in key-sorted order.
252                     // In other words, the index numbers are in order of sorted terms, so the id
253                     // for 'aardvark' comes before the id for 'ape', etc.
254                     int i = 0;
255                     int[] sortPositionPerId = new int[n];
256                     Integer[] insensitive = new Integer[n];
257                     for (int id : termIndex.values()) {
258                         sortPositionPerId[id] = i;
259                         insensitive[i] = id; // fill this so we can re-sort later, faster b/c already partially sorted
260                         i++;
261                     }
262                     ib.put(new int[n]); // NOT USED ANYMORE, JUST FOR FILE COMPATIBILITY
263                     ib.put(sortPositionPerId);
264 
265                     // Now, sort case-insensitively and write those arrays as well
266                     Arrays.sort(insensitive, new Comparator<Integer>() {
267                         @Override
268                         public int compare(Integer a, Integer b) {
269                             return collatorInsensitive.compare(terms[a], terms[b]);
270                         }
271                     });
272                     // Copy into the sortPositionPerIdInsensitive array, making sure that
273                     // identical values get identical sort positions!
274                     int[] sortPositionPerIdInsensitive = new int[n];
275                     int sortPos = 0;
276                     for (i = 0; i < n; i++) {
277                         if (i == 0
278                                 || collatorInsensitive.compare(terms[insensitive[i - 1]], terms[insensitive[i]]) != 0) {
279                             // Not identical to previous value: gets its own sort position.
280                             // If a value is identical to the previous one, it gets the same sort position.
281                             sortPos = i;
282                         }
283                         sortPositionPerIdInsensitive[insensitive[i]] = sortPos;
284                     }
285                     ib.put(new int[n]); // NOT USED ANYMORE, JUST FOR FILE COMPATIBILITY
286                     ib.put(sortPositionPerIdInsensitive);
287                 }
288             }
289         } catch (IOException e) {
290             throw BlackLabRuntimeException.wrap(e);
291         }
292     }

294     @Override
295     public String get(int index) {
296         assert index >= 0 && index < numberOfTerms : "Term index out of range (" + index + ", numterms = "
297                 + numberOfTerms + ")";
298         return terms[index];
299     }
