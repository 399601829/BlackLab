1   package nl.inl.blacklab.searches;
2   
3   import org.apache.lucene.search.Query;
4   
5   import nl.inl.blacklab.exceptions.InvalidQuery;
6   import nl.inl.blacklab.queryParser.corpusql.CorpusQueryLanguageParser;
7   import nl.inl.blacklab.search.results.SearchSettings;
8   import nl.inl.blacklab.search.results.QueryInfo;
9   import nl.inl.blacklab.search.results.SearchResult;
10  import nl.inl.blacklab.search.textpattern.TextPattern;
11  
12  /** 
13   * Empty search that just knows about its index and annotated field to search,
14   * and serves as a starting point for actual searches.
15   */
16  public class SearchEmpty extends AbstractSearch<SearchResult> {
17      
18      public SearchEmpty(QueryInfo queryInfo) {
19          super(queryInfo);
20      }
21  
22      @Override
23      protected SearchResult executeInternal() throws InvalidQuery {
24          throw new UnsupportedOperationException();
25      }
26  
27      public SearchHits find(String cql, Query filter, SearchSettings searchSettings) throws InvalidQuery {
28          return find(CorpusQueryLanguageParser.parse(cql), filter, searchSettings);
29      }
30  
31      public SearchHits find(TextPattern pattern, Query filter, SearchSettings searchSettings) {
32          return new SearchHitsFromPattern(queryInfo(), pattern, filter, searchSettings);
33      }
34      
35      public SearchDocs find(Query documentQuery) {
36          return new SearchDocsFromQuery(queryInfo(), documentQuery);
37      }
38      
39      @Override
40      public String toString() {
41          return toString("empty");
42      }
43      
44  }
