1   package nl.inl.blacklab.searches;
2   
3   import java.util.List;
4   
5   import nl.inl.blacklab.resultproperty.DocProperty;
6   import nl.inl.blacklab.resultproperty.PropertyValue;
7   import nl.inl.blacklab.search.results.DocResults;
8   import nl.inl.blacklab.search.results.QueryInfo;
9   import nl.inl.blacklab.search.results.SampleParameters;
10  
11  /** A search that produces DocResults. */
12  public abstract class SearchDocs extends SearchResults<DocResults> {
13  
14      public SearchDocs(QueryInfo queryInfo) {
15          super(queryInfo);
16      }
17      
18      /**
19       * Group hits by a property.
20       * 
21       * @param groupBy what to group by
22       * @param maxResultsToGatherPerGroup how many results to gather per group
23       * @return resulting operation
24       */
25      public SearchDocGroups group(DocProperty groupBy, int maxResultsToGatherPerGroup) {
26          return new SearchDocGroupsFromDocs(queryInfo(), this, groupBy, maxResultsToGatherPerGroup);
27      }
28  
29      /**
30       * Sort hits.
31       * 
32       * @param sortBy what to sort by
33       * @return resulting operation
34       */
35      public SearchDocs sort(DocProperty sortBy) {
36          if (sortBy == null)
37              return this;
38          return new SearchDocsSorted(queryInfo(), this, sortBy);
39      }
40      
41      /**
42       * Sample hits.
43       * 
44       * @param par how many hits to sample; seed
45       * @return resulting operation
46       */
47      public SearchDocs sample(SampleParameters par) {
48          return new SearchDocsSampled(queryInfo(), this, par);
49      }
50  
51      /**
52       * Get hits with a certain property value.
53       * 
54       * @param property property to test 
55       * @param value value to test for
56       * @return resulting operation
57       */
58      public SearchDocs filter(DocProperty property, PropertyValue value) {
59          return new SearchDocsFiltered(queryInfo(), this, property, value);
60      }
61  
62      /**
63       * Get window of hits.
64       * 
65       * @param first first hit to select
66       * @param number number of hits to select
67       * @return resulting operation
68       */
69      public SearchDocs window(int first, int number) {
70          return new SearchDocsWindow(queryInfo(), this, first, number);
71      }
72  
73      /**
74       * Facet the doc results by several properties.
75       * 
76       * Faceting is basically grouping without storing any results, just the counts. 
77       * 
78       * @param facets facets to facet on
79       * @return resulting operation
80       */
81      public SearchFacets facet(List<DocProperty> facets) {
82          return new SearchFacets(queryInfo(), this, facets);
83      }
84      
85  }
