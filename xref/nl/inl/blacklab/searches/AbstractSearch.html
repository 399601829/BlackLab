3   import java.util.concurrent.CompletableFuture;
4   import java.util.concurrent.CompletionException;
5   import java.util.concurrent.ExecutionException;
6   import java.util.concurrent.Future;
7   import java.util.function.Supplier;
8   
9   import org.apache.commons.lang3.StringUtils;
10  
11  import nl.inl.blacklab.exceptions.BlackLabRuntimeException;
12  import nl.inl.blacklab.exceptions.InterruptedSearch;
13  import nl.inl.blacklab.exceptions.InvalidQuery;
14  import nl.inl.blacklab.search.results.QueryInfo;
15  import nl.inl.blacklab.search.results.SearchResult;
16  
17  /**
18   * Abstract base class for all Search implementations,
19   * to enforce that equals() and hashCode are implemented
20   * (to ensure proper caching)
21   * 
22   * @param <R> results type, e.g. Hits
23   */
24  public abstract class AbstractSearch<R extends SearchResult> implements Search<R> {
25  	
26      private QueryInfo queryInfo;
27      
28      public AbstractSearch(QueryInfo queryInfo) {
29          this.queryInfo = queryInfo;  
30      }
31      
32      @Override
33      public Future<R> executeAsync() {
34          return getFromCache(this, () -> {
35              try {
36                  return executeInternal();
37              } catch (InvalidQuery e) {
38                  throw new CompletionException(e);
39              }
40          });
41      }
42      
43      @Override
44      public final R execute() throws InvalidQuery {
45          Future<R> future = executeAsync();
46          try {
47              return future.get();
48          } catch (ExecutionException e) {
49              if (e.getCause() instanceof CompletionException) {
50                  try {
51                      throw e.getCause().getCause();
52                  } catch (InvalidQuery e2) {
53                      throw e2;
54                  } catch (Throwable e2) {
55                      throw new AssertionError(e2);
56                  }
57              }
58              throw BlackLabRuntimeException.wrap(e.getCause());
59          } catch (InterruptedException e) {
60              throw new InterruptedSearch(e);
61          } catch (CompletionException e) {
62              try {
63                  throw e.getCause();
64              } catch (InvalidQuery e2) {
65                  throw e2;
66              } catch (Throwable e2) {
67                  throw new AssertionError(e2);
68              }
69          }
70      }
71      
72      protected abstract R executeInternal() throws InvalidQuery;
73      
74      protected Future<R> getFromCache(Search<R> search, Supplier<R> searchTask) {
75          return queryInfo.index().cache().getAsync(search, searchTask);
76      }
77      
78      protected SearchResult getFromCacheBlock(Search<R> search, Supplier<R> searchTask) throws ExecutionException {
79          return queryInfo.index().cache().get(search, searchTask);
80      }
81      
82      protected void cancelSearch(CompletableFuture<? extends SearchResult> future) {
83          queryInfo.index().cache().remove(this);
84          future.cancel(false);
85      }
86      
87      @Override
88      public QueryInfo queryInfo() {
89          return queryInfo;
90      }
91  
92      @Override
93      public int hashCode() {
94          final int prime = 31;
95          int result = 1;
96          result = prime * result + ((queryInfo == null) ? 0 : queryInfo.hashCode());
97          return result;
98      }
99  
100     @Override
101     public boolean equals(Object obj) {
102         if (this == obj)
103             return true;
104         if (obj == null)
105             return false;
106         if (getClass() != obj.getClass())
107             return false;
108         AbstractSearch<?> other = (AbstractSearch<?>) obj;
109         if (queryInfo == null) {
110             if (other.queryInfo != null)
111                 return false;
112         } else if (!queryInfo.equals(other.queryInfo))
113             return false;
114         return true;
115     }
116 
117     protected static String toString(String operation, Object... param) {
118         return operation + "(" + StringUtils.join(param, ", ") + ")";
119     }
120 	
121 }
