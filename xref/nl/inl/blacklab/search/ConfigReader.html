32  /**
33   * Reads blacklab.yaml/.json file from one of the config dirs.
34   *
35   * This file contains general BlackLab settings that apply to multiple
36   * applications, i.e. IndexTool, QueryTool, BlackLab Server, and
37   * other applications that use BlackLab.
38   *
39   * Config dirs are, in search order: $BLACKLAB_CONFIG_DIR/, $HOME/.blacklab/
40   * or /etc/blacklab/.
41   */
42  public class ConfigReader extends YamlJsonReader {
43  
44  	private static final Logger logger = LogManager.getLogger(ConfigReader.class);

47      private static List<File> configDirs;

49      /** Cache of root node of config file */
50      private static JsonNode blacklabConfig;

53      private static boolean ignoreConfigFile = false;

55      /**
56       * Do we wish to forego looking for a config file on the filesystem? Useful for repeatable testing.
57       * @param ignoreConfigFile if true, no config file will be loaded, so all setting will be at their default
58       */
59      public static void setIgnoreConfigFile(boolean ignoreConfigFile) {
60          ConfigReader.ignoreConfigFile = ignoreConfigFile;
61      }

63      /**
64       * Load the global blacklab configuration.
65       * This file configures several global settings, as well as providing default settings for any new {@link Searcher} constructed hereafter.
66       *
67       * If no explicit config file has been set by the time when the first Searcher is opened, BlackLab automatically attempts to find and load a
68       * configuration file in a number of preset locations (see {@link ConfigReader#getDefaultConfigDirs()}).
69       *
70       * Attempting to set another configuration when one is already loaded will throw an UnsupportedOperationException.
71       *
72       * @param file
73       * @throws FileNotFoundException
74       * @throws IOException
75       */
76      public static void setConfigFile(File file) throws FileNotFoundException, IOException {
77          if (ignoreConfigFile) // useful for repeatable testing
78              return;
79          
80      	if (file == null || !file.canRead())
81      		throw new FileNotFoundException("Configuration file " + file + " is unreadable.");
82  
83      	if (!FilenameUtils.isExtension(file.getName(), Arrays.asList("yaml", "yml", "json")))
84      		throw new IllegalArgumentException("Configuration file " + file + " is of an unsupported type.");
85  
86      	if (blacklabConfig != null)
87      		throw new UnsupportedOperationException("Cannot load configuration file " + file + " - another configuration file has already been loaded.");
88  
89  		boolean isJson = file.getName().endsWith(".json");
90  		try (BufferedReader reader = FileUtil.openForReading(file)) {
91  			setConfigFile(reader, isJson);
92  		}
93      }
94  
95      /**
96       * See {@link ConfigReader#setConfigFile(File)}.
97       *
98       * The reader must be closed by the user.
99       *
100      * @param reader
101      * @param isJson
102      * @throws JsonProcessingException
103      * @throws IOException
104      */
105     public static void setConfigFile(Reader reader, boolean isJson) throws JsonProcessingException, IOException{
106         if (ignoreConfigFile) // useful for repeatable testing
107             return;
108         
109     	if (blacklabConfig != null)
110     		throw new UnsupportedOperationException("Cannot load configuration file - another configuration file has already been loaded.");
111 
112     	ObjectMapper mapper = isJson ? Json.getJsonObjectMapper() : Json.getYamlObjectMapper();
113 
114     	logger.debug("Reading global BlackLab config");
115     	JsonNode parsedConfig = mapper.readTree(reader);
116 
117 	    readGlobalSettings(parsedConfig);
118 
119 	    blacklabConfig = parsedConfig;
120     }

123     public static void loadDefaultConfig() {
124     	if (blacklabConfig != null)
125     		return;
126 
127 		try {
128 			File file = FileUtil.findFile(getDefaultConfigDirs(), "blacklab", Arrays.asList("yaml", "yml", "json"));
129 			if (file != null)
130 				setConfigFile(file);
131 		} catch (IOException e) {
132 			logger.warn("Could not load default blacklab configuration file: " + e.getMessage());
133 		}
134     }

147     /**
148      * Configure the searcher according to the blacklab configuration file.
149      *
150      * @param searcher
151      * @throws IOException
152      */
153     public static void applyConfig(Searcher searcher) throws IOException {
154     	if (blacklabConfig == null)
155     		loadDefaultConfig();
156 
157     	if (blacklabConfig != null) {
158 	        readSearcherSettings(blacklabConfig, searcher);
159     	}
160     }

162     private static void readSearcherSettings(JsonNode root, Searcher searcher) {
163         obj(root, "root node");
164         Iterator<Entry<String, JsonNode>> it = root.fields();
165 
166         while (it.hasNext()) {
167             Entry<String, JsonNode> e = it.next();
168             switch (e.getKey()) {
169             case "search": readSearch(obj(e), searcher); break;
170             case "plugins":
171             case "indexing":
172             case "debug":
173             	break;
174             default:
175                 throw new IllegalArgumentException("Unknown top-level key " + e.getKey());
176             }
177         }
178     }
179 
180     private static void readSearch(ObjectNode obj, Searcher searcher) {
181         HitsSettings hitsSett = searcher.hitsSettings();
182         Iterator<Entry<String, JsonNode>> it = obj.fields();
183         while (it.hasNext()) {
184             Entry<String, JsonNode> e = it.next();
185             switch (e.getKey()) {
186             case "collator": readCollator(e, searcher); break;
187             case "contextSize": hitsSett.setContextSize(integer(e)); break;
188             case "maxHitsToRetrieve": hitsSett.setMaxHitsToRetrieve(integer(e)); break;
189             case "maxHitsToCount": hitsSett.setMaxHitsToCount(integer(e)); break;
190             default:
191                 throw new IllegalArgumentException("Unknown key " + e.getKey() + " in search section");
192             }
193         }
194     }
195 
196     private static void readCollator(Entry<String, JsonNode> e, Searcher searcher) {
197         Collator collator;
198         if (e.getValue() instanceof ObjectNode) {
199             Iterator<Entry<String, JsonNode>> it = obj(e).fields();
200             String language = null, country = null, variant = null;
201             while (it.hasNext()) {
202                 Entry<String, JsonNode> e2 = it.next();
203                 switch (e2.getKey()) {
204                 case "language": language = str(e2); break;
205                 case "country": country = str(e2); break;
206                 case "variant": variant = str(e2); break;
207                 default:
208                     throw new IllegalArgumentException("Unknown key " + e.getKey() + " in collator (must have language, can have country and variant)");
209                 }
210             }
211             if (language == null || country == null && variant != null)
212                 throw new IllegalArgumentException("Collator must have language, language+country or language+country+variant");
213             if (country == null)
214                 collator = Collator.getInstance(new Locale(language));
215             else if (variant == null)
216                 collator = Collator.getInstance(new Locale(language, country));
217             else
218                 collator = Collator.getInstance(new Locale(language, country, variant));
219         } else {
220             collator = Collator.getInstance(new Locale(str(e)));
221         }
222         searcher.setCollator(collator);
223     }

225     private static void readGlobalSettings(JsonNode root) {
226     	obj(root, "root node");
227         Iterator<Entry<String, JsonNode>> it = root.fields();
228         while (it.hasNext()) {
229             Entry<String, JsonNode> e = it.next();
230             switch (e.getKey()) {
231             case "indexing": readIndexing(obj(e)); break;
232             case "plugins": PluginManager.initPlugins(obj(e)); break;
233             case "debug": readDebug(obj(e)); break;
234             case "search":
235             	break;
236             default:
237                 throw new IllegalArgumentException("Unknown top-level key " + e.getKey());
238             }
239         }
240     }
241 
242     private static void readDebug(ObjectNode obj) {
243         Iterator<Entry<String, JsonNode>> it = obj.fields();
244         while (it.hasNext()) {
245             Entry<String, JsonNode> e = it.next();
246             switch (e.getKey()) {
247             case "trace": readTrace(obj(e)); break;
248             default:
249                 throw new IllegalArgumentException("Unknown key " + e.getKey() + " in debug section");
250             }
251         }
252     }
253 
254     private static void readTrace(ObjectNode obj) {
255         Iterator<Entry<String, JsonNode>> it = obj.fields();
256         while (it.hasNext()) {
257             Entry<String, JsonNode> e = it.next();
258             switch (e.getKey()) {
259             case "indexOpening": Searcher.setTraceIndexOpening(bool(e)); break;
260             case "optimization": Searcher.setTraceOptimization(bool(e)); break;
261             case "queryExecution": Searcher.setTraceQueryExecution(bool(e)); break;
262             default:
263                 throw new IllegalArgumentException("Unknown key " + e.getKey() + " in trace section");
264             }
265         }
266     }

268     public static void readIndexing(ObjectNode indexing) {
269         Iterator<Entry<String, JsonNode>> it = indexing.fields();
270         while (it.hasNext()) {
271             Entry<String, JsonNode> e = it.next();
272             switch (e.getKey()) {
273             case "downloadAllowed": DownloadCache.setDownloadAllowed(bool(e)); break;
274             case "downloadCacheMaxFileSizeMegs": DownloadCache.setMaxFileSizeMegs(integer(e)); break;
275             case "downloadCacheDir": DownloadCache.setDir(new File(str(e))); break;
276             case "downloadCacheSizeMegs": DownloadCache.setSizeMegs(integer(e)); break;
277             case "zipFilesMaxOpen": ZipHandleManager.setMaxOpen(integer(e)); break;
278             default:
279                 throw new IllegalArgumentException("Unknown key " + e.getKey() + " in indexing section");
280             }
281         }

284 	/**
285 	 * Return a list of directories that should be searched for BlackLab-related configuration files.
286 	 *
287 	 * May be used by applications to locate BlackLab-related configuration, such as
288 	 * input format definition files or other configuration files. IndexTool and BlackLab Server use
289 	 * this.
290 	 *
291 	 * The directories returned are (in decreasing priority):
292 	 * - $BLACKLAB_CONFIG_DIR (if env. var. is defined)
293 	 * - $HOME/.blacklab
294 	 * - /etc/blacklab
295 	 * - /vol1/etc/blacklab (legacy, will be removed)
296 	 * - /tmp (legacy, will be removed)
297 	 *
298 	 * A convenient method to use with this is {@link FileUtil#findFile(List, String, List)}.
299 	 *
300 	 * @return list of directories to search in decreasing order of priority
301 	 */
302     public static List<File> getDefaultConfigDirs() {
303         if (configDirs == null) {
304             configDirs = new ArrayList<>();
305             String strConfigDir = System.getenv("BLACKLAB_CONFIG_DIR");
306             if (strConfigDir != null && strConfigDir.length() > 0) {
307                 File configDir = new File(strConfigDir);
308                 if (configDir.exists()) {
309                     if (!configDir.canRead())
310                         logger.warn("BLACKLAB_CONFIG_DIR points to a unreadable directory: " + strConfigDir);
311                     configDirs.add(configDir);
312                 } else {
313                     logger.warn("BLACKLAB_CONFIG_DIR points to a non-existent directory: " + strConfigDir);
314                 }
315             }
316             configDirs.add(new File(System.getProperty("user.home"), ".blacklab"));
317             configDirs.add(new File("/etc/blacklab"));
318             configDirs.add(new File("/vol1/etc/blacklab")); // TODO: remove, INT-specific
319             configDirs.add(new File(System.getProperty("java.io.tmpdir")));
320         }
321         return new ArrayList<>(configDirs);
322     }
