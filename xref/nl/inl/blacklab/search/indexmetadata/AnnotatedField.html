3   /** An annotated field (formerly "complex field") */
4   public interface AnnotatedField extends Field {
5   
6   	/**
7   	 * Get the annotations for this field.
8   	 *
9   	 * Properties are returned sorted according to the displayOrder defined in the
10  	 * index metadata, if any.
11  	 *
12  	 * @return the annotations
13  	 */
14  	Annotations annotations();
15  	
16  	default Annotation mainAnnotation() {
17  	    return annotations().main();
18  	}
19  	
20  	default Annotation annotation(String name) {
21  	    return annotations().get(name);
22  	}
23  
24  	boolean hasLengthTokens();
25  
26  	boolean hasXmlTags();
27  
28      boolean hasPunctuationForwardIndex();
29  
30      String tokenLengthField();
31  
32      @Override
33      default String contentsFieldName() {
34          Annotation main = mainAnnotation();
35          AnnotationSensitivity offsetsSensitivity = main.offsetsSensitivity();
36          if (offsetsSensitivity == null)
37              offsetsSensitivity = main.sensitivity(MatchSensitivity.SENSITIVE);
38          return offsetsSensitivity.luceneField();
39      }
40  
41      /**
42       * Was the field that stores the length of this field in tokens
43       * indexed with DocValues?
44       * 
45       * @return true if it was, false if not
46       */
47      boolean hasTokenLengthDocValues();
48  
49      default boolean isDummyFieldToStoreLinkedDocuments() {
50          return annotations().isEmpty();
51      }
52  
53  }
