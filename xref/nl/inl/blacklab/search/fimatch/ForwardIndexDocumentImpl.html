1   package nl.inl.blacklab.search.fimatch;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import org.apache.lucene.index.LeafReader;
7   
8   import nl.inl.blacklab.search.fimatch.ForwardIndexAccessor.ForwardIndexAccessorLeafReader;
9   
10  /** Source of tokens for the forward index matching process. */
11  class ForwardIndexDocumentImpl extends ForwardIndexDocument {
12  
13  	/** Default size for our chunks */
14  	private static final int CHUNK_SIZE = 10;
15  
16  	/** Where to get our forward indices and forward index ids (fiids) */
17  	private ForwardIndexAccessorLeafReader fiAccessor;
18  
19  	/** Lucene document id of the document we're looking at*/
20  	private int docId;
21  
22  	/** Number of tokens in document. */
23  	private int docLengthTokens;
24  
25  	/** Chunks of the document from the forward index, for each of the properties. */
26  	private List<List<int[]>> allPropChunks = new ArrayList<>();
27  
28  	public ForwardIndexDocumentImpl(ForwardIndexAccessorLeafReader fiAccessor, int docId, LeafReader reader) {
29  		this.fiAccessor = fiAccessor;
30  		this.docId = docId;
31  		this.docLengthTokens = fiAccessor.getDocLength(docId);
32  
33  		// Create empty lists of chunks for each property
34  		for (int i = 0; i < fiAccessor.getNumberOfProperties(); i++) {
35  			allPropChunks.add(new ArrayList<int[]>());
36  		}
37  	}
38  
39  	@Override
40  	public int getToken(int propIndex, int pos) {
41  		if (pos < 0 || pos >= docLengthTokens)
42  			return -1;
43  
44  		// Get the list of chunks for the property we're interested in,
45  		// and the forward index object to get more.
46  		List<int[]> chunks = allPropChunks.get(propIndex);
47  
48  		// Where can our token be found?
49  		int whichChunk = pos / CHUNK_SIZE;
50  		int posWithinChunk = pos % CHUNK_SIZE;
51  
52  		// Make sure we have the chunk we need:
53  		// First, make sure the list is long enough.
54  		// (we fill with nulls to avoid fetching chunks we don't need)
55  		while (chunks.size() <= whichChunk)
56  			chunks.add(null);
57  		// Now, see if we have the chunk we want, and fetch it if not
58  		int[] chunk = chunks.get(whichChunk);
59  		if (chunk == null) {
60  			chunk = fetchChunk(propIndex, whichChunk);
61  			chunks.set(whichChunk, chunk);
62  		}
63  
64  		return chunk[posWithinChunk];
65  
66  	}
67  
68  	/**
69  	 * Fetch a chunk from the forward index for the specified property.
70  	 * @param propIndex which property we want a forward index chunk for
71  	 * @param number the chunk number to fetch
72  	 * @return the chunk
73  	 */
74  	protected int[] fetchChunk(int propIndex, int number) {
75  		int start = number * CHUNK_SIZE;
76  		int end = start + CHUNK_SIZE;
77  		if (end > docLengthTokens) {
78  			end = docLengthTokens;
79  		}
80  		return fiAccessor.getChunk(propIndex, docId, start, end);
81  	}
82  
83  	@Override
84  	public String getTermString(int propIndex, int termId) {
85  		return fiAccessor.getTermString(propIndex, termId);
86  	}
87  
88  	@Override
89  	public boolean termsEqual(int propIndex, int[] termId, boolean caseSensitive, boolean diacSensitive) {
90  		return fiAccessor.termsEqual(propIndex, termId, caseSensitive, diacSensitive);
91  	}
92  
93  	@Override
94  	public boolean validPos(int pos) {
95  		return pos >= 0 && pos < docLengthTokens;
96  	}
97  
98  }
