14  /**
15   * Represents both a state in an NFA, and a complete NFA
16   * with this as the starting state.
17   */
18  public class NfaStateToken extends NfaState {
19  
20  	static final String ANY_TOKEN = null;
21  
22  	/** What property we're trying to match */
23  	protected String luceneField;
24  
25  	/** Index of the property we're trying to match. Only valid after lookupPropertyNumber() called. */
26  	private int propertyNumber = -1;
27  
28  	/** The tokens this state accepts. */
29  	private Set<String> inputTokenStrings;
30  
31  	/** The tokens this state accepts. Only valid after lookupPropertNumber() called. */
32  	private MutableIntSet inputTokens = null;
33  
34  	/** Do we accept any token? */
35  	private boolean acceptAnyToken = false;
36  
37  	/** The next state if a matching token was found. */
38  	protected NfaState nextState;
39  
40  	public NfaStateToken(String luceneField, String inputToken, NfaState nextState) {
41  		this.luceneField = luceneField;
42  		inputTokenStrings = new HashSet<>();
43  		if (inputToken == null)
44  			acceptAnyToken = true;
45  		else
46  			inputTokenStrings.add(inputToken);
47  		this.nextState = nextState;
48  	}
49  
50  	public NfaStateToken(String luceneField, Set<String> inputTokens, NfaState nextState) {
51  		this.luceneField = luceneField;
52  		this.inputTokenStrings = new HashSet<>(inputTokens);
53  		this.nextState = nextState;
54  	}
55  
56  	/**
57  	 * Find all matches for this NFA in the token source.
58  	 *
59  	 * @param fiDoc where to read tokens from
60  	 * @param pos current matching position
61  	 * @param matchEnds where to collect the matches found, or null if we don't want to collect them
62  	 * @return true if any (new) matches were found, false if not
63  	 */
64  	@Override
65  	public boolean findMatchesInternal(ForwardIndexDocument fiDoc, int pos, int direction, Set<Integer> matchEnds) {
66  		// Token state. Check if it matches token from token source, and if so, continue.
67  		int actualToken = fiDoc.getToken(propertyNumber, pos);
68  		if (acceptAnyToken && actualToken >= 0 || inputTokens.contains(actualToken)) {
69  			if (nextState == null) {
70  				// null stands for the match state
71  				if (matchEnds != null)
72  					matchEnds.add(pos + direction);
73  				return true;
74  			}
75  			return nextState.findMatchesInternal(fiDoc, pos + direction, direction, matchEnds);
76  		}
77  		return false;
78  	}
79  
80  	@Override
81  	void fillDangling(NfaState state) {
82  		if (nextState == null)
83  			nextState = state;
84  	}
85  
86  	@Override
87  	NfaStateToken copyInternal(Collection<NfaState> dangling, Map<NfaState, NfaState> copiesMade) {
88  		NfaStateToken copy = new NfaStateToken(luceneField, inputTokenStrings, null);
89  		copiesMade.put(this, copy);
90  		NfaState nextStateCopy = nextState == null ? null : nextState.copy(dangling, copiesMade);
91  		copy.nextState = nextStateCopy;
92  		if (nextState == null && dangling != null)
93  			dangling.add(copy);
94  		return copy;
95  	}
96  
97  	@Override
98  	public void setNextState(int i, NfaState state) {
99  		if (i != 0)
100 			throw new RuntimeException("Token state only has one next state");
101 		nextState = state;
102 	}
103 
104 	@Override
105 	public boolean matchesEmptySequence(Set<NfaState> statesVisited) {
106 		return false;
107 	}
108 
109 	@Override
110 	public boolean hitsAllSameLength(Set<NfaState> statesVisited) {
111 		return true;
112 	}
113 
114 	@Override
115 	public int hitsLengthMin(Set<NfaState> statesVisited) {
116 		return 1;
117 	}
118 
119 	@Override
120 	public int hitsLengthMax(Set<NfaState> statesVisited) {
121 		return 1;
122 	}
123 
124 	@Override
125 	protected String dumpInternal(Map<NfaState, Integer> stateNrs) {
126 		String terms = acceptAnyToken ? "ANY" : StringUtils.join(inputTokenStrings, "|");
127 		if (terms.length() > 5000) {
128 			terms = terms.substring(0, 500) + "...";
129 		}
130 		return "TOKEN(" + terms + "," + dump(nextState, stateNrs) + ")";
131 	}
132 
133 	@Override
134 	public void lookupPropertyNumbersInternal(ForwardIndexAccessor fiAccessor, Map<NfaState, Boolean> statesVisited) {
135 		String[] comp = ComplexFieldUtil.getNameComponents(luceneField);
136 		String propertyName = comp[1];
137 		propertyNumber = fiAccessor.getPropertyNumber(propertyName);
138 		boolean caseSensitive = ComplexFieldUtil.isCaseSensitive(luceneField);
139 		boolean diacSensitive = ComplexFieldUtil.isDiacriticsSensitive(luceneField);
140 		inputTokens = new IntHashSet(); //new HashSet<>();
141 		for (String token: inputTokenStrings)  {
142 			fiAccessor.getTermNumbers(inputTokens, propertyNumber, token, caseSensitive, diacSensitive);
143 		}
144 		if (nextState != null)
145 			nextState.lookupPropertyNumbers(fiAccessor, statesVisited);
146 	}
147 
148 	@Override
149 	protected void finishInternal(Set<NfaState> visited) {
150 		if (nextState == null)
151 			nextState = match();
152 		else
153 			nextState.finish(visited);
154 	}

156 }
