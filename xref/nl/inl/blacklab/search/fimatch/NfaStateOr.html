9   public class NfaStateOr extends NfaState {
10  
11      List<NfaState> nextStates;
12  
13      private boolean clausesAllSameLength;
14  
15      public NfaStateOr(List<NfaState> nextStates, boolean clausesAllSameLength) {
16          this.nextStates = new ArrayList<>(nextStates);
17          this.clausesAllSameLength = clausesAllSameLength;
18      }
19  
20      public NfaStateOr(boolean clausesAllSameLength) {
21          this.nextStates = new ArrayList<>();
22          this.clausesAllSameLength = clausesAllSameLength;
23      }
24  
25      @Override
26      public boolean findMatchesInternal(ForwardIndexDocument fiDoc, int pos, int direction, Set<Integer> matchEnds) {
27          // OR/Split state. Find matches for all alternatives.
28          boolean result = false;
29          for (NfaState nextState : nextStates) {
30              boolean matchesFound = false;
31              matchesFound = nextState.findMatchesInternal(fiDoc, pos, direction, matchEnds);
32              if (matchesFound && (matchEnds == null || clausesAllSameLength)) {
33                  // We either don't care about the match ends, just that there are matches (matchEnds == null)
34                  // or we know we there's only one match end because all clauses are the same length.
35                  // Short-circuit.
36                  return true;
37              }
38              result |= matchesFound;
39          }
40          return result;
41      }
42  
43      @Override
44      void fillDangling(NfaState state) {
45          for (int i = 0; i < nextStates.size(); i++) {
46              if (nextStates.get(i) == null)
47                  nextStates.set(i, state);
48          }
49      }
50  
51      @Override
52      NfaStateOr copyInternal(Collection<NfaState> dangling, Map<NfaState, NfaState> copiesMade) {
53          NfaStateOr copy = new NfaStateOr(clausesAllSameLength);
54          copiesMade.put(this, copy);
55          List<NfaState> clauseCopies = new ArrayList<>();
56          boolean hasNulls = false;
57          for (NfaState nextState : nextStates) {
58              if (nextState == null)
59                  hasNulls = true;
60              else
61                  nextState = nextState.copy(dangling, copiesMade);
62              clauseCopies.add(nextState);
63          }
64          copy.nextStates.addAll(clauseCopies);
65          if (hasNulls)
66              dangling.add(copy);
67          return copy;
68      }
69  
70      @Override
71      public void setNextState(int input, NfaState state) {
72          nextStates.set(input, state);
73      }
74  
75      @Override
76      public boolean matchesEmptySequence(Set<NfaState> statesVisited) {
77          if (statesVisited.contains(this)) {
78              // We've found a cycle. Stop processing, and just return the
79              // "safest" (least-guarantee) answer. In this case: we can't
80              // guarantee that this DOESN'T match the empty sequence.
81              return true;
82          }
83          statesVisited.add(this);
84          for (NfaState nextState : nextStates) {
85              if (nextState.matchesEmptySequence(statesVisited))
86                  return true;
87          }
88          return false;
89      }
90  
91      @Override
92      public boolean hitsAllSameLength(Set<NfaState> statesVisited) {
93          if (statesVisited.contains(this)) {
94              // We've found a cycle. Stop processing, and just return the
95              // "safest" (least-guarantee) answer. In this case: we can't
96              // guarantee that hits are all the same length.
97              return false;
98          }
99          statesVisited.add(this);
100         int hitLength = -1;
101         for (NfaState nextState : nextStates) {
102             if (!nextState.hitsAllSameLength(statesVisited))
103                 return false;
104             if (hitLength != -1 && hitLength != nextState.hitsLengthMin(statesVisited))
105                 return false;
106             hitLength = nextState.hitsLengthMin(statesVisited);
107         }
108         return true;
109     }
110 
111     @Override
112     public int hitsLengthMin(Set<NfaState> statesVisited) {
113         int hitLengthMin = Integer.MAX_VALUE;
114         if (statesVisited.contains(this)) {
115             // We've found a cycle. Stop processing, and just return the
116             // "safest" (least-guarantee) answer. In this case: the smallest
117             // hit might be 0 long.
118             return 0;
119         }
120         statesVisited.add(this);
121         for (NfaState nextState : nextStates) {
122             int i = nextState.hitsLengthMin(statesVisited);
123             if (i < hitLengthMin)
124                 hitLengthMin = i;
125         }
126         return hitLengthMin;
127     }
128 
129     @Override
130     public int hitsLengthMax(Set<NfaState> statesVisited) {
131         int hitLengthMax = 0;
132         if (statesVisited.contains(this)) {
133             // We've found a cycle. Stop processing, and just return the
134             // "safest" (least-guarantee) answer. In this case: the largest
135             // hit might be "infinitely" large.
136             return Integer.MAX_VALUE;
137         }
138         statesVisited.add(this);
139         for (NfaState nextState : nextStates) {
140             int i = nextState.hitsLengthMax(statesVisited);
141             if (i > hitLengthMax)
142                 hitLengthMax = i;
143         }
144         return hitLengthMax;
145     }

147     @Override
148     protected String dumpInternal(Map<NfaState, Integer> stateNrs) {
149         StringBuilder b = new StringBuilder();
150         for (NfaState s : nextStates) {
151             if (b.length() > 0)
152                 b.append(",");
153             b.append(dump(s, stateNrs));
154         }
155         return "OR(" + b.toString() + ")";
156     }
157 
158     @Override
159     void lookupPropertyNumbersInternal(ForwardIndexAccessor fiAccessor, Map<NfaState, Boolean> statesVisited) {
160         for (NfaState s : nextStates) {
161             if (s != null)
162                 s.lookupPropertyNumbers(fiAccessor, statesVisited);
163         }
164     }
165 
166     @Override
167     protected void finishInternal(Set<NfaState> visited) {
168         for (int i = 0; i < nextStates.size(); i++) {
169             NfaState s = nextStates.get(i);
170             if (s == null)
171                 nextStates.set(i, match());
172             else
173                 s.finish(visited);
174         }
175     }
