4   import nl.inl.util.ArrayUtil;

6   public class HitPropValueMultiple extends HitPropValue {
7   	HitPropValue[] value;
8   
9   	public HitPropValueMultiple(HitPropValue[] value) {
10  		this.value = value;
11  	}
12  
13  	@Override
14  	public int compareTo(Object o) {
15  		return ArrayUtil.compareArrays(value, ((HitPropValueMultiple)o).value);
16  	}
17  
18  	@Override
19  	public int hashCode() {
20  		int result = 0;
21  		for (HitPropValue v: value) {
22  			result ^= v.hashCode();
23  		}
24  		return result;
25  	}
26  
27  	@Override
28  	public String toString() {
29  		StringBuilder b = new StringBuilder();
30  		for (HitPropValue v: value) {
31  			if (b.length() > 0)
32  				b.append(" / ");
33  			b.append(v.toString());
34  		}
35  		return b.toString();
36  	}
37  
38  	public static HitPropValueMultiple deserialize(Hits hits, String info) {
39  		String[] strValues = PropValSerializeUtil.splitMultiple(info);
40  		HitPropValue[] values = new HitPropValue[strValues.length];
41  		int i = 0;
42  		for (String strValue: strValues) {
43  			values[i] = HitPropValue.deserialize(hits, strValue);
44  			i++;
45  		}
46  		return new HitPropValueMultiple(values);
47  	}
48  
49  	@Override
50  	public String serialize() {
51  		String[] valuesSerialized = new String[value.length];
52  		for (int i = 0; i < value.length; i++) {
53  			valuesSerialized[i] = value[i].serialize();
54  		}
55  		return PropValSerializeUtil.combineMultiple(valuesSerialized);
56  	}
57  }
