41  /**
42   * Combines spans, keeping only combinations of hits that occur one after the other. The order is
43   * significant: a hit from the first span must be followed by a hit from the second.
44   *
45   * Note that this class is different from org.apache.lucene.search.spans.SpanNearQuery: it tries to
46   * make sure it generates *all* possible sequence matches. SpanNearQuery doesn't do this; once a hit
47   * is used in a SpanNearQuery match, it advances to the next hit.
48   *
49   * In the future, this class could be expanded to make the exact behaviour configurable: find all
50   * matches / find longest matches / find shortest matches / ...
51   *
52   * See SpanSequenceRaw for details on the matching process.
53   */
54  public class SpanQuerySequence extends BLSpanQueryAbstract {
55  	protected static final Logger logger = LogManager.getLogger(SpanQuerySequence.class);

69  	/**
70  	 * Flatten nested sequences in clauses array.
71  	 *
72  	 * Flattens in-place.
73  	 *
74  	 * @param clauses clauses which may need flattening
75  	 * @return true if any rewriting was done, false if not
76  	 */
77  	private static boolean flattenSequence(List<BLSpanQuery> clauses) {
78  		boolean anyRewritten = false;
79  		for (int i = 0; i < clauses.size(); i++) {
80  			BLSpanQuery child = clauses.get(i);
81  			if (child instanceof SpanQuerySequence) {
82  				clauses.remove(i);
83  				clauses.addAll(i, ((SpanQuerySequence) child).getClauses());
84  				anyRewritten = true;
85  			}
86  		}
87  		return anyRewritten;
88  	}

90  	/**
91  	 * Try to match separate start and end tags in this sequence, and convert into a
92  	 * position filter (e.g. containing) query.
93  	 *
94  	 * For example:
95  	 *  <s> []* 'bla' []* </s> ==> <s/> containing 'bla'
96  	 *
97  	 * @param clauses clauses in which to find matching tags
98  	 * @return true if any rewriting was done, false if not
99  	 */
100 	protected boolean matchingTagsToPosFilter(List<BLSpanQuery> clauses) {
101 		boolean anyRewritten = false;
102 
103 		// Try to match separate start and end tags in this sequence, and convert into a
104 		// containing query. (<s> []* 'bla' []* </s> ==> <s/> containing 'bla')
105 		for (int i = 0; i < clauses.size(); i++) {
106 			BLSpanQuery clause = clauses.get(i);
107 			if (clause instanceof SpanQueryEdge) {
108 				SpanQueryEdge start = (SpanQueryEdge) clause;
109 				if (!start.isRightEdge()) {
110 					String tagName = start.getElementName();
111 					if (tagName != null) {
112 						// Start tag found. Is there a matching end tag?
113 						for (int j = i + 1; j < clauses.size(); j++) {
114 							BLSpanQuery clause2 = clauses.get(j);
115 							if (clause2 instanceof SpanQueryEdge) {
116 								SpanQueryEdge end = (SpanQueryEdge) clause2;
117 								if (end.isRightEdge() && end.getElementName().equals(tagName)) {
118 									// Found start and end tags in sequence. Convert to containing
119 									// query.
120 									List<BLSpanQuery> search = new ArrayList<>();
121 									clauses.remove(i); // start tag
122 									for (int k = 0; k < j - i - 1; k++) {
123 										search.add(clauses.remove(i));
124 									}
125 									clauses.remove(i); // end tag
126 									boolean startAny = false;
127 									if (search.get(0) instanceof SpanQueryAnyToken) {
128 										SpanQueryAnyToken any1 = (SpanQueryAnyToken) search.get(0);
129 										if (any1.hitsLengthMin() == 0 && any1.hitsLengthMax() == MAX_UNLIMITED) {
130 											startAny = true;
131 											search.remove(0);
132 										}
133 									}
134 									boolean endAny = false;
135 									int last = search.size() - 1;
136 									if (search.get(last) instanceof SpanQueryAnyToken) {
137 										SpanQueryAnyToken any2 = (SpanQueryAnyToken) search.get(last);
138 										if (any2.hitsLengthMin() == 0 && any2.hitsLengthMax() == MAX_UNLIMITED) {
139 											endAny = true;
140 											search.remove(last);
141 										}
142 									}
143 									BLSpanQuery producer = start.getClause();
144 									BLSpanQuery filter = new SpanQuerySequence(search.toArray(new BLSpanQuery[0]));
145 									SpanQueryPositionFilter.Operation op;
146 									if (startAny) {
147 										if (endAny) {
148 											op = SpanQueryPositionFilter.Operation.CONTAINING;
149 										} else {
150 											op = SpanQueryPositionFilter.Operation.CONTAINING_AT_END;
151 										}
152 									} else {
153 										if (endAny) {
154 											op = SpanQueryPositionFilter.Operation.CONTAINING_AT_START;
155 										} else {
156 											op = SpanQueryPositionFilter.Operation.MATCHES;
157 										}
158 									}
159 									clauses.add(i, new SpanQueryPositionFilter(producer, filter, op, false));
160 									anyRewritten = true;
161 								}
162 							}
163 						}
164 					}
165 				}
166 			}
167 		}
168 		return anyRewritten;
169 	}

171 	static boolean rewriteClauses(List<BLSpanQuery> clauses, IndexReader reader) throws IOException {
172 		boolean anyRewritten = false;
173 		// Rewrite all clauses.
174 		for (int i = 0; i < clauses.size(); i++) {
175 			BLSpanQuery child = clauses.get(i);
176 			BLSpanQuery rewritten = child.rewrite(reader);
177 			if (child != rewritten) {
178 				// Replace the child with the rewritten version
179 				anyRewritten = true;
180 				clauses.set(i, rewritten);
181 			}
182 		}
183 		return anyRewritten;
184 	}

186 	static boolean optimizeClauses(List<BLSpanQuery> clauses, IndexReader reader) throws IOException {
187 		boolean anyRewritten = false;
188 		// Rewrite all clauses.
189 		for (int i = 0; i < clauses.size(); i++) {
190 			BLSpanQuery child = clauses.get(i);
191 			BLSpanQuery rewritten = child.optimize(reader);
192 			if (child != rewritten) {
193 				// Replace the child with the rewritten version
194 				anyRewritten = true;
195 				clauses.set(i, rewritten);
196 			}
197 		}
198 		return anyRewritten;
199 	}

201 	static boolean combineAdjacentClauses(List<BLSpanQuery> cl, IndexReader reader, String fieldName, Set<ClauseCombiner> combiners) {
202 
203 		boolean anyRewritten = false;
204 
205 		// Rewrite adjacent clauses according to rewriting precedence rules
206 		boolean anyRewrittenThisCycle = true;
207 		int pass = 0;
208 		while (anyRewrittenThisCycle) {
209 			if (Searcher.traceOptimization)
210 				logger.debug("combineAdj pass " + pass + ": " + StringUtils.join(cl, ", "));
211 			pass++;
212 
213 			anyRewrittenThisCycle = false;
214 
215 			// Find the highest-priority rewrite possible
216 			int highestPrio = ClauseCombiner.CANNOT_COMBINE, highestPrioIndex = -1;
217 			ClauseCombiner highestPrioCombiner = null;
218 			BLSpanQuery left, right;
219 			for (int i = 1; i < cl.size(); i++) {
220 				// See if any combiners apply, and if the priority is higher than found so far.
221 				left = cl.get(i - 1);
222 				right = cl.get(i);
223 				for (ClauseCombiner combiner: combiners) {
224 					int prio = combiner.priority(left, right, reader);
225 					if (Searcher.traceOptimization)
226 						logger.debug("  i=" + i + ", combiner=" + combiner + ", prio=" + prio);
227 					if (prio < highestPrio) {
228 						highestPrio = prio;
229 						highestPrioIndex = i;
230 						highestPrioCombiner = combiner;
231 					}
232 				}
233 			}
234 			// Any combiners found?
235 			if (highestPrio < ClauseCombiner.CANNOT_COMBINE) {
236 				// Yes, execute the highest-prio combiner
237 				if (Searcher.traceOptimization)
238 					logger.debug("  Execute combiner: " + highestPrioCombiner + ", index=" + highestPrioIndex);
239 				left = cl.get(highestPrioIndex - 1);
240 				right = cl.get(highestPrioIndex);
241 				BLSpanQuery combined = highestPrioCombiner.combine(left, right, reader);
242 				// (we used to rewrite() combined here just to be safe, but that could break optimizations later)
243 				cl.remove(highestPrioIndex);
244 				cl.set(highestPrioIndex - 1, combined);
245 				anyRewrittenThisCycle = true;
246 			}
247 			if (anyRewrittenThisCycle)
248 				anyRewritten = true;
249 		}
250 
251 		return anyRewritten;
252 	}

254 	@Override
255 	public BLSpanQuery optimize(IndexReader reader) throws IOException {
256 		super.optimize(reader);
257 		boolean canDoNfaMatching = Searcher.fromIndexReader(reader).canDoNfaMatching();
258 		boolean anyRewritten = false;
259 
260 		// Make a copy, because our methods rewrite things in-place.
261 		List<BLSpanQuery> cl = new ArrayList<>(clauses);
262 
263 		// Flatten nested sequences.
264 		// This doesn't change the query because the sequence operator is associative.
265 		anyRewritten |= flattenSequence(cl);
266 
267 		// Find matching tags and rewrite them to position filter (e.g. containing) to execute more
268 		// efficiently
269 		anyRewritten |= matchingTagsToPosFilter(cl);
270 
271 		// Try to combine adjacent clauses into more efficient ones.
272 		// We do this before rewrite (as well as after) specifically to find clauses that are slow
273 		// because
274 		// of regular expressions matching many terms (e.g. "s.*" or ".*s") and match these using an
275 		// NFA instead.
276 		// By doing it before rewriting, we save the time to expand the regex to all its matching
277 		// terms, as well
278 		// as dealing with each of these (sometimes frequent) terms, which can be significant.
279 		anyRewritten |= combineAdjacentClauses(cl, reader, getField(), ClauseCombiner.all(canDoNfaMatching));
280 
281 		// Optimize each clause, and flatten again if necessary
282 		anyRewritten |= optimizeClauses(cl, reader);
283 
284 		if (!anyRewritten) {
285 			// Nothing rewritten. If this is a sequence of length one, just return the clause;
286 			// otherwise return this object unchanged.
287 			if (cl.size() == 1)
288 				return cl.get(0);
289 			return this;
290 		}
291 		return new SpanQuerySequence(cl.toArray(new BLSpanQuery[0]));
292 	}

294 	@Override
295 	public BLSpanQuery rewrite(IndexReader reader) throws IOException {
296 		boolean canDoNfaMatching = Searcher.fromIndexReader(reader).canDoNfaMatching();
297 		boolean anyRewritten = false;
298 
299 		// Make a copy, because our methods rewrite things in-place.
300 		List<BLSpanQuery> cl = new ArrayList<>(clauses);
301 
302 		// Flatten nested sequences.
303 		// This doesn't change the query because the sequence operator is associative.
304 		anyRewritten |= flattenSequence(cl);
305 
306 		// Find matching tags and rewrite them to position filter (e.g. containing) to execute more
307 		// efficiently
308 		anyRewritten |= matchingTagsToPosFilter(cl);
309 
310 		// Try to combine adjacent clauses into more efficient ones.
311 		// We do this before rewrite (as well as after) specifically to find clauses that are slow
312 		// because
313 		// of regular expressions matching many terms (e.g. "s.*" or ".*s") and match these using an
314 		// NFA instead.
315 		// By doing it before rewriting, we save the time to expand the regex to all its matching
316 		// terms, as well
317 		// as dealing with each of these (sometimes frequent) terms, which can be significant.
318 		anyRewritten |= combineAdjacentClauses(cl, reader, getField(), ClauseCombiner.all(canDoNfaMatching));
319 
320 		// Rewrite each clause, and flatten again if necessary
321 		anyRewritten |= rewriteClauses(cl, reader);
322 		if (anyRewritten)
323 			flattenSequence(cl);
324 
325 		// Again, try to combine adjacent clauses into more efficient ones. Rewriting clauses may
326 		// have
327 		// generated new opportunities for combining clauses.
328 		anyRewritten |= combineAdjacentClauses(cl, reader, getField(), ClauseCombiner.all(canDoNfaMatching));
329 
330 		// If any part of the sequence matches the empty sequence, we must
331 		// rewrite it to several alternatives combined with OR. Do so now.
332 		List<List<BLSpanQuery>> results = makeAlternatives(cl, reader);
333 		if (results.size() == 1 && !anyRewritten) {
334 			// Nothing rewritten. If this is a sequence of length one, just return the clause;
335 			// otherwise return this object unchanged.
336 			List<BLSpanQuery> seq = results.get(0);
337 			if (seq.size() == 1)
338 				return seq.get(0);
339 			return this;
340 		}
341 		List<BLSpanQuery> orCl = new ArrayList<>();
342 		for (List<BLSpanQuery> seq: results) {
343 			if (seq.size() == 1)
344 				orCl.add(seq.get(0));
345 			else
346 				orCl.add(new SpanQuerySequence(seq.toArray(new BLSpanQuery[0])));
347 		}
348 		if (orCl.size() == 1)
349 			return orCl.get(0);
350 		return new BLSpanOrQuery(orCl.toArray(new BLSpanQuery[0])).rewrite(reader);
351 	}

353 	/**
354 	 * For possibly empty clauses, combine them with a neighbour into a binary-tree
355 	 * structure. This differs from the approach of makeAlternatives() which produces
356 	 * a OR of several longer sequences. That approach is probably more efficient with Lucene
357 	 * (because it allows more optimizations on the longer sequences produced), while this
358 	 * approach is probably more efficient for NFAs (because we don't have to follow many long
359 	 * paths in the NFA).
360 	 *
361 	 * @param cl clauses
362 	 * @param reader index reader
363 	 * @return alternatives tree
364 	 */
365 	@SuppressWarnings("unused")
366 	private static boolean makeAlternativesLocally(List<BLSpanQuery> cl, IndexReader reader) {
367 		boolean anyRewritten = false;
368 		while (true) {
369 			// Find two clauses to combine to OR, preferring to combine one that matches
370 			// the empty sequence with one that does not.
371 			int bestIndex = -1;
372 			boolean bestBothEmpty = true;
373 			for (int i = 1; i < cl.size(); i++) {
374 				BLSpanQuery left = cl.get(i - 1);
375 				BLSpanQuery right = cl.get(i);
376 				boolean leftEmpty = left.matchesEmptySequence();
377 				boolean rightEmpty = right.matchesEmptySequence();
378 				// Does either clause matcht the empty sequence, and are these two
379 				// the best candidates to combine right now?
380 				if ((leftEmpty || rightEmpty) && bestBothEmpty || (!bestBothEmpty && (!leftEmpty || !rightEmpty))) {
381 					bestBothEmpty = leftEmpty && rightEmpty;
382 					bestIndex = i;
383 				}
384 			}
385 			if (bestIndex < 0)
386 				return anyRewritten; // we're done
387 			// Combine the clauses we found
388 			BLSpanQuery left = cl.get(bestIndex - 1);
389 			BLSpanQuery right = cl.get(bestIndex);
390 			boolean leftEmpty = left.matchesEmptySequence();
391 			boolean rightEmpty = right.matchesEmptySequence();
392 			BLSpanQuery combi;
393 			BLSpanQuery both = new SpanQuerySequence(left, right);
394 			if (leftEmpty && rightEmpty) {
395 				// 4 alternatives: neither, left only, right only, or both
396 				combi = new SpanQueryRepetition(new BLSpanOrQuery(left, right, both), 0, 1);
397 			} else if (leftEmpty) {
398 				// 2 alternatives: right only, or both
399 				combi = new BLSpanOrQuery(right, both);
400 			} else {
401 				// 2 alternatives: left only, or both
402 				combi = new BLSpanOrQuery(left, both);
403 			}
404 			cl.remove(bestIndex - 1);
405 			cl.set(bestIndex - 1, combi);
406 			anyRewritten = true;
407 		}
408 	}

410 	/**
411 	 * Given translated clauses, builds several alternatives and combines them with OR.
412 	 *
413 	 * This is necessary because of how sequence matching works: first the hits in each
414 	 * of the clauses are located, then we try to detect valid sequences by looking at these
415 	 * hits. But when a clause also matches the empty sequence, you may miss valid sequence
416 	 * matches because there's no hit in the clause to combine with the hits from other clauses.
417 	 *
418 	 * @param alternatives the alternative sequences we have built so far
419 	 * @param parts translation results for each of the clauses so far
420 	 * @param reader the index reader
421 	 * @return several alternatives combined with or
422 	 * @throws IOException
423 	 */
424 	List<List<BLSpanQuery>> makeAlternatives(List<BLSpanQuery> parts, IndexReader reader) throws IOException {
425 		if (parts.size() == 1) {
426 			// Last clause in the sequence; just return it
427 			// (noEmpty() version because we will build alternatives
428 			// in the caller if the input matched the empty sequence)
429 			return Arrays.asList(Arrays.asList(parts.get(0).noEmpty().rewrite(reader)));
430 		}
431 
432 		// Recursively determine the query for the "tail" of the list,
433 		// and whether it matches the empty sequence or not.
434 		List<BLSpanQuery> partsTail = parts.subList(1, parts.size());
435 		boolean restMatchesEmpty = true;
436 		for (BLSpanQuery part: partsTail) {
437 			if (!part.matchesEmptySequence()) {
438 				restMatchesEmpty = false;
439 				break;
440 			}
441 		}
442 		List<List<BLSpanQuery>> altTail = makeAlternatives(partsTail, reader);
443 
444 		// Now, add the head part and check if that matches the empty sequence.
445 		return combine(parts.get(0), altTail, restMatchesEmpty, reader);
446 	}

448 	private static List<List<BLSpanQuery>> combine(BLSpanQuery head, List<List<BLSpanQuery>> tailAlts, boolean tailMatchesEmpty,
449 			IndexReader reader) throws IOException {
450 		List<List<BLSpanQuery>> results = new ArrayList<>();
451 		BLSpanQuery headNoEmpty = head.noEmpty().rewrite(reader);
452 		boolean headMatchesEmpty = head.matchesEmptySequence();
453 		for (List<BLSpanQuery> tailAlt: tailAlts) {
454 			// Add head in front of each tail alternative
455 			List<BLSpanQuery> n = new ArrayList<>(tailAlt);
456 			n.add(0, headNoEmpty);
457 			results.add(n);
458 
459 			// If head can be empty, also add original tail alternative
460 			if (headMatchesEmpty)
461 				results.add(tailAlt);
462 		}
463 		// If tail can be empty, also add the head separately
464 		if (tailMatchesEmpty)
465 			results.add(Arrays.asList(headNoEmpty));
466 		return results;
467 	}

471 		for (BLSpanQuery cl: clauses) {
472 			if (!cl.matchesEmptySequence())
473 				return false;
474 		}

570 
571 			// First, combine as many clauses as possible into SpansSequenceSimple,
572 			// which works for simple clauses and is the most efficient to execute.
573 			// OPT: it might be even better to favour combining low-frequency terms first,
574 			// as that minimizes useless skipping through non-matching docs.
575 			for (int i = 1; i < parts.size(); i++) {
576 				CombiPart left = parts.get(i - 1);
577 				CombiPart right = parts.get(i);
578 				CombiPart newPart = null;
579 				if (left.uniqueEnd && left.endSorted && right.startSorted && right.uniqueStart) {
580 					// We can take a shortcut because of what we know about the Spans we're
581 					// combining.
582 					SpansSequenceSimple newSpans = new SpansSequenceSimple(left.spans, right.spans);
583 					newPart = new CombiPart(newSpans, left.uniqueStart, right.uniqueEnd, left.startSorted, right.sameLength,
584 							left.sameLength && right.sameLength);
585 					parts.remove(i - 1);
586 					parts.set(i - 1, newPart);
587 					i--;
588 				}
589 			}
590 
591 			// Now, combine the rest (if any) using the more expensive SpansSequenceRaw,
592 			// that takes more complex sequences into account.
593 			while (parts.size() > 1) {
594 				CombiPart left = parts.get(0);
595 				CombiPart right = parts.get(1);
596 
597 				// Note: the spans coming from SequenceSpansRaw may not be sorted by end point.
598 				// We keep track of this and sort them manually if necessary.
599 				CombiPart newPart = null;
600 				if (!left.endSorted)
601 					left.spans = new PerDocumentSortedSpans(left.spans, PerDocumentSortedSpans.cmpEndPoint, false);
602 				if (!right.startSorted)
603 					right.spans = new PerDocumentSortedSpans(right.spans, PerDocumentSortedSpans.cmpStartPoint, false);
604 				BLSpans newSpans = new SpansSequenceRaw(left.spans, right.spans);
605 				newPart = new CombiPart(newSpans, left.uniqueStart && left.uniqueEnd && right.uniqueStart,
606 						left.uniqueEnd && right.uniqueStart && right.uniqueEnd, left.startSorted, right.sameLength,
607 						left.sameLength && right.sameLength);
608 				parts.remove(0);
609 				parts.set(0, newPart);
610 			}
611 
612 			return parts.get(0).spans;
613 		}
614 
615 	}

702 		int end = direction == 1 ? clauses.size() : -1;
703 		for (int i = start; i != end; i += direction) {

705 			if (frag == null)

768 	/**
769 	 * Either add a clause to an existing SpanQuerySequence, or create a new
770 	 * SpanQuerySequence with the two specified clauses.
771 	 *
772 	 * @param whereToInternalize existing sequence, or existing non-sequence clause
773 	 * @param clauseToInternalize clause to add to sequence or add to existing clause
774 	 * @param addToRight if true, add new clause to the right of existing; if false, to the left
775 	 * @return the expanded or newly created sequence
776 	 */
777 	public static SpanQuerySequence sequenceInternalize(BLSpanQuery whereToInternalize, BLSpanQuery clauseToInternalize,
778 			boolean addToRight) {
779 		SpanQuerySequence seq;
780 		if (whereToInternalize instanceof SpanQuerySequence) {
781 			seq = (SpanQuerySequence) whereToInternalize;
782 			seq = seq.internalizeNeighbour(clauseToInternalize, addToRight);
783 		} else {
784 			if (addToRight)
785 				seq = new SpanQuerySequence(whereToInternalize, clauseToInternalize);
786 			else
787 				seq = new SpanQuerySequence(clauseToInternalize, whereToInternalize);
788 		}
789 		return seq;
790 	}
