16  package nl.inl.blacklab.search.lucene;
17  
18  import java.io.IOException;
19  
20  import org.apache.lucene.search.spans.Spans;
21  
22  import nl.inl.blacklab.search.Span;
23  
24  /**
25   * Will be the base class for all our own Spans classes. Is able to give extra
26   * guarantees about the hits in this Spans object, such as if every hit is equal
27   * in length, if there may be duplicates, etc. This information will help us
28   * optimize certain operations, such as sequence queries, in certain cases.
29   *
30   * The default implementation is appropriate for Spans classes that return only
31   * single-term hits.
32   */
33  public abstract class BLSpans extends Spans {
34  
35      public static final int MAX_UNLIMITED = BLSpanQuery.MAX_UNLIMITED;
36  
37      /**
38       * Should we ask our clauses for captured groups? If the clauses don't capture
39       * any groups, this will be set to false to improve performance.
40       */
41      protected boolean childClausesCaptureGroups = true;
42  
43      /**
44       * Give the BLSpans tree a way to access captured groups, and the capture groups
45       * themselves a way to register themselves..
46       *
47       * subclasses should override this method, pass the context to their child
48       * clauses (if any), and either: - register the captured group they represent
49       * with the context (SpansCaptureGroup does this), OR - store the context so
50       * they can later use it to access captured groups (SpansBackreference does
51       * this)
52       *
53       * @param context the hit query context, that e.g. keeps track of captured
54       *            groups
55       */
56      public void setHitQueryContext(HitQueryContext context) {
57          int before = context.getCaptureRegisterNumber();
58          passHitQueryContextToClauses(context);
59          if (context.getCaptureRegisterNumber() == before) {
60              // Our clauses don't capture any groups; optimize
61              childClausesCaptureGroups = false;
62          }
63      }
64  
65      /**
66       * Called by setHitQueryContext() to pass the context to child clauses.
67       *
68       * @param context the hit query context, that e.g. keeps track of captured
69       *            groups
70       */
71      abstract protected void passHitQueryContextToClauses(HitQueryContext context);
72  
73      /**
74       * Get the start and end position for the captured groups contained in this
75       * BLSpans (sub)tree.
76       *
77       * @param capturedGroups an array the size of the total number of groups in the
78       *            query; the start and end positions for the groups in this subtree
79       *            will be placed in here.
80       */
81      abstract public void getCapturedGroups(Span[] capturedGroups);
82  
83      /**
84       * Advance the start position in the current doc to target or beyond.
85       *
86       * Always at least advances to the next hit, even if the current start position
87       * is already at or beyond the target.
88       *
89       * @param target target start position to advance to
90       * @return new start position, or Spans.NO_MORE_POSITIONS if we're done with
91       *         this document
92       * @throws IOException on error
93       */
94      public int advanceStartPosition(int target) throws IOException {
95          // Naive implementations; subclasses may provide a faster version.
96          int pos;
97          do {
98              pos = nextStartPosition();
99          } while (pos < target && pos != NO_MORE_POSITIONS);
100         return pos;
101     }
102 
103     @Override
104     public long cost() {
105         // returns a completely arbitrary constant value, but it's for
106         // optimizing scoring and we don't generally use that
107         return 100;
108     }
109 
110     static String inf(int max) {
111         return BLSpanQuery.inf(max);
112     }
113 
114     public static BLSpans optSortUniq(BLSpans spans, boolean sort, boolean removeDuplicates) {
115         if (spans == null)
116             return null;
117         BLSpans result = spans;
118         if (!sort && removeDuplicates) {
119             // Make already-sorted spans unique. 
120             return new SpansUnique(result);
121         }
122         if (sort) {
123             // Sort spans by document and start point, then optionally make them unique too.
124             return PerDocumentSortedSpans.startPoint(result, removeDuplicates);
125         }
126         return result;
127     }
128 }
