16  package nl.inl.blacklab.search.lucene;
17  
18  import nl.inl.blacklab.search.Hit;
19  import nl.inl.blacklab.search.Span;
20  
21  
22  /**
23   * Will be the base class for all our own Spans classes. Is able to give extra
24   * guarantees about the hits in this Spans object, such as if every
25   * hit is equal in length, if there may be duplicates, etc. This information
26   * will help us optimize certain operations, such as sequence queries, in certain
27   * cases.
28   *
29   * The default implementation is appropriate for Spans classes that return only
30   * single-term hits.
31   */
32  public abstract class BLSpans extends SpansAbstract {
33  
34  	/** Should we ask our clauses for captured groups?
35  	 *  If the clauses don't capture any groups, this will be set to false
36  	 *  to improve performance.
37  	 */
38  	protected boolean childClausesCaptureGroups = true;
39  
40  	/**
41  	 * When hit B follows hit A, is it guaranteed that B.end &gt;= A.end?
42  	 * Also, if A.end == B.end, is B.start &gt; A.start?
43  	 *
44  	 * @return true if this is guaranteed, false if not
45  	 */
46  	public boolean hitsEndPointSorted() {
47  		return true;
48  	}
49  
50  	/**
51  	 * When hit B follows hit A, is it guaranteed that B.start &gt;= A.start?
52  	 * Also, if A.start == B.start, is B.end &gt; A.end?
53  	 *
54  	 * @return true if this is guaranteed, false if not
55  	 */
56  	public boolean hitsStartPointSorted() {
57  		return true;
58  	}
59  
60  	/**
61  	 * Are all hits the same number of tokens long?
62  	 * @return true if this is guaranteed, false if not
63  	 */
64  	public boolean hitsAllSameLength() {
65  		return true;
66  	}
67  
68  	/**
69  	 * If all hits are the same number of tokens long, how long?
70  	 * @return the length if all hits are guaranteed to be the same length and that length is known, or a negative number otherwise.
71  	 */
72  	public int hitsLength() {
73  		return 1;
74  	}
75  
76  	/**
77  	 * Is it guaranteed that no two hits have the same start position?
78  	 * @return true if this is guaranteed, false if not
79  	 */
80  	public boolean hitsHaveUniqueStart() {
81  		return true;
82  	}
83  
84  	/**
85  	 * Is it guaranteed that no two hits have the same end position?
86  	 * @return true if this is guaranteed, false if not
87  	 */
88  	public boolean hitsHaveUniqueEnd() {
89  		return true;
90  	}
91  
92  	/**
93  	 * Is it guaranteed that no two hits have the same start and end position?
94  	 * @return true if this is guaranteed, false if not
95  	 */
96  	public boolean hitsAreUnique() {
97  		return true;
98  	}
99  
100 	/**
101 	 * Makes a new Hit object from the document id, start and end positions.
102 	 *
103 	 * Subclasses that already have a Hit object available should override this and return the
104 	 * existing Hit object, to avoid excessive Hit instantiations.	 *
105 	 * @return the Hit object for the current hit
106 	 */
107 	public Hit getHit() {
108 		return new Hit(doc(), start(), end());
109 	}
110 
111 	/**
112 	 * Makes a new HitSpan object from the start and end positions (no document id).
113 	 *
114 	 * Subclasses that already have a HitSpan object available could override this
115 	 * and return the existing HitSpan object, to avoid excessive HitSpan instantiations.
116 	 * (Right now, no classes use HitSpan internally, however)
117 	 *
118 	 * @return the HitSpan object for the current hit
119 	 */
120 	public Span getSpan() {
121 		return new Span(start(), end());
122 	}
123 
124 	/**
125 	 * Give the BLSpans tree a way to access captured groups, and the capture
126 	 * groups themselves a way to register themselves..
127 	 *
128 	 * subclasses should override this method, pass the context to their child
129 	 * clauses (if any), and either:
130 	 * - register the captured group they represent with the context (SpansCaptureGroup does this), OR
131 	 * - store the context so they can later use it to access captured groups (SpansBackreference does this)
132 	 *
133 	 * @param context the hit query context, that e.g. keeps track of captured groups
134 	 */
135 	public void setHitQueryContext(HitQueryContext context) {
136 		int before = context.getCaptureRegisterNumber();
137 		passHitQueryContextToClauses(context);
138 		if (context.getCaptureRegisterNumber() == before) {
139 			// Our clauses don't capture any groups; optimize
140 			childClausesCaptureGroups = false;
141 		}
142 	}
143 
144 	/**
145 	 * Called by setHitQueryContext() to pass the context to child clauses.
146 	 *
147 	 * @param context the hit query context, that e.g. keeps track of captured groups
148 	 */
149 	abstract protected void passHitQueryContextToClauses(HitQueryContext context);
150 
151 	/**
152 	 * Get the start and end position for the captured groups contained in
153 	 * this BLSpans (sub)tree.
154 	 *
155 	 * @param capturedGroups an array the size of the total number of groups in the query;
156 	 *   the start and end positions for the groups in this subtree will be placed in here.
157 	 */
158 	abstract public void getCapturedGroups(Span[] capturedGroups);
159 
160 }
