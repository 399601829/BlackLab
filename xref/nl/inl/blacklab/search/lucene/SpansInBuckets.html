16  package nl.inl.blacklab.search.lucene;
17  
18  import java.io.IOException;
19  import org.apache.lucene.search.spans.Spans;
20  
21  import nl.inl.blacklab.search.Hit;
22  import nl.inl.blacklab.search.Span;
23  
24  /**
25   * Interface to retrieve whole sequences of certain matches (in "buckets") instead of individual
26   * matches like with Spans.
27   *
28   * This is useful for efficiently processing sequences of related matches (i.e. fetch some content
29   * for all matches in one document).
30   *
31   * N.B. Note that in these classes, we avoid the term 'group' and 'grouping' because we already use
32   * these terms for the generic way of grouping spans (nl.inl.blacklab.search.grouping), while this
33   * is more focused on speed and efficiency of certain specific operations.
34   *
35   * Specifically, SpansInBuckets is designed to have random access to the contents of a bucket, but
36   * for efficiency's sake, only has sequential access to the buckets themselves. Also, SpansInBuckets
37   * uses subclassing instead of GroupIdentity objects to determine what goes in a bucket. This makes
38   * it easier to optimize.
39   *
40   * Note that SpansInBuckets assumes all hits in a bucket are from a single document.
41   */
42  public interface SpansInBuckets {
43  	/** When to reallocate arraylists to avoid holding on to too much memory */
44  	static final int ARRAYLIST_REALLOC_THRESHOLD = 1000;
45  
46  	final static int NO_MORE_BUCKETS = Spans.NO_MORE_POSITIONS;
47  
48  	public static interface BucketSpanComparator {
49  		abstract public int compare(int start1, int end1, int start2, int end2);
50  	}
51  
52  	/**
53  	 * Document id of current bucket
54  	 *
55  	 * @return Document id of current bucket
56  	 */
57  	int docID();
58  
59  	int bucketSize();
60  
61  	int startPosition(int index);
62  
63  	int endPosition(int index);
64  
65  	Hit getHit(int index);
66  
67  	/**
68  	 * Go to the next document.
69  	 *
70  	 * You still have to call nextBucket() to get to the first valid bucket.
71  	 *
72  	 * @return docID if we're at the next valid doc, NO_MORE_DOCS if we're done
73  	 * @throws IOException
74  	 */
75  	int nextDoc() throws IOException;
76  
77  	/**
78  	 * Go to the next bucket in this doc.
79  	 *
80  	 * @return docID if we're at the next valid bucket, NO_MORE_BUCKETS if we're done
81  	 * @throws IOException
82  	 */
83  	int nextBucket() throws IOException;
84  
85  	/**
86  	 * Skip to specified document id.
87  	 *
88  	 * If we're already at the target id, go to the next document (just like Spans).
89  	 *
90  	 * @param target
91  	 *            document id to skip to
92  	 * @return docID if we're at a valid document, NO_MORE_DOCS if we're done
93  	 * @throws IOException
94  	 */
95  	int advance(int target) throws IOException;
96  
97  	/**
98  	 * Pass the hit query context to the underlying BLSpans.
99  	 *
100 	 * @param context the hit query context
101 	 */
102 	void setHitQueryContext(HitQueryContext context);
103 
104 	/**
105 	 * Get the captured groups information for the current hit.
106 	 *
107 	 * @param indexInBucket what hit in the current bucket to get the information for
108 	 * @param capturedGroups where to add the captured group information
109 	 */
110 	void getCapturedGroups(int indexInBucket, Span[] capturedGroups);
111 
112 }
