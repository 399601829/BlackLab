18  import java.io.IOException;
19  
20  import nl.inl.blacklab.search.Span;
21  import nl.inl.blacklab.search.lucene.SpanQueryPositionFilter.Filter;
22  import nl.inl.blacklab.search.sequences.SpanComparatorStartPoint;
23  import nl.inl.blacklab.search.sequences.SpansInBucketsPerDocument;
24  import nl.inl.blacklab.search.sequences.SpansInBucketsPerDocumentSorted;
25  
26  import org.apache.lucene.search.spans.Spans;
27  
28  /**
29   * Finds hits from a set that contain one or more hits from the second set,
30   * or finds hits from a set that are contained by hit(s) from the second set.
31   */
32  class SpansPositionFilter extends BLSpans {
33  	static SpanComparatorStartPoint cmpStartPoint = new SpanComparatorStartPoint();
34  
35  	/** The two sets of hits to combine */
36  	private BLSpans producer;
37  
38  	/** Does the Spans object still point to a valid hit? */
39  	private boolean stillValidContainers = true;
40  
41  	private SpansInBucketsPerDocument filter;
42  
43  	/** Which index in the filter bucket did we use? (needed for getting captured groups) */
44  	private int filterIndex = -1;
45  
46  	private boolean stillValidSearch = true;
47  
48  	/**
49  	 * What filter operation to use
50  	 */
51  	private Filter op;
52  
53  	/**
54  	 * Find hits containing other hits, or contained by other hits.
55  	 *
56  	 * @param producer the hits we may be interested in
57  	 * @param filter the hits used to filter the producer hits
58  	 * @param op filter operation to use
59  	 */
60  	public SpansPositionFilter(Spans producer, Spans filter, SpanQueryPositionFilter.Filter op) {
61  		this.producer = BLSpansWrapper.optWrapSort(producer);
62  		this.op = op;
63  		stillValidContainers = true;
64  		if (!(filter instanceof BLSpans) || (filter instanceof BLSpans && ((BLSpans)filter).hitsStartPointSorted())) {
65  			// Already start point sorted; no need to sort buckets again
66  			this.filter = new SpansInBucketsPerDocument(filter);
67  		}
68  		else {
69  			// Not sorted yet; sort buckets
70  			this.filter = new SpansInBucketsPerDocumentSorted(filter, cmpStartPoint);
71  		}
72  	}
73  
74  	/**
75  	 * Find hits containing other hits.
76  	 *
77  	 * @param containers the containers we may be interested in.
78  	 * @param filter we only want containers that contain at least on hit from this filter.
79  	 */
80  	public SpansPositionFilter(Spans containers, Spans filter) {
81  		this(containers, filter, Filter.CONTAINING);
82  	}
83  
84  	/**
85  	 * @return the Lucene document id of the current hit
86  	 */
87  	@Override
88  	public int doc() {
89  		return producer.doc();
90  	}
91  
92  	/**
93  	 * @return end position of current hit
94  	 */
95  	@Override
96  	public int end() {
97  		return producer.end();
98  	}
99  
100 	/**
101 	 * Go to next span.
102 	 *
103 	 * @return true if we're at the next span, false if we're done
104 	 * @throws IOException
105 	 */
106 	@Override
107 	public boolean next() throws IOException {
108 		// Are we done yet?
109 		if (!stillValidContainers || !stillValidSearch)
110 			return false;
111 
112 		// Advance container
113 		stillValidContainers = producer.next();
114 		if (!stillValidContainers)
115 			return false; // no more containers; we're done.
116 
117 		// Find first matching container from here
118 		return synchronize();
119 	}
120 
121 	/**
122 	 * Find a container with search hits in it, starting from the current container.
123 	 *
124 	 * @return true if found, false if no such container exists (i.e. we're done)
125 	 * @throws IOException
126 	 */
127 	private boolean synchronize() throws IOException {
128 		// Find the next "valid" container, if there is one.
129 		while (true) {
130 			// Are search and container in the same document?
131 			if (filter.doc() < producer.doc()) {
132 				// No, advance search to be in the same document as the container
133 				stillValidSearch = filter.skipTo(producer.doc());
134 				if (!stillValidSearch)
135 					return false; // No more search results, we're done.
136 			}
137 
138 			// Are there search results in this document?
139 			if (filter.doc() == producer.doc()) {
140 				// Yes. See if the current container contains any of the search results.
141 				//List<Hit> filterHits = filter.getHits();
142 				switch(op) {
143 				case CONTAINING:
144 					// Looking for producer hits with a filter hit inside
145 					for (int i = 0; i < filter.bucketSize(); i++) {
146 						if (filter.start(i) >= producer.start() && filter.end(i) <= producer.end()) {
147 							// Yes, this filter hit is contained in the current producer hit.
148 							filterIndex = i; // remember for captured groups
149 							return true;
150 						}
151 					}
152 					break;
153 				case WITHIN:
154 					// Looking for producer hits contained by a filter hit
155 					for (int i = 0; i < filter.bucketSize(); i++) {
156 						if (filter.start(i) <= producer.start() && filter.end(i) >= producer.end()) {
157 							// Yes, this filter hit contains the current producer hit.
158 							filterIndex = i; // remember for captured groups
159 							return true;
160 						}
161 					}
162 					break;
163 				case STARTS_AT:
164 					// Looking for producer hits starting at a filter hit
165 					for (int i = 0; i < filter.bucketSize(); i++) {
166 						if (filter.start(i) == producer.start()) {
167 							// Yes, this filter hit starts at the current producer hit.
168 							filterIndex = i; // remember for captured groups
169 							return true;
170 						}
171 					}
172 					break;
173 				case ENDS_AT:
174 					// Looking for producer hits ending at a filter hit
175 					for (int i = 0; i < filter.bucketSize(); i++) {
176 						if (filter.end(i) == producer.end()) {
177 							// Yes, this filter hit ends at the current producer hit.
178 							filterIndex = i; // remember for captured groups
179 							return true;
180 						}
181 					}
182 					break;
183 				default:
184 					throw new RuntimeException("Unknown filter operation " + op);
185 				}
186 			}
187 
188 			// No search results found in the current container.
189 			// Advance to the next container.
190 			stillValidContainers = producer.next();
191 			if (!stillValidContainers)
192 				return false; // no more containers; we're done.
193 		}
194 	}
195 
196 	/**
197 	 * Skip to the specified document (or the first document after it containing hits)
198 	 *
199 	 * @param doc
200 	 *            the doc number to skip to (or past)
201 	 * @return true if we're still pointing to a valid hit, false if we're done
202 	 * @throws IOException
203 	 */
204 	@Override
205 	public boolean skipTo(int doc) throws IOException {
206 		// Skip both to doc
207 		stillValidContainers = producer.skipTo(doc);
208 
209 		// Find first matching container from here
210 		return synchronize();
211 	}

213 	/**
214 	 * @return start of current span
215 	 */
216 	@Override
217 	public int start() {
218 		return producer.start();
219 	}
220 
221 	@Override
222 	public String toString() {
223 		switch (op) {
224 		case CONTAINING:
225 			return "SpansContaining(" + producer + " containing " + filter + ")";
226 		case WITHIN:
227 			return "SpansContaining(" + producer + " within " + filter + ")";
228 		case STARTS_AT:
229 			return "SpansContaining(" + producer + " starts at " + filter + ")";
230 		case ENDS_AT:
231 			return "SpansContaining(" + producer + " ends at " + filter + ")";
232 		default:
233 			throw new RuntimeException("Unknown filter operation " + op);
234 		}
235 	}

272 	@Override
273 	public void passHitQueryContextToClauses(HitQueryContext context) {
274 		producer.setHitQueryContext(context);
275 		filter.setHitQueryContext(context);
276 	}
277 
278 	@Override
279 	public void getCapturedGroups(Span[] capturedGroups) {
280 		if (!childClausesCaptureGroups)
281 			return;
282 		producer.getCapturedGroups(capturedGroups);
283 		filter.getCapturedGroups(filterIndex, capturedGroups);
284 	}
