27  /**
28   * Returns all tokens that do not occur in the matches
29   * of the specified query.
30   *
31   * Each token is returned as a single hit.
32   */
33  class SpansNot extends BLSpans {
34  	/** The spans to invert */
35  	private BLSpans clause;
36  
37  	/** Have we called next() or skipTo on the clause yet? */
38  	private boolean clauseIterationStarted;
39  
40  	/** Are there more hits in our spans? */
41  	private boolean moreHitsInClause;
42  
43  	/** Are we completely done? */
44  	private boolean done;
45  
46  	/** Current document */
47  	private int currentDoc;
48  
49  	/** Current document length */
50  	private long currentDocLength;
51  
52  	/** Current token position */
53  	private int currentToken;
54  
55  	///** The Lucene index reader, for querying field length */
56  	//private AtomicReader reader;
57  
58  	/** For testing, we don't have an IndexReader available, so we use test values */
59  	private boolean useTestValues = false;
60  
61  	/** Used to get the field length in tokens for a document */
62  	DocFieldLengthGetter lengthGetter;
63  
64  	/** How much to subtract from length (for ignoring closing token) */
65  	private int subtractFromLength;
66  
67  	/** Highest document id plus one */
68  	private int maxDoc;
69  
70  	/** Documents that haven't been deleted */
71  	private Bits liveDocs;
72  
73  	/** For testing, we don't have an IndexReader available, so we use test values.
74  	 *
75  	 *  The test values are: there are 3 documents (0, 1 and 2) and each is 5 tokens long.
76  	 *
77  	 *  @param test whether or not we want to use test values
78  	 */
79  	void setTest(boolean test) {
80  		useTestValues = test;
81  		if (useTestValues)
82  			maxDoc = 3;
83  		lengthGetter.setTest(test);
84  	}

86  	/**
87  	 * Constructs a SpansNot.
88  	 * @param ignoreLastToken if true, we assume the last token is always a special closing token and ignore it
89  	 * @param reader the index reader, for getting field lengths
90  	 * @param fieldName the field name, for getting field lengths
91  	 * @param clause the clause to invert, or null if we want all tokens
92  	 */
93  	public SpansNot(boolean ignoreLastToken, AtomicReader reader, String fieldName, Spans clause) {
94  		//this.reader = reader;
95  		maxDoc = reader == null ? -1 : reader.maxDoc();
96  		liveDocs = reader == null ? null : MultiFields.getLiveDocs(reader);
97  		subtractFromLength = ignoreLastToken ? 1 : 0;
98  		this.lengthGetter = new DocFieldLengthGetter(reader, fieldName);
99  		this.clause = clause == null ? null : BLSpansWrapper.optWrap(clause);
100 
101 		done = false;
102 		moreHitsInClause = true;
103 		clauseIterationStarted = false;
104 		currentDoc = -1;
105 		currentDocLength = -1;
106 		currentToken = -1;
107 	}

110 	 * @return the Lucene document id of the current hit

114 		return currentDoc;
115 	}

118 	 * @return end position of current hit
119 	 */
120 	@Override
121 	public int end() {
122 		return currentToken + 1;
123 	}

125 	/**
126 	 * Go to next span.
127 	 *
128 	 * @return true if we're at the next span, false if we're done
129 	 * @throws IOException
130 	 */
131 	@Override
132 	public boolean next() throws IOException {
133 		if (done)
134 			return false;
135 
136 		// Advance token
137 		currentToken++;
138 
139 		boolean foundValidToken = false;
140 		while (!foundValidToken) {
141 
142 			// Which of 3 situations are we in?
143 			if (currentDoc < 0) {
144 				// A - We haven't started yet.
145 				//     Go to first document and first hit in clause.
146 				if (!nextDoc()) {
147 					return false;
148 				}
149 				moreHitsInClause = clause == null ? false : clause.next();
150 				clauseIterationStarted = true;
151 
152 				// Loop again to determine if we are at a valid token or not.
153 
154 			} else if (moreHitsInClause && clause.doc() == currentDoc)  {
155 
156 				// B - Spans is at currentDoc.
157 				//     Look at hit, adjust currentToken
158 
159 				// Current hit beyond currentToken?
160 				if (clause.start() > currentToken) {
161 
162 					// Yes. currentToken is fine to produce.
163 					foundValidToken = true;
164 
165 				} else {
166 					// No; advance currentToken past this hit if necessary
167 					if (clause.end() > currentToken) {
168 						// (note that end is the first word not in the hit)
169 						currentToken = clause.end();
170 					}
171 
172 					// Now go to next hit and loop again, until we hit the
173 					// then-part above.
174 					moreHitsInClause = clause.next();
175 					clauseIterationStarted = true;
176 				}
177 
178 			} else {
179 
180 				// C - Spans is depleted or is pointing beyond current doc.
181 				//     Either produce next token (because it's obviously not in
182 				//     the spans matches), or move to next doc if we're done with this doc.
183 				if (currentToken < currentDocLength) {
184 
185 					// Token is fine to produce.
186 					foundValidToken = true;
187 
188 				} else {
189 					// We're done in this document, on to the next
190 					if (!nextDoc()) {
191 						// Done with all documents.
192 						return false;
193 					}
194 				}
195 			}
196 		}
197 		return true;
198 	}
199 
200 	/**
201 	 * Go to the next document in the index.
202 	 *
203 	 * Advances currentDoc until it finds a non-deleted document,
204 	 * then determines length and sets currentToken to 0.
205 	 *
206 	 * @return true if next document was found, false if there are no more documents.
207 	 */
208 	private boolean nextDoc() {
209 		if (currentDoc >= maxDoc)
210 			return false;
211 		boolean currentDocIsDeletedDoc;
212 		do {
213 			currentDoc++;
214 			currentDocIsDeletedDoc = liveDocs != null && !liveDocs.get(currentDoc);
215  		} while (currentDoc < maxDoc && currentDocIsDeletedDoc);
216 		if (currentDoc > maxDoc)
217 			throw new RuntimeException("currentDoc > maxDoc!!");
218 		if (currentDoc == maxDoc) {
219 			done = true;
220 			return false; // no more docs; we're done
221 		}
222 		currentDocLength = lengthGetter.getFieldLength(currentDoc) - subtractFromLength;
223 		currentToken = 0;
224 		return true;
225 	}
226 
227 	/**
228 	 * Skip to the specified document (or the first document after it containing hits).
229 	 *
230 	 * @param doc
231 	 *            the doc number to skip to (or past)
232 	 * @return true if we're still pointing to a valid hit, false if we're done
233 	 * @throws IOException
234 	 */
235 	@Override
236 	public boolean skipTo(int doc) throws IOException {
237 		if (currentDoc >= maxDoc || doc >= maxDoc) {
238 			currentDoc = maxDoc;
239 			moreHitsInClause = false;
240 			if (doc >= maxDoc)
241 				System.err.println("SpansNot.skipTo: tried to skip to " + doc + ", but maxDoc = " + maxDoc);
242 			return false;
243 		}
244 
245 		// If it's not already (past) there, skip clause
246 		// to doc (or beyond if there's no hits in doc)
247 		if (moreHitsInClause && (!clauseIterationStarted || clause.doc() < doc)) {
248 			moreHitsInClause = clause == null ? false : clause.skipTo(doc);
249 			clauseIterationStarted = true;
250 		}
251 
252 		if (currentDoc >= doc) {
253 			// We can't skip to it because we're already there or beyond.
254 			// But, as per spec, skipTo always at least advances to the next match.
255 			return next();
256 		}
257 
258 		currentDoc = doc - 1;
259 		if (!nextDoc())
260 			return false;
261 
262 		// Put us at first valid hit
263 		currentToken = -1;
264 		return next();
265 	}
