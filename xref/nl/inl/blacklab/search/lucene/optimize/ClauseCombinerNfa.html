3   import org.apache.logging.log4j.LogManager;
4   import org.apache.logging.log4j.Logger;
5   import org.apache.lucene.index.IndexReader;
6   
7   import nl.inl.blacklab.search.Searcher;
8   import nl.inl.blacklab.search.fimatch.ForwardIndexAccessor;
9   import nl.inl.blacklab.search.fimatch.NfaTwoWay;

13  /**
14   * Converts one clause to an NFA and uses the other as the anchor in a
15   * FISEQ operation to match using the forward index.
16   */
17  public class ClauseCombinerNfa extends ClauseCombiner {
18  
19  	protected static final Logger logger = LogManager.getLogger(ClauseCombinerNfa.class);
20  
21  	private static final int FORWARD_PRIORITY = 10000000;
22  
23  	/** NOTE: this used to be twice the value of FORWARD_PRIORITY, in an attempt to
24  	 *  prevent building NFAs in different directions, resulting in a suboptimal result,
25  	 *  but that actually causes good optimization opportunities to be skipped altogether. */
26  	private static final int BACKWARD_PRIORITY = 10000000;
27  
28  	/**
29  	 * The default value of nfaThreshold.
30  	 */
31  	public static final long DEFAULT_NFA_THRESHOLD = 900;
32  
33  	/**
34  	 * The maximum value of nfaFactor, meaning "make as many NFAs as possible".
35  	 */
36  	public static final long MAX_NFA_MATCHING = Long.MAX_VALUE;
37  
38  	/**
39  	 * The minimum value of nfaFactor, meaning "make no NFAs".
40  	 */
41  	public static final long NO_NFA_MATCHING = 0;
42  
43  	/**
44  	 * Indicates how expensive fetching a lot of term positions from Lucene is;
45  	 * Used to calculate the cost of "regular" matching.
46  	 */
47  	private static final long TERM_FREQ_PENALTY_FACTOR = 500;
48  
49  	/**
50  	 * What we multiply our calculated cost ratio by to get an integer in a reasonable
51  	 * range.
52  	 */
53  	private static final long COST_RATIO_CONSTANT_FACTOR = 1000;
54  
55  	/**
56  	 * The ratio of estimated numbers of hits that we use to decide
57  	 * whether or not to try NFA-matching with two clauses / subsequences.
58  	 * The lower the number, the more we use NFA-matching.
59  	 */
60  	private static long nfaThreshold = DEFAULT_NFA_THRESHOLD;
61  
62  	public static void setNfaThreshold(long nfaThreshold) {
63  		ClauseCombinerNfa.nfaThreshold = nfaThreshold;
64  	}
65  
66  	public static void setNfaMatchingEnabled(boolean doNfaMatching) {
67  		boolean doingNfaMatching = nfaThreshold != NO_NFA_MATCHING;
68  		if (doNfaMatching != doingNfaMatching)
69  			nfaThreshold = doNfaMatching ? DEFAULT_NFA_THRESHOLD : NO_NFA_MATCHING;
70  	}
71  
72  	private static long getFactor(BLSpanQuery left, BLSpanQuery right, IndexReader reader) {
73  		if (nfaThreshold == NO_NFA_MATCHING)
74  			return 0;
75  		boolean leftEmpty = left.matchesEmptySequence();
76  		boolean rightEmpty = right.matchesEmptySequence();
77  		long numLeft = Math.max(1, left.reverseMatchingCost(reader));
78  		long numRight = Math.max(1, right.reverseMatchingCost(reader));
79  		long seqReverseCost = Math.min(numLeft, numRight) + (numLeft + numRight) / TERM_FREQ_PENALTY_FACTOR;
80  		int fiCostLeft = left.forwardMatchingCost();
81  		int fiCostRight = right.forwardMatchingCost();
82  		long costNfaToReverseForward = COST_RATIO_CONSTANT_FACTOR * numLeft * fiCostRight / seqReverseCost;
83  		long costNfaToReverseBackward = COST_RATIO_CONSTANT_FACTOR * numRight * fiCostLeft / seqReverseCost;
84  		boolean leftNfa = left.canMakeNfa();
85  		boolean rightNfa = right.canMakeNfa();
86  		boolean backwardPossible = leftNfa && !rightEmpty;
87  		boolean forwardPossible = rightNfa && !leftEmpty;
88  		//fp1 bp1 rf242624 rb2568 fil5 fir1 nl27114064 nr57411
89  		//factor == -2569, abs(factor) > nfaThreshold (2000)
90  		if (Searcher.traceOptimization) {
91  			logger.debug(String.format("   fp%d bp%d rf%d rb%d fil%d fir%d nl%d nr%d",
92  				forwardPossible ? 1 : 0,
93  				backwardPossible ? 1 : 0,
94  				costNfaToReverseForward,
95  				costNfaToReverseBackward,
96  				fiCostLeft,
97  				fiCostRight,
98  				numLeft,
99  				numRight
100 			));
101 		}
102 		if (forwardPossible || backwardPossible) {
103 			// Possible.
104 			if (forwardPossible && backwardPossible) {
105 				// Both possible; choose the best
106 				if (costNfaToReverseBackward >= costNfaToReverseForward) {
107 					// Forward
108 					return costNfaToReverseForward + 1;
109 				}
110 				// Backward
111 				return -costNfaToReverseBackward - 1;
112 			}
113 			// One possibility; which one?
114 			if (forwardPossible)
115 				return costNfaToReverseForward + 1;
116 			return -costNfaToReverseBackward - 1;
117 		}
118 		return 0; // not possible
119 	}
120 
121 	@Override
122 	public int priority(BLSpanQuery left, BLSpanQuery right, IndexReader reader) {
123 		if (nfaThreshold == NO_NFA_MATCHING) {
124 			if (Searcher.traceOptimization) logger.debug("   nfa matching switched off");
125 			return CANNOT_COMBINE;
126 		}
127 		long factor = getFactor(left, right, reader);
128 		if (factor == 0) {
129 			if (Searcher.traceOptimization) logger.debug("   factor == 0");
130 			return CANNOT_COMBINE;
131 		}
132 		long absFactor = Math.abs(factor);
133 		if (absFactor > nfaThreshold) {
134 			if (Searcher.traceOptimization) logger.debug("   factor == " + factor + ", abs(factor) > nfaThreshold ("+nfaThreshold+")");
135 			return CANNOT_COMBINE;
136 		}
137 		return factor > 0 ? FORWARD_PRIORITY - (int)(10000 / absFactor) : BACKWARD_PRIORITY - (int)(10000 / absFactor);
138 	}
139 
140 	@Override
141 	public BLSpanQuery combine(BLSpanQuery left, BLSpanQuery right, IndexReader reader) {
142 		// Could we make an NFA out of this clause?
143 		long factor = getFactor(left, right, reader);
144 		if (factor == 0)
145 			throw new UnsupportedOperationException("Cannot combine " + left + " and " + right);
146 		if (factor > 0) {
147 			// Forward (i.e. left is anchor, right is NFA)
148 			if (left instanceof SpanQueryFiSeq && ((SpanQueryFiSeq)left).getDirection() == 1) {
149 				// Existing forward FISEQ; add NFA to it (re-use fiAccessor so properties get same index).
150 				return ((SpanQueryFiSeq)left).appendNfa(right);
151 			}
152 			// New FISEQ.
153 			ForwardIndexAccessor fiAccessor = ForwardIndexAccessor.fromSearcher(Searcher.fromIndexReader(reader), right.getField());
154 			NfaTwoWay nfaTwoWay = right.getNfaTwoWay(fiAccessor, 1);
155 			return new SpanQueryFiSeq(left, false, nfaTwoWay, right, 1, fiAccessor);
156 		}
157 
158 		// Backward (i.e. right is anchor, left is NFA)
159 		if (right instanceof SpanQueryFiSeq && ((SpanQueryFiSeq)right).getDirection() == -1) {
160 			// Existing backward FISEQ; add NFA to it (re-use fiAccessor so properties get same index).
161 			return ((SpanQueryFiSeq)right).appendNfa(left);
162 		}
163 		// New FISEQ.
164 		ForwardIndexAccessor fiAccessor = ForwardIndexAccessor.fromSearcher(Searcher.fromIndexReader(reader), left.getField());
165 		NfaTwoWay nfaTwoWay = left.getNfaTwoWay(fiAccessor, -1);
166 		return new SpanQueryFiSeq(right, true, nfaTwoWay, left, -1, fiAccessor);
167 
168 	}
