15  /**
16   * Converts one clause to an NFA and uses the other as the anchor in a FISEQ
17   * operation to match using the forward index.
18   */
19  public class ClauseCombinerNfa extends ClauseCombiner {
20  
21      protected static final Logger logger = LogManager.getLogger(ClauseCombinerNfa.class);
22  
23      private static final int FORWARD_PRIORITY = 10_000_000;
24  
25      // NOTE: BACKWARD_PRIORITY has slightly different value to prevent FindBugs "same code for two branches" warning
26      /**
27       * NOTE: this used to be twice the value of FORWARD_PRIORITY, in an attempt to
28       * prevent building NFAs in different directions, resulting in a suboptimal
29       * result, but that actually causes good optimization opportunities to be
30       * skipped altogether.
31       */
32      private static final int BACKWARD_PRIORITY = 10_000_001;
33  
34      /**
35       * The default value of nfaThreshold.
36       */
37      public static final long DEFAULT_NFA_THRESHOLD = 900;
38  
39      /**
40       * The maximum value of nfaFactor, meaning "make as many NFAs as possible".
41       */
42      public static final long MAX_NFA_MATCHING = Long.MAX_VALUE;
43  
44      /**
45       * The minimum value of nfaFactor, meaning "make no NFAs".
46       */
47      public static final long NO_NFA_MATCHING = 0;
48  
49      /**
50       * Indicates how expensive fetching a lot of term positions from Lucene is; Used
51       * to calculate the cost of "regular" matching.
52       * 
53       * Higher values means "regular" (reverse) matching is considered relatively cheaper.
54       */
55      private static final long TERM_FREQ_DIVIDER = 500;
56  
57      /**
58       * What we multiply our calculated cost ratio by to get an integer in a
59       * reasonable range.
60       */
61      private static final long COST_RATIO_CONSTANT_FACTOR = 1000;
62  
63      /**
64       * The ratio of estimated numbers of hits that we use to decide whether or not
65       * to try NFA-matching with two clauses / subsequences. The lower the number,
66       * the more we use NFA-matching.
67       */
68      private static long nfaThreshold = DEFAULT_NFA_THRESHOLD;
69  
70      public static void setNfaThreshold(long nfaThreshold) {
71          ClauseCombinerNfa.nfaThreshold = nfaThreshold;
72      }
73  
74      public static long getNfaThreshold() {
75          return ClauseCombinerNfa.nfaThreshold;
76      }
77  
78      public static void setNfaMatchingEnabled(boolean doNfaMatching) {
79          boolean doingNfaMatching = nfaThreshold != NO_NFA_MATCHING;
80          if (doNfaMatching != doingNfaMatching)
81              nfaThreshold = doNfaMatching ? DEFAULT_NFA_THRESHOLD : NO_NFA_MATCHING;
82      }
83  
84      private static long getFactor(BLSpanQuery left, BLSpanQuery right, IndexReader reader) {
85          if (nfaThreshold == NO_NFA_MATCHING)
86              return 0;
87          boolean leftEmpty = left.matchesEmptySequence();
88          boolean rightEmpty = right.matchesEmptySequence();
89          long numLeft = Math.max(1, left.reverseMatchingCost(reader));
90          long numRight = Math.max(1, right.reverseMatchingCost(reader));
91          long seqReverseCost = Math.min(numLeft, numRight) + (numLeft + numRight) / TERM_FREQ_DIVIDER;
92          int fiCostLeft = left.forwardMatchingCost();
93          int fiCostRight = right.forwardMatchingCost();
94          long costNfaToReverseForward = COST_RATIO_CONSTANT_FACTOR * numLeft * fiCostRight / seqReverseCost;
95          long costNfaToReverseBackward = COST_RATIO_CONSTANT_FACTOR * numRight * fiCostLeft / seqReverseCost;
96          boolean leftNfa = left.canMakeNfa();
97          boolean rightNfa = right.canMakeNfa();
98          boolean backwardPossible = leftNfa && !rightEmpty;
99          boolean forwardPossible = rightNfa && !leftEmpty;
100         //fp1 bp1 rf242624 rb2568 fil5 fir1 nl27114064 nr57411
101         //factor == -2569, abs(factor) > nfaThreshold (2000)
102         if (BlackLabIndexImpl.traceOptimization()) {
103             left.log(LogLevel.CHATTY, String.format("(CCNFA: fp%d bp%d rf%d rb%d fil%d fir%d nl%d nr%d)",
104                     forwardPossible ? 1 : 0,
105                     backwardPossible ? 1 : 0,
106                     costNfaToReverseForward,
107                     costNfaToReverseBackward,
108                     fiCostLeft,
109                     fiCostRight,
110                     numLeft,
111                     numRight));
112         }
113         if (forwardPossible || backwardPossible) {
114             // Possible.
115             if (forwardPossible && backwardPossible) {
116                 // Both possible; choose the best
117                 if (costNfaToReverseBackward >= costNfaToReverseForward) {
118                     // Forward
119                     return costNfaToReverseForward + 1;
120                 }
121                 // Backward
122                 return -costNfaToReverseBackward - 1;
123             }
124             // One possibility; which one?
125             if (forwardPossible)
126                 return costNfaToReverseForward + 1;
127             return -costNfaToReverseBackward - 1;
128         }
129         return 0; // not possible
130     }

132     @Override
133     public int priority(BLSpanQuery left, BLSpanQuery right, IndexReader reader) {
134         if (nfaThreshold == NO_NFA_MATCHING) {
135             if (BlackLabIndexImpl.traceOptimization())
136                 left.log(LogLevel.DETAIL, "(CCNFA: nfa matching switched off)");
137             return CANNOT_COMBINE;
138         }
139         long factor = getFactor(left, right, reader);
140         if (factor == 0) {
141             if (BlackLabIndexImpl.traceOptimization())
142                 left.log(LogLevel.DETAIL, "(CCNFA: factor == 0)");
143             return CANNOT_COMBINE;
144         }
145         long absFactor = Math.abs(factor);
146         if (absFactor > nfaThreshold) {
147             if (BlackLabIndexImpl.traceOptimization())
148                 left.log(LogLevel.DETAIL, "(CCNFA: factor == " + factor + ", abs(factor) > nfaThreshold (" + nfaThreshold + "))");
149             return CANNOT_COMBINE;
150         }
151         return factor > 0 ? FORWARD_PRIORITY - (int) (10_000 / absFactor)
152                 : BACKWARD_PRIORITY - (int) (10_000 / absFactor);
153     }
154 
155     @Override
156     public BLSpanQuery combine(BLSpanQuery left, BLSpanQuery right, IndexReader reader) {
157         // Could we make an NFA out of this clause?
158         long factor = getFactor(left, right, reader);
159         if (factor == 0)
160             throw new UnsupportedOperationException("Cannot combine " + left + " and " + right);
161         if (factor > 0) {
162             // Forward (i.e. left is anchor, right is NFA)
163             if (left instanceof SpanQueryFiSeq && ((SpanQueryFiSeq) left).getDirection() == 1) {
164                 // Existing forward FISEQ; add NFA to it (re-use fiAccessor so properties get same index).
165                 return ((SpanQueryFiSeq) left).appendNfa(right);
166             }
167             // New FISEQ.
168             ForwardIndexAccessor fiAccessor = ForwardIndexAccessor.fromIndex(BlackLab.fromIndexReader(reader),
169                     right.getField());
170             NfaTwoWay nfaTwoWay = right.getNfaTwoWay(fiAccessor, 1);
171             return new SpanQueryFiSeq(left, false, nfaTwoWay, right, 1, fiAccessor);
172         }
173 
174         // Backward (i.e. right is anchor, left is NFA)
175         if (right instanceof SpanQueryFiSeq && ((SpanQueryFiSeq) right).getDirection() == SpanQueryFiSeq.DIR_TO_LEFT) {
176             // Existing backward FISEQ; add NFA to it (re-use fiAccessor so properties get same index).
177             return ((SpanQueryFiSeq) right).appendNfa(left);
178         }
179         // New FISEQ.
180         ForwardIndexAccessor fiAccessor = ForwardIndexAccessor.fromIndex(BlackLab.fromIndexReader(reader),
181                 left.getField());
182         NfaTwoWay nfaTwoWay = left.getNfaTwoWay(fiAccessor, SpanQueryFiSeq.DIR_TO_LEFT);
183         return new SpanQueryFiSeq(right, true, nfaTwoWay, left, SpanQueryFiSeq.DIR_TO_LEFT, fiAccessor);
184 
185     }
