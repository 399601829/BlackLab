12  /**
13   * Represents the current "execution context" for executing a TextPattern query.
14   * Inside a query, this context may change: a different property may
15   * be "selected" to search in, the case sensitivity setting may change, etc. This
16   * object is passed to the translation methods and keeps track of this context.
17   */
18  public class QueryExecutionContext {
19  	/** The searcher object, representing the BlackLab index */
20  	public Searcher searcher;
21  
22  	/** The (complex) field to search */
23  	public String fieldName;
24  
25  	/** The property to search */
26  	public String propName;
27  
28  	/** Search case-sensitive?
29  	 *
30  	 * NOTE: depending on available alternatives in the index, this will choose
31  	 * the most appropriate one.
32  	 */
33  	public boolean caseSensitive;
34  
35  	/** Search diacritics-sensitive?
36  	 *
37  	 * NOTE: depending on available alternatives in the index, this will choose
38  	 * the most appropriate one.
39  	 */
40  	public boolean diacriticsSensitive;
41  
42  	/**
43  	 * Construct a query execution context object.
44  	 * @param searcher the searcher object
45  	 * @param fieldName the (complex) field to search
46  	 * @param propName the property to search
47  	 * @param caseSensitive whether search defaults to case-sensitive
48  	 * @param diacriticsSensitive whether search defaults to diacritics-sensitive
49  	 */
50  	public QueryExecutionContext(Searcher searcher, String fieldName, String propName, boolean caseSensitive, boolean diacriticsSensitive) {
51  		this.searcher = searcher;
52  		this.fieldName = fieldName;
53  		this.propName = propName;
54  		this.caseSensitive = caseSensitive;
55  		this.diacriticsSensitive = diacriticsSensitive;
56  	}

58  	/**
59  	 * Return a new query execution context with a different property selected.
60  	 * @param newPropName the property to select
61  	 * @return the new context
62  	 */
63  	public QueryExecutionContext withProperty(String newPropName) {
64  		return new QueryExecutionContext(searcher, fieldName, newPropName, caseSensitive, diacriticsSensitive);
65  	}

67  	public QueryExecutionContext withSensitive(boolean caseSensitive_, boolean diacriticsSensitive_) {

71  	public String optDesensitize(String value) {
72  
73  		final String s = ComplexFieldUtil.SENSITIVE_ALT_NAME;
74  		final String i = ComplexFieldUtil.INSENSITIVE_ALT_NAME;
75  		final String ci = ComplexFieldUtil.CASE_INSENSITIVE_ALT_NAME;
76  		final String di = ComplexFieldUtil.DIACRITICS_INSENSITIVE_ALT_NAME;
77  
78  		String[] parts = ComplexFieldUtil.getNameComponents(luceneField());
79  
80  		String alt = parts.length >= 3 ? parts[2] : "";
81  		if (alt.equals(s)) {
82  			// Don't desensitize
83  			return value;
84  		}
85  		if (alt.equals(i)) {
86  			// Fully desensitize;
87  			return StringUtil.removeAccents(value).toLowerCase();
88  		}
89  		if (alt.equals(ci)) {
90  			// Only case-insensitive
91  			return value.toLowerCase();
92  		}
93  		if (alt.equals(di)) {
94  			// Only diacritics-insensitive
95  			return StringUtil.removeAccents(value);
96  		}
97  
98  		// Unknown alternative; don't change value
99  		return value;
100 	}

102 	/**
103 	 * Return alternatives for the current field/prop that
104 	 * exist and are appropriate for our current settings.
105 	 *
106 	 * @return the alternatives that exist, in order of appropriateness
107 	 */
108 	private String[] getAlternatives() {
109 
110 		if (searcher == null) {
111 			// Test
112 			if (caseSensitive)
113 				return new String[] {"s", "i"};
114 			return new String[] {"i", "s"};
115 		}
116 
117 		final String s = ComplexFieldUtil.SENSITIVE_ALT_NAME;
118 		final String i = ComplexFieldUtil.INSENSITIVE_ALT_NAME;
119 		final String ci = ComplexFieldUtil.CASE_INSENSITIVE_ALT_NAME;
120 		final String di = ComplexFieldUtil.DIACRITICS_INSENSITIVE_ALT_NAME;
121 
122 		ComplexFieldDesc cfd = searcher.getIndexStructure().getComplexFieldDesc(fieldName);
123 		if (cfd == null)
124 			return null;
125 
126 		// Find the property
127 		PropertyDesc pd = cfd.getPropertyDesc(propName);
128 		SensitivitySetting sensitivity = pd.getSensitivity();
129 //		Collection<String> availableAlternatives = Collections.emptyList();
130 //		if (pd != null) {
131 //			availableAlternatives = pd.getAlternatives();
132 //		}
133 
134 		// New alternative naming scheme (every alternative has a name)
135 		List<String> validAlternatives = new ArrayList<>();
136 		if (!caseSensitive && !diacriticsSensitive) {
137 			// search insensitive if available
138 			if (sensitivity != SensitivitySetting.ONLY_SENSITIVE)
139 				validAlternatives.add(i);
140 			if (sensitivity != SensitivitySetting.ONLY_INSENSITIVE)
141 				validAlternatives.add(s);
142 		} else if (caseSensitive && diacriticsSensitive) {
143 			// search fully-sensitive if available
144 			if (sensitivity != SensitivitySetting.ONLY_INSENSITIVE)
145 				validAlternatives.add(s);
146 			if (sensitivity != SensitivitySetting.ONLY_SENSITIVE)
147 				validAlternatives.add(i);
148 		} else if (!diacriticsSensitive) {
149 			// search case-sensitive if available
150 			if (sensitivity == SensitivitySetting.CASE_AND_DIACRITICS_SEPARATE)
151 				validAlternatives.add(di);
152 			if (sensitivity != SensitivitySetting.ONLY_INSENSITIVE)
153 				validAlternatives.add(s);
154 			if (sensitivity != SensitivitySetting.ONLY_SENSITIVE)
155 				validAlternatives.add(i);
156 		} else {
157 			// search diacritics-sensitive if available
158 			if (sensitivity == SensitivitySetting.CASE_AND_DIACRITICS_SEPARATE)
159 				validAlternatives.add(ci);
160 			if (sensitivity != SensitivitySetting.ONLY_SENSITIVE)
161 				validAlternatives.add(i);
162 			if (sensitivity != SensitivitySetting.ONLY_INSENSITIVE)
163 				validAlternatives.add(s);
164 		}
165 		return validAlternatives.toArray(new String[] {});
166 	}

177 	/**
178 	 * Returns the correct current Lucene field name to use, based on the complex field name,
179 	 * property name and list of alternatives.
180 	 * @param includeAlternative if true, also includes the default alternative at the end of the field name (alternatives determine
181 	 *   stuff like case-/diacritics-sensitivity).
182 	 * @return null if field, property or alternative not found; valid Lucene field name otherwise
183 	 */
184 	public String luceneField(boolean includeAlternative) {
185 
186 		// Determine available alternatives based on sensitivity preferences.
187 		String[] alternatives = includeAlternative ? getAlternatives() : null;
188 
189 		if (searcher == null) {
190 			// Mostly for testing. Don't check, just combine field parts.
191 			if (alternatives == null || alternatives.length == 0)
192 				return ComplexFieldUtil.propertyField(fieldName, propName);
193 			return ComplexFieldUtil.propertyField(fieldName, propName, alternatives[0]);
194 		}
195 
196 		// Find the field and the property.
197 		ComplexFieldDesc cfd = searcher.getIndexStructure().getComplexFieldDesc(fieldName);
198 		if (cfd == null)
199 			return null;
200 
201 		if (ComplexFieldUtil.isBookkeepingSubfield(propName)) {
202 			// Not a property but a bookkeeping subfield (prob. starttag/endtag); ok, return it
203 			// (can be removed when old field naming scheme is removed)
204 			return ComplexFieldUtil.bookkeepingField(fieldName, propName);
205 		}
206 
207 		// Find the property
208 		PropertyDesc pd = cfd.getPropertyDesc(propName);
209 		if (pd == null)
210 			return ComplexFieldUtil.propertyField(fieldName, propName); // doesn't exist? use plain property name
211 
212 		if (alternatives == null || alternatives.length == 0) {
213 			// Don't use any alternatives
214 			return ComplexFieldUtil.propertyField(fieldName, propName);
215 		}
216 
217 		// Find the first available alternative to use
218 		for (String alt: alternatives) {
219 			if (pd.hasAlternative(alt)) {
220 				// NOTE: is this loop necessary at all? getAlternatives() only
221 				//  returns available alternatives, so the first one should always
222 				//  be okay, right?
223 				return ComplexFieldUtil.propertyField(fieldName, propName, alt);
224 			}
225 		}
226 
227 		// No valid alternative found. Use plain property.
228 		// NOTE: should never happen, and doesn't make sense anymore as there are
229 		// no 'plain properties' anymore.
230 		return ComplexFieldUtil.propertyField(fieldName, propName);
231 	}
