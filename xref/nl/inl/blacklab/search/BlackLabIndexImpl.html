123     /**
124      * Cut a few words from a string.
125      *
126      * Note, this just splits on whitespace and glues words back with space. Might
127      * not work very well in all cases, but it's not likely to be used anyway (we
128      * generally don't cut a few words from a metadata field).
129      *
130      * @param content the string to cut from
131      * @param startAtWord first word to include
132      * @param endAtWord first word not to include
133      * @return the cut string
134      */
135     static String getWordsFromString(String content, int startAtWord,
136             int endAtWord) {
137         if (startAtWord == -1 && endAtWord == -1)
138             return content;
139         // We want specific words from the field; quick-n-dirty way to do this
140         // (will probably never be used, but let's try to be generic)
141         String[] words = content.split("\\s+");
142         if (startAtWord == -1)
143             startAtWord = 0;
144         if (endAtWord == -1)
145             endAtWord = words.length;
146         StringBuilder b = new StringBuilder();
147         for (int i = startAtWord; i < endAtWord; i++) {
148             if (b.length() > 0)
149                 b.append(" ");
150             b.append(words[i]);
151         }
152         return b.toString();
153     }

185     /**
186      * Get a value from the manifest file, if available.
187      *
188      * @param key key to get the value for, e.g. "Build-Time".
189      * @param defaultValue value to return if no manifest found or key not found
190      * @return value from the manifest, or the default value if not found
191      */
192     static String valueFromManifest(String key, String defaultValue) {
193         try {
194             URL res = BlackLabIndexImpl.class.getResource(BlackLabIndexImpl.class.getSimpleName() + ".class");
195             URLConnection conn = res.openConnection();
196             if (!(conn instanceof JarURLConnection)) {
197                 // Not running from a JAR, no manifest to read
198                 return defaultValue;
199             }
200             JarURLConnection jarConn = (JarURLConnection) res.openConnection();
201             Manifest mf = jarConn.getManifest();
202             String value = null;
203             if (mf != null) {
204                 Attributes atts = mf.getMainAttributes();
205                 if (atts != null) {
206                     value = atts.getValue(key);
207                 }
208             }
209             return value == null ? defaultValue : value;
210         } catch (IOException e) {
211             throw new BlackLabRuntimeException("Could not read '" + key + "' from manifest", e);
212         }
213     }

220      * @return the analyzer, or null if the name wasn't recognized

222     public static Analyzer analyzerInstance(String analyzerName) {
223         analyzerName = analyzerName.toLowerCase();
224         if (analyzerName.equals("whitespace")) {

231             return NONTOKENIZING_ANALYZER;
232         }
233         return null;

341     /**
342      * Open an index.
343      *
344      * @param indexDir the index directory
345      * @param indexMode if true, open in index mode; if false, open in search mode.
346      * @param createNewIndex if true, delete existing index in this location if it
347      *            exists.
348      * @param config input format config to use as template for index structure /
349      *            metadata (if creating new index)
350      * @throws IndexTooOld if the index is too old to be opened by this BlackLab version
351      * @throws ErrorOpeningIndex if the index couldn't be opened
352      */
353     BlackLabIndexImpl(BlackLabEngine blackLab, File indexDir, boolean indexMode, boolean createNewIndex, ConfigInputFormat config) throws ErrorOpeningIndex {
354         this.blackLab = blackLab;
355         searchSettings = SearchSettings.defaults();
356         try {
357             this.indexMode = indexMode;
358 
359             try {
360                 BlackLab.applyConfigToIndex(this);
361             } catch (InvalidConfiguration e) {
362                 throw new InvalidConfiguration(e.getMessage() + " (BlackLab configuration file)", e.getCause());
363             }
364 
365             openIndex(indexDir, indexMode, createNewIndex);
366 
367             // Determine the index structure
368             if (traceIndexOpening)
369                 logger.debug("  Determining index structure...");
370             IndexMetadataImpl indexMetadataImpl = new IndexMetadataImpl(reader, indexDir, createNewIndex, config);
371             indexMetadata = indexMetadataImpl;
372             if (indexMode)
373                 indexMetadataWriter = indexMetadataImpl;
374             else
375                 indexMetadata.freeze();
376 
377             finishOpeningIndex(indexDir, indexMode, createNewIndex);
378         } catch (IOException e) {
379             throw new ErrorOpeningIndex("Could not open index: " + indexDir, e);
380         }
381     }

460     public void forEachDocument(DocTask task) {
461         final int maxDoc = reader().maxDoc();
462         final Bits liveDocs = MultiFields.getLiveDocs(reader());
463         for (int docId = 0; docId < maxDoc; docId++) {
464             if (liveDocs == null || liveDocs.get(docId)) {
465                 task.perform(doc(docId));
466             }
467         }
468     }

518     /**
519      * Register a ContentStore as a content accessor.
520      *
521      * This tells the BlackLabIndex how the content of different fields may be accessed.
522      * This is used for making concordances, for example. Some fields are stored in
523      * the Lucene index, while others may be stored on the file system, a database,
524      * etc.
525      *
526      * A ContentStore is a filesystem-based way to access the contents.
527      *
528      * @param field the field for which this is the content accessor
529      * @param contentStore the ContentStore object by which to access the content
530      *
531      */
532     protected void registerContentStore(Field field, ContentStore contentStore) {
533         contentStores.put(field, contentStore);
534         
535         // Start reading the content store's TOC in the background, so it doesn't
536         // trigger on the first search
537         blackLab.initializationExecutorService().execute(new Runnable() {
538             @Override
539             public void run() {
540                 //logger.debug("START initialize CS: " + field.name());
541                 contentStore.initialize();
542                 //logger.debug("END   initialize CS: " + field.name());
543             }
544         });
545     }

595     protected void openIndex(File indexDir, boolean indexMode, boolean createNewIndex)
596             throws IOException, CorruptIndexException, LockObtainFailedException {
597         if (!indexMode && createNewIndex)
598             throw new BlackLabRuntimeException("Cannot create new index, not in index mode");
599 
600         if (!createNewIndex) {
601             if (!indexMode || VersionFile.exists(indexDir)) {
602                 if (!BlackLabIndex.isIndex(indexDir)) {
603                     throw new IllegalArgumentException("Not a BlackLab index, or wrong version! "
604                             + VersionFile.report(indexDir));
605                 }
606             }
607         }
608 
609         if (traceIndexOpening)
610             logger.debug("Constructing BlackLabIndex...");
611 
612         if (indexMode) {
613             if (traceIndexOpening)
614                 logger.debug("  Opening IndexWriter...");
615             indexWriter = openIndexWriter(indexDir, createNewIndex, null);
616             if (traceIndexOpening)
617                 logger.debug("  Opening corresponding IndexReader...");
618             reader = DirectoryReader.open(indexWriter, false);
619         } else {
620             // Open Lucene index
621             if (traceIndexOpening)
622                 logger.debug("  Following symlinks...");
623             Path indexPath = indexDir.toPath();
624             while (Files.isSymbolicLink(indexPath)) {
625                 // Resolve symlinks, as FSDirectory.open() can't handle them
626                 indexPath = Files.readSymbolicLink(indexPath);
627             }
628             if (traceIndexOpening)
629                 logger.debug("  Opening IndexReader...");
630             reader = DirectoryReader.open(FSDirectory.open(indexPath));
631         }
632         this.indexLocation = indexDir;
633 
634 //      logger.debug("TOTAL TERM FREQ contents%lemma@i: " + reader.getSumTotalTermFreq("contents%lemma@i"));
635 //      logger.debug("TOTAL TERM FREQ test: " + reader.getSumTotalTermFreq("test"));
636 
637     }

639     protected void finishOpeningIndex(File indexDir, boolean indexMode, boolean createNewIndex)
640             throws IOException, CorruptIndexException, LockObtainFailedException, ErrorOpeningIndex {
641         isEmptyIndex = indexMetadata.isNewIndex();
642 
643         // TODO: we need to create the analyzer before opening the index, because
644         //   we can't change the analyzer attached to the IndexWriter (and passing a different
645         //   analyzer in addDocument() went away in Lucene 5.x).
646         //   For now, if we're in index mode, we re-open the index with the analyzer we determined.
647         if (traceIndexOpening)
648             logger.debug("  Creating analyzers...");
649         createAnalyzers();
650 
651         if (indexMode) {
652             // Re-open the IndexWriter with the analyzer we've created above (see comment above)
653             if (traceIndexOpening)
654                 logger.debug("  Re-opening IndexWriter with newly created analyzers...");
655             reader.close();
656             reader = null;
657             indexWriter.close();
658             indexWriter = null;
659             indexWriter = openIndexWriter(indexDir, createNewIndex, analyzer);
660             if (traceIndexOpening)
661                 logger.debug("  IndexReader too...");
662             reader = DirectoryReader.open(indexWriter, false);
663         }
664 
665         // Register ourselves in the mapping from IndexReader to BlackLabIndex,
666         // so we can find the corresponding BlackLabIndex object from within Lucene code
667         blackLab.registerSearcher(reader, this);
668 
669         // Detect and open the ContentStore for the contents field
670         if (!createNewIndex) {
671             if (traceIndexOpening)
672                 logger.debug("  Determining main contents field name...");
673             AnnotatedField mainContentsField = indexMetadata.mainAnnotatedField();
674             if (mainContentsField == null) {
675                 if (!indexMode) {
676                     if (!isEmptyIndex)
677                         throw new BlackLabRuntimeException("Could not detect main contents field");
678                 }
679             }
680 
681             // Register content stores
682             if (traceIndexOpening)
683                 logger.debug("  Opening content stores...");
684             for (AnnotatedField field: indexMetadata.annotatedFields()) {
685                 if (field.hasContentStore()) {
686                     File dir = new File(indexDir, "cs_" + field.name());
687                     if (dir.exists()) {
688                         if (traceIndexOpening)
689                             logger.debug("    " + dir + "...");
690                         registerContentStore(field, ContentStore.open(dir, indexMode, false));
691                     }
692                 }
693             }
694         }
695 
696         if (traceIndexOpening)
697             logger.debug("  Opening IndexSearcher...");
698         indexSearcher = new IndexSearcher(reader);
699 
700         // Make sure large wildcard/regex expansions succeed
701         if (traceIndexOpening)
702             logger.debug("  Setting maxClauseCount...");
703         BooleanQuery.setMaxClauseCount(100_000);
704 
705         // Open the forward indices
706         if (!createNewIndex) {
707             if (traceIndexOpening)
708                 logger.debug("  Opening forward indices...");
709             for (AnnotatedField field: annotatedFields()) {
710                 for (Annotation annotation: field.annotations()) {
711                     if (annotation.hasForwardIndex()) {
712                         // This annotation has a forward index. Make sure it is open.
713                         if (traceIndexOpening)
714                             logger.debug("    " + annotation.luceneFieldPrefix() + "...");
715                         annotationForwardIndex(annotation);
716                     }
717                 }
718             }
719         }
720     }

727     private void createAnalyzers() {
728         Map<String, Analyzer> fieldAnalyzers = new HashMap<>();
729         fieldAnalyzers.put("fromInputFile", analyzerInstance("nontokenizing"));
730         Analyzer baseAnalyzer = analyzerInstance(indexMetadata.metadataFields().defaultAnalyzerName());
731         for (MetadataField field: indexMetadata.metadataFields()) {
732             String analyzerName = field.analyzerName();
733             if (field.type() == FieldType.UNTOKENIZED)
734                 analyzerName = "nontokenizing";
735             if (analyzerName.length() > 0 && !analyzerName.equalsIgnoreCase("default")) {
736                 Analyzer fieldAnalyzer = analyzerInstance(analyzerName);
737                 if (fieldAnalyzer == null) {
738                     logger.error("Unknown analyzer name " + analyzerName + " for field " + field.name());
739                 } else {
740                     if (fieldAnalyzer != baseAnalyzer)
741                         fieldAnalyzers.put(field.name(), fieldAnalyzer);
742                 }
743             }
744         }
745 
746         analyzer = new PerFieldAnalyzerWrapper(baseAnalyzer, fieldAnalyzers);
747     }

749     @Override
750     public void close() {
751         try {
752             if (blackLab != null) {
753                 blackLab.removeSearcher(this);
754                 blackLab = null;
755             }
756 
757             if (reader != null) {
758                 reader.close();
759                 reader = null;
760             }
761             if (indexWriter != null) {
762                 indexWriter.commit();
763                 indexWriter.close();
764                 indexWriter = null;
765             }
766 
767             if (contentStores != null) {
768                 contentStores.close();
769                 contentStores = null;
770             }
771 
772             // Close the forward indices
773             if (forwardIndices != null) {
774                 for (ForwardIndex fi : forwardIndices.values()) {
775                     fi.close();
776                 }
777                 forwardIndices = null;
778             }
779 
780         } catch (IOException e) {
781             throw BlackLabRuntimeException.wrap(e);
782         }

838     @Override
839     public IndexWriter openIndexWriter(File indexDir, boolean create, Analyzer useAnalyzer) throws IOException,
840             CorruptIndexException, LockObtainFailedException {
841         if (!indexDir.exists() && create) {
842             if (!indexDir.mkdir())
843                 throw new BlackLabRuntimeException("Could not create dir: " + indexDir);
844         }
845         Path indexPath = indexDir.toPath();
846         while (Files.isSymbolicLink(indexPath)) {
847             // Resolve symlinks, as FSDirectory.open() can't handle them
848             indexPath = Files.readSymbolicLink(indexPath);
849         }
850         Directory indexLuceneDir = FSDirectory.open(indexPath);
851         if (useAnalyzer == null)
852             useAnalyzer = new BLDutchAnalyzer();
853         IndexWriterConfig config = LuceneUtil.getIndexWriterConfig(useAnalyzer, create);
854         IndexWriter writer = new IndexWriter(indexLuceneDir, config);
855 
856         if (create)
857             VersionFile.write(indexDir, "blacklab", "2");
858         else {
859             if (!BlackLabIndex.isIndex(indexDir)) {
860                 throw new IllegalArgumentException("Not a BlackLab index, or wrong type or version! "
861                         + VersionFile.report(indexDir) + ": " + indexDir);
862             }
863         }
864 
865         return writer;
866     }

878     protected void deleteFromForwardIndices(Document d) {
879         // Delete this document in all forward indices
880         for (Map.Entry<AnnotatedField, ForwardIndex> e : forwardIndices.entrySet()) {
881             AnnotatedField field = e.getKey();
882             ForwardIndex fi = e.getValue();
883             for (Annotation annotation: field.annotations()) {
884                 if (annotation.hasForwardIndex())
885                     fi.get(annotation).deleteDocumentByLuceneDoc(d);
886             }
887         }

908     @Override
909     public void delete(Query q) {
910         logger.debug("Delete query: " + q);
911         if (!indexMode)
912             throw new BlackLabRuntimeException("Cannot delete documents, not in index mode");
913         try {
914             // Open a fresh reader to execute the query
915             try (IndexReader freshReader = DirectoryReader.open(indexWriter, false)) {
916                 // Execute the query, iterate over the docs and delete from FI and CS.
917                 IndexSearcher s = new IndexSearcher(freshReader);
918                 Weight w = s.createNormalizedWeight(q, false);
919                 logger.debug("Doing delete. Number of leaves: " + freshReader.leaves().size());
920                 for (LeafReaderContext leafContext : freshReader.leaves()) {
921                     Bits liveDocs = leafContext.reader().getLiveDocs();
922                     
923                     Scorer scorer = w.scorer(leafContext);
924                     if (scorer == null) {
925                         logger.debug("  No hits in leafcontext");
926                         continue; // no matching documents
927                     }
928 
929                     // Iterate over matching docs
930                     DocIdSetIterator it = scorer.iterator();
931                     logger.debug("  Iterate over matching docs in leaf");
932                     while (true) {
933                         int docId = it.nextDoc();
934                         if (docId == DocIdSetIterator.NO_MORE_DOCS)
935                             break;
936                         if (liveDocs != null && !liveDocs.get(docId)) {
937                             // already deleted.
938                             continue;
939                         }
940                         docId += leafContext.docBase;
941                         Document d = freshReader.document(docId);
942                         logger.debug("    About to delete docId " + docId + ", fromInputFile=" + d.get("fromInputFile") + " from FI and CS");
943 
944                         deleteFromForwardIndices(d);
945 
946                         // Delete this document in all content stores
947                         contentStores.deleteDocument(d);
948                     }
949                 }
950             } finally {
951                 reader.close();
952             }
953 
954             // Finally, delete the documents from the Lucene index
955             logger.debug("  Delete docs from Lucene index");
956             indexWriter.deleteDocuments(q);
957 
958         } catch (IOException e) {
959             throw BlackLabRuntimeException.wrap(e);
960         }
961     }
