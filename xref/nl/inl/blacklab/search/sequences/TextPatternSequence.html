27  /**
28   * A sequence of patterns. The patterns specified may be any pattern, and may themselves be
29   * sequences if desired.

31  public class TextPatternSequence extends TextPatternAnd {
32  	public TextPatternSequence(TextPattern... clauses) {
33  		super(clauses);
34  	}
35  
36  	@Override
37  	public <T> T translate(TextPatternTranslator<T> translator, QueryExecutionContext context) {
38  		List<T> chResults = new ArrayList<T>();
39  
40  		// Keep track of which clauses can match the empty sequence. Use this to build alternatives
41  		// at the end. (see makeAlternatives)
42  		List<Boolean> matchesEmptySeq = new ArrayList<Boolean>();
43  
44  		// To deal with wildcard tokens at the end of a sequence. See below.
45  		TextPatternAnyToken previousAnyTokensPart = null;
46  
47  		// Translate the clauses from back to front, because this makes it easier to
48  		// deal with TextPatternAnyToken (see below).
49  		for (int i = clauses.size() - 1; i >= 0; i--) {
50  			TextPattern cl = clauses.get(i);
51  
52  			if (cl instanceof TextPatternAnyToken) {
53  				// These are a special case, because we cannot translate these into a
54  				// query by themselves. We need to combine them with a query to the left
55  				// or to the right.
56  				// (We used to prefer the query to the right, as this prevented us from
57  				// accidentally going past the end of the document (old version of
58  				// SpansExpansionRaw), but this issue has since been fixed. We still
59  				// go through the sequence from right to left for this reason, however)
60  				TextPatternAnyToken any = (TextPatternAnyToken) cl;
61  				if (chResults.size() > 0) {
62  					// Yes, we have a query to the right. Use that.
63  					T rightNeighbour = chResults.remove(0);
64  					boolean rightNeighbourMatchesEmpty = matchesEmptySeq.remove(0);
65  					T result = translator.expand(context, rightNeighbour, true, any.min, any.max);
66  					chResults.add(0, result);
67  					matchesEmptySeq
68  							.add(0, rightNeighbourMatchesEmpty && any.matchesEmptySequence());
69  				} else {
70  					// No, we don't have a query to the right, use the one to the left.
71  					// Of course, we haven't seen that one yet, so save the pattern and
72  					// handle it in the next loop iteration
73  
74  					if (previousAnyTokensPart != null) {
75  						// We already encountered a matchall which we haven't
76  						// yet processed. Simply add the two together.
77  						int min = previousAnyTokensPart.min + any.min;
78  						int max = previousAnyTokensPart.max + any.max;
79  						if (previousAnyTokensPart.max == -1 || any.max == -1)
80  							max = -1; // infinite expansion
81  						previousAnyTokensPart = new TextPatternAnyToken(min, max);
82  					}
83  					else
84  						previousAnyTokensPart = any;
85  				}
86  				continue; // done with the AnyToken pattern, on to the next
87  			}
88  
89  			// Translate this part of the sequence
90  			T translated = cl.translate(translator, context);
91  			boolean translatedMatchesEmpty = cl.matchesEmptySequence();
92  
93  			// If a wildcard part (TextPatternAnyToken) was found to the right of this,
94  			// expand this part to the right. This only happens if the AnyToken part is
95  			// at the end of the sequence.
96  			if (previousAnyTokensPart != null) {
97  				translated = translator.expand(context, translated, false, previousAnyTokensPart.min,
98  						previousAnyTokensPart.max);
99  				if (translatedMatchesEmpty)
100 					translatedMatchesEmpty = previousAnyTokensPart.matchesEmptySequence();
101 				previousAnyTokensPart = null;
102 			}
103 
104 			// Insert at start of list, to preserve query order
105 			chResults.add(0, translated);
106 			matchesEmptySeq.add(0, translatedMatchesEmpty);
107 		}
108 
109 		// Did the whole sequence consist of any-token parts?
110 		if (chResults.size() == 0 && previousAnyTokensPart != null) {
111 			// Yes.
112 			// Translate as-is (don't use expansion). Probably less efficient,
113 			// but it's the only way to resolve this type of query.
114 			chResults.add(previousAnyTokensPart.translate(translator, context));
115 		}
116 
117 		// Is it still a sequence, or just one part?
118 		if (chResults.size() == 1)
119 			return chResults.get(0); // just one part, return that
120 
121 		// Multiple parts; create sequence object
122 		return makeAlternatives(translator, context, chResults, matchesEmptySeq);
123 	}
124 
125 	/**
126 	 * Given translated clauses, builds several alternatives and combines them with OR.
127 	 *
128 	 * This is necessary because of how sequence matching works: first the hits in each
129 	 * of the clauses are located, then we try to detect valid sequences by looking at these
130 	 * hits. But when a clause also matches the empty sequence, you may miss valid sequence
131 	 * matches because there's no hit in the clause to combine with the hits from other clauses.
132 	 *
133 	 * @param <T> type to translate to
134 	 * @param translator translator
135 	 * @param context the query execution context
136 	 * @param chResults translation results for each of the clauses so far
137 	 * @param matchesEmptySeq whether each of the clauses matches the empty sequence
138 	 * @return several alternatives combined with or
139 	 */
140 	public <T> T makeAlternatives(TextPatternTranslator<T> translator, QueryExecutionContext context,
141 			List<T> chResults, List<Boolean> matchesEmptySeq) {
142 		if (chResults.size() == 1) {
143 			// Last clause in the sequence; just return it
144 			// FIXME: what if this matches the empty sequence?
145 			//        e.g. "the" "quick"?
146 			//        should match both "the quick" and "the",
147 			//        but right now, just matches "the quick".
148 			//        To fix this, we should check that if the "tail" end can
149 			//        match the empty sequence, not just the "head".
150 			return chResults.get(0);
151 		}
152 
153 		// Recursively determine the query for the "tail" of the list
154 		List<T> resultsRest = chResults.subList(1, chResults.size());
155 		List<Boolean> emptyRest = matchesEmptySeq.subList(1, matchesEmptySeq.size());
156 		T rest = makeAlternatives(translator, context, resultsRest, emptyRest);
157 		boolean restMatchesEmpty = true;
158 		for (int i = 1; i < matchesEmptySeq.size(); i++) {
159 			if (!matchesEmptySeq.get(i)) {
160 				restMatchesEmpty = false;
161 				break;
162 			}
163 		}
164 
165 		// Now, add the head part and check if it could match the empty sequence
166 		T firstPart = chResults.get(0);
167 		boolean firstPartMatchesEmpty = matchesEmptySeq.get(0);
168 		T result = translator.sequence(context, Arrays.asList(firstPart, rest));
169 		if (firstPartMatchesEmpty) {
170 			List<T> alternatives;
171 			if (restMatchesEmpty) {
172 				// Both can match empty sequence. Generate 2 additional alternatives
173 				// (only the first part, only the rest part (both empty doesn't make sense))
174 				alternatives = Arrays.asList(
175 					result,
176 					translator.sequence(context, Arrays.asList(rest)),
177 					translator.sequence(context, Arrays.asList(firstPart)));
178 			} else {
179 				// Yes, head matches empty sequence. Also include sequence without the head.
180 				alternatives = Arrays.asList(result,
181 						translator.sequence(context, Arrays.asList(rest)));
182 			}
183 			result = translator.or(context, alternatives);
184 		} else if (restMatchesEmpty) {
185 			// Yes, rest matches empty sequence. Also include sequence without the rest.
186 			List<T> alternatives = Arrays.asList(
187 				result,
188 				translator.sequence(context, Arrays.asList(firstPart)));
189 			result = translator.or(context, alternatives);
190 		}
191 		return result;
192 	}

194 	@Override
195 	public String toString() {
196 		StringBuilder b = new StringBuilder();
197 		for (TextPattern cl : clauses) {
198 			if (b.length() > 0)
199 				b.append(" ");
200 			b.append(cl.toString());
201 		}
202 		return b.toString();
203 	}
204 
205 	@Override
206 	public boolean matchesEmptySequence() {
207 		for (TextPattern cl : clauses) {
208 			if (!cl.matchesEmptySequence())
209 				return false;
210 		}
211 		return true;
212 	}
