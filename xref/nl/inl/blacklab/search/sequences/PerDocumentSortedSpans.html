16  package nl.inl.blacklab.search.sequences;
17  
18  import java.io.IOException;
19  import java.util.Collection;
20  import java.util.Comparator;
21  
22  import nl.inl.blacklab.search.Hit;
23  import nl.inl.blacklab.search.Span;
24  import nl.inl.blacklab.search.lucene.BLSpans;
25  import nl.inl.blacklab.search.lucene.BLSpansWrapper;
26  import nl.inl.blacklab.search.lucene.HitQueryContext;
27  
28  import org.apache.lucene.search.spans.Spans;
29  
30  /**
31   * Sort the given Spans per document, according to the given comparator.
32   */
33  public class PerDocumentSortedSpans extends BLSpans {
34  
35  	static Comparator<Hit> cmpStartPoint = new SpanComparatorStartPoint();
36  
37  	static Comparator<Hit> cmpEndPoint = new SpanComparatorEndPoint();
38  
39  	protected BLSpans source;
40  
41  	private int curDoc = -1, curStart = -1, curEnd = -1;
42  
43  	private SpansInBuckets bucketedSpans;
44  
45  	private boolean eliminateDuplicates;
46  
47  	private int prevStart, prevEnd;
48  
49  	private int indexInBucket = -2; // -2 == not started yet; -1 == just started a bucket
50  
51  	/** Sort hits by end point instead of by start point? */
52  	private boolean sortByEndPoint;
53  
54  	public PerDocumentSortedSpans(Spans src, boolean sortByEndPoint, boolean eliminateDuplicates) {
55  		this.source = BLSpansWrapper.optWrap(src);
56  
57  		// Wrap a HitsPerDocument and show it to the client as a normal, sequential Spans.
58  		this.sortByEndPoint = sortByEndPoint;
59  		Comparator<Hit> comparator = null;
60  		if (sortByEndPoint) {
61  			if (!source.hitsEndPointSorted())
62  				comparator = cmpEndPoint;
63  		} else {
64  			if (!source.hitsStartPointSorted())
65  				comparator = cmpStartPoint;
66  		}
67  		bucketedSpans = new SpansInBucketsPerDocumentSorted(src, comparator);
68  
69  		this.eliminateDuplicates = eliminateDuplicates;
70  	}
71  
72  	@Override
73  	public int doc() {
74  		return curDoc;
75  	}
76  
77  	@Override
78  	public int start() {
79  		return curStart;
80  	}
81  
82  	@Override
83  	public int end() {
84  		return curEnd;
85  	}
86  
87  	@Override
88  	public Hit getHit() {
89  		return bucketedSpans.getHit(indexInBucket);
90  	}
91  
92  	@Override
93  	public boolean next() throws IOException {
94  		do {
95  			if (indexInBucket >= 0) {
96  				prevStart = bucketedSpans.start(indexInBucket);
97  				prevEnd = bucketedSpans.end(indexInBucket);
98  			} else {
99  				prevStart = prevEnd = -1;
100 			}
101 			if (indexInBucket == -2 || indexInBucket == bucketedSpans.bucketSize() - 1) {
102 				if (!bucketedSpans.next())
103 					return false;
104 				prevStart = prevEnd = -1;
105 				indexInBucket = -1;
106 			}
107 			indexInBucket++;
108 			curDoc = bucketedSpans.doc();
109 			curStart = bucketedSpans.start(indexInBucket);
110 			curEnd = bucketedSpans.end(indexInBucket);
111 		} while (eliminateDuplicates && prevStart == curStart && prevEnd == curEnd);
112 		return true;
113 	}
114 
115 	@Override
116 	public boolean skipTo(int target) throws IOException {
117 		int oldDoc = bucketedSpans.doc();
118 		if (!bucketedSpans.skipTo(target))
119 			return false;
120 		if (oldDoc != bucketedSpans.doc())
121 			prevStart = prevEnd = -1;
122 		indexInBucket = -1;
123 		return next();
124 	}
125 
126 	@Override
127 	public Collection<byte[]> getPayload() {
128 		// not used
129 		return null;
130 	}
131 
132 	@Override
133 	public boolean isPayloadAvailable() {
134 		// not used
135 		return false;
136 	}
137 
138 	@Override
139 	public String toString() {
140 		return bucketedSpans.toString();
141 	}
142 
143 	@Override
144 	public boolean hitsAllSameLength() {
145 		return source.hitsAllSameLength();
146 	}
147 
148 	@Override
149 	public int hitsLength() {
150 		return source.hitsLength();
151 	}
152 
153 	@Override
154 	public boolean hitsHaveUniqueStart() {
155 		return source.hitsHaveUniqueStart();
156 	}
157 
158 	@Override
159 	public boolean hitsHaveUniqueEnd() {
160 		return source.hitsHaveUniqueEnd();
161 	}
162 
163 	@Override
164 	public boolean hitsAreUnique() {
165 		return source.hitsAreUnique() || eliminateDuplicates;
166 	}
167 
168 	@Override
169 	public boolean hitsEndPointSorted() {
170 		return sortByEndPoint ? true : hitsAllSameLength();
171 	}
172 
173 	@Override
174 	public boolean hitsStartPointSorted() {
175 		return sortByEndPoint ? hitsAllSameLength() : true;
176 	}
177 
178 	@Override
179 	public void passHitQueryContextToClauses(HitQueryContext context) {
180 		bucketedSpans.setHitQueryContext(context);
181 	}
182 
183 	@Override
184 	public void getCapturedGroups(Span[] capturedGroups) {
185 		bucketedSpans.getCapturedGroups(indexInBucket, capturedGroups);
186 	}
187 }
