55              }
56          }

65      @Override
66      public void characterOffsets(Field field, int[] startsOfWords, int[] endsOfWords,
67              boolean fillInDefaultsIfNotFound) {
68  
69          if (startsOfWords.length == 0)
70              return; // nothing to do
71          try {
72              // Determine lowest and highest word position we'd like to know something about.
73              // This saves a little bit of time for large result sets.
74              int minP = -1, maxP = -1;
75              int numStarts = startsOfWords.length;
76              int numEnds = endsOfWords.length;
77              for (int i = 0; i < numStarts; i++) {
78                  if (startsOfWords[i] < minP || minP == -1)
79                      minP = startsOfWords[i];
80                  if (startsOfWords[i] > maxP)
81                      maxP = startsOfWords[i];
82              }
83              for (int i = 0; i < numEnds; i++) {
84                  if (endsOfWords[i] < minP || minP == -1)
85                      minP = endsOfWords[i];
86                  if (endsOfWords[i] > maxP)
87                      maxP = endsOfWords[i];
88              }
89              if (minP < 0 || maxP < 0)
90                  throw new BlackLabRuntimeException("Can't determine min and max positions");
91  
92              String fieldPropName = field.offsetsField();
93  
94              org.apache.lucene.index.Terms terms = index.reader().getTermVector(id, fieldPropName);
95              if (terms == null)
96                  throw new IllegalArgumentException("Field " + fieldPropName + " in doc " + id + " has no term vector");
97              if (!terms.hasPositions())
98                  throw new IllegalArgumentException(
99                          "Field " + fieldPropName + " in doc " + id + " has no character postion information");
100 
101             //int lowestPos = -1, highestPos = -1;
102             int lowestPosFirstChar = -1, highestPosLastChar = -1;
103             int total = numStarts + numEnds;
104             boolean[] done = new boolean[total]; // NOTE: array is automatically initialized to zeroes!
105             int found = 0;
106 
107             // Iterate over terms
108             TermsEnum termsEnum = terms.iterator();
109             while (termsEnum.next() != null) {
110                 PostingsEnum dpe = termsEnum.postings(null, PostingsEnum.POSITIONS);
111 
112                 // Iterate over docs containing this term (NOTE: should be only one doc!)
113                 while (dpe.nextDoc() != DocIdSetIterator.NO_MORE_DOCS) {
114                     // Iterate over positions of this term in this doc
115                     int positionsRead = 0;
116                     int numberOfPositions = dpe.freq();
117                     while (positionsRead < numberOfPositions) {
118                         int position = dpe.nextPosition();
119                         if (position == -1)
120                             break;
121                         positionsRead++;
122 
123                         // Keep track of the lowest and highest char pos, so
124                         // we can fill in the character positions we didn't find
125                         int startOffset = dpe.startOffset();
126                         if (startOffset < lowestPosFirstChar || lowestPosFirstChar == -1) {
127                             lowestPosFirstChar = startOffset;
128                         }
129                         int endOffset = dpe.endOffset();
130                         if (endOffset > highestPosLastChar) {
131                             highestPosLastChar = endOffset;
132                         }
133 
134                         // We've calculated the min and max word positions in advance, so
135                         // we know we can skip this position if it's outside the range we're interested in.
136                         // (Saves a little time for large result sets)
137                         if (position < minP || position > maxP) {
138                             continue;
139                         }
140 
141                         for (int m = 0; m < numStarts; m++) {
142                             if (!done[m] && position == startsOfWords[m]) {
143                                 done[m] = true;
144                                 startsOfWords[m] = startOffset;
145                                 found++;
146                             }
147                         }
148                         for (int m = 0; m < numEnds; m++) {
149                             if (!done[numStarts + m] && position == endsOfWords[m]) {
150                                 done[numStarts + m] = true;
151                                 endsOfWords[m] = endOffset;
152                                 found++;
153                             }
154                         }
155 
156                         // NOTE: we might be tempted to break here if found == total,
157                         // but that would foul up our calculation of highestPosLastChar and
158                         // lowestPosFirstChar.
159                     }
160                 }
161 
162             }
163             if (found < total) {
164                 if (!fillInDefaultsIfNotFound)
165                     throw new BlackLabRuntimeException("Could not find all character offsets!");
166 
167                 if (lowestPosFirstChar < 0 || highestPosLastChar < 0)
168                     throw new BlackLabRuntimeException("Could not find default char positions!");
169 
170                 for (int m = 0; m < numStarts; m++) {
171                     if (!done[m])
172                         startsOfWords[m] = lowestPosFirstChar;
173                 }
174                 for (int m = 0; m < numEnds; m++) {
175                     if (!done[numStarts + m])
176                         endsOfWords[m] = highestPosLastChar;
177                 }
178             }
179 
180         } catch (IOException e) {
181             throw BlackLabRuntimeException.wrap(e);
182         }
183     }

185     /**
186      * Get character positions from a list of hits.
187      *
188      * @param hits the hits for which we wish to find character positions
189      * @return a list of HitSpan objects containing the character positions for the
190      *         hits.
191      */
192     private List<HitCharSpan> getCharacterOffsets(Hits hits) {
193         int[] starts = new int[hits.size()];
194         int[] ends = new int[hits.size()];
195         Iterator<Hit> hitsIt = hits.iterator();
196         for (int i = 0; i < starts.length; i++) {
197             Hit hit = hitsIt.next(); // hits.get(i);
198             starts[i] = hit.start();
199             ends[i] = hit.end() - 1; // end actually points to the first word not in the hit, so
200                                    // subtract one
201         }
202 
203         characterOffsets(hits.field(), starts, ends, true);
204 
205         List<HitCharSpan> hitspans = new ArrayList<>(starts.length);
206         for (int i = 0; i < starts.length; i++) {
207             hitspans.add(new HitCharSpan(starts[i], ends[i]));
208         }
209         return hitspans;
210     }

212     /**
213      * Convert start/end word positions to char positions.
214      *
215      * @param docId Lucene Document id
216      * @param field field to use
217      * @param startAtWord where to start getting the content (-1 for start of
218      *            document, 0 for first word)
219      * @param endAtWord where to end getting the content (-1 for end of document)
220      * @return the start and end char position as a two element int array (with any
221      *         -1's preserved)
222      */
223     private int[] startEndWordToCharPos(Field field, int startAtWord, int endAtWord) {
224         if (startAtWord == -1 && endAtWord == -1) {
225             // No need to translate anything
226             return new int[] { -1, -1 };
227         }
228 
229         // Translate word pos to char pos and retrieve content
230         // NOTE: this boolean stuff is a bit iffy, but is necessary because
231         // getCharacterOffsets doesn't handle -1 to mean start/end of doc.
232         // We should probably fix that some time.
233         boolean startAtStartOfDoc = startAtWord == -1;
234         boolean endAtEndOfDoc = endAtWord == -1;
235         int[] starts = { startAtStartOfDoc ? 0 : startAtWord };
236         int[] ends = { endAtEndOfDoc ? starts[0] : endAtWord };
237         characterOffsets(field, starts, ends, true);
238         if (startAtStartOfDoc)
239             starts[0] = -1;
240         if (endAtEndOfDoc)
241             ends[0] = -1;
242         return new int[] { starts[0], ends[0] };
243     }

245     @Override
246     public String contentsByCharPos(Field field, int startAtChar, int endAtChar) {
247         Document d = luceneDoc();
248         if (!field.hasContentStore()) {
249             // No special content accessor set; assume a stored field
250             return d.get(field.contentsFieldName()).substring(startAtChar, endAtChar);
251         }
252         return index.contentAccessor(field).getSubstringsFromDocument(d, new int[] { startAtChar }, new int[] { endAtChar })[0];
253     }

255     @Override
256     public String contents(Field field, int startAtWord, int endAtWord) {
257         Document d = luceneDoc();
258         if (!field.hasContentStore()) {
259             // No special content accessor set; assume a stored field
260             String content = d.get(field.contentsFieldName());
261             if (content == null)
262                 throw new IllegalArgumentException("Field not found: " + field.name());
263             return BlackLabIndexImpl.getWordsFromString(content, startAtWord, endAtWord);
264         }
265 
266         int[] startEnd = startEndWordToCharPos(field, startAtWord, endAtWord);
267         return index.contentAccessor(field).getSubstringsFromDocument(d, new int[] { startEnd[0] }, new int[] { startEnd[1] })[0];
268     }

270     @Override
271     public String highlightContent(Hits hits, int startAtWord, int endAtWord) {
272 
273         // Convert word positions to char positions
274         int lastWord = endAtWord < 0 ? endAtWord : endAtWord - 1; // if whole content, don't subtract one
275         AnnotatedField field = hits.field();
276         int[] startEndCharPos = startEndWordToCharPos(field, startAtWord, lastWord);
277 
278         // Get content by char positions
279         int startAtChar = startEndCharPos[0];
280         int endAtChar = startEndCharPos[1];
281         String content = contentsByCharPos(field, startAtChar, endAtChar);
282 
283         boolean wholeDocument = startAtWord == -1 && endAtWord == -1;
284         boolean mustFixUnbalancedTags = !wholeDocument;
285 
286         // Do we have anything to highlight, or do we have an XML fragment that needs balancing?
287         ResultsStats hitsStats = hits.hitsStats();
288         if (hitsStats.processedAtLeast(1) || mustFixUnbalancedTags) {
289             // Find the character offsets for the hits and highlight
290             List<HitCharSpan> hitspans = null;
291             if (hitsStats.processedAtLeast(1)) // if hits == null, we still want the highlighter to make it well-formed
292                 hitspans = getCharacterOffsets(hits);
293             XmlHighlighter hl = new XmlHighlighter();
294             hl.setUnbalancedTagsStrategy(index.defaultUnbalancedTagsStrategy());
295             if (startAtChar == -1)
296                 startAtChar = 0;
297             content = hl.highlight(content, hitspans, startAtChar);
298         }
299         return content;
300     }

302     /**
303      * Get a number of substrings from a certain field in a certain document.
304      *
305      * For larger documents, this is faster than retrieving the whole content first
306      * and then cutting substrings from that.
307      *
308      * @param d the document
309      * @param field the field
310      * @param starts start positions of the substring we want
311      * @param ends end positions of the substring we want; correspond to the starts
312      *            array.
313      * @return the substrings
314      */
315     private String[] getSubstringsFromDocument(Document d, Field field, int[] starts,
316             int[] ends) {
317         if (!field.hasContentStore()) {
318             String[] content;
319             // No special content accessor set; assume a non-annotated stored field
320             String fieldContent = d.get(field.contentsFieldName());
321             content = new String[starts.length];
322             for (int i = 0; i < starts.length; i++) {
323                 content[i] = fieldContent.substring(starts[i], ends[i]);
324             }
325             return content;
326         }
327         // Content accessor set. Use it to retrieve the content.
328         return index.contentAccessor(field).getSubstringsFromDocument(d, starts, ends);
329     }

331     @Override
332     public List<Concordance> makeConcordancesFromContentStore(Field field, int[] startsOfWords,
333             int[] endsOfWords, XmlHighlighter hl) {
334         // Determine starts and ends
335         int n = startsOfWords.length / 2;
336         int[] starts = new int[n];
337         int[] ends = new int[n];
338         for (int i = 0, j = 0; i < startsOfWords.length; i += 2, j++) {
339             starts[j] = startsOfWords[i];
340             ends[j] = endsOfWords[i + 1];
341         }
342 
343         // Retrieve 'em all
344         Document d = luceneDoc();
345         String[] content = getSubstringsFromDocument(d, field, starts, ends);
346 
347         // Cut 'em up
348         List<Concordance> rv = new ArrayList<>();
349         for (int i = 0, j = 0; i < startsOfWords.length; i += 2, j++) {
350             // Put the concordance in the Hit object
351             int absLeft = startsOfWords[i];
352             int absRight = endsOfWords[i + 1];
353             int relHitLeft = startsOfWords[i + 1] - absLeft;
354             int relHitRight = endsOfWords[i] - absLeft;
355             String currentContent = content[j];
356 
357             // Determine context and build concordance.
358             // Note that hit text may be empty for hits of length zero,
359             // such as a search for open tags (which have a location but zero length,
360             // like a search for a word has a length 1)
361             String hitText = relHitRight < relHitLeft ? ""
362                     : currentContent.substring(relHitLeft,
363                             relHitRight);
364             String leftContext = currentContent.substring(0, relHitLeft);
365             String rightContext = currentContent.substring(relHitRight, absRight - absLeft);
366 
367             // Make each fragment well-formed
368             hitText = hl.makeWellFormed(hitText);
369             leftContext = hl.makeWellFormed(leftContext);
370             rightContext = hl.makeWellFormed(rightContext);
371 
372             rv.add(new Concordance(new String[] { leftContext, hitText, rightContext }));
373         }
374         return rv;
375     }

377     @Override
378     public int hashCode() {
379         final int prime = 31;
380         int result = 1;
381         result = prime * result + id;
382         result = prime * result + ((index == null) ? 0 : index.hashCode());
383         return result;
384     }

386     @Override
387     public boolean equals(Object obj) {
388         if (this == obj)
389             return true;
390         if (obj == null)
391             return false;
392         if (getClass() != obj.getClass())
393             return false;
394         DocImpl other = (DocImpl) obj;
395         if (id != other.id)
396             return false;
397         if (index == null) {
398             if (other.index != null)
399                 return false;
400         } else if (!index.equals(other.index))
401             return false;
402         return true;
403     }
