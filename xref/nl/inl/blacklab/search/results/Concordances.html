17  /** Concordances for a list of hits. */
18  public class Concordances {
19  
20      /**
21       * The concordances, if they have been retrieved.
22       *
23       * NOTE: when making concordances from the forward index, this will always be
24       * null, because Kwics will be used internally. This is only used when making
25       * concordances from the content store (the old default).
26       */
27      private Map<Hit, Concordance> concordances;
28      
29      Kwics kwics = null;
30  
31      /**
32       * @param hits
33       */
34      Concordances(Hits hits, ConcordanceType type, ContextSize contextSize) {
35          if (contextSize.left() < 0)
36              throw new IllegalArgumentException("contextSize cannot be negative");
37          if (type == ConcordanceType.FORWARD_INDEX) {
38              kwics = new Kwics(hits, contextSize);
39          }
40      
41          // Get the concordances
42          concordances = retrieveConcordancesFromContentStore(hits, contextSize);
43      }
44  
45      /**
46       * Return the concordance for the specified hit.
47       *
48       * The first call to this method will fetch the concordances for all the hits in
49       * this Hits object. So make sure to select an appropriate HitsWindow first:
50       * don't call this method on a Hits set with >1M hits unless you really want to
51       * display all of them in one go.
52       *
53       * @param h the hit
54       * @return concordance for this hit
55       */
56      public Concordance get(Hit h) {
57          if (kwics != null)
58              return kwics.get(h).toConcordance();
59          return concordances.get(h);
60      }
61  
62      /**
63       * Retrieves the concordance information (left, hit and right context) for a
64       * number of hits in the same document from the ContentStore.
65       *
66       * NOTE1: it is assumed that all hits in this Hits object are in the same
67       * document!
68       * 
69       * @param hits hits to make concordance for
70       * @param field field to make conc for
71       * @param wordsAroundHit number of words left and right of hit to fetch
72       * @param conc where to add the concordances
73       * @param hl
74       */
75      private synchronized static void makeConcordancesSingleDocContentStore(Hits hits, ContextSize wordsAroundHit,
76              Map<Hit, Concordance> conc,
77              XmlHighlighter hl) {
78          if (hits.size() == 0)
79              return;
80          QueryInfo queryInfo = hits.queryInfo();
81          Doc doc = queryInfo.index().doc(hits.get(0).doc());
82          int arrayLength = hits.size() * 2;
83          int[] startsOfWords = new int[arrayLength];
84          int[] endsOfWords = new int[arrayLength];
85  
86          // Determine the first and last word of the concordance, as well as the
87          // first and last word of the actual hit inside the concordance.
88          int startEndArrayIndex = 0;
89          int contextSz = wordsAroundHit.left();
90          for (Hit hit : hits) {
91              int hitStart = hit.start();
92              int hitEnd = hit.end() - 1;
93  
94              int start = hitStart - contextSz;
95              if (start < 0)
96                  start = 0;
97              int end = hitEnd + contextSz;
98  
99              startsOfWords[startEndArrayIndex] = start;
100             startsOfWords[startEndArrayIndex + 1] = hitStart;
101             endsOfWords[startEndArrayIndex] = hitEnd;
102             endsOfWords[startEndArrayIndex + 1] = end;
103 
104             startEndArrayIndex += 2;
105         }
106 
107         // Get the relevant character offsets (overwrites the startsOfWords and endsOfWords
108         // arrays)
109         AnnotatedField field = queryInfo.field();
110         doc.characterOffsets(field, startsOfWords, endsOfWords, true);
111 
112         // Make all the concordances
113         List<Concordance> newConcs = doc.makeConcordancesFromContentStore(field, startsOfWords, endsOfWords, hl);
114         for (int i = 0; i < hits.size(); i++) {
115             conc.put(hits.get(i), newConcs.get(i));
116         }
117     }
118 
119     /**
120      * Generate concordances from content store (slower).
121      *
122      * @param hits hits for which to generate concordances
123      * @param contextSize how many words around the hit to retrieve
124      * @return the concordances
125      */
126     private static Map<Hit, Concordance> retrieveConcordancesFromContentStore(Hits hits, ContextSize contextSize) {
127         XmlHighlighter hl = new XmlHighlighter(); // used to make fragments well-formed
128         QueryInfo queryInfo = hits.queryInfo();
129         hl.setUnbalancedTagsStrategy(queryInfo.index().defaultUnbalancedTagsStrategy());
130         // Group hits per document
131         MutableIntObjectMap<List<Hit>> hitsPerDocument = IntObjectMaps.mutable.empty();
132         for (Hit key: hits) {
133             List<Hit> hitsInDoc = hitsPerDocument.get(key.doc());
134             if (hitsInDoc == null) {
135                 hitsInDoc = new ArrayList<>();
136                 hitsPerDocument.put(key.doc(), hitsInDoc);
137             }
138             hitsInDoc.add(key);
139         }
140         Map<Hit, Concordance> conc = new HashMap<>();
141         for (List<Hit> l : hitsPerDocument.values()) {
142             Hits hitsInThisDoc = Hits.fromList(queryInfo, l);
143             Concordances.makeConcordancesSingleDocContentStore(hitsInThisDoc, contextSize, conc, hl);
144         }
145         return conc;
146     }
