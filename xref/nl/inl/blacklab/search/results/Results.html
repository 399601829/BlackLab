38      // Perform simple generic sampling operation
39      protected static <T> List<T> doSample(Results<T> source, SampleParameters sampleParameters) {
40          // We can later provide an optimized version that uses a HitsSampleCopy or somesuch
41          // (this class could save memory by only storing the hits we're interested in)
42          
43          List<T> results = new ArrayList<>();
44          
45          Random random = new Random(sampleParameters.seed());
46          int numberOfHitsToSelect = sampleParameters.numberOfHits(source.size());
47          if (numberOfHitsToSelect > source.size())
48              numberOfHitsToSelect = source.size(); // default to all hits in this case
49          // Choose the hits
50          Set<Integer> chosenHitIndices = new TreeSet<>();
51          for (int i = 0; i < numberOfHitsToSelect; i++) {
52              // Choose a hit we haven't chosen yet
53              int hitIndex;
54              do {
55                  hitIndex = random.nextInt(source.size());
56              } while (chosenHitIndices.contains(hitIndex));
57              chosenHitIndices.add(hitIndex);
58          }
59          
60          // Add the hits in order of their index
61          for (Integer hitIndex : chosenHitIndices) {
62              T hit = source.get(hitIndex);
63              results.add(hit);
64          }
65          return results;
66      }

68      protected static <T> List<T> doWindow(Results<T> results, int first, int number) {
69          if (first < 0 || first != 0 && !results.resultsProcessedAtLeast(first + 1)) {
70              //throw new BlackLabRuntimeException("First hit out of range");
71              return Collections.emptyList();
72          }
73      
74          // Auto-clamp number
75          int actualSize = number;
76          if (!results.resultsProcessedAtLeast(first + actualSize))
77              actualSize = results.size() - first;
78      
79          // Make sublist (copy results from List.subList() to avoid lingering references large lists)
80          return new ArrayList<T>(results.resultsSubList(first, first + actualSize));
81      }

83      protected static <T> List<T> doFilter(Results<T> results, ResultProperty<T> property, PropertyValue value) {
84          return results.stream().filter(g -> property.get(g).equals(value)).collect(Collectors.toList());
85      }

87      protected static <P extends ResultProperty<T>, T> List<T> doSort(Results<T> results, P sortProp) {
88          results.ensureAllResultsRead();
89          List<T> sorted = new ArrayList<>(results.resultsList());
90          sorted.sort(sortProp);
91          return sorted;
92      }

249     /**
250      * Return an iterator over these hits.
251      *
252      * @return the iterator
253      */
254     @Override
255     public Iterator<T> iterator() {
256         // Construct a custom iterator that iterates over the hits in the hits
257         // list, but can also take into account the Spans object that may not have
258         // been fully read. This ensures we don't instantiate Hit objects for all hits
259         // if we just want to display the first few.
260         return new Iterator<T>() {
261         
262             int index = -1;
263         
264             @Override
265             public boolean hasNext() {
266                 // Do we still have hits in the hits list?
267                 ensureResultsRead(index + 2);
268                 return results.size() >= index + 2;
269             }
270         
271             @Override
272             public T next() {
273                 // Check if there is a next, taking unread hits from Spans into account
274                 if (hasNext()) {
275                     index++;
276                     return results.get(index);
277                 }
278                 throw new NoSuchElementException();
279             }
280         
281             @Override
282             public void remove() {
283                 throw new UnsupportedOperationException();
284             }
285         
286         };
287     }

291      *
292      * @param i index of the desired hit
293      * @return the hit, or null if it's beyond the last hit

295     public synchronized T get(int i) {
296         ensureResultsRead(i + 1);
297         if (i >= results.size())
298             return null;
299         return results.get(i);

413     /**
414      * Get part of the list of results.
415      * 
416      * Clients shouldn't use this. Used internally for certain performance-sensitive
417      * operations like sorting.
418      * 
419      * The returned list is a view backed by the results list.
420      * 
421      * If toIndex is out of range, no exception is thrown, but a smaller list is returned.
422      * 
423      * @return the list of hits
424      */
425     protected List<T> resultsSubList(int fromIndex, int toIndex) {
426         ensureResultsRead(toIndex);
427         if (toIndex > results.size())
428             toIndex = results.size();
429         return results.subList(fromIndex, toIndex);
430     }
