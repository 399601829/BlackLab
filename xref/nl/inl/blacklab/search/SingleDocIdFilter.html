1   package nl.inl.blacklab.search;
2   
3   import org.apache.lucene.index.LeafReaderContext;
4   import org.apache.lucene.search.DocIdSet;
5   import org.apache.lucene.search.Filter;
6   import org.apache.lucene.util.Bits;
7   
8   /**
9    * A Filter that only matches a single Lucene document id.
10   *
11   * Used for finding hits in a single document (for highlighting).
12   */
13  public class SingleDocIdFilter extends Filter {
14  
15  	private int luceneDocId;
16  
17  	public SingleDocIdFilter(int luceneDocId) {
18  		this.luceneDocId = luceneDocId;
19  	}
20  
21  	@Override
22  	public DocIdSet getDocIdSet(LeafReaderContext ctx, Bits bits) {
23  		// Check that id could be in this segment, and bits allows this doc id
24  		if (luceneDocId >= ctx.docBase && (bits == null || bits.get(luceneDocId - ctx.docBase))) {
25  			// ctx is a single segment, so use docBase to adjust the id
26  			return new SingleDocIdSet(luceneDocId - ctx.docBase);
27  		}
28  		return DocIdSet.EMPTY;
29  	}
30  
31  	@Override
32  	public String toString(String field) {
33  		return "SingleDocIdFilter(" + luceneDocId + ")";
34  	}
35  
36  }
