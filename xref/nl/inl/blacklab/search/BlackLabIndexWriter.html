1   package nl.inl.blacklab.search;
2   
3   import java.io.File;
4   import java.io.IOException;
5   
6   import org.apache.lucene.analysis.Analyzer;
7   import org.apache.lucene.index.CorruptIndexException;
8   import org.apache.lucene.index.IndexWriter;
9   import org.apache.lucene.search.Query;
10  import org.apache.lucene.store.LockObtainFailedException;
11  
12  import nl.inl.blacklab.search.indexmetadata.AnnotatedField;
13  import nl.inl.blacklab.search.indexmetadata.Annotation;
14  import nl.inl.blacklab.search.indexmetadata.IndexMetadataWriter;
15  
16  public interface BlackLabIndexWriter extends BlackLabIndex {
17  
18      /**
19       * Call this to roll back any changes made to the index this session. Calling
20       * close() will automatically commit any changes. If you call this method, then
21       * call close(), no changes will be committed.
22       */
23      void rollback();
24  
25      /**
26       * Get information about the structure of the BlackLab index.
27       *
28       * @return the structure object
29       */
30      IndexMetadataWriter metadataWriter();
31  
32      IndexWriter openIndexWriter(File indexDir, boolean create, Analyzer useAnalyzer)
33              throws IOException, CorruptIndexException, LockObtainFailedException;
34  
35      IndexWriter writer();
36  
37      /**
38       * Deletes documents matching a query from the BlackLab index.
39       *
40       * This deletes the documents from the Lucene index, the forward indices and the
41       * content store(s).
42       * 
43       * @param q the query
44       */
45      void delete(Query q);
46  
47      /**
48       * Get existing annotation or create new one.
49       * 
50       * @param field field for which this annotation is
51       * @param annotName annotation name
52       * @return annotation
53       */
54      Annotation getOrCreateAnnotation(AnnotatedField field, String annotName);
55  
56      /**
57       * Is the indexer still open?
58       * 
59       * It can be closed unexpectedly if e.g. the GC overhead limit is exceeded.
60       * If that happened, we should stop indexing. 
61       * 
62       * @return true if the indexer was closed, false if not
63       */
64      boolean isOpen();
65  
66  }
