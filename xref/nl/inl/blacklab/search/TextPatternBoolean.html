16  package nl.inl.blacklab.search;
17  
18  import java.util.ArrayList;
19  import java.util.List;
20  
21  import org.apache.lucene.search.BooleanClause.Occur;
22  
23  /**
24   * A TextPattern that performs document-level combining of boolean clauses. Modeled after Lucene's
25   * BooleanQuery, each clause can get the MUST, SHOULD or MUST NOT label attached.
26   *
27   * MUST-clauses must all occur in the document. At least one of all SHOULD clauses must occur in the
28   * document. None of the MUST NOT clauses must occur in the document.
29   *
30   * For documents that satisfy these criteria, all the MUST and SHOULD hits are reported.
31   */
32  public class TextPatternBoolean extends TextPattern {
33  
34  	private List<TextPattern> must = new ArrayList<TextPattern>();
35  
36  	private List<TextPattern> should = new ArrayList<TextPattern>();
37  
38  	private List<TextPattern> mustNot = new ArrayList<TextPattern>();
39  
40  	public TextPatternBoolean(boolean disableCoord) {
41  		// do nothing (disableCoord is ignored as we don't use scoring)
42  	}
43  
44  	@Override
45  	public <T> T translate(TextPatternTranslator<T> translator, QueryExecutionContext context) {
46  
47  		// First, translate clauses into complete TextPattern. Then translate that.
48  		TextPattern translated = null;
49  
50  		// MUST and SHOULD
51  		TextPattern tpMust = null, tpShould = null;
52  		if (must.size() > 0) {
53  			// Build a TextPattern that combines all MUST queries with AND
54  			tpMust = new TextPatternDocLevelAnd(must.toArray(new TextPattern[0]));
55  		}
56  		if (should.size() > 0) {
57  			// Build a TextPattern that combines all SHOULD queries with OR
58  			tpShould = new TextPatternOr(should.toArray(new TextPattern[0]));
59  		}
60  		if (tpMust == null && tpShould == null)
61  			throw new RuntimeException("Query must contain included terms (cannot just exclude)");
62  		else if (tpMust != null && tpShould != null) {
63  			// Require all MUST queries and at least one of the SHOULD queries
64  			translated = new TextPatternDocLevelAnd(tpMust, tpShould);
65  		} else {
66  			translated = tpMust == null ? tpShould : tpMust;
67  		}
68  
69  		// MUST NOT
70  		if (mustNot.size() > 0) {
71  			// Build a TextPattern that combines all MUST NOT queries with AND
72  			translated = new TextPatternDocLevelAndNot(translated, new TextPatternDocLevelAnd(
73  					mustNot.toArray(new TextPattern[0])));
74  		}
75  
76  		return translated.translate(translator, context);
77  	}
78  
79  	public void add(TextPattern query, Occur occur) {
80  		add(new TPBooleanClause(query, occur));
81  	}
82  
83  	public void add(TPBooleanClause clause) {
84  		switch (clause.getOccur()) {
85  		case MUST:
86  			must.add(clause.getQuery());
87  			break;
88  		case SHOULD:
89  			should.add(clause.getQuery());
90  			break;
91  		case MUST_NOT:
92  			mustNot.add(clause.getQuery());
93  			break;
94  		}
95  	}
96  
97  	@Override
98  	public Object clone() {
99  		try {
100 			TextPatternBoolean clone = (TextPatternBoolean) super.clone();
101 
102 			// copy list of children so we can modify it independently
103 			clone.must = new ArrayList<TextPattern>(must);
104 			clone.should = new ArrayList<TextPattern>(should);
105 			clone.mustNot = new ArrayList<TextPattern>(mustNot);
106 
107 			return clone;
108 		} catch (CloneNotSupportedException e) {
109 			throw new RuntimeException("Clone not supported: " + e.getMessage());
110 		}
111 	}

114 }
