1   package nl.inl.blacklab.analysis;
2   
3   import org.apache.lucene.analysis.util.CharTokenizer;
4   import org.apache.lucene.util.AttributeFactory;
5   
6   /**
7    * A simple tokenizer for Dutch texts. Basically the whitespace tokenizer
8    * with a few exceptional punctuation characters that are included in tokens.
9    *
10   * These are the exceptions:
11   * * apostrophes (e.g. zo'n, da's: apostrophes at the beginning or end of a token will be filtered out later)
12   * * dashes (e.g. ex-man, multi-)
13   * * periods (e.g. a.u.b., N.B.; will be filtered out later)
14   * * parens and brackets (e.g. bel(len), (pre)cursor; will be filtered out later)
15   */
16  public class BLDutchTokenizer extends CharTokenizer {
17  
18  	public BLDutchTokenizer() {
19  		super();
20  	}
21  
22  	public BLDutchTokenizer(AttributeFactory factory) {
23  		super(factory);
24  	}
25  
26  	@Override
27  	protected boolean isTokenChar(int c) {
28  
29  		if (Character.isWhitespace(c))
30  			return false;
31  
32  		if (Character.isLetter(c) || Character.isDigit(c))
33  			return true;
34  
35  		if (c == '\'' || c == '-' || c == '.' || c == '(' || c == '[' || c == ')' || c == ']') {
36  			/* These are exceptions to the rule that non-letters are not token chars.
37  			 *
38  			 * An example of each:
39  			 * * zo'n, da's     (apostrophes at the beginning or end of a token will be filtered out later)
40  			 * * ex-man, multi-
41  			 * * a.u.b., N.B.   (periods will be filtered out later)
42  			 * * bel(len),      (pre)cursor (parens and brackets will be filtered out later)
43  			 */
44  			return true;
45  		}
46  
47  		// Some non-exceptional punctuation character. Not part of the token.
48  		return false;
49  	}
50  
51  }
