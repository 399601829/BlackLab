1   package nl.inl.blacklab.analysis;
2   
3   import org.apache.lucene.analysis.util.CharTokenizer;
4   import org.apache.lucene.util.AttributeFactory;
5   
6   /**
7    * A simple tokenizer for Dutch texts. Basically the whitespace tokenizer with a
8    * few exceptional punctuation characters that are included in tokens.
9    *
10   * These are the exceptions: * apostrophes (e.g. zo'n, da's: apostrophes at the
11   * beginning or end of a token will be filtered out later) * dashes (e.g.
12   * ex-man, multi-) * periods (e.g. a.u.b., N.B.; will be filtered out later) *
13   * parens and brackets (e.g. bel(len), (pre)cursor; will be filtered out later)
14   */
15  public class BLDutchTokenizer extends CharTokenizer {
16  
17      public BLDutchTokenizer() {
18          super();
19      }
20  
21      public BLDutchTokenizer(AttributeFactory factory) {
22          super(factory);
23      }
24  
25      @Override
26      protected boolean isTokenChar(int c) {
27  
28          if (Character.isWhitespace(c))
29              return false;
30  
31          if (Character.isLetter(c) || Character.isDigit(c))
32              return true;
33  
34          /* These are exceptions to the rule that non-letters are not token chars.
35          *
36          * An example of each:
37          * 
38          * <ul>
39          * <li>zo'n, da's     (apostrophes at the beginning or end of a token will be filtered out later)
40          * <li>ex-man, multi-
41          * <li>a.u.b., N.B.   (periods will be filtered out later)
42          * <li>bel(len),      (pre)cursor (parens and brackets will be filtered out later)
43          * </ul>
44          * 
45          * Any other characters is some non-exceptional punctuation character, so not part of the token.
46          */
47          return c == '\'' || c == '-' || c == '.' || c == '(' || c == '[' || c == ')' || c == ']';
48      }
49  
50  }
