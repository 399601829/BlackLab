52  	static Map<String, String> indexerParam = new TreeMap<>();
53  
54  	public static void main(String[] args) throws Exception {
55  		// If the current directory contains indexer.properties, read it
56  		File propFile = new File(".", "indexer.properties");
57  		if (propFile.canRead())
58  			readParametersFromPropertiesFile(propFile);
59  
60  		// Parse command line
61  		int maxDocsToIndex = 0;
62  		File indexDir = null, inputDir = null;
63  		String glob = "*";
64  		String docFormat = null;
65  		boolean createNewIndex = false;
66  		String command = "";
67  		Set<String> commands = new HashSet<>(Arrays.asList("add", "create", "delete"));
68  		boolean addingFiles = true;
69  		String deleteQuery = null;
70  		boolean useThreads = false;
71  		List<File> linkedFileDirs = new ArrayList<>();
72  		for (int i = 0; i < args.length; i++) {
73  			String arg = args[i].trim();
74  			if (arg.startsWith("---")) {
75  				String name = arg.substring(3);
76  				if (i + 1 == args.length) {
77  					System.err.println("Passing parameter to indexer: argument needed!");
78  					usage();
79  					return;
80  				}
81  				i++;
82  				String value = args[i];
83  				indexerParam.put(name, value);
84  			} else if (arg.startsWith("--")) {
85  				String name = arg.substring(2);
86  				switch(name) {
87  				case "threads":
88  					useThreads = true;
89  					break;
90  				case "nothreads":
91  					useThreads = false;
92  					break;
93  				case "linked-file-dir":
94                      if (i + 1 == args.length) {
95                          System.err.println("--linked-file-dir option needs argument");
96                          usage();
97                          return;
98                      }
99  				    linkedFileDirs.add(new File(args[i + 1]));
100 				    i++;
101 				    break;
102 				case "maxdocs":
103 					if (i + 1 == args.length) {
104 						System.err.println("--maxdocs option needs argument");
105 						usage();
106 						return;
107 					}
108 					try {
109 						maxDocsToIndex = Integer.parseInt(args[i + 1]);
110 						i++;
111 					} catch (NumberFormatException e) {
112 						System.err.println("--maxdocs option needs integer argument");
113 						usage();
114 						return;
115 					}
116 					break;
117 				case "create":
118 					System.err.println("Option --create is deprecated; use create command (--help for details)");
119 					createNewIndex = true;
120 					break;
121 				case "indexparam":
122 					if (i + 1 == args.length) {
123 						System.err.println("--indexparam option needs argument");
124 						usage();
125 						return;
126 					}
127 					propFile = new File(args[i + 1]);
128 					if (!propFile.canRead()) {
129 						System.err.println("Cannot read " + propFile);
130 						usage();
131 						return;
132 					}
133 					readParametersFromPropertiesFile(propFile);
134 					i++;
135 					break;
136 				case "help":
137 					usage();
138 					return;
139 				default: {
140 					System.err.println("Unknown option --" + name);
141 					usage();
142 					return;
143 				}
144 				}
145 			} else {
146 				if (command.length() == 0 && commands.contains(arg)) {
147 					command = arg;
148 					addingFiles = command.equals("add") || command.equals("create");
149 				} else if (indexDir == null) {
150 					indexDir = new File(arg);
151 				} else if (addingFiles && inputDir == null) {
152 					if (arg.startsWith("\"") && arg.endsWith("\"")) {
153 						// Trim off extra quotes needed to pass file glob to
154 						// Windows JVM.
155 						arg = arg.substring(1, arg.length() - 1);
156 					}
157 					if (arg.contains("*") || arg.contains("?") || new File(arg).isFile()) {
158 						// Contains file glob. Separate the two components.
159 						int n = arg.lastIndexOf('/', arg.length() - 2);
160 						if (n < 0)
161 							n = arg.lastIndexOf('\\', arg.length() - 2);
162 						if (n < 0) {
163 							glob = arg;
164 							inputDir = new File(".");
165 						} else {
166 							glob = arg.substring(n + 1);
167 							inputDir = new File(arg.substring(0, n));
168 						}
169 					} else {
170 						inputDir = new File(arg);
171 					}
172 				} else if (addingFiles && docFormat == null) {
173 					docFormat = arg;
174 				} else if (command.equals("delete") && deleteQuery == null) {
175 					deleteQuery = arg;
176 				} else {
177 					System.err.println("Too many arguments!");
178 					usage();
179 					return;
180 				}
181 			}
182 		}
183 		if (indexDir == null) {
184 			System.err.println("No index dir given.");
185 			usage();
186 			return;
187 		}
188 
189 		// Check the command
190 		if (command.length() == 0) {
191 			System.err.println("No command specified; specify 'create' or 'add'. (--help for details)");
192 			usage();
193 			return;
194 			//System.err.println("No command specified; assuming \"add\" (--help for details)");
195 			//command = "add";
196 		}
197 		if (command.equals("delete")) {
198 			commandDelete(indexDir, deleteQuery);
199 			return;
200 		}
201 		if (command.equals("create"))
202 			createNewIndex = true;
203 
204 		// We're adding files. Do we have an input dir/file and file format name?
205 		if (inputDir == null) {
206 			System.err.println("No input dir given.");
207 			usage();
208 			return;
209 		}
210 
211 
212 		// Init log4j
213 		LogUtil.setupBasicLoggingConfig();
214 
215 		File indexDirParent = indexDir.getAbsoluteFile().getParentFile();
216 		File inputDirParent = inputDir.getAbsoluteFile().getParentFile();
217 		List<File> dirs = new ArrayList<>(Arrays.asList(new File("."), inputDir, indexDir));
218 		if (inputDirParent != null)
219 		    dirs.add(2, inputDirParent);
220 		if (indexDirParent != null && !dirs.contains(indexDirParent))
221 			dirs.add(indexDirParent);
222 		propFile = FileUtil.findFile(dirs, "indexer", Arrays.asList("properties"));
223 		if (propFile != null && propFile.canRead())
224 			readParametersFromPropertiesFile(propFile);
225 
226 		File indexTemplateFile = null;
227 		if (createNewIndex) {
228 			indexTemplateFile = FileUtil.findFile(dirs, "indextemplate", Arrays.asList("json", "yaml", "yml"));
229 		}
230 
231 		String op = createNewIndex ? "Creating new" : "Appending to";
232 		String strGlob = File.separator;
233 		if (glob != null && glob.length() > 0 && !glob.equals("*")) {
234 			strGlob += glob;
235 		}
236 		System.out.println(op + " index in " + indexDir + File.separator + " from " + inputDir + strGlob +
237 		                   (docFormat != null ? " (using format " + docFormat + ")" : "(using autodetected format)"));
238 		if (!indexerParam.isEmpty()) {
239 			System.out.println("Indexer parameters:");
240 			for (Map.Entry<String,String> e: indexerParam.entrySet()) {
241 				System.out.println("  " + e.getKey() + ": " + e.getValue());
242 			}
243 		}
244 
245 		// Make sure BlackLab can find our format configuration files
246 		// (by default, it will already look in $BLACKLAB_CONFIG_DIR/formats, $HOME/.blacklab/formats
247 		//  and /etc/blacklab/formats, but we also want it to look in the current dir, the input dir,
248 		//  and the parent(s) of the input and index dirs)
249 		File currentWorkingDir = new File(System.getProperty("user.dir"));
250         List<File> formatDirs = new ArrayList<>(Arrays.asList(currentWorkingDir, inputDirParent, inputDir));
251 		if (!formatDirs.contains(indexDirParent))
252 		    formatDirs.add(indexDirParent);
253 
254 		DocumentFormats.registerFormatsInDirectories(formatDirs);
255 
256 		if (docFormat == null) {
257 			System.err.println("No document format given; trying to detect it from the index...");
258 		} else if (docFormat.equals("teip4")) {
259 			System.err.println("'teip4' is deprecated, use 'tei' for either TEI P4 or P5.");
260 			docFormat = "tei";
261 		}
262 
263 		// Create the indexer and index the files
264         DownloadCache.setDownloadAllowed(true); // allow downloading linked (metadata) documents
265 		if (!createNewIndex || indexTemplateFile == null || !indexTemplateFile.canRead()) {
266 			indexTemplateFile = null;
267 		}
268 		Indexer indexer;
269 		try {
270 			indexer = new Indexer(indexDir, createNewIndex, docFormat, indexTemplateFile);
271 			if (useThreads)
272 				indexer.setUseThreads(true);
273 		} catch (DocumentFormatException e1) {
274 			System.err.println(e1.getMessage());
275 			System.err.println("Please specify a correct format on the command line.");
276 			usage();
277 			return;
278 		}
279 		if (createNewIndex)
280 			indexer.getSearcher().getIndexStructure().setDocumentFormat(docFormat);
281 		indexer.setIndexerParam(indexerParam);
282 		if (maxDocsToIndex > 0)
283 			indexer.setMaxNumberOfDocsToIndex(maxDocsToIndex);
284         indexer.setLinkedFileDirs(linkedFileDirs);
285 		try {
286 			if (glob.contains("*") || glob.contains("?")) {
287 				// Real wildcard glob
288 				indexer.index(inputDir, glob);
289 			} else {
290 				// Single file.
291 				indexer.index(new File(inputDir, glob), "*");
292 			}
293 		} catch (Exception e) {
294 			System.err.println("An error occurred, aborting indexing (changes will be rolled back). Error details follow.");
295 			e.printStackTrace();
296 			indexer.rollback();
297 		} finally {
298 			// Close the index.
299 			indexer.close();
300 		}
301 	}
302 
303 	private static void readParametersFromPropertiesFile(File propFile) {
304 		Properties p = readPropertiesFromFile(propFile);
305 		for (Map.Entry<Object, Object> e: p.entrySet()) {
306 			indexerParam.put(e.getKey().toString(), e.getValue().toString());
307 		}
308 	}
309 
310 	private static void commandDelete(File indexDir, String deleteQuery) throws IOException,
311 			org.apache.lucene.queryparser.classic.ParseException, CorruptIndexException {
312 		if (deleteQuery == null) {
313 			System.err.println("No delete query given.");
314 			usage();
315 			return;
316 		}
317 		Searcher searcher = Searcher.openForWriting(indexDir, false);
318 		try {
319 			System.out.println("Doing delete: " + deleteQuery);
320 			searcher.delete(LuceneUtil.parseLuceneQuery(deleteQuery, searcher.getAnalyzer(), null));
321 		} finally {
322 			searcher.close();
323 		}
324 	}

326 	private static void usage() {
327 		System.err.flush();
328 		System.out.flush();
329 		System.out
330 				.println("Usage:\n"
331 						+ "  IndexTool {add|create} [options] <indexdir> <inputdir> <format>\n"
332 						+ "  IndexTool delete <indexdir> <filterQuery>\n"
333 						+ "\n"
334 						+ "Options:\n"
335 						+ "  --maxdocs <n>          Stop after indexing <n> documents\n"
336                         + "  --linked-file-dir <d>  Look in directory <d> for linked (e.g. metadata) files\n"
337                         + "  --[no]threads          Enable/disable multithreaded indexing (experimental)\n"
338                         + "                         Currently disabled by default. Once properly tested,\n"
339                         + "                         we will change the default to enabled.\n"
340                         + "\n"
341                         + "Deprecated options (not needed anymore with .yaml format configs):\n"
342 						+ "  --indexparam <file>    Read properties file with parameters for DocIndexer\n"
343 						+ "                         (NOTE: even without this option, if the current\n"
344 						+ "                         directory, the input or index directory (or its parent)\n"
345 						+ "                         contain a file named indexer.properties, these are passed\n"
346 						+ "                         to the indexer)\n"
347 						+ "  ---<name> <value>      Pass parameter to DocIndexer class\n"
348 						+ "  ---meta-<name> <value> Add an extra metadata field to documents indexed.\n"
349 						+ "                         You can also add a property named meta-<name> to your\n"
350 						+ "                         indexer.properties file. This field is stored untokenized.\n"
351 						+ "\n"
352 						+ "Available input format configurations:");
353 		for (Format format: DocumentFormats.getFormats()) {
354             String name = format.getId();
355 		    String displayName = format.getDisplayName();
356 		    String desc = format.getDescription();
357 		    String url = format.getHelpUrl();
358 		    if (!url.isEmpty())
359 		        url = "\n      (see " + url + ")";
360             if (displayName.length() > 0)
361                 displayName = " (" + displayName + ")";
362             if (desc.length() > 0) {
363                 desc = "\n      " + StringUtil.join(StringUtil.wrap(desc, 75), "\n      ");
364             }
365 			System.out.println("  " + name + displayName + desc + url);
366 		}
367 	}

369 	/**
370 	 * Read Properties from the specified file
371 	 *
372 	 * @param file
373 	 *            the file to read
374 	 * @return the Properties read
375 	 */
376 	public static Properties readPropertiesFromFile(File file) {
377 		try {
378 			if (!file.isFile()) {
379 				throw new IllegalArgumentException("Property file " + file.getCanonicalPath()
380 						+ " does not exist or is not a regular file!");
381 			}
382 
383 			try (Reader in = new BufferedReader(new InputStreamReader(new FileInputStream(file), "iso-8859-1"))) {
384 				Properties properties = new Properties();
385 				properties.load(in);
386 				return properties;
387 			}
388 		} catch (Exception e) {
389 			throw ExUtil.wrapRuntimeException(e);
390 		}
391 	}
