8   /**
9    * Retrieves and displays a document from a BlackLab content store.
10   */
11  public class GetDocFromContentStore {
12  
13  	public static void main(String[] args) throws Exception {
14  
15  		if (args.length != 2) {
16  			System.err.println("Usage: GetDocFromContentStore <contentStoreDir> <docId>");
17  			return;
18  		}
19  
20  		File csDir = new File(args[0]);
21  		int id = Integer.parseInt(args[1]);
22  
23  		ContentStore cs = new ContentStoreDirZip(csDir);
24  		String content = cs.retrieve(id);
25  		System.out.println(content);
26  	}
27  }
