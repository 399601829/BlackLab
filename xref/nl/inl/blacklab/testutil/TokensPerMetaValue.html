20  /**
21   * Determine the number of tokens in the subcorpus defined by each of the
22   * metadatafield values. (Only for those metadata fields that have a limited
23   * number of values, all of which were captured in the index metadata file).
24   */
25  public class TokensPerMetaValue {
26  
27      public static void main(String[] args) throws ParseException, ErrorOpeningIndex {
28  
29          String indexDir = "/home/jan/blacklab/gysseling/index";
30          if (args.length >= 1)
31              indexDir = args[0];
32          String annotatedFieldName = "contents";
33          if (args.length >= 2)
34              annotatedFieldName = args[1];
35  
36          try (BlackLabIndex index = BlackLab.open(new File(indexDir))) {
37              // Loop over all metadata fields
38              IndexMetadata indexMetadata = index.metadata();
39              System.out.println("field\tvalue\tnumberOfDocs\tnumberOfTokens");
40              for (MetadataField field: indexMetadata.metadataFields()) {
41                  // Check if this field has only a few values
42                  if (field.isValueListComplete().equals(ValueListComplete.YES)) {
43                      // Loop over the values
44                      for (Map.Entry<String, Integer> entry : field.valueDistribution().entrySet()) {
45                          // Determine token count for this value
46                          Query filter = LuceneUtil.parseLuceneQuery("\"" + entry.getKey().toLowerCase() + "\"",
47                                  new BLDutchAnalyzer(), field.name());
48                          DocResults docs = index.queryDocuments(filter);
49                          int totalNumberOfTokens = docs.intSum(new DocPropertyAnnotatedFieldLength(index, annotatedFieldName));
50                          System.out.println(field.name() + "\t" + entry.getKey() + "\t" + entry.getValue() + "\t"
51                                  + totalNumberOfTokens);
52                      }
53                  }
54              }
55          }
56      }
57  }
