18  /**
19   * Determine the number of tokens in the subcorpus defined by each of the metadatafield values.
20   * (Only for those metadata fields that have a limited number of values, all of which were captured
21   * in the index metadata file).
22   */
23  public class TokensPerMetaValue {
24  
25  	public static void main(String[] args) throws IOException, ParseException {
26  
27  		String indexDir = "/home/jan/blacklab/gysseling/index";
28  		if (args.length >= 1)
29  			indexDir = args[0];
30  		String complexFieldName = "contents";
31  		if (args.length >= 2)
32  			complexFieldName = args[1];
33  
34  		Searcher searcher = Searcher.open(new File(indexDir));
35  		try {
36  			// Loop over all metadata fields
37  			IndexStructure indexStructure = searcher.getIndexStructure();
38  			System.out.println("field\tvalue\tnumberOfDocs\tnumberOfTokens");
39  			for (String metaFieldName: indexStructure.getMetadataFields()) {
40  				// Check if this field has only a few values
41  				MetadataFieldDesc fd = indexStructure.getMetadataFieldDesc(metaFieldName);
42  				if (fd.isValueListComplete()) {
43  					// Loop over the values
44  					for (Map.Entry<String, Integer> entry: fd.getValueDistribution().entrySet()) {
45  						// Determine token count for this value
46  						String fieldName = fd.getName();
47  						Query filter = LuceneUtil.parseLuceneQuery("\"" + entry.getKey().toLowerCase() + "\"", new BLDutchAnalyzer(), fieldName);
48  						DocResults docs = searcher.queryDocuments(filter);
49  						int totalNumberOfTokens = docs.intSum(new DocPropertyComplexFieldLength(complexFieldName));
50  						System.out.println(fieldName + "\t" + entry.getKey() + "\t" + entry.getValue() + "\t" + totalNumberOfTokens);
51  					}
52  				}
53  			}
54  		} finally {
55  			searcher.close();
56  		}
57  	}
58  }
