27      public static void main(String[] args) throws ErrorOpeningIndex, InvalidQuery {
28          if (args.length == 0) {
29              System.err.println("Please specify index directory.");
30              System.exit(1);
31          }
32          File indexDir = new File(args[0]);
33          System.out.println("Opening index " + indexDir + "...");
34          try (BlackLabIndex index = BlackLab.open(indexDir)) {
35                  
36              FutureSearchResultCache cache = new FutureSearchResultCache();
37              cache.setTrace(true);
38              index.setCache(cache); //new SearchCacheDebug());
39              
40              String cqlLemmaSchip = "[lemma=\"schip\"]";
41              Annotation annotLemma = index.mainAnnotatedField().annotation("lemma");
42              MetadataField titleField = index.metadata().metadataFields().special("title");
43              
44              System.out.println("\nFirst 20 hits for 'schip':");
45              Hits hits = index.search()
46                      .find(cqlLemmaSchip, null, index.searchSettings())
47                      .window(0, 20)
48                      .execute();
49              Kwics kwics = hits.kwics(null);
50              Annotation word = hits.field().annotation("word");
51              hits.forEach(hit -> {
52                  Kwic kwic = kwics.get(hit);
53                  String left = StringUtils.join(kwic.left(word), " ");
54                  String match = StringUtils.join(kwic.match(word), " ");
55                  String right = StringUtils.join(kwic.right(word), " ");
56                  System.out.println("- " + left + " [" + match + "] " + right);
57              });
58  
59              System.out.println("\nFirst 10 document results for 'schip':");
60              index.search()
61                      .find(cqlLemmaSchip, null, index.searchSettings())
62                      .docs(3)
63                      .window(0, 10)
64                      .execute()
65                      .forEach(doc -> {
66                          Document document = doc.identity().luceneDoc();
67                          String title = document.get(titleField.name());
68                          System.out.println("- " + title + " (" + doc + ")"); 
69                      });
70              System.out.flush();
71  
72              System.out.println("\nFirst 10 document results for title:test :");
73              index.search()
74                      .find(new TermQuery(new Term("title", "test")))
75                      .window(0, 10)
76                      .execute()
77                      .forEach(doc -> {
78                          Document document = doc.identity().luceneDoc();
79                          String title = document.get(titleField.name());
80                          System.out.println("- " + title + " (" + doc + ")"); 
81                      });
82              System.out.flush();
83  
84              System.out.println("\nFirst 10 document results for title:test :");
85              index.search()
86                      .find(new TermQuery(new Term("title", "test")))
87                      .window(0, 10)
88                      .execute()
89                      .forEach(doc -> {
90                          Document document = doc.identity().luceneDoc();
91                          String title = document.get(titleField.name());
92                          System.out.println("- " + title + " (" + doc + ")"); 
93                      });
94              System.out.flush();
95  
96              System.out.println("\nCount number of hits for 'schip': " + index
97                      .search()
98                      .find(cqlLemmaSchip, null, index.searchSettings())
99                      .count()
100                     .execute()
101                     .processedTotal());
102 
103             System.out.println("\nCount different spellings for 'schip': ");
104             index
105                     .search()
106                     .find(cqlLemmaSchip, null, index.searchSettings())
107                     .group(new HitPropertyHitText(index), 3)
108                     .execute()
109                     .forEach(group -> {
110                         System.out.println("- " + group.identity() + " (stored " + group.numberOfStoredResults() + " of " + group.size() + ") ");
111                     });
112 
113             System.out.println("\nCollocations for 'waterval': ");
114             TermFrequencyList colls = index.search()
115                     .find("[lemma=\"waterval\"]", null, index.searchSettings())
116                     .collocations(annotLemma, ContextSize.get(10), MatchSensitivity.INSENSITIVE)
117                     //.window(0, 10)
118                     .execute();
119             for (int i = 0; i < 10 && i < colls.size(); i++) {
120                 System.out.println("- " + colls.get(i));
121             }
122         }
123     }
