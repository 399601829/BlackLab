1   package nl.inl.blacklab.indexers.preprocess;
2   
3   import java.io.InputStream;
4   import java.io.OutputStream;
5   import java.io.PushbackInputStream;
6   import java.nio.charset.Charset;
7   import java.util.Set;
8   
9   public interface ConvertPlugin extends Plugin {
10  
11  	/**
12  	 * Get a list of file extensions or formats (not strictly formatIdentifiers, though there might be overlap) that this converter can convert from.
13  	 * Is purely for descriptive purposes, and has no exact relation to the strings passed into {@link ConvertPlugin#perform(InputStream, Charset, String, OutputStream)},
14  	 * so exact spelling, punctuation or capitalization is not of technical importance.
15  	 *
16  	 * This overlaps somewhat with the purpose of getDescription() but it's important we have a consistent way of describing converter capabilties.
17  	 *
18  	 * @return the list of formats and file extensions supported by this converter.
19  	 */
20  	public Set<String> getInputFormats();
21  
22  	/**
23  	 * Get the name of the format this plugin will convert files into.
24  	 * Contrary to {@link ConvertPlugin#getInputFormats()}, this format must match a format accepted by a {@link TagPlugin}.
25  	 * Should be as descriptive as possible (e.g. just "xml" is not exact enough,
26  	 * because xml can mean different things based on the contents of the file, so instead use "folia" or "tei", or whatever contents the file happens to have).
27  	 *
28  	 * @return the format this convertplugin outputs.
29  	 */
30  	public String getOutputFormat();
31  
32  	/**
33  	 * Can this converter convert this file
34  	 *
35  	 * @param is stream containing a pushback buffer of at least 251 characters
36  	 * @param cs (optional) charset of the inputstream, if this is a text (non-binary) file type
37  	 * @param inputFormat
38  	 * @return true if this file can be converted into this plugin's outputFormat
39  	 */
40  	public boolean canConvert(PushbackInputStream is, Charset cs, String inputFormat);
41  
42  	/**
43  	 * Perform on a text file.
44  	 *
45  	 * @param is input. Should not be closed by the implementation.
46  	 * @param cs as inputFormat, but for the charset of the inputStream, not always meaningful, but required for some implementations that transform textual data.
47  	 * @param inputFormat arbitrary string describing input data, can be file extension, or more semantic. Usage may vary, so acceptable values must be coordinated between callers and implementations.
48  	 * @param os output. Should not be closed by the implementation.
49  	 * @throws PluginException
50  	 */
51  	public void perform(InputStream is, Charset cs, String inputFormat, OutputStream os) throws PluginException;
52  }
