1   package nl.inl.blacklab.indexers.preprocess;
2   
3   import java.io.Reader;
4   import java.io.Writer;
5   
6   import nl.inl.blacklab.exceptions.PluginException;
7   import nl.inl.blacklab.index.DocIndexer;
8   
9   public interface TagPlugin extends Plugin {
10  
11      /**
12       * Get the name of the format this plugin can tag. Should be as descriptive as
13       * possible (e.g. just "xml" is not exact enough, because xml can mean different
14       * things based on the contents of the file, so instead use "folia" or "tei", or
15       * whatever contents the file happens to have). see
16       * {@link ConvertPlugin#getOutputFormat()}.
17       *
18       * @return the format this convertplugin outputs.
19       */
20      String getInputFormat();
21  
22      /**
23       * Get the name of the format this plugin tags into. This should match a
24       * {@link nl.inl.blacklab.index.DocIndexerFactory.Format} so that the tagged
25       * data can then be indexed by a {@link DocIndexer}
26       *
27       * @return the formatIdentifier
28       */
29      String getOutputFormatIdentifier();
30  
31      /**
32       * Unfortunate side-effect of docIndexers requiring a filename to do their work.
33       * 
34       * @param inputFileName
35       * @return a valid file name for an indexable file of the type as returned by
36       *         {@link TagPlugin#getOutputFormatIdentifier()}
37       */
38      String getOutputFileName(String inputFileName);
39  
40      /**
41       * Perform on a text file.
42       *
43       * @param reader input. Should not be closed by the implementation.
44       * @param writer output. Should not be closed by the implementation.
45       * @throws PluginException
46       */
47      void perform(Reader reader, Writer writer) throws PluginException;
48  }
