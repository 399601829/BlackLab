30      protected static String replaceDollarRefs(String pattern, List<String> replacements) {
31          if (pattern != null) {
32              int i = 1;
33              for (String replacement : replacements) {
34                  pattern = pattern.replace("$" + i, replacement);
35                  i++;
36              }
37          }
38          return pattern;
39      }

41      public static DocIndexerConfig fromConfig(ConfigInputFormat config) {
42          DocIndexerConfig docIndexer=null;
43          switch (config.getFileType()) {
44          case XML:
45              for (ConfigInputFormat.FileTypeOption fto : ConfigInputFormat.FileTypeOption.fromConfig(config, ConfigInputFormat.FileType.XML)) {
46                  if (fto== ConfigInputFormat.FileTypeOption.SAXONICA) {
47                      docIndexer=new DocIndexerSaxon();
48                      break;
49                  }
50              }
51              if (docIndexer==null) {
52                  docIndexer=new DocIndexerXPath();
53              }
54              break;
55          case TABULAR:
56              docIndexer = new DocIndexerTabular();
57              break;
58          case TEXT:
59              docIndexer = new DocIndexerPlainText();
60              break;
61          case CHAT:
62              docIndexer = new DocIndexerChat();
63              break;
64          default:
65              throw new InvalidInputFormatConfig(
66                      "Unknown file type: " + config.getFileType() + " (use xml, tabular, text or chat)");
67          }
68  
69          docIndexer.setConfigInputFormat(config);
70  
71          if (config.getConvertPluginId() != null || config.getTagPluginId() != null) {
72              return new DocIndexerConvertAndTag(docIndexer, config);
73          } else {
74              return docIndexer;
75          }
76      }

103     @SuppressWarnings("deprecation")
104     protected void init() {
105         if (inited)
106             return;
107         inited = true;
108         setStoreDocuments(config.shouldStore());
109         for (ConfigAnnotatedField af : config.getAnnotatedFields().values()) {
110 
111             // Define the properties that make up our annotated field
112             if (af.isDummyForStoringLinkedDocuments())
113                 continue;
114             List<ConfigAnnotation> annotations = new ArrayList<>(af.getAnnotationsFlattened().values());
115             if (annotations.isEmpty())
116                 throw new InvalidInputFormatConfig("No annotations defined for field " + af.getName());
117             ConfigAnnotation mainAnnotation = annotations.get(0);
118             AnnotatedFieldWriter fieldWriter = new AnnotatedFieldWriter(af.getName(), mainAnnotation.getName(),
119                     getSensitivitySetting(mainAnnotation), false);
120             addAnnotatedField(fieldWriter);
121 
122             AnnotationWriter annotStartTag = fieldWriter.addAnnotation(null, AnnotatedFieldNameUtil.TAGS_ANNOT_NAME,
123                     getSensitivitySetting(AnnotatedFieldNameUtil.TAGS_ANNOT_NAME), true);
124             annotStartTag.setHasForwardIndex(false);
125 
126             // Create properties for the other annotations
127             for (int i = 1; i < annotations.size(); i++) {
128                 ConfigAnnotation annot = annotations.get(i);
129                 if (!annot.isForEach())
130                     fieldWriter.addAnnotation(annot, annot.getName(), getSensitivitySetting(annot), false);
131             }
132             for (ConfigStandoffAnnotations standoff : af.getStandoffAnnotations()) {
133                 for (ConfigAnnotation annot : standoff.getAnnotations().values()) {
134                     fieldWriter.addAnnotation(annot, annot.getName(), getSensitivitySetting(annot), false);
135                 }
136             }
137             if (!fieldWriter.hasAnnotation(AnnotatedFieldNameUtil.PUNCTUATION_ANNOT_NAME)) {
138                 // Hasn't been created yet. Create it now.
139                 fieldWriter.addAnnotation(null, AnnotatedFieldNameUtil.PUNCTUATION_ANNOT_NAME,
140                         getSensitivitySetting(AnnotatedFieldNameUtil.PUNCTUATION_ANNOT_NAME), false);
141             }
142             if (docWriter != null) {
143                 IndexMetadataImpl indexMetadata = (IndexMetadataImpl)docWriter.indexWriter().metadataWriter();
144                 indexMetadata.registerAnnotatedField(fieldWriter);
145             }
146 
147         }

160     protected String processString(String result, List<ConfigProcessStep> process, Map<String, String> mapValues) {
161         for (ConfigProcessStep step : process) {
162             String method = step.getMethod();
163             Map<String, String> param = step.getParam();
164             switch (method) {
165             case "replace":
166                 result = opReplace(result, param);
167                 break;
168             case "default":
169                 result = opDefault(result, param);
170                 break;
171             case "append":
172                 result = opAppend(result, param);
173                 break;
174             case "split":
175                 result = opSplit(result, param).get(0);
176                 break;
177             case "chatFormatAgeToMonths":
178                 result = opChatFormatAgeToMonths(result);
179                 break;
180             case "strip":
181                 result = opStrip(result, param);
182                 break;
183             case "parsePos":
184             {
185                 // Get individual feature out of a part of speech string like "NOU(gender=f,number=p)"
186                 String field = param.containsKey("field") ? param.get("field") : "_";
187                 result = opParsePartOfSpeech(result, field);
188                 break;
189             }
190             default:
191                 // In the future, we'll support user plugins here
192                 throw new UnsupportedOperationException("Unknown processing step method " + method);
193             }
194         }
195         if (mapValues != null && !mapValues.isEmpty()) {
196             // Finally, apply any value mappings specified.
197             String mappedResult = mapValues.get(result);
198             if (mappedResult != null)
199                 result = mappedResult;
200         }
201         return result;
202     }

204     /**
205      * process linked documents when configured. An xPath processor can be provided,
206      * it will retrieve information from the document to construct a path to a linked document.
207      * @param ld
208      * @param xpathProcessor
209      */
210     protected void processLinkedDocument( ConfigLinkedDocument ld, Function<String, String> xpathProcessor) {
211         // Resolve linkPaths to get the information needed to fetch the document
212         List<String> results = new ArrayList<>();
213         for (ConfigLinkValue linkValue : ld.getLinkValues()) {
214             String result = "";
215             String valuePath = linkValue.getValuePath();
216             String valueField = linkValue.getValueField();
217             if (valuePath != null) {
218                 // Resolve value using XPath
219                 result = xpathProcessor.apply(valuePath);
220                 if (result == null || result.isEmpty()) {
221                     switch (ld.getIfLinkPathMissing()) {
222                         case IGNORE:
223                             break;
224                         case WARN:
225                             docWriter.listener()
226                                     .warning("Link path " + valuePath + " not found in document " + documentName);
227                             break;
228                         case FAIL:
229                             throw new BlackLabRuntimeException("Link path " + valuePath + " not found in document " + documentName);
230                     }
231                 }
232             } else if (valueField != null) {
233                 // Fetch value from Lucene doc
234                 result = getMetadataField(valueField).get(0);
235             }
236             result = processString(result, linkValue.getProcess(), null);
237             results.add(result);
238         }
239 
240         // Substitute link path results in inputFile, pathInsideArchive and documentPath
241         String inputFile = replaceDollarRefs(ld.getInputFile(), results);
242         String pathInsideArchive = replaceDollarRefs(ld.getPathInsideArchive(), results);
243         String documentPath = replaceDollarRefs(ld.getDocumentPath(), results);
244 
245         try {
246             // Fetch and index the linked document
247             indexLinkedDocument(inputFile, pathInsideArchive, documentPath, ld.getInputFormatIdentifier(),
248                     ld.shouldStore() ? ld.getName() : null);
249         } catch (Exception e) {
250             String moreInfo = "(inputFile = " + inputFile;
251             if (pathInsideArchive != null)
252                 moreInfo += ", pathInsideArchive = " + pathInsideArchive;
253             if (documentPath != null)
254                 moreInfo += ", documentPath = " + documentPath;
255             moreInfo += ")";
256             switch (ld.getIfLinkPathMissing()) {
257                 case IGNORE:
258                 case WARN:
259                     docWriter.listener().warning("Could not find or parse linked document for " + documentName + moreInfo
260                             + ": " + e.getMessage());
261                     break;
262                 case FAIL:
263                     throw new BlackLabRuntimeException("Could not find or parse linked document for " + documentName + moreInfo, e);
264             }
265         }
266     }

268     protected List<String> processStringMultipleValues(String input, List<ConfigProcessStep> process, Map<String, String> mapValues) {
269         List<String> result = Arrays.asList(input);
270 
271         for (ConfigProcessStep step : process) {
272             String method = step.getMethod();
273             Map<String, String> param = step.getParam();
274 
275             switch (method) {
276             case "replace":
277                 for (int i = 0; i < result.size(); ++i) {
278                     result.set(i, opReplace(result.get(i), param));
279                 }
280                 break;
281             case "default":
282                 for (int i = 0; i < result.size(); ++i) {
283                     result.set(i, opDefault(result.get(i), param));
284                 }
285                 break;
286             case "append":
287                 for (int i = 0; i < result.size(); ++i) {
288                     result.set(i, opAppend(result.get(i), param));
289                 }
290                 break;
291             case "split":
292                 result = result.stream()
293                     .flatMap(r -> opSplit(r, param).stream())
294                     .collect(Collectors.toCollection(ArrayList<String>::new));
295                 break;
296             case "chatFormatAgeToMonths":
297                 for (int i = 0; i < result.size(); ++i) {
298                     result.set(i, opChatFormatAgeToMonths(result.get(i)));
299                 }
300                 break;
301             case "strip":
302                 for (int i = 0; i < result.size(); ++i) {
303                     result.set(i, opStrip(result.get(i), param));
304                 }
305                 break;
306             case "parsePos":
307             {
308                 // Get individual feature out of a part of speech string like "NOU(gender=f,number=p)"
309                 String field = param.containsKey("field") ? param.get("field") : "_";
310                 for (int i = 0; i < result.size(); ++i) {
311                     result.set(i, opParsePartOfSpeech(result.get(i), field));
312                 }
313                 break;
314             }
315             default:
316                 // In the future, we'll support user plugins here
317                 throw new UnsupportedOperationException("Unknown processing step method " + method);
318             }
319         }
320         if (mapValues != null && !mapValues.isEmpty()) {
321             // Finally, apply any value mappings specified.
322             for (int i = 0; i < result.size(); ++i) {
323                 String mappedResult = mapValues.get(result.get(i));
324                 if (mappedResult != null)
325                     result.set(i, mappedResult);
326             }
327         }
328         return result;
329     }

331     static String opParsePartOfSpeech(String result, String field) {
332         // Trim character/string from beginning and end
333         result = result.trim();
334         if (field.equals("_")) {
335             //  Get main pos: A(b=c,d=e) -> A
336             return result.replaceAll("^([^\\(]+)(\\s*\\(.*\\))?$", "$1");
337         } else {
338             //  Get feature: A(b=c,d=e) -> e  (if field == d)
339             String featuresString = result.replaceAll("^[^\\(]+(\\s*\\((.*)\\))?$", "$2");
340             return Arrays.stream(featuresString.split(","))
341                 .map(feat -> feat.split("="))
342                 .filter(featParts -> featParts[0].trim().equals(field))
343                 .map(featParts -> featParts[1].trim())
344                 .findFirst()
345                 .orElse("");
346         }
347     }

349     static String opStrip(String result, Map<String, String> param) {
350         // Trim character/string from beginning and end
351         String stripChars = param.containsKey("chars") ? param.get("chars") : " ";
352         result = StringUtils.strip(result, stripChars);
353         return result;
354     }

356     protected String opChatFormatAgeToMonths(String result) {
357         // 1;10.30 => 1 jaar, 10 maanden, 30 dagen => afgerond 23 maanden?
358         String[] parts = result.split("[;\\.]", 3);
359         int years = 0;
360         int months = 0;
361         int days = 0;
362         try {
363             years = Integer.parseInt(parts[0]);
364             months = parts.length <= 1 ? 0 : Integer.parseInt(parts[1]);
365             days = parts.length <= 2 ? 0 : Integer.parseInt(parts[2]);
366         } catch (NumberFormatException e) {
367             warn("action 'opChatFormatAgeToMonths': illegal value " + result);
368         }
369         return Integer.toString(years * 12 + months + (days > 14 ? 1 : 0) );
370     }

372     /**
373      * Split the result string on a separator and return one or all parts.
374      *
375      * @param result the string to split
376      * @param param
377      * <pre>
378      * - "separator" for the separator (defaults to ;),
379      * - "keep" for the keep index, accepts numbers or the special string "all" (defaults to -1)
380      *      if "keep" <= 0 returns the first part.
381      *      if "keep" > number of splits return empty string.
382      *      if "keep" == "all" return all parts
383      * </pre>
384      * @return
385      */
386     private List<String> opSplit(String result, Map<String, String> param) {
387         // Split on a separator regex and keep one or all parts (first part by default)
388         String separator = param.getOrDefault("separator", ";");
389         String keep = param.getOrDefault("keep", "-1").toLowerCase();
390         String[] parts = result.split(separator, -1);
391 
392         if (keep.equals("all")) {
393             return Arrays.asList(parts);
394         }
395 
396         int i = -1;
397         try { i = Integer.parseInt(keep); } catch (NumberFormatException e) {}
398         if (i < 0) {
399             warn("action 'split', parameter 'keep': must be at least 1");
400             i = 0;
401         }
402 
403         return Arrays.asList(i < parts.length ? parts[i] : "");
404     }

406     /**
407      * Appends a constant value, or the value of a metadata field to the result string.
408      *
409      * @param result the input string
410      * @param param
411      * <pre>
412      * - "separator" for the separator (defaults to " ")
413      * - "field" for the metadata field whose value will be appended
414      * - "value" for a constant value ("field" takes precedence if it exists)
415      * </pre>
416      * @return
417      */
418     private String opAppend(String result, Map<String, String> param) {
419         String separator = param.containsKey("separator") ? param.get("separator") : " ";
420         String field = param.get("field");
421         String value;
422         if (field != null)
423             value = StringUtils.join(getMetadataField(field), separator);
424         else
425             value = param.get("value");
426         if (value != null && value.length() > 0) {
427             if (result.length() > 0)
428                 result += separator;
429             result += value;
430         }
431         return result;
432     }

434     /**
435      * Optionally replace an empty result with a constant value, or the value of a metadata field.
436      *
437      * @param result the input string
438      * @param param
439      * <pre>
440      * - "field" for the metadata field whose value will be used
441      * - "separator" to join the metadata field if it contains multiple values (defaults to ;)
442      * - "value" for a constant value ("field" takes precedence if it exists)
443      * </pre>
444      * @return
445      */
446     private String opDefault(String result, Map<String, String> param) {
447         if (result.length() == 0) {
448             String field = param.get("field");
449             String value;
450             String sep = param.getOrDefault("separator", ";");
451             if (field != null)
452                 value = StringUtils.join(getMetadataField(field), sep);
453             else
454                 value = param.get("value");
455             if (value != null)
456                 result = value;
457         }
458         return result;
459     }

461     /**
462      * Perform a regex replace on result. Allows group references.
463      *
464      * @param result
465      * @param param
466      * <pre>
467      * - "find" for the regex
468      * - "replace" the replacement string
469      * </pre>
470      * @return
471      */
472     private static String opReplace(String result, Map<String, String> param) {
473         String find = param.get("find");
474         String replace = param.get("replace");
475         if (find == null || replace == null)
476             throw new InvalidInputFormatConfig("replace needs parameters find and replace");
477         try {
478             result = result.replaceAll(find, replace);
479         } catch (PatternSyntaxException e) {
480             throw new InvalidInputFormatConfig("Syntax error in replace regex: " + find);
481         }
482         return result;
483     }

485     /**
486      * If any processing steps were defined for this metadata field, apply them now.
487      *
488      * This is used for non-XML formats, where we don't actively seek out the
489      * metadata but encounter it as we go.
490      *
491      * @param name metadata field name
492      * @param value metadata field value
493      * @return processed value (or orifinal value if not found / no processing steps
494      *         defined)
495      */
496     protected String processMetadataValue(String name, String value) {
497         ConfigMetadataField f = config.getMetadataField(name);
498         if (f != null) {
499             value = processString(value, f.getProcess(), f.getMapValues());
500         }
501         return value;
