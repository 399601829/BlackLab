39      /** Tabular types we support */
40      enum Type {
41          CSV,
42          TSV;
43  
44          public static Type fromStringValue(String str) {
45              switch (str.toUpperCase()) {
46              case "TDF":
47                  return TSV;
48              case "EXCEL":
49                  return CSV;
50              }
51              return valueOf(str.toUpperCase());
52          }
53  
54          public String stringValue() {
55              return toString().toLowerCase();
56          }
57      }

59      /**
60       * Regex for recognizing open or close tag and capturing tag name and attributes
61       * part
62       */
63      final static Pattern REGEX_TAG = Pattern.compile("^\\s*<\\s*(/\\s*)?(\\w+)((\\b[^>]+)?)>\\s*$");
64  
65      /** Single- or double-quoted attribute in a tag */
66      static final Pattern REGEX_ATTR = Pattern.compile("\\b(\\w+)\\s*=\\s*(\"[^\"]*\"|'[^']*')");
67  
68      private static Map<String, String> getAttr(String group) {
69          if (group == null)
70              return Collections.emptyMap();
71          String strAttrDef = group.trim();
72          Matcher m = REGEX_ATTR.matcher(strAttrDef);
73          Map<String, String> attributes = new LinkedHashMap<>();
74          while (m.find()) {
75              String key = m.group(1);
76              String value = m.group(2);
77              value = value.substring(1, value.length() - 1); // chop quotes
78              attributes.put(key, value);
79          }
80          return attributes;
81      }

106     @Override
107     public void setConfigInputFormat(ConfigInputFormat config) {
108         if (config.getAnnotatedFields().size() > 1)
109             throw new InvalidInputFormatConfig("Tabular type can only have 1 annotated field");
110         super.setConfigInputFormat(config);
111         Map<String, String> opt = config.getFileTypeOptions();
112         Type type = opt.containsKey("type") ? Type.fromStringValue(opt.get("type")) : Type.CSV;
113         //ConfigTabularOptions tab = config.getTabularOptions();
114         switch (type) {
115         case TSV:
116             tabularFormat = CSVFormat.TDF;
117             break;
118         case CSV:
119             tabularFormat = CSVFormat.EXCEL;
120             break;
121         default:
122             throw new InvalidInputFormatConfig("Unknown tabular type " + opt.get("type") + " (use csv or tsv)");
123         }
124         if (opt.containsKey("columnNames") && opt.get("columnNames").equalsIgnoreCase("true"))
125             tabularFormat = tabularFormat.withFirstRecordAsHeader();
126         if (opt.containsKey("delimiter") && opt.get("delimiter").length() > 0)
127             tabularFormat = tabularFormat.withDelimiter(opt.get("delimiter").charAt(0));
128         if (opt.containsKey("quote") && opt.get("quote").length() > 0)
129             tabularFormat = tabularFormat.withQuote(opt.get("quote").charAt(0));
130         else
131             tabularFormat = tabularFormat.withQuote(null); // disable quotes altogether
132         allowSeparatorsAfterInlineTags = false;
133         if (opt.containsKey("allowSeparatorsAfterInlineTags")
134                 && opt.get("allowSeparatorsAfterInlineTags").equalsIgnoreCase("true"))
135             allowSeparatorsAfterInlineTags = true;
136         hasInlineTags = opt.containsKey("inlineTags") && opt.get("inlineTags").equalsIgnoreCase("true");
137         hasGlueTags = opt.containsKey("glueTags") && opt.get("glueTags").equalsIgnoreCase("true");
138         if (opt.containsKey("multipleValuesSeparator"))
139             multipleValuesSeparator = opt.get("multipleValuesSeparator");
140     }

157     @Override
158     public void setDocument(Reader reader) {
159         try {
160             inputReader = reader instanceof BufferedReader ? (BufferedReader) reader : new BufferedReader(reader);
161             records = tabularFormat.parse(inputReader);
162         } catch (IOException e) {
163             throw BlackLabRuntimeException.wrap(e);
164         }

177     @Override
178     public void index() throws MalformedInputFile, PluginException, IOException {
179         super.index();
180 
181         // If a documentPath was specified, look for that as the document tag
182         boolean lookForDocumentTags = !config.getDocumentPath().equals("/");
183 
184         if (!lookForDocumentTags)
185             startDocument();
186 
187         // Do we need to look for tags at all?
188         boolean lookForTags = !lookForDocumentTags || hasInlineTags || hasGlueTags;
189 
190         // Are we inside a document now?
191         boolean inDocument = !lookForDocumentTags;
192 
193         csvData = new StringBuilder();
194         try (CSVPrinter p = new CSVPrinter(csvData, tabularFormat)) {
195             // For the configured annotated field...
196             for (ConfigAnnotatedField annotatedField : config.getAnnotatedFields().values()) {
197                 setCurrentAnnotatedFieldName(annotatedField.getName());
198 
199                 // For each token position
200                 for (CSVRecord record : records) {
201                     if (record.size() == 0)
202                         continue; // skip empty lines
203 
204                     // If this format contains tags...
205                     if (lookForTags && (record.size() == 1 || allowSeparatorsAfterInlineTags)) {
206                         // Is this a tag line instead of a token line?
207                         String possibleTag = record.get(0);
208                         Matcher m = REGEX_TAG.matcher(possibleTag);
209                         if (m.find()) {
210                             // It's a document tag, an inline tag or a glue tag
211                             boolean isOpenTag = m.group(1) == null;
212                             String tagName = m.group(2);
213                             String rest = m.group(3).trim();
214                             boolean selfClosing = rest.endsWith("/");
215                             if (!isOpenTag && selfClosing)
216                                 throw new MalformedInputFile("Close tag must not also end with /: " + tagName);
217                             if (selfClosing)
218                                 rest = rest.substring(0, rest.length() - 1);
219                             Map<String, String> attributes = getAttr(rest);
220 
221                             if (lookForDocumentTags && tagName.equals(config.getDocumentPath())) {
222                                 // Document tag.
223                                 if (inDocument && isOpenTag)
224                                     throw new MalformedInputFile("Found document open tag inside document");
225                                 if (!inDocument && !isOpenTag)
226                                     throw new MalformedInputFile(
227                                             "Found document close tag outside of document");
228                                 if (isOpenTag) {
229                                     // Start a new document and add attributes as metadata fields
230                                     inDocument = true;
231                                     startDocument();
232                                     for (Entry<String, String> e : attributes.entrySet()) {
233                                         String value = processMetadataValue(e.getKey(), e.getValue());
234                                         addMetadataField(e.getKey(), value);
235                                     }
236                                 } else {
237                                     endDocument();
238                                     inDocument = false;
239                                 }
240                             } else if (hasGlueTags && tagName.equals(GLUE_TAG_NAME)) {
241                                 // Glue tag. Don't add default punctuation when adding next word.
242                                 if (!attributes.isEmpty())
243                                     warn("Glue tag has attributes: " + attributes.toString());
244                                 setPreventNextDefaultPunctuation();
245                             } else {
246                                 inlineTag(tagName, isOpenTag, attributes);
247                                 if (selfClosing)
248                                     inlineTag(tagName, !isOpenTag, null);
249                             }
250                             continue;
251                         }
252                     }
253 
254                     // It's a regular token
255                     beginWord();
256                     if (isStoreDocuments()) {
257                         p.printRecord(record);
258                     }
259 
260                     // For each annotation
261                     for (ConfigAnnotation annotation : annotatedField.getAnnotationsFlattened().values()) {
262                         // Either column number of name
263                         String value;
264                         if (annotation.isValuePathInteger()) {
265                             int i = annotation.getValuePathInt() - 1;
266                             if (i < record.size())
267                                 value = record.get(i);
268                             else
269                                 value = "";
270                         } else {
271                             if (record.isMapped(annotation.getValuePath()))
272                                 value = record.get(annotation.getValuePath());
273                             else
274                                 value = "";
275                         }
276                         value = processString(value, annotation.getProcess(), null);
277                         if (annotation.isMultipleValues()) {
278                             // Multiple values possible. Split on multipleValuesSeparator.
279                             boolean first = true;
280                             List<String> values = Arrays.asList(value.split(multipleValuesSeparator, -1));
281                             if (!annotation.isAllowDuplicateValues()) {
282                                 // Discard any duplicate values from the list
283                                 values = values.stream().distinct().collect(Collectors.toList());
284                             }
285                             for (String v : values) {
286                                 annotation(annotation.getName(), v, first ? 1 : 0, null);
287                                 first = false;
288                             }
289                         } else {
290                             // Single value.
291                             annotation(annotation.getName(), value, 1, null);
292                         }
293                     }
294                     endWord();
295                 }
296             }
297         }
298 
299         if (!lookForDocumentTags)
300             endDocument();
301     }
