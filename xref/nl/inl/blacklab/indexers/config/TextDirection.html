3   /**
4    * Text direction: left-to-right or right-to-left (and possible future values?)
5    */
6   public enum TextDirection {
7       LEFT_TO_RIGHT("ltr", "lefttoright"),
8       RIGHT_TO_LEFT("rtl", "righttoleft");
9   
10      private String[] codes;
11  
12      TextDirection(String... codes) {
13          this.codes = codes;
14      }
15  
16      /**
17       * Get the primary string code for this TextDirection.
18       * 
19       * @return string code
20       */
21      public String getCode() {
22          return this.codes[0];
23      }
24  
25      /**
26       * Does the specified code match this value?
27       * 
28       * @param code code to match, e.g. "LTR" or "right-to-left"
29       * @return true if it matches, false if not
30       */
31      public boolean matchesCode(String code) {
32          code = code.toLowerCase().replaceAll("[^a-z]", ""); // remove any spaces, dashes, etc.
33          for (String checkCode : codes) {
34              if (checkCode.equals(code))
35                  return true;
36          }
37          return false;
38      }
39  
40      /**
41       * Return a TextDirection from a string code (e.g. "LTR", "right-to-left", etc.)
42       *
43       * @param code textDirection code to recognize
44       * @return corresponding TextDirection
45       */
46      public static TextDirection fromCode(String code) {
47          for (TextDirection d : TextDirection.values()) {
48              if (d.matchesCode(code))
49                  return d;
50          }
51          return LEFT_TO_RIGHT; // default value
52      }
53  }
