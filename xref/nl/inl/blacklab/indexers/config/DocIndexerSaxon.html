39          } catch (IOException | XPathException | SAXException e) {
40              throw BlackLabRuntimeException.wrap(e);

44      @Override
45      public void index() throws MalformedInputFile, PluginException, IOException {
46          super.index();
47  
48          try {
49              for (NodeInfo doc : saxonHelper.findNodes(config.getDocumentPath(),contents)) {
50                  indexDocument(doc);
51              }
52          } catch (XPathExpressionException e) {
53              throw new InvalidConfiguration(e.getMessage() + String.format("; when indexing file: %s", documentName), e);
54          }
55      }

57      /**
58       * Index document from the current node.
59       */
60      protected void indexDocument(NodeInfo doc) throws XPathExpressionException {
61  
62          startDocument();
63  
64          // For each configured annotated field...
65          for (ConfigAnnotatedField annotatedField : config.getAnnotatedFields().values()) {
66              if (!annotatedField.isDummyForStoringLinkedDocuments())
67                  processAnnotatedField(doc,annotatedField);
68          }
69  
70          // For each configured metadata block..
71          for (ConfigMetadataBlock b : config.getMetadataBlocks()) {
72              processMetadataBlock(doc, b);
73          }
74  
75          // For each linked document...
76          for (ConfigLinkedDocument ld : config.getLinkedDocuments().values()) {
77              Function<String, String> xpathProcessor = xpath -> {
78                  try {
79                      return saxonHelper.getValue(xpath,doc);
80                  } catch (XPathExpressionException e) {
81                      throw new InvalidConfiguration(e.getMessage() + String.format("; when indexing file: %s", documentName), e);
82                  }
83              };
84              processLinkedDocument(ld, xpathProcessor);
85          }
86  
87          endDocument();
88      }

91      protected void processAnnotatedField(NodeInfo doc, ConfigAnnotatedField annotatedField)
92              throws XPathExpressionException {
93          setCurrentAnnotatedFieldName(annotatedField.getName());
94  
95          /*
96          Woorden kunnen in "inline tags zitten"
97  
98          Tussen woorden kunnen leestekens zitten
99  
100         Verzamelen en plaatsen m.b.v. NodeInfo#compareOrder
101 
102          */
103 
104         List<NodeInfo> puncts = saxonHelper.findNodes(annotatedField.getPunctPath(), doc);
105 
106         List<NodeInfo> inlines = new ArrayList<>(500);
107 
108         for (ConfigInlineTag inl : annotatedField.getInlineTags()) {
109             inlines.addAll(saxonHelper.findNodes(inl.getPath(),doc));
110         }
111         SaxonHelper.documentOrder(puncts);
112         SaxonHelper.documentOrder(inlines);
113 
114         for (NodeInfo container : saxonHelper.findNodes(annotatedField.getContainerPath(),doc)) {
115             int wNum=0;
116             Map<Integer,List<NodeInfo>> inlinesToEnd = new HashMap<>();
117             for (NodeInfo word : saxonHelper.findNodes(annotatedField.getWordsPath(),container)) {
118                 wNum++;
119                 /*
120                 hier gaan we kijken of er inline tags of leestekens voor dit woord zitten
121                 - indexeren opentag
122                 - onthouden na welk word gesloten moet worden
123                 - verwijderen uit lijst met inlines en leestekens
124                 - indexeren word(s)
125                 - indexeren sluittag(s) na het juiste woord
126 
127                  */
128                 List<NodeInfo> precedingPuncts = new ArrayList<>(3); // hierin verzamelen we puncts zodat we SNEL weten of een NodeInfo een punct is
129                 List<NodeInfo> preceding = getPreceding(puncts, inlines, word, precedingPuncts);
130 
131                 for (NodeInfo punctOrInline : preceding) {
132                     if (precedingPuncts.contains(punctOrInline)) {
133                         String punct = punctOrInline.getStringValue();
134                         punctuation(punct==null||punct.isEmpty()?" ":punct);
135                         if (!puncts.remove(punctOrInline)) {
136                             throw new BlackLabRuntimeException(String.format("punct not deleted %s",punctOrInline.toShortString()));
137                         }
138                     } else {
139                         // check if this word is within the inline, if so this word will always be the first word in the inline
140                         // because we remove the inline after it has been processed
141                         AxisIterator descendants = punctOrInline.iterateAxis(Axis.DESCENDANT.getAxisNumber());
142                         boolean isDescendant = false;
143                         NodeInfo next;
144                         while ((next = descendants.next()) != null) {
145                             if (next.equals(word)) {
146                                 isDescendant = true;
147                                 break;
148                             }
149                         }
150                         if (isDescendant) {
151                             Map<String,String> atts = new HashMap<>(3);
152                             AxisIterator attributes = punctOrInline.iterateAxis(Axis.ATTRIBUTE.getAxisNumber());
153                             while ((next = attributes.next()) != null) {
154                                 atts.put(next.getDisplayName(),next.getStringValue());
155                             }
156                             inlineTag(punctOrInline.getDisplayName(), true, atts);
157                             int increment =
158                                     Integer.parseInt(saxonHelper
159                                             .getValue("count("+annotatedField.getWordsPath()+")",punctOrInline))
160                                             -1;
161                             if (inlinesToEnd.containsKey(wNum+increment)) {
162                                 inlinesToEnd.get(wNum+increment).add(0,punctOrInline);
163                             } else {
164                                 List<NodeInfo> toEnd =  new ArrayList<>(3);
165                                 toEnd.add(punctOrInline);
166                                 inlinesToEnd.put(wNum+increment, toEnd);
167                             }
168 //                            System.out.println("inline "+punctOrInline.toShortString()+" for " +wNum + " should close after " + (wNum+increment));
169                         }
170                         if (!inlines.remove(punctOrInline)) {
171                             throw new BlackLabRuntimeException(String.format("not deleted %s",punctOrInline.toShortString()));
172                         }
173                     }
174 
175                 }
176                 charPos = saxonHelper.getStartPos(word);
177                 beginWord();
178                 for (Map.Entry<String, ConfigAnnotation> an : annotatedField.getAnnotations().entrySet()) {
179                     ConfigAnnotation annotation = an.getValue();
180                     // TODO we may need to support multiple values here
181                     String value = saxonHelper.getValue(annotation.getValuePath(),word);
182                     annotation(annotation.getName(),value,1,null);
183                 }
184                 charPos = saxonHelper.getEndPos(word);
185                 endWord();
186 //                System.out.println("looking for inlines to close after " +wNum);
187                 if (inlinesToEnd.containsKey(wNum)) {
188                     for (NodeInfo toEnd : inlinesToEnd.get(wNum)) {
189 //                        System.out.println("closing inline "+toEnd.toShortString()+" after " +wNum);
190                         inlineTag(toEnd.getDisplayName(),false,null);
191                     }
192                     inlinesToEnd.remove(wNum);
193                 }
194             }
195             if (!inlinesToEnd.isEmpty()) {
196                 throw new BlackLabRuntimeException(String.format("unclosed inlines left: %s ", inlinesToEnd.values()));
197             }
198 
199         }
200     }

202     /**
203      * return punctuations and inlines occurring before a word
204      * @param puncts
205      * @param inlines
206      * @param word
207      * @return
208      */
209     private List<NodeInfo> getPreceding(List<NodeInfo> puncts, List<NodeInfo> inlines, NodeInfo word, List<NodeInfo> precedingPuncts) {
210         List<NodeInfo> preceding = new ArrayList<>();
211         for (NodeInfo pi : puncts) {
212             if (word!=null&&word.compareOrder(pi)!=1) {
213                 break;
214             }
215             preceding.add(pi);
216             precedingPuncts.add(pi);
217         }
218         for (NodeInfo pi : inlines) {
219             if (word!=null&&word.compareOrder(pi)!=1) {
220                 break;
221             }
222             preceding.add(pi);
223         }
224         SaxonHelper.documentOrder(preceding);
225         return preceding;
226     }

229     protected void processMetadataBlock(NodeInfo doc, ConfigMetadataBlock b) throws XPathExpressionException {
230 
231         for (NodeInfo header : saxonHelper.findNodes(b.getContainerPath(),doc)) {
232             List<ConfigMetadataField> fields = b.getFields();
233             for (int i = 0; i < fields.size(); i++) { // NOTE: fields may be added during loop, so can't iterate
234                 ConfigMetadataField f = fields.get(i);
235 
236                 // Metadata field configs without a valuePath are just for
237                 // adding information about fields captured in forEach's,
238                 // such as extra processing steps
239                 if (f.getValuePath() == null || f.getValuePath().isEmpty())
240                     continue;
241 
242                 // Capture whatever this configured metadata field points to
243                 if (f.isForEach()) {
244                     // "forEach" metadata specification
245                     // (allows us to capture many metadata fields with 3 XPath expressions)
246                     for (NodeInfo forEach : saxonHelper.findNodes(f.getForEachPath(),header)) {
247                         // Find the fieldName and value for this forEach match
248                         String origFieldName = saxonHelper.getValue(f.getName(),forEach);
249                         String fieldName = AnnotatedFieldNameUtil.sanitizeXmlElementName(origFieldName);
250                         if (!origFieldName.equals(fieldName)) {
251                             DocIndexerXPath.warnSanitized(origFieldName, fieldName);
252                         }
253                         b.getOrCreateField(fieldName);
254 
255                         // Multiple matches will be indexed at the same position.
256                         processValue(forEach, f);
257                     }
258                 } else {
259                     // Regular metadata field; just the fieldName and an XPath expression for the value
260                     // Multiple matches will be indexed at the same position.
261                     processValue(header, f);
262                 }
263             }
264         }
265 
266     }

268     private void processValue(NodeInfo header, ConfigMetadataField f) throws XPathExpressionException {
269         for (Object val : saxonHelper.find(f.getValuePath(),header)) {
270             if (val instanceof NodeInfo) {
271                 String unprocessedValue = saxonHelper.getValue(".", val);
272                 for (String value : processStringMultipleValues(unprocessedValue, f.getProcess(), f.getMapValues())) {
273                     addMetadataField(f.getName(), value);
274                 }
275             } else {
276                 String metadataValue = processString(String.valueOf(val), f.getProcess(), f.getMapValues());
277                 addMetadataField(f.getName(), metadataValue);
278 
279             }
280         }
