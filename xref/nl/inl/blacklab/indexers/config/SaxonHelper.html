98      /**
99       * translation of recorded line and column number to character position in the document
100      * @param lineNumber
101      * @param columnNumber
102      * @return
103      */
104     private int getCharPos(int lineNumber, int columnNumber) {
105         int charsOnline = cumulativeColsPerLine.get(lineNumber) -
106                 (lineNumber == 1 ? 0 : cumulativeColsPerLine.get(lineNumber - 1));
107         return cumulativeColsPerLine.get(lineNumber)
108                 - (charsOnline - columnNumber)
109                 + (lineNumber - 1) * lineEnd
110                 - 1; // saxon notifies columnNumber just after tag
111     }

122     SaxonHelper(Reader reader, ConfigInputFormat blConfig) throws IOException, SAXException, XPathException {
123         // characters needed for calculating positions
124         document = IOUtils.toCharArray(reader);
125         CharArrayReader stream = new CharArrayReader(document);
126         AtomicInteger line = new AtomicInteger();
127         AtomicInteger cols = new AtomicInteger();
128         // determine cumulative number of chars per line
129         IOUtils.lineIterator(stream).forEachRemaining(l ->
130                 cumulativeColsPerLine.put(line.incrementAndGet(), cols.addAndGet(l.length())));
131         stream.reset();
132         if (cumulativeColsPerLine.size() > 1) {
133             int i = -1;
134             while ((i = stream.read()) != -1) {
135                 if (i == '\r') {
136                     // windows file
137                     lineEnd = 2;
138                     break;
139                 }
140             }
141             stream.reset();
142         }
143         // make sure our content handler doesn't get overwritten by saxon
144         MyContentHandler myContentHandler = new MyContentHandler();
145         XMLReader xmlReader = XMLReaderFactory.createXMLReader();
146         xmlReader.setContentHandler(myContentHandler);
147         XMLReader wrapper = new MyXMLReader(xmlReader);
148         // regular parsing with line numbering enabled
149         InputSource inputSrc = new InputSource(stream);
150         Source source = new SAXSource(wrapper, inputSrc);
151         Configuration config = ((XPathFactoryImpl) X_PATH_FACTORY_THREAD_LOCAL.get()).getConfiguration();
152         config.setLineNumbering(true);
153         contents = config.buildDocumentTree(source);
154         if (document.length / 2 > MAXDOCSIZEINMEMORY) {
155 //            System.out.println("Disk cache for document, length " + (chars.length / 2));
156             documentDiskCache = File.createTempFile("blDocToIndex",null);
157             documentDiskCache.deleteOnExit();
158             IOUtils.write(document, new FileWriter(documentDiskCache));
159             document = null;
160         }
161         // setup namespace aware xpath that will compile xpath expressions
162         xPath = X_PATH_FACTORY_THREAD_LOCAL.get().newXPath();
163         if (blConfig.isNamespaceAware()) {
164             namespaces.add("xml", "http://www.w3.org/XML/1998/namespace");
165             for (Map.Entry<String, String> e : blConfig.getNamespaces().entrySet()) {
166                 namespaces.add(e.getKey(), e.getValue());
167             }
168             xPath.setNamespaceContext(namespaces);
169         }
170     }

301         private Deque<StartEndPos> elStack = new ArrayDeque<>();
302 
303         @Override
304         public void startElement(String uri, String localName, String qName, Attributes atts) throws SAXException {
305             int end = getCharPos(locator.getLineNumber(), locator.getColumnNumber());
306             int begin = end;
307             // now look back for the < character
308             for (int i = end - 1; i > 0; i--) {
309                 if ('<' == document[i]) {
310                     begin = i;
311                     break;
312                 }
313             }
314             // NOTE more testing needed for self closing tags
315             if (begin == end) {
316                 throw new BlackLabRuntimeException(String.format("No '<' found for %s at line %d, col %d, charpos %d",
317                         qName, locator.getLineNumber(), locator.getColumnNumber(), end));
318             }
319             StartEndPos startEndPos = new StartEndPos(begin);
320             elStack.push(startEndPos);
321             startEndPosMap.put(end, startEndPos);
322             saxonHandler.startElement(uri, localName, qName, atts);
323         }
324 
325         @Override
326         public void endElement(String uri, String localName, String qName) throws SAXException {
327             elStack.pop().setEndPos(getCharPos(locator.getLineNumber(), locator.getColumnNumber()));
328             saxonHandler.endElement(uri, localName, qName);
329         }

386     /**
387      * Create XPathExpression and cache.
388      *
389      * @param xpathExpr the xpath expression
390      * @return the XPathExpression
391      */
392     private XPathExpression acquireXPathExpression(String xpathExpr) {
393         XPathExpression xPathExpression = compiledXPaths.get(xpathExpr);
394         if (xPathExpression == null) {
395             try {
396                 xPathExpression = xPath.compile(xpathExpr);
397                 compiledXPaths.put(xpathExpr, xPathExpression);
398             } catch (XPathExpressionException e) {
399                 throw new BlackLabRuntimeException("Error in XPath expression " + xpathExpr + " : " + e.getMessage(), e);
400             }
401         }
402         return xPathExpression;
403     }

405     /**
406      * Find results in a context, return a list of Objects. This approach is usefull if you don't know
407      * the return type(s) in advance. This works for all return types of an xPath, also the ones that
408      * return for example one boolean. Often a List&lt;NodeInfo> will be returned.
409      *
410      * @param xPath
411      * @param context
412      * @return
413      */
414     List<?> find(String xPath, Object context) throws XPathExpressionException {
415         return (List<?>) acquireXPathExpression(xPath).evaluate(context,XPathConstants.NODESET);
416     }
417 
418     /**
419      * Calls {@link #find(String, Object)} and casts the result to List&lt;NodeInfo>, NOTE that the
420      * resulting list may still contain Objects that are no NodeInfo, due to the way collections work,
421      * Collections.checkedList won't help here.
422      * @param xPath
423      * @param context
424      * @return
425      * @throws XPathExpressionException
426      */
427     @SuppressWarnings("unchecked")
428     List<NodeInfo> findNodes(String xPath, Object context) throws XPathExpressionException {
429         return (List<NodeInfo>) find(xPath,context);
430     }
431 
432     /**
433      * return a string representation of an xpath result, using {@link NodeInfo#getStringValue()} or
434      * String.valueOf. Handling multiple results should be done in xPath, for example concat.
435      * @param xPath
436      * @param context
437      * @return
438      * @throws XPathExpressionException
439      * @throws InvalidConfiguration when the xpath returns multiple results
440      */
441     String getValue(String xPath, Object context) throws XPathExpressionException {
442         List<?> list = find(xPath, context);
443         if (list.size()==1) {
444             Object o = list.get(0);
445             if (o instanceof NodeInfo) {
446                 return ((NodeInfo) o).getStringValue();
447             } else {
448                 return String.valueOf(o);
449             }
450         } else {
451             if (list.isEmpty())
452                 return "";
453             else
454                     throw new InvalidConfiguration(
455                             String.format("list %s contains multiple values, change your xpath %s to return one result or concatenate",
456                                     list.stream().map(o -> o instanceof NodeInfo ? ((NodeInfo) o).toShortString() : String.valueOf(o))
457                                             .collect(Collectors.toList()),xPath));
458         }
459     }

529     /**
530      * returns the document as a string, sets all state in the helper to null
531      * @return
532      */
533     String getDocument(boolean clean) {
534         char[] rv = document;
535         try {
536             if (document ==null) {
537                 rv = IOUtils.toCharArray(new FileReader(documentDiskCache));
538             }
539         } catch (IOException e) {
540             throw new BlackLabRuntimeException("unable to read document cache from disk");
541         }
542         if (clean) clean();
543         return new String(rv);
544     }
