48      /**
49       *
50       * @param r
51       * @param isJson
52       * @param cfg
53       * @param finder responsible for getting (optionally locating/loading) other
54       *            configs that this config depends on. (for config keys "baseFormat"
55       *            and "inputFormat")
56       * @throws IOException
57       * @throws InvalidInputFormatConfig if the file is not a valid config
58       */
59      public static void read(Reader r, boolean isJson, ConfigInputFormat cfg,
60              Function<String, Optional<ConfigInputFormat>> finder) throws IOException {
61          ObjectMapper mapper = isJson ? Json.getJsonObjectMapper() : Json.getYamlObjectMapper();
62  
63          JsonNode root;
64          try {
65              root = mapper.readTree(r);
66          } catch (JsonParseException e) {
67              throw new InvalidInputFormatConfig("Could not parse config file: " + e.getMessage());
68          }
69          read(root, cfg, finder);
70      }

73       *
74       * @param file
75       * @param cfg
76       * @param finder responsible for getting (optionally locating/loading) other
77       *            configs that this config depends on. ("baseFormat" and
78       *            "inputFormat")

80       * @throws InvalidInputFormatConfig if the file is not a valid config
81       */
82      public static void read(File file, ConfigInputFormat cfg, Function<String, Optional<ConfigInputFormat>> finder)
83              throws IOException {
84          read(FileUtil.openForReading(file), file.getName().endsWith(".json"), cfg, finder);
85          cfg.setReadFromFile(file);
86      }

88      protected static void read(JsonNode root, ConfigInputFormat cfg,
89              Function<String, Optional<ConfigInputFormat>> finder) {
90          obj(root, "root node");
91          Iterator<Entry<String, JsonNode>> it = root.fields();
92          while (it.hasNext()) {
93              Entry<String, JsonNode> e = it.next();
94              switch (e.getKey()) {
95              case "displayName":
96                  cfg.setDisplayName(str(e));
97                  break;
98              case "description":
99                  cfg.setDescription(str(e));
100                 break;
101             case "helpUrl":
102                 cfg.setHelpUrl(str(e));
103                 break;
104             case "baseFormat": {
105                 String formatIdentifier = str(e);
106                 if (finder == null)
107                     throw new InvalidInputFormatConfig(
108                             "Format depends on base format " + formatIdentifier + " but no BaseFormatFinder provided.");
109 
110                 ConfigInputFormat baseFormat = finder
111                         .apply(formatIdentifier)
112                         .orElseThrow(() -> new InvalidInputFormatConfig(
113                                 "Base format " + formatIdentifier + " not found for format " + cfg.getName()));
114 
115                 cfg.setBaseFormat(baseFormat);
116                 break;
117             }
118             case "type":
119                 cfg.setType(str(e));
120                 break;
121             case "fileType":
122                 cfg.setFileType(FileType.fromStringValue(str(e)));
123                 break;
124             case "fileTypeOptions":
125                 readFileTypeOptions(e, cfg);
126                 break;
127 //            case "tabularOptions": cfg.setTabularOptions(readTabularOptions(e)); break;
128             case "corpusConfig":
129                 readCorpusConfig(e, cfg.getCorpusConfig());
130                 break;
131             case "namespaces":
132                 readStringMap(e, cfg.namespaces);
133                 break;
134             case "documentPath":
135                 cfg.setDocumentPath(str(e));
136                 break;
137             case "store":
138                 cfg.setStore(bool(e));
139                 break;
140             case "indexFieldAs":
141                 readStringMap(e, cfg.indexFieldAs);
142                 break;
143             case "annotatedFields":
144                 readAnnotatedFields(e, cfg);
145                 break;
146             case "metadataDefaultAnalyzer":
147                 cfg.setMetadataDefaultAnalyzer(str(e));
148                 break;
149             case "metadataDefaultUnknownCondition":
150                 cfg.setMetadataDefaultUnknownCondition(UnknownCondition.fromStringValue(str(e)));
151                 break;
152             case "metadataDefaultUnknownValue":
153                 cfg.setMetadataDefaultUnknownValue(str(e));
154                 break;
155             case "metadata":
156                 readMetadata(e, cfg);
157                 break;
158             case "linkedDocuments":
159                 readLinkedDocuments(e, cfg);
160                 break;
161             case "convertPlugin":
162                 cfg.setConvertPluginId(str(e));
163                 break;
164             case "tagPlugin":
165                 cfg.setTagPluginId(str(e));
166                 break;
167             case "isVisible":
168                 cfg.setVisible(bool(e));
169                 break;
170             default:
171                 throw new InvalidInputFormatConfig("Unknown top-level key " + e.getKey());
172             }
173         }
174 
175         // Ensure that if we have any linked documents we want to store (like metadata), there exists an
176         // annotated field where we can store it (even if it has no annotations).
177         for (ConfigLinkedDocument ld: cfg.getLinkedDocuments().values()) {
178             if (ld.shouldStore() && cfg.getAnnotatedField(ld.getName()) == null) {
179                 // Field doesn't exit yet. Create a dummy field for it.
180                 cfg.addAnnotatedField(ConfigAnnotatedField.createDummyForStoringLinkedDocument(ld.getName()));
181             }
182         }
183     }

185     private static void readCorpusConfig(Entry<String, JsonNode> ccEntry, ConfigCorpus corpusConfig) {
186         ObjectNode node = obj(ccEntry.getValue(), "");
187         Iterator<Entry<String, JsonNode>> it = node.fields();
188         while (it.hasNext()) {
189             Entry<String, JsonNode> e = it.next();
190             switch (e.getKey()) {
191             case "displayName":
192                 corpusConfig.setDisplayName(str(e));
193                 break;
194             case "description":
195                 corpusConfig.setDescription(str(e));
196                 break;
197             case "contentViewable":
198                 corpusConfig.setContentViewable(bool(e));
199                 break;
200             case "textDirection":
201                 corpusConfig.setTextDirection(TextDirection.fromCode(str(e)));
202                 break;
203             case "specialFields":
204                 readStringMap(e, corpusConfig.specialFields);
205                 break;
206             case "metadataFieldGroups":
207                 readMetadataFieldGroups(e, corpusConfig);
208                 break;
209             case "annotationGroups":
210                 readAllAnnotationGroups(e, corpusConfig);
211                 break;
212             default:
213                 throw new InvalidInputFormatConfig("Unknown key " + e.getKey() + " in corpusConfig");
214             }
215         }

262                     break;
263                 default:
264                     throw new InvalidInputFormatConfig(
265                             "Unknown key " + e.getKey() + " in metadata field group " + g.getName());
266                 }
267             }
268             cfg.addMetadataFieldGroup(g);
269         }

317             ConfigAnnotatedField af = cfg.getOrCreateAnnotatedField(fieldName);
318             while (itField.hasNext()) {
319                 Entry<String, JsonNode> e = itField.next();

323                     break;

326                     break;
327                 case "containerPath":
328                     af.setContainerPath(str(e));
329                     break;
330                 case "wordPath":
331                     af.setWordPath(str(e));
332                     break;
333                 case "tokenPositionIdPath":
334                     af.setTokenPositionIdPath(str(e));
335                     break;

370     protected static ConfigAnnotation readAnnotation(ConfigAnnotation parentAnnot, JsonNode a) {
371         Iterator<Entry<String, JsonNode>> itAnnotation = obj(a, "annotation").fields();
372         ConfigAnnotation annot = new ConfigAnnotation();
373         while (itAnnotation.hasNext()) {
374             Entry<String, JsonNode> e = itAnnotation.next();
375             boolean isSubannotation = parentAnnot != null;
376             switch (e.getKey()) {
377             case "name":
378                 String name = str(e);
379                 String fullName = parentAnnot == null ? name : parentAnnot.getName() + AnnotatedFieldNameUtil.SUBANNOTATION_FIELD_PREFIX_SEPARATOR + name;
380                 annot.setName(warnSanitizeXmlElementName(fullName));
381                 break;
382             case "value":
383                 annot.setValuePath(fixedStringToXpath(str(e)));
384                 break;
385             case "valuePath":
386                 annot.setValuePath(str(e));
387                 break;
388             case "captureValuePaths":
389                 ArrayNode paths = (ArrayNode) e.getValue();
390                 paths.iterator().forEachRemaining((t) -> {
391                     annot.addCaptureValuePath(t.asText());
392                 });
393                 break;
394             case "forEachPath":
395                 if (!isSubannotation)
396                     throw new InvalidInputFormatConfig("Only subannotations may have forEachPath/namePath");
397                 annot.setForEachPath(str(e));
398                 break;
399             case "namePath":
400                 if (!isSubannotation)
401                     throw new InvalidInputFormatConfig("Only subannotations may have forEachPath/namePath");
402                 annot.setName(str(e));
403                 break;
404             case "process":
405                 annot.setProcess(readProcess(e));
406                 break;
407             case "displayName":
408                 annot.setDisplayName(str(e));
409                 break;
410             case "description":
411                 annot.setDescription(str(e));
412                 break;
413             case "basePath":
414                 annot.setBasePath(str(e));
415                 break;
416             case "sensitivity":
417                 if (isSubannotation)
418                     throw new InvalidInputFormatConfig("Subannotations may not have their own sensitivity settings");
419                 annot.setSensitivity(SensitivitySetting.fromStringValue(str(e)));
420                 break;
421             case "uiType":
422                 annot.setUiType(str(e));
423                 break;
424             case "subannotations":
425                 if (isSubannotation)
426                     throw new InvalidInputFormatConfig("Subannotations may not have their own subannotations");
427                 readSubAnnotations(e, annot);
428                 break;
429             case "forwardIndex":
430                 annot.setForwardIndex(bool(e));
431                 break;
432             case "multipleValues":
433                 annot.setMultipleValues(bool(e));
434                 break;
435             case "allowDuplicateValues":
436                 annot.setAllowDuplicateValues(bool(e));
437                 break;
438             case "captureXml":
439                 annot.setCaptureXml(bool(e));
440                 break;
441             case "isInternal":
442                 annot.setInternal(bool(e));
443                 break;
444             default:
445                 throw new InvalidInputFormatConfig(
446                         "Unknown key " + e.getKey() + " in annotation " + StringUtils.defaultString(annot.getName()));
447             }
448         }
449         return annot;
450     }

452     /**
453      * Convert a fixed string value to an XPath expression yielding that value.
454      *
455      * @param s fixed string the XPath should evaluate to
456      * @return XPath expression
457      */
458     public static String fixedStringToXpath(String s) {
459         return "\"" + s.replaceAll("\\\\", "\\\\").replaceAll("\"", "\\\"") + "\"";
460     }

512     private static void readMetadata(Entry<String, JsonNode> mdEntry, ConfigInputFormat cfg) {
513         JsonNode node = mdEntry.getValue();
514         if (node instanceof ObjectNode) {
515             // Single metadata block
516             readMetadataBlock(node, cfg);
517         } else if (node instanceof ArrayNode) {
518             // List of metadata blocks
519             Iterator<JsonNode> itAnnotations = node.elements();
520             while (itAnnotations.hasNext()) {
521                 JsonNode as = itAnnotations.next();
522                 readMetadataBlock(as, cfg);
523             }
524         } else {
525             throw new InvalidInputFormatConfig("Wrong node type for metadata (must be object or array)");
526         }
527     }

529     private static void readMetadataBlock(JsonNode as, ConfigInputFormat cfg) {
530         ConfigMetadataBlock b = cfg.createMetadataBlock();
531         Iterator<Entry<String, JsonNode>> it = obj(as, "metadata block").fields();
532         while (it.hasNext()) {
533             Entry<String, JsonNode> e = it.next();
534             switch (e.getKey()) {
535             case "containerPath":
536                 b.setContainerPath(str(e));
537                 break;
538             case "defaultAnalyzer":
539                 b.setDefaultAnalyzer(str(e));
540                 break;
541             case "fields":
542                 readMetadataFields(e, b);
543                 break;
544             default:
545                 throw new InvalidInputFormatConfig("Unknown key " + e.getKey() + " in metadata block");
546             }
547         }
548     }

557     private static void readMetadataFields(Entry<String, JsonNode> mfsEntry, ConfigMetadataBlock b) {
558         Iterator<JsonNode> itFields = array(mfsEntry).elements();
559         while (itFields.hasNext()) {
560             JsonNode fld = itFields.next();
561             String name = fld.has("name") ? fld.get("name").asText() : null;
562             ConfigMetadataField f;
563             boolean existingField = false;
564             if (name != null) {
565                 // If field exists, modify existing field.
566                 // This is mostly because of forward references to fields; the field instance
567                 // would be created by the reference, and the field properties will be added when
568                 // they are parsed later in the file.
569                 f = b.getOrCreateField(warnSanitizeXmlElementName(name));
570                 existingField = true;
571             } else
572                 f = new ConfigMetadataField();
573             Iterator<Entry<String, JsonNode>> itField = obj(fld, "metadata field").fields();
574             while (itField.hasNext()) {
575                 Entry<String, JsonNode> e = itField.next();
576                 switch (e.getKey()) {
577                 case "name":
578                     f.setName(warnSanitizeXmlElementName(str(e)));
579                     break;
580                 case "namePath":
581                     f.setName(str(e));
582                     break;
583                 case "value":
584                     f.setValuePath(fixedStringToXpath(str(e)));
585                     break;
586                 case "valuePath":
587                     f.setValuePath(str(e));
588                     break;
589                 case "forEachPath":
590                     f.setForEachPath(str(e));
591                     break;
592                 case "process":
593                     f.setProcess(readProcess(e));
594                     break;
595                 case "mapValues":
596                     Map<String, String> mapValues = new HashMap<>();
597                     readStringMap(e, mapValues);
598                     f.setMapValues(mapValues);
599                     break;
600                 case "displayName":
601                     f.setDisplayName(str(e));
602                     break;
603                 case "description":
604                     f.setDescription(str(e));
605                     break;
606                 case "type":
607                     f.setType(FieldType.fromStringValue(str(e)));
608                     break;
609                 case "uiType":
610                     f.setUiType(str(e));
611                     break;
612                 case "unknownCondition":
613                     f.setUnknownCondition(UnknownCondition.fromStringValue(str(e)));
614                     break;
615                 case "unknownValue":
616                     f.setUnknownValue(str(e));
617                     break;
618                 case "analyzer":
619                     f.setAnalyzer(str(e));
620                     break;
621                 case "displayOrder":
622                     List<String> fields = new ArrayList<>();
623                     readStringList(e, fields);
624                     f.addDisplayOrder(fields);
625                     break;
626                 case "displayValues":
627                     Map<String, String> values = new HashMap<>();
628                     readStringMap(e, values);
629                     f.addDisplayValues(values);
630                     break;
631                 default:
632                     throw new InvalidInputFormatConfig(
633                             "Unknown key " + e.getKey() + " in metadata field " + StringUtils.defaultString(f.getName(), "(unnamed)"));
634                 }
635             }
636             if (!existingField)
637                 b.addMetadataField(f);
638         }
639     }

641     private static void readLinkedDocuments(Entry<String, JsonNode> ldsEntry, ConfigInputFormat cfg) {
642         Iterator<Entry<String, JsonNode>> itLinkedDocs = obj(ldsEntry).fields();
643         while (itLinkedDocs.hasNext()) {
644             Entry<String, JsonNode> linkedDoc = itLinkedDocs.next();
645             ConfigLinkedDocument ld = cfg.getOrCreateLinkedDocument(linkedDoc.getKey());
646             Iterator<Entry<String, JsonNode>> itLinkedDoc = obj(linkedDoc).fields();
647             while (itLinkedDoc.hasNext()) {
648                 Entry<String, JsonNode> e = itLinkedDoc.next();
649                 switch (e.getKey()) {
650                 case "store":
651                     ld.setStore(bool(e));
652                     break;
653                 case "linkValues":
654                     readLinkValues(e, ld);
655                     break;
656                 case "ifLinkPathMissing":
657                     ld.setIfLinkPathMissing(MissingLinkPathAction.fromStringValue(str(e)));
658                     break;
659                 case "inputFile":
660                     ld.setInputFile(str(e));
661                     break;
662                 case "pathInsideArchive":
663                     ld.setPathInsideArchive(str(e));
664                     break;
665                 case "documentPath":
666                     ld.setDocumentPath(str(e));
667                     break;
668                 case "inputFormat":
669                     readInputFormat(ld, e);
670                     break;
671                 default:
672                     throw new InvalidInputFormatConfig(
673                             "Unknown key " + e.getKey() + " in linked document " + ld.getName());
674                 }
675             }
676         }

679     protected static void readInputFormat(ConfigLinkedDocument ld, Entry<String, JsonNode> e) {
680         // Resolve the inputFormat right now, instead of potentially failing later when the format is actually needed at some point during indexing
681         String formatIdentifier = str(e);
682         Format format = DocumentFormats.getFormat(formatIdentifier);
683         if (format == null)
684             throw new InvalidInputFormatConfig(
685                     "Unknown input format " + str(e) + " in linked document " + ld.getName());
686 
687         ld.setInputFormatIdentifier(formatIdentifier);
688     }

690     private static void readLinkValues(Entry<String, JsonNode> lvsEntry, ConfigLinkedDocument ld) {
691         Iterator<JsonNode> itLinkValues = array(lvsEntry).elements();
692         while (itLinkValues.hasNext()) {
693             JsonNode linkValue = itLinkValues.next();
694             ConfigLinkValue lv = new ConfigLinkValue();
695             Iterator<Entry<String, JsonNode>> itLinkValue = obj(linkValue, "link value").fields();
696             while (itLinkValue.hasNext()) {
697                 Entry<String, JsonNode> e = itLinkValue.next();
698                 switch (e.getKey()) {
699                 case "value":
700                     lv.setValuePath(fixedStringToXpath(str(e)));
701                     break;
702                 case "valuePath":
703                     lv.setValuePath(str(e));
704                     break;
705                 case "valueField":
706                     lv.setValueField(str(e));
707                     break;
708                 case "process":
709                     lv.setProcess(readProcess(e));
710                     break;
711                 default:
712                     throw new InvalidInputFormatConfig(
713                             "Unknown key " + e.getKey() + " in linked document " + ld.getName());
714                 }
715             }
716             ld.addLinkValue(lv);
717         }
718     }

720     private static List<ConfigProcessStep> readProcess(Entry<String, JsonNode> prEntry) {
721         Iterator<JsonNode> itSteps = array(prEntry).elements();
722         List<ConfigProcessStep> p = new ArrayList<>();
723         while (itSteps.hasNext()) {
724             JsonNode step = itSteps.next();
725             ConfigProcessStep s = new ConfigProcessStep();
726             Iterator<Entry<String, JsonNode>> itStep = obj(step, "processing step").fields();
727             while (itStep.hasNext()) {
728                 Entry<String, JsonNode> e = itStep.next();
729                 switch (e.getKey()) {
730                 case "action":
731                     s.setMethod(str(e));
732                     break;
733                 default:
734                     s.addParam(e.getKey(), str(e));
735                     break;
736                 }
737             }
738             p.add(s);
739         }
740         return p;
741     }
