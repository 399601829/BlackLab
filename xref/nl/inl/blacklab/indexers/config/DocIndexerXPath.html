117     /**

121     private Map<AutoPilot, String> autoPilotsInUse = new HashMap<>();

123     /**
124      * Create AutoPilot and declare namespaces on it.
125      *
126      * @param xpathExpr xpath expression for the AutoPilot
127      * @return the AutoPilot
128      * @throws XPathParseException
129      */
130     private AutoPilot acquireAutoPilot(String xpathExpr) throws XPathParseException {
131         AutoPilot ap = compiledXPaths.remove(xpathExpr);
132         if (ap == null) {
133             ap = new AutoPilot(nav);
134             if (config.isNamespaceAware()) {
135                 ap.declareXPathNameSpace("xml", "http://www.w3.org/XML/1998/namespace"); // builtin
136                 for (Entry<String, String> e : config.getNamespaces().entrySet()) {
137                     ap.declareXPathNameSpace(e.getKey(), e.getValue());
138                 }
139             }
140             try {
141                 ap.selectXPath(xpathExpr);
142             } catch (XPathParseException e) {
143                 throw new BlackLabRuntimeException("Error in XPath expression " + xpathExpr + " : " + e.getMessage(), e);
144             }
145         } else {
146             ap.resetXPath();
147         }
148         autoPilotsInUse.put(ap, xpathExpr);
149         return ap;
150     }
151 
152     private void releaseAutoPilot(AutoPilot ap) {
153         String xpathExpr = autoPilotsInUse.remove(ap);
154         compiledXPaths.put(xpathExpr, ap);
155     }

157     @Override
158     public void index() throws MalformedInputFile, PluginException, IOException {
159         super.index();
160 
161         if (inputDocument.length > 0) { // VTD doesn't like empty documents
162             // Parse use VTD-XML
163             vg = new VTDGen();
164             vg.setDoc(inputDocument);
165             // Whitespace in between elements is normally ignored,
166             // but we explicitly allow whitespace in between elements to be collected here.
167             // This allows punctuation xpath to match this whitespace, in case punctuation/whitespace in the document isn't contained in a dedicated element or attribute.
168             // This doesn't mean that this whitespace is always used, it just enables the punctuation xpath to find this whitespace if it explicitly matches it.
169             vg.enableIgnoredWhiteSpace(true);
170             try {
171                 vg.parse(config.isNamespaceAware());
172 
173                 nav = vg.getNav();
174 
175                 // Find all documents
176                 AutoPilot documents = acquireAutoPilot(config.getDocumentPath());
177                 while (documents.evalXPath() != -1) {
178                     indexDocument();
179                 }
180                 releaseAutoPilot(documents);
181             } catch (VTDException e) {
182                 throw new MalformedInputFile("Error indexing file: " + documentName, e);
183             }
184         }
185     }

187     /**
188      * Index document from the current node.
189      *
190      * @throws VTDException on XPath parse (or other) error
191      */
192     protected void indexDocument() throws VTDException {
193 
194         startDocument();
195 
196         // For each configured annotated field...
197         for (ConfigAnnotatedField annotatedField : config.getAnnotatedFields().values()) {
198             if (!annotatedField.isDummyForStoringLinkedDocuments())
199                 processAnnotatedField(annotatedField);
200         }
201 
202         // For each configured metadata block..
203         for (ConfigMetadataBlock b : config.getMetadataBlocks()) {
204             processMetadataBlock(b);
205         }
206 
207         // For each linked document...
208         for (ConfigLinkedDocument ld : config.getLinkedDocuments().values()) {
209             Function<String, String> xpathProcessor = xpath -> {
210                 // Resolve value using XPath
211                 AutoPilot apLinkPath = null;
212                 String result = null;
213                 try {
214                     apLinkPath = acquireAutoPilot(xpath);
215                     result = apLinkPath.evalXPathToString();
216                     if (result == null || result.isEmpty()) {
217                         switch (ld.getIfLinkPathMissing()) {
218                             case IGNORE:
219                                 break;
220                             case WARN:
221                                 docWriter.listener()
222                                         .warning("Link path " + xpath + " not found in document " + documentName);
223                                 break;
224                             case FAIL:
225                                 throw new BlackLabRuntimeException("Link path " + xpath + " not found in document " + documentName);
226                         }
227                     }
228                     releaseAutoPilot(apLinkPath);
229                 } catch (XPathParseException e) {
230                     throw new InvalidConfiguration(e.getMessage() + String.format("; when indexing file: %s", documentName), e);
231                 }
232                 return result;
233             };
234             processLinkedDocument(ld, xpathProcessor);
235         }
236 
237         endDocument();
238     }

240     protected void processAnnotatedField(ConfigAnnotatedField annotatedField)
241             throws XPathParseException, XPathEvalException, NavException, VTDException {
242         Map<String, Integer> tokenPositionsMap = new HashMap<>();
243 
244         // Determine some useful stuff about the field we're processing
245         // and store in instance variables so our methods can access them
246         setCurrentAnnotatedField(annotatedField);
247 
248         // Precompile XPaths for words, evalToString, inline tags, punct and (sub)annotations
249         AutoPilot words = acquireAutoPilot(annotatedField.getWordsPath());
250         AutoPilot apEvalToString = acquireAutoPilot(".");
251         List<AutoPilot> apsInlineTag = new ArrayList<>();
252         for (ConfigInlineTag inlineTag : annotatedField.getInlineTags()) {
253             AutoPilot apInlineTag = acquireAutoPilot(inlineTag.getPath());
254             apsInlineTag.add(apInlineTag);
255         }
256         AutoPilot apPunct = null;
257         if (annotatedField.getPunctPath() != null)
258             apPunct = acquireAutoPilot(annotatedField.getPunctPath());
259         String tokenPositionIdPath = annotatedField.getTokenPositionIdPath();
260         AutoPilot apTokenPositionId = null;
261         if (tokenPositionIdPath != null) {
262             apTokenPositionId = acquireAutoPilot(tokenPositionIdPath);
263         }
264 
265         // For each body element...
266         // (there's usually only one, but there's no reason to limit it)
267         navpush();
268         AutoPilot bodies = acquireAutoPilot(annotatedField.getContainerPath());
269         AnnotatedFieldWriter annotatedFieldWriter = getAnnotatedField(annotatedField.getName());
270         while (bodies.evalXPath() != -1) {
271 
272             // First we find all inline elements (stuff like s, p, b, etc.) and store
273             // the locations of their start and end tags in a sorted list.
274             // This way, we can keep track of between which words these tags occur.
275             // For end tags, we will update the payload of the start tag when we encounter it,
276             // just like we do in our SAX parsers.
277             List<InlineObject> tagsAndPunct = new ArrayList<>();
278             for (AutoPilot apInlineTag : apsInlineTag) {
279                 navpush();
280                 apInlineTag.resetXPath();
281                 while (apInlineTag.evalXPath() != -1) {
282                     collectInlineTag(tagsAndPunct);
283                 }
284                 navpop();
285             }
286             setAddDefaultPunctuation(true);
287             if (apPunct != null) {
288                 // We have punctuation occurring between word tags (as opposed to
289                 // punctuation that is tagged as a word itself). Collect this punctuation.
290                 setAddDefaultPunctuation(false);
291                 navpush();
292                 apPunct.resetXPath();
293                 while (apPunct.evalXPath() != -1) {
294                     apEvalToString.resetXPath();
295                     String punct = apEvalToString.evalXPathToString();
296                     // If punctPath matches an empty tag, replace it with a space.
297                     // Deals with e.g. <lb/> (line break) tags in TEI.
298                     if (punct.isEmpty())
299                         punct = " ";
300                     collectPunct(tagsAndPunct, punct);
301                 }
302                 navpop();
303             }
304             tagsAndPunct.sort(Comparator.naturalOrder());
305             Iterator<InlineObject> inlineObjectsIt = tagsAndPunct.iterator();
306             InlineObject nextInlineObject = inlineObjectsIt.hasNext() ? inlineObjectsIt.next() : null;
307 
308             // Now, find all words, keeping track of what inline objects occur in between.
309             navpush();
310             words.resetXPath();
311 
312             // first find all words and sort the list -- words are returned out of order when they are at different nesting levels
313             // since the xpath spec doesn't enforce any order, there's nothing we can do
314             // so record their positions, sort the list, then restore the position and carry on
315             List<Pair<Integer, BookMark>> wordPositions = new ArrayList<>();
316             while (words.evalXPath() != -1) {
317                 BookMark b = new BookMark(nav);
318                 b.setCursorPosition();
319                 wordPositions.add(Pair.of(nav.getCurrentIndex(), b));
320             }
321             wordPositions.sort((a, b) -> a.getKey().compareTo(b.getKey()));
322 
323             for (Pair<Integer, BookMark> wordPosition : wordPositions) {
324                 wordPosition.getValue().setCursorPosition();
325 
326                 // Capture tokenPositionId for this token position?
327                 if (apTokenPositionId != null) {
328                     apTokenPositionId.resetXPath();
329                     String tokenPositionId = apTokenPositionId.evalXPathToString();
330                     tokenPositionsMap.put(tokenPositionId, getCurrentTokenPosition());
331                 }
332 
333                 // Does an inline object occur before this word?
334                 long wordFragment = nav.getContentFragment();
335                 int wordOffset = (int) wordFragment;
336                 while (nextInlineObject != null && wordOffset >= nextInlineObject.getOffset()) {
337                     // Yes. Handle it.
338                     if (nextInlineObject.type() == InlineObjectType.PUNCTUATION)
339                         punctuation(nextInlineObject.getText());
340                     else
341                         inlineTag(nextInlineObject.getText(), nextInlineObject.type() == InlineObjectType.OPEN_TAG,
342                                 nextInlineObject.getAttributes());
343                     nextInlineObject = inlineObjectsIt.hasNext() ? inlineObjectsIt.next() : null;
344                 }
345 
346                 fragPos = FragmentPosition.BEFORE_OPEN_TAG;
347                 beginWord();
348 
349                 // For each configured annotation...
350                 int lastValuePosition = -1; // keep track of last value position so we can update lagging annotations
351                 for (ConfigAnnotation annotation : annotatedField.getAnnotations().values()) {
352                     processAnnotation(annotation, null);
353                     AnnotationWriter annotWriter = getAnnotation(annotation.getName());
354                     int lvp = annotWriter.lastValuePosition();
355                     if (lastValuePosition < lvp) {
356                         lastValuePosition = lvp;
357                     }
358                 }
359 
360                 fragPos = FragmentPosition.AFTER_CLOSE_TAG;
361                 endWord();
362                 
363                 // Add empty values to all lagging annotations
364                 for (AnnotationWriter prop: annotatedFieldWriter.annotationWriters()) {
365                     while (prop.lastValuePosition() < lastValuePosition) {
366                         prop.addValue("");
367                         if (prop.hasPayload())
368                             prop.addPayload(null);
369                     }
370                 }
371             }
372             navpop();
373 
374             // Handle any inline objects after the last word
375             while (nextInlineObject != null) {
376                 if (nextInlineObject.type() == InlineObjectType.PUNCTUATION)
377                     punctuation(nextInlineObject.getText());
378                 else
379                     inlineTag(nextInlineObject.getText(), nextInlineObject.type() == InlineObjectType.OPEN_TAG,
380                             nextInlineObject.getAttributes());
381                 nextInlineObject = inlineObjectsIt.hasNext() ? inlineObjectsIt.next() : null;
382             }
383 
384         }
385         navpop();
386 
387         // For each configured standoff annotation...
388         for (ConfigStandoffAnnotations standoff : annotatedField.getStandoffAnnotations()) {
389             // For each instance of this standoff annotation..
390             navpush();
391             AutoPilot apStandoff = acquireAutoPilot(standoff.getPath());
392             AutoPilot apTokenPos = acquireAutoPilot(standoff.getRefTokenPositionIdPath());
393             while (apStandoff.evalXPath() != -1) {
394 
395                 // Determine what token positions to index these values at
396                 navpush();
397                 List<Integer> tokenPositions = new ArrayList<>();
398                 apTokenPos.resetXPath();
399                 while (apTokenPos.evalXPath() != -1) {
400                     apEvalToString.resetXPath();
401                     String tokenPositionId = apEvalToString.evalXPathToString();
402                     Integer integer = tokenPositionsMap.get(tokenPositionId);
403                     if (integer == null)
404                         warn("Unresolved reference to token position: '" + tokenPositionId + "'");
405                     else
406                         tokenPositions.add(integer);
407                 }
408                 navpop();
409 
410                 for (ConfigAnnotation annotation : standoff.getAnnotations().values()) {
411                     processAnnotation(annotation, tokenPositions);
412                 }
413             }
414             releaseAutoPilot(apStandoff);
415             releaseAutoPilot(apTokenPos);
416             navpop();
417         }
418 
419         releaseAutoPilot(words);
420         releaseAutoPilot(apEvalToString);
421         for (AutoPilot ap : apsInlineTag) {
422             releaseAutoPilot(ap);
423         }
424         if (apPunct != null)
425             releaseAutoPilot(apPunct);
426         if (apTokenPositionId != null)
427             releaseAutoPilot(apTokenPositionId);
428         releaseAutoPilot(bodies);
429     }

431     protected void navpush() {
432         nav.push();
433         fragPosStack.add(fragPos);
434         fragPos = FragmentPosition.BEFORE_OPEN_TAG;
435     }
436 
437     protected void navpop() {
438         nav.pop();
439         fragPos = fragPosStack.remove(fragPosStack.size() - 1);
440     }

442     protected void processMetadataBlock(ConfigMetadataBlock b)
443             throws XPathParseException, XPathEvalException, NavException {
444         // For each instance of this metadata block...
445         navpush();
446         AutoPilot apMetadataBlock = acquireAutoPilot(b.getContainerPath());
447         while (apMetadataBlock.evalXPath() != -1) {
448 
449             // For each configured metadata field...
450             List<ConfigMetadataField> fields = b.getFields();
451             for (int i = 0; i < fields.size(); i++) { // NOTE: fields may be added during loop, so can't iterate
452                 ConfigMetadataField f = fields.get(i);
453 
454                 // Metadata field configs without a valuePath are just for
455                 // adding information about fields captured in forEach's,
456                 // such as extra processing steps
457                 if (f.getValuePath() == null || f.getValuePath().isEmpty())
458                     continue;
459 
460                 // Capture whatever this configured metadata field points to
461                 AutoPilot apMetadata = acquireAutoPilot(f.getValuePath());
462                 if (f.isForEach()) {
463                     // "forEach" metadata specification
464                     // (allows us to capture many metadata fields with 3 XPath expressions)
465                     navpush();
466                     AutoPilot apMetaForEach = acquireAutoPilot(f.getForEachPath());
467                     AutoPilot apFieldName = acquireAutoPilot(f.getName());
468                     while (apMetaForEach.evalXPath() != -1) {
469                         // Find the fieldName and value for this forEach match
470                         apFieldName.resetXPath();
471                         String origFieldName = apFieldName.evalXPathToString();
472                         String fieldName = AnnotatedFieldNameUtil.sanitizeXmlElementName(origFieldName);
473                         if (!origFieldName.equals(fieldName)) {
474                             warnSanitized(origFieldName, fieldName);
475                         }
476                         ConfigMetadataField metadataField = b.getOrCreateField(fieldName);
477 
478                         apMetadata.resetXPath();
479 
480                         // Multiple matches will be indexed at the same position.
481                         AutoPilot apEvalToString = acquireAutoPilot(".");
482                         while (apMetadata.evalXPath() != -1) {
483                             apEvalToString.resetXPath();
484                             String unprocessedValue = apEvalToString.evalXPathToString();
485                             for (String value : processStringMultipleValues(unprocessedValue, f.getProcess(), null)) {
486                                 // Also execute process defined for named metadata field, if any
487                                 for (String processedValue : processStringMultipleValues(value, metadataField.getProcess(), metadataField.getMapValues())) {
488                                     addMetadataField(fieldName, processedValue);
489                                 }
490                             }
491                         }
492                         releaseAutoPilot(apEvalToString);
493                     }
494                     releaseAutoPilot(apMetaForEach);
495                     releaseAutoPilot(apFieldName);
496                     navpop();
497                 } else {
498                     // Regular metadata field; just the fieldName and an XPath expression for the value
499                     // Multiple matches will be indexed at the same position.
500                     AutoPilot apEvalToString = acquireAutoPilot(".");
501                     try {
502                         while (apMetadata.evalXPath() != -1) {
503                             apEvalToString.resetXPath();
504                             String unprocessedValue = apEvalToString.evalXPathToString();
505                             for (String value : processStringMultipleValues(unprocessedValue, f.getProcess(), f.getMapValues())) {
506                                 addMetadataField(f.getName(), value);
507                             }
508                         }
509                     } catch(XPathEvalException e) {
510                         /*
511                         An xpath like string(@value) will make evalXPath() fail.
512                         There is no good way to check wether this exception will occur
513                         When the exception occurs we try to evaluate the xpath as string
514                         NOTE: an xpath with dot like: string(.//tei:availability[1]/@status='free') may fail silently!!
515                          */
516                         if (logger.isDebugEnabled()) {
517                             logger.debug(String.format("An xpath with a dot like %s may fail silently and may have to be replaced by one like %s",
518                                     "string(.//tei:availability[1]/@status='free')",
519                                     "string(//tei:availability[1]/@status='free')"));
520                         }
521                         String metadataValue = apMetadata.evalXPathToString();
522                         metadataValue = processString(metadataValue, f.getProcess(), f.getMapValues());
523                         addMetadataField(f.getName(), metadataValue);
524                     }                    releaseAutoPilot(apEvalToString);
525                 }
526                 releaseAutoPilot(apMetadata);
527             }
528 
529         }
530         releaseAutoPilot(apMetadataBlock);
531         navpop();
532     }

543     /**
544      * Process an annotation at the current position.
545      *
546      * @param annotation annotation to process
547      * @param indexAtPositions if null: index at the current position; otherwise,
548      *            index at all these positions
549      * @throws VTDException on XPath error
550      */
551     protected void processAnnotation(ConfigAnnotation annotation, List<Integer> indexAtPositions) throws VTDException {
552         String basePath = annotation.getBasePath();
553         if (basePath != null) {
554             // Basepath given. Navigate to the (first) matching element and evaluate the other XPaths from there.
555             navpush();
556             AutoPilot apBase = acquireAutoPilot(basePath);
557             apBase.evalXPath();
558             releaseAutoPilot(apBase);
559         }
560         try {
561             String valuePath = annotation.getValuePath();
562             if (valuePath == null) {
563                 // No valuePath given. Assume this will be captures using forEach.
564                 return;
565             }
566 
567             // See if we want to capture any values and substitute them into the XPath
568             int i = 1;
569             for (String captureValuePath : annotation.getCaptureValuePaths()) {
570                 AutoPilot apCaptureValuePath = acquireAutoPilot(captureValuePath);
571                 String value = apCaptureValuePath.evalXPathToString();
572                 releaseAutoPilot(apCaptureValuePath);
573                 valuePath = valuePath.replace("$" + i, value);
574                 i++;
575             }
576 
577             // Find matches for this annotation.
578             String annotValue = findAnnotationMatches(annotation, valuePath, indexAtPositions, null);
579 
580             // For each configured subannotation...
581             Set<String> alreadySeen = new HashSet<>(); // keep track of which annotation have multiple values so we can use the correct position increment 
582             for (ConfigAnnotation subAnnot : annotation.getSubAnnotations()) {
583                 // Subannotation configs without a valuePath are just for
584                 // adding information about subannotations captured in forEach's,
585                 // such as extra processing steps
586                 if (subAnnot.getValuePath() == null || subAnnot.getValuePath().isEmpty())
587                     continue;
588 
589                 // Capture this subannotation value
590                 AutoPilot apValue = acquireAutoPilot(subAnnot.getValuePath());
591                 if (subAnnot.isForEach()) {
592                     // "forEach" subannotation specification
593                     // (allows us to capture multiple subannotations with 3 XPath expressions)
594                     navpush();
595                     AutoPilot apForEach = acquireAutoPilot(subAnnot.getForEachPath());
596                     AutoPilot apName = acquireAutoPilot(subAnnot.getName());
597                     alreadySeen.clear();
598                     while (apForEach.evalXPath() != -1) {
599                         // Find the name and value for this forEach match
600                         apName.resetXPath();
601                         apValue.resetXPath();
602 
603                         String name = apName.evalXPathToString();
604                         String subannotationName = annotation.getName() + AnnotatedFieldNameUtil.SUBANNOTATION_FIELD_PREFIX_SEPARATOR + name;
605                         ConfigAnnotation actualSubAnnot = annotation.getSubAnnotation(subannotationName);
606 
607                         String value = null;
608                         if (actualSubAnnot != null) {
609                             value = actualSubAnnot.isCaptureXml()? apValue.evalXPath() != -1 ? getXml(apValue) : "" : apValue.evalXPathToString(); 
610                             value = processString(value, subAnnot.getProcess(), null);
611                             // Also apply process defined in named subannotation, if any
612                             value = processString(value, actualSubAnnot.getProcess(), null);
613                         } else {
614                             value = subAnnot.isCaptureXml()? apValue.evalXPath() != -1 ? getXml(apValue) : "" : apValue.evalXPathToString();
615                             value = processString(value, subAnnot.getProcess(), null);
616                         }
617                         if (!alreadySeen.contains(subannotationName)) {
618                             // First occurrence of this annotation
619                             annotation(subannotationName, value, 1, indexAtPositions);
620                             alreadySeen.add(subannotationName);
621                         } else {
622                             // Subsequent occurrence of this annotation
623                             annotation(subannotationName, value, 0, indexAtPositions);
624                         }
625                     }
626                     releaseAutoPilot(apForEach);
627                     releaseAutoPilot(apName);
628                     navpop();
629                 } else {
630                     // Regular subannotation; just the fieldName and an XPath expression for the value
631                     String subValuePath = subAnnot.getValuePath();
632                     String reuseValue = subValuePath.equals(valuePath) && subAnnot.isCaptureXml() == annotation.isCaptureXml() ? annotValue : null;
633                     findAnnotationMatches(subAnnot, subValuePath, indexAtPositions, reuseValue);
634                 }
635                 releaseAutoPilot(apValue);
636             }
637 
638         } finally {
639             if (basePath != null) {
640                 // We pushed when we navigated to the base element; pop now.
641                 navpop();
642             }
643         }
644     }

646     protected String findAnnotationMatches(ConfigAnnotation annotation, String valuePath,
647             List<Integer> indexAtPositions, String reuseValueFromParentAnnot)
648             throws XPathParseException, XPathEvalException, NavException {
649         String annotValueForReuse = null;
650         boolean evalXml = annotation.isCaptureXml();
651 
652         if (reuseValueFromParentAnnot == null) {
653             navpush();
654             AutoPilot apValuePath = acquireAutoPilot(valuePath);
655             if (annotation.isMultipleValues()) {
656                 // If we don't want duplicates, keep track of the values we've indexed
657                 Set<String> valuesAlreadyIndexed = null;
658                 if (!annotation.isAllowDuplicateValues())
659                     valuesAlreadyIndexed = new HashSet<>();
660                 
661                 // Multiple matches will be indexed at the same position.
662                 AutoPilot apValue = acquireAutoPilot(".");
663                 boolean firstValue = true;
664                 while (apValuePath.evalXPath() != -1) {
665                     apValue.resetXPath();
666                     String unprocessedValue = evalXml ? apValue.evalXPath() != -1 ? getXml(apValue) : "" : apValue.evalXPathToString();
667                     for (String value : processStringMultipleValues(unprocessedValue, annotation.getProcess(), null)) {
668                         if (valuesAlreadyIndexed == null || !valuesAlreadyIndexed.contains(value.toLowerCase())) {
669                             int increment = firstValue ? 1 : 0;
670                             annotation(annotation.getName(), value, increment, indexAtPositions);
671                             firstValue = false;
672                             if (valuesAlreadyIndexed != null) {
673                                 // Keep track of values seen so we can discard duplicates
674                                 valuesAlreadyIndexed.add(value.toLowerCase());
675                             }
676                         }
677                     }
678                 }
679                 releaseAutoPilot(apValue);
680 
681                 // No annotations have been added, the result of the xPath query must have been empty.
682                 if (firstValue) {
683                     // Add default value(s)
684                     for (String value : processStringMultipleValues("", annotation.getProcess(), null)) {
685                         int increment = firstValue ? 1 : 0;
686                         annotation(annotation.getName(), value, increment, indexAtPositions);
687                     }
688                 }
689             } else {
690                 // Single value expected
691                 annotValueForReuse = evalXml ? apValuePath.evalXPath() != -1 ? getXml(apValuePath) : "" : apValuePath.evalXPathToString();
692                 String annotValue = processString(annotValueForReuse, annotation.getProcess(), null);
693                 annotation(annotation.getName(), annotValue, 1, indexAtPositions);
694             }
695             releaseAutoPilot(apValuePath);
696             navpop();
697         } else {
698             // We can reuse the value from the parent annotation, with different processing
699             annotValueForReuse = reuseValueFromParentAnnot;
700             String annotValue = processString(annotValueForReuse, annotation.getProcess(), null);
701             annotation(annotation.getName(), annotValue, 1, indexAtPositions);
702         }
703         return annotValueForReuse; // so subannotations can reuse it if they use the same valuePath
704     }

706     @Override
707     public void indexSpecificDocument(String documentXPath) {
708         super.indexSpecificDocument(documentXPath);
709 
710         try {
711             // Parse use VTD-XML
712             vg = new VTDGen();
713             vg.setDoc(inputDocument);
714             vg.parse(config.isNamespaceAware());
715 
716             nav = vg.getNav();
717 
718             boolean docDone = false;
719             if (documentXPath != null) {
720                 // Find our specific document
721                 AutoPilot documents = acquireAutoPilot(documentXPath);
722                 while (documents.evalXPath() != -1) {
723                     if (docDone)
724                         throw new BlackLabRuntimeException(
725                                 "Document link " + documentXPath + " matched multiple documents in " + documentName);
726                     indexDocument();
727                     docDone = true;
728                 }
729                 releaseAutoPilot(documents);
730             } else {
731                 // Process whole file; must be 1 document
732                 AutoPilot documents = acquireAutoPilot(config.getDocumentPath());
733                 while (documents.evalXPath() != -1) {
734                     if (docDone)
735                         throw new BlackLabRuntimeException(
736                                 "Linked file contains multiple documents (and no document path given) in "
737                                         + documentName);
738                     indexDocument();
739                     docDone = true;
740                 }
741                 releaseAutoPilot(documents);
742             }
743         } catch (Exception e1) {
744             throw BlackLabRuntimeException.wrap(e1);
745         }
746     }

748     /**
749      * Add open and close InlineObject objects for the current element to the list.
750      *
751      * @param inlineObject list to add the new open/close tag objects to
752      * @throws NavException
753      */
754     private void collectInlineTag(List<InlineObject> inlineObject) throws NavException {
755         // Get the element and content fragments
756         // (element fragment = from start of start tag to end of end tag;
757         //  content fragment = from end of start tag to start of end tag)
758         long elementFragment = nav.getElementFragment();
759         int startTagOffset = (int) elementFragment;
760         int endTagOffset;
761         long contentFragment = nav.getContentFragment();
762         if (contentFragment == -1) {
763             // Empty (self-closing) element.
764             endTagOffset = startTagOffset;
765         } else {
766             // Regular element with separate open and close tags.
767             int contentOffset = (int) contentFragment;
768             int contentLength = (int) (contentFragment >> 32);
769             int contentEnd = contentOffset + contentLength;
770             endTagOffset = contentEnd;
771         }
772 
773         // Find element name
774         int currentIndex = nav.getCurrentIndex();
775         String elementName = dedupe(nav.toString(currentIndex));
776 
777         // Add the inline tags to the list
778         InlineObject openTag = new InlineObject(elementName, startTagOffset, InlineObjectType.OPEN_TAG,
779                 getAttributes());
780         InlineObject closeTag = new InlineObject(elementName, endTagOffset, InlineObjectType.CLOSE_TAG, null);
781         openTag.setMatchingTag(closeTag);
782         closeTag.setMatchingTag(openTag);
783         inlineObject.add(openTag);
784         inlineObject.add(closeTag);
785     }

788      * Add InlineObject for a punctuation text node.
789      *
790      * @param inlineObjects list to add the punct object to
791      * @param text
792      * @throws NavException
793      */
794     private void collectPunct(List<InlineObject> inlineObjects, String text) throws NavException {
795         int i = nav.getCurrentIndex();
796         int offset = nav.getTokenOffset(i);
797 //		int length = nav.getTokenLength(i);
798 
799         // Make sure we only keep 1 copy of identical punct texts in memory
800         text = dedupe(StringUtil.normalizeWhitespace(text));
801 
802         // Add the punct to the list
803         inlineObjects.add(new InlineObject(text, offset, InlineObjectType.PUNCTUATION, null));
804     }

806     /**
807      * Gets attribute map for current element
808      */
809     private Map<String, String> getAttributes() {
810         navpush();
811         AutoPilot apAttr = new AutoPilot(nav);
812         apAttr.selectAttr("*");
813         int i = -1;
814         Map<String, String> attr = new HashMap<>();
815         try {
816             while ((i = apAttr.iterateAttr()) != -1) {
817                 String name = nav.toString(i);
818                 String value = nav.toString(i + 1);
819                 attr.put(name, value);
820             }
821         } catch (NavException e) {
822             throw BlackLabRuntimeException.wrap(e);
823         }
824         navpop();
825         return attr;
826     }

849     @Override
850     protected int getCharacterPosition() {
851         // VTD-XML provides no way of getting the current character position,
852         // only the byte position.
853         // In order to keep track of character position (which we need for Lucene's term vector),
854         // we fetch the bytes processed since this method was last called, convert them to a String,
855         // and use the string length to adjust the character position.
856         // Note that this only works if this method is called for increasing byte positions,
857         // which is true because we only use it for word tags.
858         try {
859             int currentByteOffset = getCurrentByteOffset();
860             if (currentByteOffset > lastCharPositionByteOffset) {
861                 int length = currentByteOffset - lastCharPositionByteOffset;
862                 String str = new String(inputDocument, lastCharPositionByteOffset, length, StandardCharsets.UTF_8);
863                 lastCharPosition += str.length();
864                 lastCharPositionByteOffset = currentByteOffset;
865             }
866             return lastCharPosition;
867         } catch (NavException e) {
868             throw BlackLabRuntimeException.wrap(e);
869         }
870     }
871 
872     protected int getCurrentByteOffset() throws NavException {
873         if (fragPos == FragmentPosition.BEFORE_OPEN_TAG || fragPos == FragmentPosition.AFTER_CLOSE_TAG) {
874             long elFrag = nav.getElementFragment();
875             int elOffset = (int) elFrag;
876             if (fragPos == FragmentPosition.AFTER_CLOSE_TAG) {
877                 int elLength = (int) (elFrag >> 32);
878                 return elOffset + elLength;
879             }
880             return elOffset;
881         }
882         long contFrag = nav.getContentFragment();
883         int contOffset = (int) contFrag;
884         if (fragPos == FragmentPosition.BEFORE_CLOSE_TAG) {
885             int contLength = (int) (contFrag >> 32);
886             return contOffset + contLength;
887         }
888         return contOffset;
889     }

896     /** Get the raw xml from the document at the current position 
897      * @throws NavException */
898     private static String getXml(AutoPilot ap) throws NavException {
899         long frag = ap.getNav().getContentFragment();
900         if (frag == -1) {
901             return "";
902         }
903 
904         int offset = (int) frag;
905         int length = (int) (frag >> 32);
906 
907         return ap.getNav().toRawString(offset, length);
908     }
