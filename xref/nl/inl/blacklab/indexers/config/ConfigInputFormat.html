54      /** file type options for a FileType */
55      public enum FileTypeOption {
56          VTD(FileType.XML, Constants.PROCESSING), SAXONICA(FileType.XML, Constants.PROCESSING);
57          private final FileType fileType;
58          private final String key;
59  
60          FileTypeOption(FileType fileType, String key) {
61              this.fileType = fileType;
62              this.key = key;
63          }
64  
65          public FileType getFileType() {
66              return fileType;
67          }
68  
69          public String getKey() {
70              return key;
71          }
72  
73          public static FileTypeOption byKeyValue(String key, String value) {
74              for (FileTypeOption fto : values()) {
75                  if (fto.getKey().equals(key) && fto.name().equalsIgnoreCase(value)) {
76                      return fto;
77                  }
78              }
79              return null;
80          }
81  
82          public static List<FileTypeOption> forFileType(FileType fileType) {
83              return Arrays.stream(values()).filter(fto -> fto.fileType == fileType).collect(Collectors.toList());
84          }
85  
86          public static List<FileTypeOption> fromConfig (ConfigInputFormat config, FileType ft) {
87              return config.getFileTypeOptions().entrySet().stream()
88                      .filter(opt -> byKeyValue(opt.getKey(),opt.getValue()) !=null && byKeyValue(opt.getKey(),opt.getValue()).fileType==ft)
89                      .map(opt -> byKeyValue(opt.getKey(),opt.getValue())).collect(Collectors.toList());
90          }

221     /**
222      * Copy everything except name, displayName and description from the specified
223      * format.
224      * 
225      * @param baseFormat format to copy from
226      */
227     public void setBaseFormat(ConfigInputFormat baseFormat) {
228         type = baseFormat.getType();
229         fileType = baseFormat.getFileType();
230         if (baseFormat.getFileTypeOptions() != null)
231             fileTypeOptions.putAll(baseFormat.getFileTypeOptions());
232 //        if (baseFormat.getTabularOptions() != null)
233 //            tabularOptions = baseFormat.getTabularOptions().copy();
234         corpusConfig = baseFormat.corpusConfig.copy();
235         namespaces.putAll(baseFormat.getNamespaces());
236         documentPath = baseFormat.getDocumentPath();
237         store = baseFormat.shouldStore();
238         indexFieldAs.putAll(baseFormat.getIndexFieldAs());
239         metadataDefaultAnalyzer = baseFormat.getMetadataDefaultAnalyzer();
240         metadataDefaultUnknownCondition = baseFormat.getMetadataDefaultUnknownCondition();
241         metadataDefaultUnknownValue = baseFormat.getMetadataDefaultUnknownValue();
242         for (ConfigMetadataBlock b : baseFormat.getMetadataBlocks()) {
243             addMetadataBlock(b.copy());
244         }
245         for (ConfigAnnotatedField f : baseFormat.getAnnotatedFields().values()) {
246             addAnnotatedField(f.copy());
247         }
248         linkedDocuments.putAll(baseFormat.getLinkedDocuments());
249         setVisible(baseFormat.isVisible());
250     }

252     /**
253      * Validate this configuration.
254      */
255     public void validate() {
256         String t = "input format";
257         req(name, t, "name");
258         req(documentPath, t, "documentPath");
259 //        if (tabularOptions != null)
260 //            tabularOptions.validate();
261         for (ConfigMetadataBlock b : metadataBlocks)
262             b.validate();
263         for (ConfigAnnotatedField af : annotatedFields.values()) {
264             if (fileType != FileType.XML)
265                 af.setWordPath("N/A"); // prevent validation error
266             af.validate();
267         }
268         for (ConfigLinkedDocument ld : linkedDocuments.values())
269             ld.validate();
270     }

275     }

482     public static String stripExtensions(String fileName) {
483         String name = fileName.replaceAll("\\.(ya?ml|json)$", "");
484         if (name.endsWith(".blf"))
485             return name.substring(0, name.length() - 4);
486         return name;
487     }

493     public BufferedReader getFormatFile() {
494         try {
495             if (readFromFile == null)
496                 return null;
497 
498             if (readFromFile.getPath().startsWith("$BLACKLAB_JAR")) {
499                 InputStream stream = DocumentFormats.class.getClassLoader()
500                         .getResourceAsStream("formats/" + getName() + ".blf.yaml");
501                 return new BufferedReader(new InputStreamReader(stream, StandardCharsets.UTF_8));
502             }
503             return FileUtil.openForReading(readFromFile);
504         } catch (FileNotFoundException e) {
505             throw BlackLabRuntimeException.wrap(e);
506         }
507     }
