157     protected AnnotatedFieldWriter getMainAnnotatedField() {
158         if (mainAnnotatedField == null) {
159             // The "main annotated field" is the field that stores the document content id for now.
160             // (We will change this eventually so the document content id is not stored with a annotated field
161             // but as a metadata field instead.)
162             // The main annotated field is a field named "contents" or, if that does not exist, the first
163             // annotated field
164             for (AnnotatedFieldWriter field : annotatedFields.values()) {
165                 if (mainAnnotatedField == null)
166                     mainAnnotatedField = field;
167                 else if (field.name().equals("contents"))
168                     mainAnnotatedField = field;
169             }
170         }
171         return mainAnnotatedField;
172     }

224     /**
225      * Index a linked document.
226      *
227      * @param inputFile where the linked document can be found (file or http(s)
228      *            reference)
229      * @param pathInsideArchive if input file is an archive: the path to the file we
230      *            need inside the archive
231      * @param documentPath XPath to the specific linked document we need
232      * @param inputFormatIdentifier input format of the linked document
233      * @param storeWithName if set, store the linked document and store the id to it
234      *            in a field with this name with "Cid" (content id) appended to it
235      * @throws IOException on error
236      */
237     protected void indexLinkedDocument(String inputFile, String pathInsideArchive, String documentPath,
238             String inputFormatIdentifier, String storeWithName) throws IOException {
239         // Fetch the input file (either by downloading it to a temporary location, or opening it from disk)
240         File f = resolveFileReference(inputFile);
241 
242         // Get the data
243         byte[] data;
244         String completePath = inputFile;
245         if (inputFile.endsWith(".zip") || inputFile.endsWith(".tar") || inputFile.endsWith(".tar.gz")
246                 || inputFile.endsWith(".tgz")) {
247             // It's an archive. Unpack the right file from it.
248             completePath += "/" + pathInsideArchive;
249             data = FileProcessor.fetchFileFromArchive(f, pathInsideArchive);
250         } else {
251             // Regular file.
252             try (InputStream is = new FileInputStream(f)) {
253                 data = IOUtils.toByteArray(is);
254             } catch (IOException e) {
255                 throw BlackLabRuntimeException.wrap(e);
256             }
257         }
258         if (data == null) {
259             throw new BlackLabRuntimeException("Error reading linked document");
260         }
261 
262         // Index the data
263         try (DocIndexer docIndexer = DocumentFormats.get(inputFormatIdentifier, docWriter, completePath, data,
264                 Indexer.DEFAULT_INPUT_ENCODING)) {
265             if (docIndexer instanceof DocIndexerBase) {
266                 @SuppressWarnings("resource")
267                 DocIndexerBase ldi = (DocIndexerBase) docIndexer;
268                 ldi.indexingIntoExistingLuceneDoc = true;
269                 ldi.currentLuceneDoc = currentLuceneDoc;
270                 ldi.metadataFieldValues = metadataFieldValues;
271                 if (storeWithName != null) {
272                     // If specified, store in this content store and under this name instead of the default
273                     ldi.contentStoreName = storeWithName;
274                 }
275                 ldi.indexSpecificDocument(documentPath);
276             } else {
277                 throw new BlackLabRuntimeException("Linked document indexer must be subclass of DocIndexerBase, but is "
278                         + docIndexer.getClass().getName());
279             }
280         }
281 
282     }

294     /**
295      * Given a URL or file reference, either download to a temp file or find file
296      * and return it.
297      *
298      * @param inputFile URL or (relative) file reference
299      * @return the file
300      * @throws IOException
301      */
302     protected File resolveFileReference(String inputFile) throws IOException {
303         if (inputFile.startsWith("http://") || inputFile.startsWith("https://")) {
304             return DownloadCache.downloadFile(inputFile);
305         }
306         if (inputFile.startsWith("file://"))
307             inputFile = inputFile.substring(7);
308         File f = docWriter.linkedFile(inputFile);
309         if (f == null)
310             throw new FileNotFoundException("Referenced file not found");
311         if (!f.canRead())
312             throw new IOException("Cannot read referenced file " + f);
313         return f;
314     }

350     protected void startDocument() {
351 
352         traceln("START DOCUMENT");
353         if (!indexingIntoExistingLuceneDoc) {
354             currentLuceneDoc = new Document();
355             addMetadataField("fromInputFile", documentName);
356             addMetadataFieldsFromParameters(); // DEPRECATED for these types of indexer, but still supported for now
357         }
358         if (docWriter != null && !indexingIntoExistingLuceneDoc)
359             docWriter.listener().documentStarted(documentName);
360     }
361 
362     protected void endDocument() {
363         traceln("END DOCUMENT");
364 
365         for (AnnotatedFieldWriter field : getAnnotatedFields().values()) {
366             AnnotationWriter propMain = field.mainAnnotation();
367 
368             // Make sure all the annotations have an equal number of values.
369             // See what annotation has the highest position
370             // (in practice, only starttags and endtags should be able to have
371             // a position one higher than the rest)
372             int lastValuePos = 0;
373             for (AnnotationWriter prop : field.annotationWriters()) {
374                 if (prop.lastValuePosition() > lastValuePos)
375                     lastValuePos = prop.lastValuePosition();
376             }
377 
378             // Make sure we always have one more token than the number of
379             // words, so there's room for any tags after the last word, and we
380             // know we should always skip the last token when matching.
381             if (propMain.lastValuePosition() == lastValuePos)
382                 lastValuePos++;
383 
384             // Add empty values to all lagging annotations
385             for (AnnotationWriter prop : field.annotationWriters()) {
386                 while (prop.lastValuePosition() < lastValuePos) {
387                     prop.addValue("");
388                     if (prop.hasPayload())
389                         prop.addPayload(null);
390                     if (prop == propMain) {
391                         field.addStartChar(getCharacterPosition());
392                         field.addEndChar(getCharacterPosition());
393                     }
394                 }
395             }
396             // Store the different annotations of the annotated field that
397             // were gathered in lists while parsing.
398             field.addToLuceneDoc(currentLuceneDoc);
399 
400             // Add the field with all its annotations to the forward index
401             addToForwardIndex(field);
402 
403         }
404 
405         if (isStoreDocuments()) {
406             storeDocument();
407         }
408 
409         if (docWriter != null) {
410             // If there's an external metadata fetcher, call it now so it can
411             // add the metadata for this document and (optionally) store the
412             // metadata
413             // document in the content store (and the corresponding id in the
414             // Lucene doc)
415             MetadataFetcher m = getMetadataFetcher();
416             if (m != null) {
417                 m.addMetadata();
418             }
419 
420         }
421 
422         if (!indexingIntoExistingLuceneDoc)
423             addMetadataToDocument();
424         try {
425             // Add Lucene doc to indexer, if not existing already
426             if (docWriter != null && !indexingIntoExistingLuceneDoc)
427                 docWriter.add(currentLuceneDoc);
428         } catch (Exception e) {
429             throw BlackLabRuntimeException.wrap(e);
430         }
431 
432         for (AnnotatedFieldWriter annotatedField : getAnnotatedFields().values()) {
433             // Reset annotated field for next document
434             // don't reuse buffers, they're still referenced by the lucene doc.
435             annotatedField.clear(!indexingIntoExistingLuceneDoc);
436         }
437 
438         // Report progress
439         if (docWriter != null) {
440             reportCharsProcessed();
441             reportTokensProcessed();
442         }
443         if (docWriter != null && !indexingIntoExistingLuceneDoc)
444             docWriter.listener().documentDone(documentName);
445 
446         currentLuceneDoc = null;
447 
448         // Stop if required
449         if (docWriter != null) {
450             if (!docWriter.continueIndexing())
451                 throw new MaxDocsReached();
452         }
453 
454         uniqueStrings.clear();
455     }

457     /**
458      * Store the entire document at once.
459      *
460      * Subclasses that simply capture the entire document can use this in their
461      * storeDocument implementation.
462      * 
463      * @param document document to store
464      */
465     protected void storeWholeDocument(String document) {
466         // Finish storing the document in the document store,
467         // retrieve the content id, and store that in Lucene.
468         // (Note that we do this after adding the dummy token, so the character
469         // positions for the dummy token still make (some) sense)
470         String contentIdFieldName;
471         String contentStoreName = getContentStoreName();
472         if (contentStoreName == null) {
473             AnnotatedFieldWriter main = getMainAnnotatedField();
474             if (main == null) {
475                 contentStoreName = "metadata";
476                 contentIdFieldName = "metadataCid";
477             } else {
478                 contentStoreName = main.name();
479                 contentIdFieldName = AnnotatedFieldNameUtil.contentIdField(main.name());
480             }
481         } else {
482             contentIdFieldName = contentStoreName + "Cid";
483         }
484         int contentId = -1;
485         if (docWriter != null) {
486             ContentStore contentStore = docWriter.contentStore(contentStoreName);
487             contentId = contentStore.store(document);
488         }
489         currentLuceneDoc.add(new IntField(contentIdFieldName, contentId, Store.YES));
490     }

499     protected void inlineTag(String tagName, boolean isOpenTag, Map<String, String> attributes) {
500         if (isOpenTag) {
501             trace("<" + tagName + ">");
502 
503             int lastStartTagPos = propTags().lastValuePosition();
504             int currentPos = getCurrentTokenPosition();
505             int posIncrement = currentPos - lastStartTagPos;
506             propTags().addValue(tagName, posIncrement);
507             propTags().addPayload(null);
508             int startTagIndex = propTags().lastValueIndex();
509             openInlineTags.add(new OpenTagInfo(tagName, startTagIndex));
510 
511             for (Entry<String, String> e : attributes.entrySet()) {
512                 // Index element attribute values
513                 String name = e.getKey();
514                 String value = e.getValue();
515                 propTags().addValue("@" + name.toLowerCase() + "__" + value.toLowerCase(), 0);
516                 propTags().addPayload(null);
517             }
518 
519         } else {
520             traceln("</" + tagName + ">");
521 
522             int currentPos = getCurrentTokenPosition();
523 
524             // Add payload to start tag annotation indicating end position
525             if (openInlineTags.isEmpty())
526                 throw new MalformedInputFile("Close tag " + tagName + " found, but that tag is not open");
527             OpenTagInfo openTag = openInlineTags.remove(openInlineTags.size() - 1);
528             if (!openTag.name.equals(tagName))
529                 throw new MalformedInputFile(
530                         "Close tag " + tagName + " found, but " + openTag.name + " expected");
531             byte[] payload = ByteBuffer.allocate(4).putInt(currentPos).array();
532             propTags().setPayloadAtIndex(openTag.index, new BytesRef(payload));
533         }
534     }

557      * calls {@link #getCharacterPosition()}

559     protected void endWord() {
560         String punct;
561         if (punctuation.length() == 0)
562             punct = addDefaultPunctuation && !preventNextDefaultPunctuation ? " " : "";
563         else
564             punct = punctuation.toString();
565 
566         preventNextDefaultPunctuation = false;
567         // Normalize once more in case we hit more than one adjacent punctuation
568         propPunct().addValue(StringUtil.normalizeWhitespace(punct));
569         addEndChar(getCharacterPosition());
570         wordsDone++;
571         if (wordsDone > 0 && wordsDone % 5000 == 0) {
572             reportCharsProcessed();
573             reportTokensProcessed();
574         }
575         if (punctuation.length() > 10_000)
576             punctuation = new StringBuilder(); // let's not hold on to this much memory
577         else
578             punctuation.setLength(0);
579     }

581     /**
582      * Index an annotation.
583      *
584      * Can be used to add annotation(s) at the current position (indexAtPositions ==
585      * null), or to add annotations at specific positions (indexAtPositions contains
586      * positions). The latter is used for standoff annotations.
587      *
588      * Also called for subannotations (with the value already prepared)
589      *
590      * @param name annotation name
591      * @param value annotation value
592      * @param increment if indexAtPosition == null: the token increment to use
593      * @param indexAtPositions if null: index at the current position; otherwise:
594      *            index at these positions
595      */
596     protected void annotation(String name, String value, int increment, List<Integer> indexAtPositions) {
597         AnnotationWriter annotation = getAnnotation(name);
598         if (annotation != null) {
599             if (indexAtPositions == null) {
600                 if (name.equals("word"))
601                     trace(value + " ");
602                 annotation.addValue(value, increment);
603             } else {
604                 for (Integer position : indexAtPositions) {
605                     annotation.addValueAtPosition(value, position);
606                 }
607             }
608         } else {
609             // Annotation not declared; report, but keep going
610             if (!skippedAnnotations.contains(name)) {
611                 skippedAnnotations.add(name);
612                 logger.error(documentName + ": skipping undeclared annotation " + name);
613             }
614         }
615     }

624     @Override
625     public final void reportCharsProcessed() {
626         final int charsDone = getCharacterPosition();
627         final int charsDoneSinceLastReport;
628         if (charsDoneAtLastReport > charsDone)
629             charsDoneSinceLastReport = charsDone;
630         else
631             charsDoneSinceLastReport = charsDone - charsDoneAtLastReport;
632 
633         docWriter.listener().charsDone(charsDoneSinceLastReport);
634         charsDoneAtLastReport = charsDone;
635     }

637     /**
638      * Report the change in wordsDone since the last report
639      */
640     @Override
641     public final void reportTokensProcessed() {
642         final int wordsDoneSinceLastReport;
643         if (wordsDoneAtLastReport > wordsDone) // wordsDone reset by child class? report everything then
644             wordsDoneSinceLastReport = wordsDone;
645         else
646             wordsDoneSinceLastReport = wordsDone - wordsDoneAtLastReport;
647 
648         docWriter.listener().tokensDone(wordsDoneSinceLastReport);
649         wordsDoneAtLastReport = wordsDone;
650     }

652     /**
653      * Get the external metadata fetcher for this indexer, if any.
654      *
655      * The metadata fetcher can be configured through the "metadataFetcherClass"
656      * parameter.
657      *
658      * @return the metadata fetcher if any, or null if there is none.
659      */
660     protected MetadataFetcher getMetadataFetcher() {
661         if (metadataFetcher == null) {
662             @SuppressWarnings("deprecation")
663             String metadataFetcherClassName = getParameter("metadataFetcherClass");
664             if (metadataFetcherClassName != null) {
665                 try {
666                     Class<? extends MetadataFetcher> metadataFetcherClass = Class.forName(metadataFetcherClassName)
667                             .asSubclass(MetadataFetcher.class);
668                     Constructor<? extends MetadataFetcher> ctor = metadataFetcherClass.getConstructor(DocIndexer.class);
669                     metadataFetcher = ctor.newInstance(this);
670                 } catch (ReflectiveOperationException e) {
671                     throw BlackLabRuntimeException.wrap(e);
672                 }
673             }
674         }
675         return metadataFetcher;
676     }
