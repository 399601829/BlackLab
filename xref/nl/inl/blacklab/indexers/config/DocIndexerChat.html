213     /**
214      * Slice a string like Python.
215      *
216      * Negative indices are counted from the end of the string. Indices out of range
217      * result in an empty string, not an exception.
218      *
219      * @param str string to slice
220      * @param start starting point
221      * @param end end point
222      * @return slice
223      */
224     private static String slice(String str, int start, int end) {
225         if (start < 0) {
226             start += str.length();
227             if (start < 0)
228                 start = 0;
229         }
230         if (end < 0) {
231             end += str.length();
232             if (end < 0)
233                 end = 0;
234         }
235         if (start > str.length())
236             start = str.length();
237         if (end > str.length())
238             end = str.length();
239         return str.substring(start, end);
240     }

317     // Convert date string to month number
318     private int getMonths(String age) {
319         // input format is 3;6.14 (y;m.d)
320         // also accept y.m.d and y;m;d and y.m;d with a warning or any separators for that matter
321         String cleanAge = age.trim();
322         boolean errorFound = false;
323         boolean warningNeeded = false;
324         String monthStr = "";
325         String yearStr = "";
326         String[] parts = cleanAge.split(SEPS, -1);
327         // print(age, thelist, file = logfile)
328         // print(input("continue?"), file = logfile)
329         if (parts.length >= 1) {
330             yearStr = parts[0];
331             if (!yearStr.matches("[0-9]+"))
332                 errorFound = true;
333         }
334         if (parts.length >= 2) {
335             monthStr = parts[1];
336             if (!monthStr.matches("[0-9]{1,2}"))
337                 errorFound = true;
338         }
339         if (parts.length < 1 || parts.length > 3)
340             errorFound = true;
341         int result;
342         if (!errorFound) {
343             if (!cleanAge.matches(AGE_REGEX))
344                 warningNeeded = true;
345             int year = Integer.parseInt(yearStr);
346             int month = monthStr.isEmpty() ? 0 : Integer.parseInt(monthStr);
347             if (month < 0 || month > 11)
348                 log("Warning: Illegal month value in age(0< = m< = 11): " + cleanAge);
349             result = 12 * year + month;
350         } else {
351             result = 0;
352             log("Error: uninterpretable age value: " + cleanAge + ". No months attribute computed");
353         }
354         if (warningNeeded)
355             log("Warning: Illegal age syntax for " + cleanAge + ". Syntax must be y;m.d");
356         return result;
357     }

419     private Date normalizeDate(String str) {
420         Date date;
421         try {
422             date = DateUtils.parseDate(str, locale, new String[] { "d-M-Y", "dd-MMM-yyyy" });
423         } catch (ParseException e) {
424             try {
425                 date = DateUtils.parseDate(str, usLocale, new String[] { "d-M-Y", "dd-MMM-yyyy" });
426             } catch (ParseException e1) {
427                 log("Date " + str + " cannot be interpreted");
428                 throw BlackLabRuntimeException.wrap(e1);
429             }
430         }
431         return date;
432     }

525     @SuppressWarnings("unchecked")
526     private void startBlock() {
527         Map<String, String> blockMetadata = new HashMap<>();
528         for (Entry<String, Object> entry : metadata.entrySet()) {
529             String el = entry.getKey();
530             if (DO_NOT_PRINT_IN_HEADERS.contains(el)) {
531                 // (pass)
532             } else if (ALL_HEADERS.contains(el)) {
533                 Object curval = metadata.get(el);
534                 if (curval instanceof Date) {
535                     blockMetadata.put(despaceMetadataName(el), toIsoFormat((Date) curval));
536                 } else if (curval instanceof String || curval instanceof Integer) {
537                     blockMetadata.put(despaceMetadataName(el), curval.toString());
538                 } else {
539                     log("startBlock: unknown type for " + el + " = " + curval);
540                 }
541                 if (!PRINT_IN_HEADERS.contains(el))
542                     log("unknown metadata element encountered: " + el);
543             }
544         }
545         if (metadata.containsKey("uttid"))
546             blockMetadata.put("uttid", metadata.get("uttid").toString());
547         if (metadata.containsKey("origutt"))
548             blockMetadata.put("origutt", metadata.get("origutt").toString());
549         String curcode = "";
550         if (metadata.containsKey("speaker")) {
551             curcode = metadata.get("speaker").toString();
552             blockMetadata.put("speaker", curcode);
553             Map<String, Object> participants = (Map<String, Object>) metadata.get("participants");
554             if (participants != null && participants.containsKey(curcode)) {
555                 Map<String, Object> codeMap = (Map<String, Object>) participants.get(curcode);
556                 for (Entry<String, Object> entry : codeMap.entrySet()) {
557                     String el = entry.getKey();
558                     blockMetadata.put(despaceMetadataName(el), codeMap.get(el).toString());
559                 }
560             }
561         }
562         if (metadata.containsKey("id")) {
563             Map<String, Object> mdid = (Map<String, Object>) metadata.get("id");
564             if (mdid != null) {
565                 Map<String, Object> curcodeMap = (Map<String, Object>) mdid.get(curcode);
566                 if (curcodeMap != null) {
567                     for (Entry<String, Object> entry : curcodeMap.entrySet()) {
568                         String el = entry.getKey();
569                         Object curval = curcodeMap.get(el);
570                         if (curval instanceof Date) {
571                             blockMetadata.put(despaceMetadataName(el), toIsoFormat((Date) curcodeMap.get(el)));
572                         } else if (curval instanceof String || curval instanceof Integer) {
573                             blockMetadata.put(despaceMetadataName(el), curcodeMap.get(el).toString());
574                         } else {
575                             log("startBlock: unknown type for " + el + " = " + curval);
576                         }
577                     }
578                 }
579             }
580         }
581 
582         inlineTag(blockTagName, true, blockMetadata);
583     }

664             } else if (cleanHeaderLine.equals("@end")) {
665                 // (pass)
666             } else if (cleanHeaderLine.equals("@blank")) {
667                 // (pass)
668             } else {
669                 log("Warning: unknown header " + headerLine + " encountered in line " + lineNumber);
670             }
671 
672         } else {
673             String headerName = headerLine.substring(1, headerNameEnd);
674             String entry = headerLine.substring(headerNameEnd + 1);
675             String cleanEntry = entry.trim();
676             List<String> entryList = Arrays.asList(cleanEntry.split(",", -1));
677             String cleanHeaderName = headerName.trim();
678             String cleanHeaderNameBase = slice(cleanHeaderName, 0, -3).trim();
679             String headerParameter = slice(cleanHeaderName, -3);
680             cleanHeaderName = cleanHeaderName.toLowerCase();
681             if (cleanHeaderName.equals("font")) {
682                 // (pass)
683             } else if (cleanHeaderName.equals("languages")) {
684                 metadata.put("languages", entryList);
685             } else if (cleanHeaderName.equals("colorwords")) {
686                 metadata.put("colorwords", entryList);
687             } else if (cleanHeaderName.equals("options")) {
688                 // (pass)
689             } else if (cleanHeaderName.equals("participants")) {
690                 treatParticipants(entryList, metadata);
691             } else if (cleanHeaderName.equals("id")) {
692                 treatId(entry, metadata);
693             } else if (cleanHeaderName.equals("date")) {
694                 metadata.put(cleanHeaderName, normalizeDate(cleanEntry));
695             } else if (SIMPLE_HEADERNAMES.contains(cleanHeaderName)) {
696                 metadata.put(cleanHeaderName, cleanEntry);
697             } else if (SKIP_HEADER_NAMES.contains(cleanHeaderName)) {
698                 // (pass)
699             } else if (SIMPLE_INT_HEADERNAMES.contains(cleanHeaderName)) {
700                 int i;
701                 try {
702                     i = Integer.parseInt(cleanEntry);
703                 } catch (NumberFormatException e) {
704                     log("Warning: couldn't parse integer for header " + cleanHeaderName + ": '" + cleanEntry
705                             + "'. Using -1.");
706                     i = -1;
707                 }
708                 metadata.put(cleanHeaderName, i);
709             } else if (SIMPLE_COUNTER_HEADERS.contains(cleanHeaderName)) {
710                 counter.put(cleanHeaderName, counter.get(cleanHeaderName) + 1);
711                 metadata.put(cleanHeaderName, counter.get(cleanHeaderName));
712             } else if (PARTICIPANT_SPECIFIC_HEADERS.contains(cleanHeaderNameBase)) {
713                 if (!metadata.containsKey("id"))
714                     metadata.put("id", new HashMap<String, Object>());
715                 Map<String, Object> mdid = (Map<String, Object>) metadata.get("id");
716                 if (!mdid.containsKey(headerParameter))
717                     mdid.put(headerParameter, new HashMap<String, String>());
718                 Map<String, Object> hp = (Map<String, Object>) mdid.get(headerParameter);
719                 if (cleanHeaderNameBase.equals("birth of")) {
720                     Date date = normalizeDate(cleanEntry);
721                     hp.put(cleanHeaderNameBase, date);
722                 } else if (cleanHeaderNameBase.equals("age of")) {
723                     // print("<{}>".format(cleanentry), file = logfile)

725                     hp.put("age", cleanEntry);
726                     int months = getMonths(cleanEntry);
727                     if (months != 0)
728                         hp.put("months", months);
729                 } else {
730                     hp.put(cleanHeaderNameBase, cleanEntry);
731                 }
732 
733             } else {
734                 log("Warning: unknown metadata element encountered: " + cleanHeaderName);
735             }
736         }
