16  package nl.inl.blacklab.perdocument;
17  
18  import java.util.ArrayList;
19  import java.util.Collection;
20  import java.util.Collections;
21  import java.util.Comparator;
22  import java.util.HashMap;
23  import java.util.Iterator;
24  import java.util.List;
25  import java.util.Map;
26  
27  import nl.inl.blacklab.search.Searcher;
28  import nl.inl.blacklab.search.grouping.HitPropValue;
29  
30  /**
31   * Applies grouping to the results in a DocResults object.
32   */
33  public class DocGroups implements Iterable<DocGroup> {
34  	Map<HitPropValue, DocGroup> groups = new HashMap<HitPropValue, DocGroup>();
35  
36  	List<DocGroup> orderedGroups = new ArrayList<DocGroup>();
37  
38  	private Searcher searcher;
39  
40  	private int largestGroupSize = 0;
41  
42  	private int totalResults = 0;
43  
44  	private DocProperty groupBy;
45  
46  	/**
47  	 * The DocResults we were created from
48  	 */
49  	private DocResults docResults;
50  
51  	/**
52  	 * Constructor. Fills the groups from the given document results.
53  	 *
54  	 * @param docResults
55  	 *            the results to group.
56  	 * @param groupBy
57  	 *            the criterium to group on.
58  	 * @deprecated use DocResults.groupedBy(docProperty). This constructor
59  	 *      	   will be made package private eventually.
60  	 */
61  	@Deprecated
62  	public DocGroups(DocResults docResults, DocProperty groupBy) {
63  		this.docResults = docResults;
64  		searcher = docResults.getSearcher();
65  		this.groupBy = groupBy;
66  		//Thread currentThread = Thread.currentThread();
67  		Map<HitPropValue, List<DocResult>> groupLists = new HashMap<HitPropValue, List<DocResult>>();
68  		for (DocResult r : docResults) {
69  			HitPropValue groupId = groupBy.get(r);
70  			List<DocResult> group = groupLists.get(groupId);
71  			if (group == null) {
72  				group = new ArrayList<DocResult>();
73  				groupLists.put(groupId, group);
74  			}
75  			group.add(r);
76  			if (group.size() > largestGroupSize)
77  				largestGroupSize = group.size();
78  			totalResults++;
79  		}
80  		for (Map.Entry<HitPropValue, List<DocResult>> e: groupLists.entrySet()) {
81  			DocGroup docGroup = new DocGroup(searcher, e.getKey(), e.getValue());
82  			groups.put(e.getKey(), docGroup);
83  			orderedGroups.add(docGroup);
84  		}
85  	}
86  
87  	public Collection<DocGroup> getGroups() {
88  		return Collections.unmodifiableCollection(orderedGroups);
89  	}
90  
91  	public DocGroup getGroup(HitPropValue groupId) {
92  		return groups.get(groupId);
93  	}
94  
95  	/**
96  	 * Order the groups based on the specified group property.
97  	 *
98  	 * @param prop the property to sort on
99  	 * @param sortReverse if true, perform reverse sort
100 	 * @deprecated renamed to sort()
101 	 */
102 	@Deprecated
103 	public void sortGroups(DocGroupProperty prop, boolean sortReverse) {
104 		sort(prop, sortReverse);
105 	}
106 
107 	/**
108 	 * Order the groups based on the specified group property.
109 	 *
110 	 * @param prop the property to sort on
111 	 * @param sortReverse if true, perform reverse sort
112 	 */
113 	public void sort(DocGroupProperty prop, boolean sortReverse) {
114 		Comparator<DocGroup> comparator = new ComparatorDocGroupProperty(prop, sortReverse,
115 				searcher.getCollator());
116 		Collections.sort(orderedGroups, comparator);
117 	}
118 
119 	public void sort(DocGroupProperty prop) {
120 		sort(prop, false);
121 	}
122 
123 	@Override
124 	public Iterator<DocGroup> iterator() {
125 		return getGroups().iterator();
126 	}
127 
128 	public int numberOfGroups() {
129 		return groups.size();
130 	}
131 
132 	public int getLargestGroupSize() {
133 		return largestGroupSize;
134 	}
135 
136 	public int getTotalResults() {
137 		return totalResults;
138 	}
139 
140 	public DocProperty getGroupCriteria() {
141 		return groupBy;
142 	}
143 
144 	public DocResults getOriginalDocResults() {
145 		return docResults;
146 	}
147 
148 }
