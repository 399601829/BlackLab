16  package nl.inl.blacklab.perdocument;
17  
18  import nl.inl.blacklab.search.grouping.HitPropValueInt;
19  
20  /**
21   * For grouping DocResult objects on the number of hits. This would put documents with 1 hit in a
22   * group, documents with 2 hits in another group, etc.
23   */
24  public class DocPropertyNumberOfHits extends DocProperty {
25  	@Override
26  	public HitPropValueInt get(DocResult result) {
27  		return new HitPropValueInt(result.getNumberOfHits());
28  	}
29  
30  	/**
31  	 * Compares two docs on this property
32  	 * @param a first doc
33  	 * @param b second doc
34  	 * @return 0 if equal, negative if a < b, positive if a > b.
35  	 */
36  	@Override
37  	public int compare(DocResult a, DocResult b) {
38  		if (reverse)
39  			return b.getNumberOfHits() - a.getNumberOfHits();
40  		return a.getNumberOfHits() - b.getNumberOfHits();
41  	}
42  
43  	@Override
44  	public boolean defaultSortDescending() {
45  		return !reverse;
46  	}
47  
48  	@Override
49  	public String getName() {
50  		return "number of hits";
51  	}
52  
53  	public static DocPropertyNumberOfHits deserialize() {
54  		return new DocPropertyNumberOfHits();
55  	}
56  
57  	@Override
58  	public String serialize() {
59  		return serializeReverse() + "numhits";
60  	}
61  
62  }
