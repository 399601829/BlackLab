27  /**
28   * A hit property for grouping on the context of the hit. Requires
29   * HitConcordances as input (so we have the hit text available).
30   */
31  public class HitPropertyRightContext extends HitPropertyContextBase {
32  
33      static HitPropertyRightContext deserializeProp(BlackLabIndex index, AnnotatedField field, String info) {
34          return deserializeProp(HitPropertyRightContext.class, index, field, info);
35      }
36  
37      HitPropertyRightContext(HitPropertyRightContext prop, Results<Hit> hits, Contexts contexts, boolean invert) {
38          super(prop, hits, contexts, invert);
39      }
40  
41      public HitPropertyRightContext(BlackLabIndex index, Annotation annotation, MatchSensitivity sensitivity, ContextSize contextSize) {
42          super("right context", "right", index, annotation, sensitivity, contextSize);
43      }
44  
45      public HitPropertyRightContext(BlackLabIndex index, Annotation annotation, MatchSensitivity sensitivity) {
46          this(index, annotation, sensitivity, null);
47      }
48  
49      public HitPropertyRightContext(BlackLabIndex index, Annotation annotation) {
50          this(index, annotation, null, null);
51      }
52  
53      public HitPropertyRightContext(BlackLabIndex index, MatchSensitivity sensitivity) {
54          this(index, null, sensitivity, null);
55      }
56  
57      public HitPropertyRightContext(BlackLabIndex index) {
58          this(index, null, null, null);
59      }
60  
61      @Override
62      public HitProperty copyWith(Results<Hit> newHits, Contexts contexts, boolean invert) {
63          return new HitPropertyRightContext(this, newHits, contexts, invert);
64      }
65  
66      @Override
67      public PropertyValueContextWords get(Hit result) {
68          int[] context = contexts.get(result);
69          //int contextHitStart = context[Contexts.CONTEXTS_HIT_START_INDEX];
70          int contextRightStart = context[Contexts.RIGHT_START_INDEX];
71          int contextLength = context[Contexts.LENGTH_INDEX];
72  
73          // Copy the desired part of the context
74          int n = contextLength - contextRightStart;
75          if (n <= 0)
76              return new PropertyValueContextWords(index, annotation, sensitivity, new int[0], false);
77          int[] dest = new int[n];
78          int contextStart = contextLength * contextIndices.get(0) + Contexts.NUMBER_OF_BOOKKEEPING_INTS;
79          System.arraycopy(context, contextStart + contextRightStart, dest, 0, n);
80          return new PropertyValueContextWords(index, annotation, sensitivity, dest, false);
81      }
82  
83      @Override
84      public int compare(Hit a, Hit b) {
85          int[] ca = contexts.get(a);
86          int caRightStart = ca[Contexts.RIGHT_START_INDEX];
87          int caLength = ca[Contexts.LENGTH_INDEX];
88          int[] cb = contexts.get(b);
89          int cbRightStart = cb[Contexts.RIGHT_START_INDEX];
90          int cbLength = cb[Contexts.LENGTH_INDEX];
91  
92          // Compare the right context for these two hits
93          int contextIndex = contextIndices.get(0);
94          int ai = caRightStart;
95          int bi = cbRightStart;
96          while (ai < caLength && bi < cbLength) {
97              int cmp = terms.compareSortPosition(
98                      ca[contextIndex * caLength + ai + Contexts.NUMBER_OF_BOOKKEEPING_INTS],
99                      cb[contextIndex * cbLength + bi + Contexts.NUMBER_OF_BOOKKEEPING_INTS], sensitivity);
100             if (cmp != 0)
101                 return reverse ? -cmp : cmp;
102             ai++;
103             bi++;
104         }
105         // One or both ran out, and so far, they're equal.
106         if (ai >= caLength) {
107             if (bi < cbLength) {
108                 // b longer than a => a < b
109                 return reverse ? 1 : -1;
110             }
111             return 0; // same length; a == b
112         }
113         return reverse ? -1 : 1; // a longer than b => a > b
114     }
115 
116 }
