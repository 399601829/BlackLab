36  /**
37   * Abstract base class for a property of a hit, like document title, hit text,
38   * right context, etc.
39   */
40  public abstract class HitProperty implements ResultProperty<Hit> {
41      protected static final Logger logger = LogManager.getLogger(HitProperty.class);
42  
43      public static HitProperty deserialize(Results<Hit> hits, String serialized) {
44          return deserialize(hits.index(), hits.field(), serialized);
45      }
46  
47      /**
48       * Convert the String representation of a HitProperty back into the HitProperty
49       * 
50       * @param index our index
51       * @param field field we're searching
52       * @param serialized the serialized object
53       * @return the HitProperty object, or null if it could not be deserialized
54       */
55      public static HitProperty deserialize(BlackLabIndex index, AnnotatedField field, String serialized) {
56          if (PropertySerializeUtil.isMultiple(serialized)) {
57              boolean reverse = false;
58              if (serialized.startsWith("-(") && serialized.endsWith(")")) {
59                  reverse = true;
60                  serialized = serialized.substring(2, serialized.length() - 1);
61              }
62              HitProperty result = HitPropertyMultiple.deserializeProp(index, field, serialized);
63              if (reverse)
64                  result = result.reverse();
65              return result;
66          }
67      
68          String[] parts = PropertySerializeUtil.splitPartFirstRest(serialized);
69          String type = parts[0].toLowerCase();
70          boolean reverse = false;
71          if (type.length() > 0 && type.charAt(0) == '-') {
72              reverse = true;
73              type = type.substring(1);
74          }
75          String info = parts.length > 1 ? parts[1] : "";
76          HitProperty result;
77          switch (type) {
78          case "decade":
79              result = HitPropertyDocumentDecade.deserializeProp(index, ResultProperty.ignoreSensitivity(info));
80              break;
81          case "docid":
82              result = HitPropertyDocumentId.deserializeProp();
83              break;
84          case "field":
85              result = HitPropertyDocumentStoredField.deserializeProp(index, ResultProperty.ignoreSensitivity(info));
86              break;
87          case "hit":
88              result = HitPropertyHitText.deserializeProp(index, field, info);
89              break;
90          case "left":
91              result = HitPropertyLeftContext.deserializeProp(index, field, info);
92              break;
93          case "right":
94              result = HitPropertyRightContext.deserializeProp(index, field, info);
95              break;
96          case "wordleft":
97              result = HitPropertyWordLeft.deserializeProp(index, field, info);
98              break;
99          case "wordright":
100             result = HitPropertyWordRight.deserializeProp(index, field, info);
101             break;
102         case "context":
103             result = HitPropertyContextWords.deserializeProp(index, field, info);
104             break;
105         case "hitposition":
106             result = HitPropertyHitPosition.deserializeProp();
107             break;
108         case "doc":
109             result = HitPropertyDoc.deserializeProp(index);
110             break;
111             
112         case "fieldlen":
113             throw new BlackLabRuntimeException("Grouping hit results by " + type + " is not yet supported");
114             /*result = HitPropertyDocumentAnnotatedFieldLength.deserialize(ResultProperty.ignoreSensitivity(info));
115             break;*/
116             
117         case "numhits":
118             throw new BlackLabRuntimeException("Cannot group hit results by " + type);
119             
120         default:
121             logger.debug("Unknown HitProperty '" + type + "'");
122             return null;
123         }
124         if (reverse)
125             result = result.reverse();
126         return result;
127     }

129     /** The Hits object we're looking at */
130     protected Results<Hit> hits;
131 
132     /** Reverse comparison result or not? */
133     protected boolean reverse;
134 
135     /** Hit contexts, if any */
136     protected Contexts contexts = null;
137 
138     /**
139      * For HitProperties that need context, the context indices that correspond to
140      * the context(s) they need in the result set. (in the same order as reported by
141      * needsContext()).
142      */
143     List<Integer> contextIndices;
144 
145     public HitProperty() {
146         this.hits = null;
147         this.reverse = sortDescendingByDefault();
148     }
149 
150     /**
151      * Copy a HitProperty, with some optional changes.
152      * 
153      * @param prop property to copy
154      * @param hits new hits to use, or null to inherit
155      * @param contexts new contexts to use, or null to inherit
156      * @param invert true to invert the previous sort order; false to keep it the same
157      */
158     HitProperty(HitProperty prop, Results<Hit> hits, Contexts contexts, boolean invert) {
159         this.hits = hits == null ? prop.hits : hits;
160         this.reverse = prop.reverse;
161         if (invert)
162             this.reverse = !this.reverse;
163         this.setContexts(contexts); // this will initialize contextIndices to default value...
164         if (prop.contextIndices != null)
165             this.contextIndices = prop.contextIndices; // ...but if we already had different values, use those
166     }
167 
168     /**
169      * Is the default for this property to sort descending?
170      * 
171      * This is usually a good default for "group size" or "number of hits".
172      * 
173      * @return whether to sort descending by default
174      */
175     protected boolean sortDescendingByDefault() {
176         return false;
177     }
178 
179     /**
180      * Set contexts to use.
181      * 
182      * Only ever used by constructor, so doesn't break immutability.
183      * 
184      * @param contexts contexts to use, or null for none
185      */
186     protected HitProperty setContexts(Contexts contexts) {
187         if (needsContext() != null) {
188             this.contexts = contexts;
189     
190             // Unless the client sets different context indices, assume we got the ones we wanted in the correct order
191             if (contexts != null)
192                 this.contextIndices = IntStream.range(0, contexts.size()).boxed().collect(Collectors.toList());
193         }
194         return this;
195     }
196 
197     /**
198      * For HitProperties that need context, sets the context indices that correspond
199      * to the context(s) they need in the result set.
200      * 
201      * Only needed if the context indices differ from the assumed default of 0, 1,
202      * 2, ...
203      * 
204      * Only called from the {@link HitPropertyMultiple#copyWith(Hits, Contexts, boolean)}, so doesn't break immutability.
205      * 
206      * @param contextIndices the indices, in the same order as reported by
207      *            needsContext().
208      */
209     protected void setContextIndices(List<Integer> contextIndices) {
210         if (this.contextIndices == null)
211             this.contextIndices = new ArrayList<>();
212         else
213             this.contextIndices.clear();
214         this.contextIndices.addAll(contextIndices);
215     }

217     @Override
218     public abstract PropertyValue get(Hit hit);
219 
220     @Override
221     public int compare(Hit a, Hit b) {
222         PropertyValue hitPropValueA = get(a);
223         PropertyValue hitPropValueB = get(b);
224         return hitPropValueA.compareTo(hitPropValueB);
225     }
226 
227     /**
228      * Retrieve context from which field(s) prior to sorting/grouping on this
229      * property?
230      * 
231      * @return null if no context is required, the fieldnames otherwise
232      */
233     public List<Annotation> needsContext() {
234         return null;
235     }
236 
237     /**
238      * If this property need context(s), how large should they be?
239      * 
240      * @param index index, so we can find the default context size if we need to 
241      * @return required context size
242      */
243     public ContextSize needsContextSize(BlackLabIndex index) {
244         return index.defaultContextSize();
245     }

247     @Override
248     public abstract String name();
249 
250     @Override
251     public abstract String serialize();
252 
253     /**
254      * Used by subclasses to add a dash for reverse when serializing
255      * 
256      * @return either a dash or the empty string
257      */
258     @Override
259     public String serializeReverse() {
260         return reverse ? "-" : "";
261     }
262     
263     @Override
264     public HitProperty reverse() {
265         return copyWith(hits, contexts, true);
266     }
267 
268     /**
269      * Produce a copy of this HitProperty object with a different Hits and Contexts
270      * object.
271      *
272      * @param newHits new Hits to use
273      * @param contexts new Contexts to use, or null for none
274      * @return the new HitProperty object
275      */
276     public HitProperty copyWith(Results<Hit> newHits, Contexts contexts) {
277         return copyWith(newHits, contexts, false);
278     }
279 
280     /**
281      * Produce a copy of this HitProperty object with a different Hits and Contexts
282      * object.
283      *
284      * @param newHits new Hits to use, or null to inherit
285      * @param contexts new Contexts to use, or null to inherit
286      * @param invert true if we should invert the previous sort order; false to keep it the same
287      * @return the new HitProperty object
288      */
289     public abstract HitProperty copyWith(Results<Hit> newHits, Contexts contexts, boolean invert);
290 
291     @Override
292     public boolean isReverse() {
293         return reverse;
294     }
295 
296     @Override
297     public String toString() {
298         return serialize();
299     }
300 
301     @Override
302     public int hashCode() {
303         final int prime = 31;
304         int result = 1;
305         result = prime * result + (reverse ? 1231 : 1237);
306         return result;
307     }
308 
309     @Override
310     public boolean equals(Object obj) {
311         if (this == obj)
312             return true;
313         if (obj == null)
314             return false;
315         if (getClass() != obj.getClass())
316             return false;
317         HitProperty other = (HitProperty) obj;
318         if (reverse != other.reverse)
319             return false;
320         return true;
321     }
