4   package nl.inl.util;
5   
6   import java.util.ArrayList;
7   import java.util.List;
8   
9   /**
10   * Keeps a simple pool of free resource objects.
11   *
12   * This is useful when it's relatively expensive to create an object, and we often
13   * need one for a short time.
14   *
15   * Note that this is a very basic resource pool.
16   * @param <T> resource type to pool
17   */
18  public abstract class SimpleResourcePool<T> {
19  	/**
20  	 * The maximum number of unused resource objects in the free pool at any one time. Once the
21  	 * pool is full, additional resource objects are dropped to be garbage-collected instead of
22  	 * returned to the pool.
23  	 */
24  	private int maxFreePoolSize;
25  
26  	/**
27  	 * Switch to disable pooling.
28  	 */
29  	private boolean neverPool;
30  
31  	/** List of free resource objects */
32  	private List<T> freePool;
33  
34  	/**
35  	 * Initializes the free pool.
36  	 *
37  	 * @param maxFreePoolSize
38  	 *            size of the resource pool. If 0, disables pooling.
39  	 */
40  	public SimpleResourcePool(int maxFreePoolSize) {
41  		this.maxFreePoolSize = maxFreePoolSize;
42  		freePool = new ArrayList<>();
43  		neverPool = (maxFreePoolSize == 0);
44  	}
45  
46  	/**
47  	 * Cleanup the resources in the free pool.
48  	 */
49  	public void close() {
50  		clear();
51  	}
52  
53  	/**
54  	 * Destroy all objects in the free pool.
55  	 */
56  	public synchronized void clear() {
57  		for (T resource: freePool) {
58  			destroyResource(resource);
59  		}
60  	}
61  
62  	/**
63  	 * Create a new resource object.
64  	 * @return the new resource.
65  	 */
66  	public abstract T createResource();
67  
68  	/**
69  	 * Destroy a resource object.
70  	 * @param resource the resource to destroy.
71  	 */
72  	public void destroyResource(T resource) {
73  		// Default: do nothing
74  	}
75  
76  	/**
77  	 * Sets whether or not we want to do any pooling at all.
78  	 * If not, always creates a new resource object and never returns
79  	 * it to the free pool.
80  	 *
81  	 * @param poolingEnabled if true, pool free resources.
82  	 */
83  	public synchronized void setPoolingEnabled(boolean poolingEnabled) {
84  		this.neverPool = !poolingEnabled;
85  	}
86  
87  	/**
88  	 * Retrieves a resource object from the free pool, or creates a new one if none are available
89  	 *
90  	 * @return the resource object
91  	 */
92  	public synchronized T acquire() {
93  		try {
94  			if (neverPool || freePool.size() == 0) {
95  				return createResource();
96  			}
97  			return freePool.remove(0);
98  		} catch (Exception e) {
99  			throw ExUtil.wrapRuntimeException(e);
100 		}
101 	}
102 
103 	/**
104 	 * Returns a resource object to the free pool.
105 	 *
106 	 * @param resource
107 	 */
108 	public synchronized void release(T resource) {
109 		if (neverPool)
110 			return;
111 		synchronized (this) {
112 			if (freePool.size() < maxFreePoolSize) {
113 				freePool.add(resource);
114 			} else {
115 				destroyResource(resource);
116 			}
117 		}
118 	}
119 
120 }
