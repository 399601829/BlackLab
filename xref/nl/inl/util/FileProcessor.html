24   * Process (trees of) files, which may include archives that we want to

30      public interface FileHandler {
31          /**
32           * Handle a directory.
33           * <p>
34           * Called for all processed child (and descendant if
35           * {@link FileProcessor#isRecurseSubdirs()} directories of the input file,
36           * excluding the input directory itself. NOTE: This is only called for regular
37           * directories, and not for archives or processed directories within archives.
38           * NOTE: {@link FileProcessor#pattGlob} is NOT applied to directories. So the
39           * directory names may not match the provided pattern.
40           * <p>
41           * This function may be called in multiple threads when FileProcessor was
42           * created with thread support (see
43           * {@link FileProcessor#FileProcessor(boolean, boolean, boolean)})
44           *
45           * @param dir the directory
46           * @throws Exception these will be passed to
47           *             {@link ErrorHandler#errorOccurred(Throwable, String, File)}
48           */
49          void directory(File dir) throws Exception;
50  
51          /**
52           * Handle a file stream.
53           * <p>
54           * Called for all processed files that match the {@link FileProcessor#pattGlob},
55           * including the input file. Not called for archives if
56           * {@link FileProcessor#isProcessArchives()} is true (though it will then be
57           * called for files within those archives).
58           * <p>
59           * This function may be called in multiple threads when FileProcessor was
60           * created with thread support (see
61           * {@link FileProcessor#FileProcessor(boolean, boolean, boolean)}) <br>
62           * NOTE: the InputStream should be closed by the implementation.
63           *
64           * @param path filename, including path inside archives (if the file is within
65           *            an archive)
66           * @param is
67           * @param file (optional, if known) the file from which the InputStream was
68           *            built, or - if the InputStream is a file within an archive - the
69           *            archive.
70           * @throws Exception these will be passed to
71           *             {@link ErrorHandler#errorOccurred(Throwable, String, File)}
72           */
73          void file(String path, InputStream is, File file) throws Exception;
74  
75          // Regular file(File f) function is omitted on purpose.
76          // As we process regular files as well as "virtual" files (entries in archives and the like) in the same manner.
77          // This means in some cases there is no actual file backing up the data
78      }

121     public static byte[] fetchFileFromArchive(File f, final String pathInsideArchive) {
122         if (f.getName().endsWith(".gz") || f.getName().endsWith(".tgz")) {
123             // We have to process the whole file, we can't do random access.
124             PathCapturingFileHandler fileCapturer = new PathCapturingFileHandler() {
125                 byte[] file;
126 
127                 @Override
128                 public void directory(File dir) throws Exception {
129                     //
130                 }
131 
132                 @Override
133                 public void file(String path, InputStream is, File archive) throws Exception {
134                     if (path.endsWith(pathInsideArchive))
135                         this.file = IOUtils.toByteArray(is);
136                 }
137 
138                 @Override
139                 public byte[] getFile() {
140                     return file;
141                 }
142             };
143             try (FileProcessor proc = new FileProcessor(1, false, true)) {
144                 proc.setFileHandler(fileCapturer);
145                 proc.processFile(f, "");
146             } catch (FileNotFoundException e) {
147                 throw BlackLabRuntimeException.wrap(e);
148             }
149 
150             // FileProcessor must have completed/be closed before result is available
151             return fileCapturer.getFile();
152         } else if (f.getName().endsWith(".zip")) {
153             // We can do random access. Fetch the file we want.
154             try {
155                 ZipFile z = ZipHandleManager.openZip(f);
156                 ZipEntry e = z.getEntry(pathInsideArchive);
157                 if (e == null) {
158                     throw new BlackLabRuntimeException("Linked document " + pathInsideArchive + " not found in archive " + f);
159                 }
160                 try (InputStream is = z.getInputStream(e)) {
161                     return IOUtils.toByteArray(is);
162                 }
163             } catch (IOException e) {
164                 throw BlackLabRuntimeException.wrap(e);
165             }
166         } else {
167             throw new UnsupportedOperationException("Unsupported archive type: " + f.getName());
168         }
169     }

204     /**
205      * FileProcessor operates in two distinct stages: - The traversal of
206      * directories/archives, this is done on the "main" thread (i.e. the thread that
207      * initially called processFile/processInputStream) - Handling of all
208      * files/entries, this is usually done asynchronously by our Handler.
209      *
210      * If an exception occurs in the handling stage, we want to stop all ongoing and
211      * queued handlers, but also stop the the main thread if it's still busy
212      * traversing and creating more handlers. The problem is that the main can't
213      * directly act on exceptions thrown in handlers, as the exception is thrown
214      * asynchronously.
215      *
216      * So we need a way to signal the main thread to cease all work: - aborting all
217      * handlers/tasks is easy, we can shut down the ExcecutorService directly from
218      * the handler thread when the exception occurs. - aborting the main thread will
219      * require setting some flag and some manual checking on its part we could call
220      * Thread.interrupt() on the main thread, but this would require the handlers to
221      * keep a reference to the main thread so instead just use this flag that the
222      * main thread checks while it's performing work.
223      */
224     private volatile boolean closed = false; // TODO: should this be AtomicBoolean?
225 
226     /**
227      * Separate from closed to allow aborting even while already closed or closing
228      * This happens when an error occurs while processing remainder of queue, it's
229      * also useful to allow aborting when closing unexpectedly takes a long time.
230      */
231     private boolean aborted = false;

233     public FileProcessor(final int numberOfThreadsToUse, boolean recurseSubdirs, boolean processArchives) {
234         this.recurseSubdirs = recurseSubdirs;
235         this.processArchives = processArchives;
236         setFileNameGlob("*");
237 
238         // We always use an ExecutorService to call our handlers to simplify our code
239         // When not using threads, the service is just a fancy wrapper around doing task.run() directly inside the calling thread.
240         if (numberOfThreadsToUse > 1) {
241             // NOTE: we need to create our own executor instead of using Executors.newFixedThreadPool()
242             // Because that implementation has an unbounded job queue by default, we run the risk that we queue 50k jobs and eat up all memory
243             // So we provide our own queue that will block when it's full.
244             // We could just provide a default LinkedBlockingDeque, but the default behavior for the executor is to reject all jobs while the queue is full.
245             // To get around this, override the queue.offer() function used internally by the executor to queue jobs,
246             // and make it blocking (instead of returning false instantly, which would make the executor reject the job)
247             int cpuCores = Runtime.getRuntime().availableProcessors();
248             int actualThreadsToUse = Math.max(1, Math.min(cpuCores - 1, numberOfThreadsToUse)); // no more than (cores-1), but at least 1
249             executor = new ThreadPoolExecutor(actualThreadsToUse, actualThreadsToUse, Integer.MAX_VALUE, TimeUnit.DAYS,
250                 new LinkedBlockingDeque<Runnable>(50) {
251                     @Override
252                     public boolean offer(Runnable r) {
253                         try {
254                             return offer(r, Integer.MAX_VALUE, TimeUnit.DAYS);
255                         } catch (InterruptedException e) {
256                             return false;
257                         }
258                     }
259                 }
260             );
261 
262             // Never throw RejectedExecutionException in the main thread
263             // (this can rarely happen when the FileProcessor shut down from another thread (usually a task thread that encountered an exception?)
264             // just in between checking state and submitting)
265             ((ThreadPoolExecutor) executor).setRejectedExecutionHandler(new ThreadPoolExecutor.DiscardPolicy());
266         } else {
267             executor = new MainThreadExecutorService((r, e) -> {
268                 /* swallow RejectedExecutionExceptions, same as above. */ });
269         }
270     }

346     /**
347      * Process a file or directory.
348      *
349      * If this file is a directory, all child files will be processed, files within
350      * subdirectories will only be processed if {@link #isRecurseSubdirs()} is true.
351      * For rules on how files are processed, regarding archives etc, see
352      * {@link #processInputStream(String, InputStream, File)}.
353      *
354      * @param file file, directory or archive to process
355      * @param pathSoFar path so far, so we can report the path to the file
356      * @throws FileNotFoundException
357      */
358     public void processFile(File file, String pathSoFar) throws FileNotFoundException {
359         if (!file.exists())
360             throw new FileNotFoundException("Input file or dir not found: " + file);
361 
362         if (closed)
363             return;
364 
365         if (file.isDirectory()) { // Even if recurseSubdirs is false, we should process all direct children
366             pathSoFar += "/" + file.getName();
367             for (File childFile : FileUtil.listFilesSorted(file)) {
368                 if (closed)
369                     return;
370 
371                 // Report
372                 if (childFile.isDirectory()) {
373                     CompletableFuture.runAsync(makeRunnable(() -> fileHandler.directory(childFile)), executor)
374                             .exceptionally(e -> reportAndAbort(e, childFile.toString(), childFile));
375                 }
376 
377                 if (recurseSubdirs || !childFile.isDirectory())
378                     processFile(childFile, pathSoFar);
379             }
380         } else {
381             processInputStream(pathSoFar + "/" + file.getName(), new FileInputStream(file), file);
382         }
383     }

385     /**
386      * Process from an InputStream, which may be an archive or a regular file.
387      *
388      * Archives (.zip and .tar.gz) will only be processed if
389      * {@link #isProcessArchives()} is true. GZipped files (.gz) will be unpacked
390      * regardless. Note that all files within archives will be processed, regardless
391      * of whether they match {@link FileProcessor#pattGlob}
392      *
393      * @param path filename, optionally including path to the file or path within an
394      *            archive
395      * @param is the stream
396      * @param file (optional) the file from which the InputStream was built, or - if
397      *            the InputStream is a file within an archive - the archive. This is
398      *            only used for reporting to FileHandler and ErrorHandler
399      */
400     public void processInputStream(String path, InputStream is, File file) {
401         if (closed)
402             return;
403 
404         TarGzipReader.FileHandler handler = (pathInArchive, streamInArchive) -> {
405             processInputStream(pathInArchive, streamInArchive, file);
406             return !closed; // quit processing the archive if we've received an error in the meantime
407         };
408 
409         if (isProcessArchives() && path.endsWith(".tar.gz") || path.endsWith(".tgz")) {
410             TarGzipReader.processTarGzip(path, is, handler);
411         } else if (isProcessArchives() && path.endsWith(".zip")) {
412             TarGzipReader.processZip(path, is, handler);
413         } else if (path.endsWith(".gz")) {
414             TarGzipReader.processGzip(path, is, handler);
415         } else if (!skipFile(path) && getFileNamePattern().matcher(path).matches()) {
416             CompletableFuture.runAsync(makeRunnable(() -> fileHandler.file(path, is, file)), executor)
417                     .exceptionally(e -> reportAndAbort(e, path, file));
418         }
419     }

421     /**
422      * Callback for when handler throws an exception. Report it, and if it's
423      * irrecoverable, abort.
424      * {@link ErrorHandler#errorOccurred(Throwable, String, File)}
425      *
426      * @param e
427      * @param path
428      * @param f
429      * @return always null, has return type to enable use as exception handler in
430      *         CompletableFuture
431      */
432     private synchronized Void reportAndAbort(Throwable e, String path, File f) {
433         if (e instanceof CompletionException) // async exception
434             e = e.getCause();
435 
436         // Only report the first fatal exception
437         if (!aborted) {
438             if (errorHandler == null) {
439                 System.err.println("WARNING: No errorHandler set for FileProcessor!");
440                 e.printStackTrace(System.err);
441             }
442             if (errorHandler == null || !errorHandler.errorOccurred(e, path, f)) {
443                 abort();
444             }
445         }
446 
447         return null;
448     }

450     /**
451      * Like {@link FileProcessor#close()} but immediately abort all running handler
452      * tasks and cancel any pending tasks.
453      *
454      * Subsequent calls to close, processFile or processInputStream will have no
455      * effect.
456      */
457     // this function can't be synchronized on (this) or we couldn't abort from an async handler while the main thread is working/waiting on close().
458     public void abort() {
459         synchronized (this) {
460             if (aborted)
461                 return;
462             closed = true;
463             aborted = true;
464         }
465 
466         executor.shutdownNow();
467     }
468 
469     /**
470      * Close the executor and wait until all running and pending handler tasks have
471      * completed. Calling close() while processFile or processInputStream is in
472      * progress will cause them to skip all remaining files. Files for which a task
473      * has already been put in the queue will still be processed as normal.
474      *
475      * Subsequent calls to close, processFile or processInputStream will have no
476      * effect.
477      */
478     @Override
479     public void close() {
480         synchronized (this) {
481             if (closed)
482                 return;
483             closed = true;
484         }
485 
486         try {
487             executor.shutdown();
488             // Outside the synchronized block to allow calling abort() while waiting for close() to complete
489             // This is used by tasks that threw a fatal exception
490             executor.awaitTermination(Long.MAX_VALUE, TimeUnit.NANOSECONDS);
491         } catch (InterruptedException e) {
492             throw new BlackLabRuntimeException("Interrupted while waiting for processing threads to finish", e);
493         }
494     }

496     /*
497      * Bit of boilerplate to allow using submitting tasks that throw checked exceptions with CompletableFuture.
498      * Wrap the task in a runnable that catches the checked exception and rethrows it in an unchecked manner.
499      * The exception is then caught in the future and made available (using for example CompletableFuture::completeExceptionally)
500      *
501      * see https://blog.jooq.org/2012/09/14/throw-checked-exceptions-like-runtime-exceptions-in-java/
502      */
503     @SuppressWarnings("unchecked")
504     private static <T extends Exception> void rethrowUnchecked(Exception t) throws T {
505         throw (T) t;
506     }

508     @FunctionalInterface
509     private interface ThrowingRunnable<E extends Throwable> {
510         void call() throws E;
511     }
512 
513     private static <E extends Exception> Runnable makeRunnable(ThrowingRunnable<E> c) {
514         return () -> {
515             try {
516                 c.call();
517             } catch (Exception e) {
518                 rethrowUnchecked(e);
519             }
520         };
521     }
