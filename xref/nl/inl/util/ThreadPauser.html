1   package nl.inl.util;
2   
3   import nl.inl.blacklab.search.Pausible;
4   
5   public interface ThreadPauser extends Pausible {
6   
7       static ThreadPauser create() {
8           return new ThreadPauserImpl();
9       }
10  
11      @Override
12      void pause(boolean paused);
13  
14      @Override
15      boolean isPaused();
16  
17      /**
18       * If the thread we're controlling is supposed to be paused, wait until it is 
19       * unpaused or it's interrupted.
20       *
21       * @throws InterruptedException if thread was interrupted from elsewhere (e.g. load manager)
22       */
23      void waitIfPaused() throws InterruptedException;
24  
25      /**
26           * How long has this job been paused for currently?
27          *
28          * This does not include any previous pauses.
29          *
30          * @return number of ms since the job was paused, or 0 if not paused
31          */
32      long currentPauseLength();
33  
34      /**
35          * How long has this job been running currently?
36          *
37          * This does not include any previous running phases.
38          *
39          * @return number of ms since the job was set to running, or 0 if not running
40          */
41      long currentRunPhaseLength();
42  
43      /**
44          * How long has this job been paused in total?
45          *
46          * @return total number of ms the job has been paused
47          */
48      long pausedTotal();
49  
50  }
