16  package nl.inl.util;
17  
18  import java.text.Normalizer;
19  import java.util.regex.Matcher;
20  import java.util.regex.Pattern;
21  
22  import org.apache.commons.lang3.StringUtils;
23  
24  /**
25   * A collection of String-related utility methods and regular expression
26   * patterns.
27   */
28  public final class StringUtil {
29      /** nbsp character (decimal 160 = hex A0) */
30      public static final char CHAR_NON_BREAKING_SPACE = '\u00A0';
31  
32      /** Matches whitespace. */
33      private static final Pattern PATT_WHITESPACE = Pattern.compile("\\s+");
34  
35      /** Whitespace and/or punctuation at end */
36      private static final Pattern PATT_WS_PUNCT_AT_END = Pattern.compile("[\\p{P}\\s]+$");
37  
38      /** Whitespace and/or punctuation at start */
39      private static final Pattern PATT_WS_PUNCT_AT_START = Pattern.compile("^[\\p{P}\\s]+");
40  
41      private static final Pattern PATT_REGEX_CHARACTERS = Pattern.compile("([\\|\\\\\\?\\*\\+\\(\\)\\[\\]\\-\\^\\$\\{\\}\\.])");
42  
43      private static final Pattern PATT_DIACRITICAL_MARKS = Pattern.compile("\\p{InCombiningDiacriticalMarks}+");
44  
45      private StringUtil() {
46      }
47  
48      /**
49       * Escape regex special characters
50       * 
51       * (Pattern.quote() also does this, but this method is needed if you use a different regex
52       *  engine from Java's, such as with Lucene)
53       *
54       * @param termStr the string to escape characters in
55       * @return the escaped string
56       */
57      public static String escapeRegexCharacters(String termStr) {
58          Matcher m = PATT_REGEX_CHARACTERS.matcher(termStr);
59          termStr = m.replaceAll("\\\\$1");
60          return termStr;
61      }
62  
63      /**
64       * Replace adjacent whitespace characters with a single space
65       *
66       * @param s source string
67       * @return the result
68       */
69      public static String normalizeWhitespace(String s) {
70          Matcher m = PATT_WHITESPACE.matcher(s);
71          return m.replaceAll(" ");
72      }
73  
74      /**
75       * <p>
76       * Removes diacritics (~= accents) from a string. The case will not be altered.
77       * </p>
78       * <p>
79       * For instance, '&agrave;' will be replaced by 'a'.
80       * </p>
81       * <p>
82       * Note that ligatures will be left as is.
83       * </p>
84       *
85       * <pre>
86       * StringUtils.stripAccents(null)                = null
87       * StringUtils.stripAccents("")                  = ""
88       * StringUtils.stripAccents("control")           = "control"
89       * StringUtils.stripAccents("&eacute;clair")     = "eclair"
90       * </pre>
91       *
92       * NOTE: this method was copied from Apache StringUtils. The only change is
93       * precompiling the regular expression for efficiency.
94       *
95       * @param input String to be stripped
96       * @return input text with diacritics removed
97       *
98       * @since 3.0
99       */
100     // See also Lucene's ASCIIFoldingFilter (Lucene 2.9) that replaces accented characters by their unaccented equivalent (and uncommitted bug fix: https://issues.apache.org/jira/browse/LUCENE-1343?focusedCommentId=12858907&page=com.atlassian.jira.plugin.system.issuetabpanels%3Acomment-tabpanel#action_12858907).
101     public static String stripAccents(final String input) {
102         if (input == null) {
103             return null;
104         }
105         final StringBuilder decomposed = new StringBuilder(Normalizer.normalize(input, Normalizer.Form.NFD));
106         convertRemainingAccentCharacters(decomposed);
107         // Note that this doesn't correctly remove ligatures...
108         return PATT_DIACRITICAL_MARKS.matcher(decomposed).replaceAll(StringUtils.EMPTY);
109     }
110 
111     private static void convertRemainingAccentCharacters(StringBuilder decomposed) {
112         for (int i = 0; i < decomposed.length(); i++) {
113             if (decomposed.charAt(i) == '\u0141') {
114                 decomposed.deleteCharAt(i);
115                 decomposed.insert(i, 'L');
116             } else if (decomposed.charAt(i) == '\u0142') {
117                 decomposed.deleteCharAt(i);
118                 decomposed.insert(i, 'l');
119             }
120         }
121     }
122 
123     /**
124      * Remove any punctuation and whitespace at the start and end of input.
125      *
126      * @param input the input string
127      * @return the string without punctuation or whitespace at the edges.
128      */
129     public static String trimWhitespaceAndPunctuation(String input) {
130         input = PATT_WS_PUNCT_AT_END.matcher(input).replaceAll("");
131         input = PATT_WS_PUNCT_AT_START.matcher(input).replaceAll("");
132         return input;
133     }

135     /**
136      * A lowercase letter followed by an uppercase one, both matched in groups.
137      */
138     static Pattern lcaseUcase = Pattern.compile("(\\p{Ll})(\\p{Lu})");
139 
140 
141     /**
142      * Convert a string from a camel-case "identifier" style to a human-readable
143      * version, by putting spaces between words, uppercasing the first letter and
144      * lowercasing the rest.
145      *
146      * E.g. myCamelCaseString becomes "My camel case string".
147      *
148      * @param camelCaseString a string in camel case, i.e. multiple capitalized
149      *            words glued together.
150      * @param dashesToSpaces if true, also converts dashes and underscores to spaces
151      * @return a human-readable version of the input string
152      */
153     public static String camelCaseToDisplayable(String camelCaseString, boolean dashesToSpaces) {
154         String spaceified = camelCaseString;
155         spaceified = lcaseUcase.matcher(spaceified).replaceAll("$1 $2");
156         if (dashesToSpaces)
157             spaceified = spaceified.replaceAll("[\\-_]", " ");
158         return StringUtils.capitalize(spaceified.toLowerCase());
159     }
160 
161     /**
162      * For a number n, return a string like "nth".
163      *
164      * So 1 returns "1st", 2 returns "2nd", and so on.
165      *
166      * @param docNumber number
167      * @return ordinal for that number
168      */
169     public static String ordinal(int docNumber) {
170         final String[] ordSuffix = { "th", "st", "nd", "rd" };
171         int i = docNumber;
172         if (i < 1 || i > 3)
173             i = 0;
174         return docNumber + ordSuffix[i];
175     }
176 
177     /**
178      * Convert wildcard string to regex string.
179      *
180      * Adapted from: http://stackoverflow.com/a/28758377
181      *
182      * @param wildcard wildcard pattern
183      * @return equivalent regex pattern
184      */
185     public static String wildcardToRegex(String wildcard) {
186         StringBuffer s = new StringBuffer(wildcard.length());
187         s.append('^');
188         for (int i = 0, is = wildcard.length(); i < is; i++) {
189             char c = wildcard.charAt(i);
190             switch (c) {
191             case '*':
192                 s.append(".*");
193                 break;
194             case '?':
195                 s.append(".");
196                 break;
197             // escape special regexp-characters
198             case '^': // escape character in cmd.exe
199             case '(':
200             case ')':
201             case '[':
202             case ']':
203             case '$':
204             case '.':
205             case '{':
206             case '}':
207             case '|':
208             case '\\':
209                 s.append("\\");
210                 s.append(c);
211                 break;
212             default:
213                 s.append(c);
214                 break;
215             }
216         }
217         s.append('$');
218         return s.toString();
219     }
